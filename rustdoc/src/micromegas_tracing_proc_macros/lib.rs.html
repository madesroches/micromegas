1//! `log_fn` and `span_fn` procedural macros
2//!
3//! Injects instrumentation into sync and async functions.
4//! `span_fn` supports sync functions, async functions, and async trait methods automatically.
5
6// crate-specific lint exceptions:
7//#![allow()]
8
9use proc_macro::TokenStream;
10use proc_macro2::Literal;
11use quote::quote;
12use syn::{
13    ItemFn, ReturnType, Type, TypePath,
14    parse::{Parse, ParseStream, Result},
15    parse_macro_input, parse_quote,
16};
17
18struct TraceArgs {
19    alternative_name: Option<Literal>,
20}
21
22impl Parse for TraceArgs {
23    fn parse(input: ParseStream<'_>) -> Result<Self> {
24        if input.is_empty() {
25            Ok(Self {
26                alternative_name: None,
27            })
28        } else {
29            Ok(Self {
30                alternative_name: Some(Literal::parse(input)?),
31            })
32        }
33    }
34}

36/// Check if the function returns a Future (indicating it's an async trait method)
37fn returns_future(function: &ItemFn) -> bool {
38    match &function.sig.output {
39        ReturnType::Type(_, ty) => is_future_type(ty),
40        ReturnType::Default => false,
41    }
42}
43
44/// Check if a type is a Future type (Pin<Box<dyn Future>> or impl Future)
45fn is_future_type(ty: &Type) -> bool {
46    match ty {
47        // Check for Pin<Box<dyn Future<...>>>
48        Type::Path(TypePath { path, .. }) => {
49            if let Some(last_segment) = path.segments.last()
50                && last_segment.ident == "Pin"
51            {
52                // This is the pattern async-trait generates
53                return true;
54            }
55            false
56        }
57        // Check for impl Future<...>
58        Type::ImplTrait(impl_trait) => impl_trait.bounds.iter().any(|bound| {
59            if let syn::TypeParamBound::Trait(trait_bound) = bound
60                && let Some(segment) = trait_bound.path.segments.last()
61            {
62                return segment.ident == "Future";
63            }
64            false
65        }),
66        _ => false,
67    }
68}

70/// span_fn: trace the execution of sync functions, async functions, and async trait methods
71#[proc_macro_attribute]
72pub fn span_fn(args: TokenStream, input: TokenStream) -> TokenStream {
73    let args = parse_macro_input!(args as TraceArgs);
74    let mut function = parse_macro_input!(input as ItemFn);
75
76    let function_name = args
77        .alternative_name
78        .map_or(function.sig.ident.to_string(), |n| n.to_string());
79
80    if returns_future(&function) {
81        // Case 1: Async trait method (after #[async_trait] transformation)
82        // Function returns Pin<Box<dyn Future<Output = T>>> and has no async keyword
83        let stmts = &function.block.stmts;
84
85        // Extract and instrument the async block from Box::pin(async move { ... })
86        if stmts.len() == 1
87            && let syn::Stmt::Expr(syn::Expr::Call(call_expr)) = &stmts[0]
88            && call_expr.args.len() == 1
89        {
90            let async_block = &call_expr.args[0];
91
92            // Replace the function body with instrumented version
93            function.block = parse_quote! {
94                {
95                    static_span_desc!(_SCOPE_DESC, concat!(module_path!(), "::", #function_name));
96                    Box::pin(InstrumentedFuture::new(
97                        #async_block,
98                        &_SCOPE_DESC
99                    ))
100                }
101            };
102        } else {
103            // For complex async functions that don't match the simple Box::pin pattern,
104            // wrap the entire body in an async block and instrument it
105            let original_block = &function.block;
106            function.block = parse_quote! {
107                {
108                    static_span_desc!(_SCOPE_DESC, concat!(module_path!(), "::", #function_name));
109                    Box::pin(InstrumentedFuture::new(
110                        async move #original_block,
111                        &_SCOPE_DESC
112                    ))
113                }
114            };
115        }
116    } else if function.sig.asyncness.is_some() {
117        // Case 2: Regular async function
118        let original_block = &function.block;
119        let output_type = match &function.sig.output {
120            syn::ReturnType::Type(_, ty) => quote! { #ty },
121            syn::ReturnType::Default => quote! { () },
122        };
123
124        // Remove async and change return type to impl Future
125        function.sig.asyncness = None;
126        function.sig.output = parse_quote! { -> impl std::future::Future<Output = #output_type> };
127        function.block = parse_quote! {
128            {
129                static_span_desc!(_SCOPE_DESC, concat!(module_path!(), "::", #function_name));
130                let fut = async move #original_block;
131                InstrumentedFuture::new(fut, &_SCOPE_DESC)
132            }
133        };
134    } else {
135        // Case 3: Regular sync function
136        function.block.stmts.insert(
137            0,
138            parse_quote! {
139                span_scope!(_METADATA_FUNC, concat!(module_path!(), "::", #function_name));
140            },
141        );
142    }
143
144    TokenStream::from(quote! {
145        #function
146    })
147}

149/// log_fn: log the execution of a function
150#[proc_macro_attribute]
151pub fn log_fn(args: TokenStream, input: TokenStream) -> TokenStream {
152    assert!(args.is_empty());
153    let mut function = parse_macro_input!(input as ItemFn);
154    let function_name = function.sig.ident.to_string();
155
156    function.block.stmts.insert(
157        0,
158        parse_quote! {
159            trace!(#function_name);
160        },
161    );
162    TokenStream::from(quote! {
163        #function
164    })
165}
