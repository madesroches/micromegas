16/// Fetch JWKS from the OIDC provider using openidconnect's built-in discovery
17async fn fetch_jwks(issuer_url: &IssuerUrl) -> Result<Arc<CoreJsonWebKeySet>> {
18    // Create HTTP client with SSRF protection (no redirects)
19    let http_client = reqwest::ClientBuilder::new()
20        .redirect(reqwest::redirect::Policy::none())
21        .build()
22        .map_err(|e| anyhow!("Failed to create HTTP client: {e:?}"))?;
23
24    // Use openidconnect's built-in OIDC discovery
25    let metadata = CoreProviderMetadata::discover_async(issuer_url.clone(), &http_client)
26        .await
27        .map_err(|e| {
28            anyhow!(
29                "Failed to discover OIDC metadata from {}: {e:?}",
30                issuer_url
31            )
32        })?;
33
34    // Fetch JWKS from jwks_uri
35    let jwks_uri = metadata.jwks_uri();
36    let jwks: CoreJsonWebKeySet = http_client
37        .get(jwks_uri.url().as_str())
38        .send()
39        .await
40        .map_err(|e| anyhow!("Failed to fetch JWKS from {}: {e:?}", jwks_uri))?
41        .json()
42        .await
43        .map_err(|e| anyhow!("Failed to parse JWKS: {e:?}"))?;
44
45    Ok(Arc::new(jwks))
46}

48/// JWKS cache for an OIDC issuer
49///
50/// Caches JSON Web Key Sets with automatic TTL expiration.
51/// Uses moka for thread-safe caching with atomic cache miss handling.
52struct JwksCache {
53    issuer_url: IssuerUrl,
54    cache: Cache<String, Arc<CoreJsonWebKeySet>>,
55}
56
57impl JwksCache {
58    /// Create a new JWKS cache
59    fn new(issuer_url: IssuerUrl, ttl: Duration) -> Self {
60        let cache = Cache::builder().time_to_live(ttl).build();
61
62        Self { issuer_url, cache }
63    }
64
65    /// Get the JWKS, fetching from the issuer if not cached
66    async fn get(&self) -> Result<Arc<CoreJsonWebKeySet>> {
67        let issuer_url = self.issuer_url.clone();
68
69        self.cache
70            .try_get_with(
71                "jwks".to_string(),
72                async move { fetch_jwks(&issuer_url).await },
73            )
74            .await
75            .map_err(|e| anyhow!("Failed to fetch JWKS: {e:?}"))
76    }
77}

128/// Audience can be either a string or an array of strings in OIDC tokens
129#[derive(Debug, Serialize, Deserialize)]
130#[serde(untagged)]
131enum Audience {
132    Single(String),
133    Multiple(Vec<String>),
134}
135
136impl Audience {
137    fn contains(&self, aud: &str) -> bool {
138        match self {
139            Audience::Single(s) => s == aud,
140            Audience::Multiple(v) => v.iter().any(|a| a == aud),
141        }
142    }

145/// JWT Claims from OIDC ID token
146#[derive(Debug, Serialize, Deserialize)]
147struct Claims {
148    /// Issuer - identifies the principal that issued the JWT
149    iss: String,
150    /// Subject - identifies the principal that is the subject of the JWT
151    sub: String,
152    /// Audience - identifies the recipients that the JWT is intended for
153    /// Can be either a single string or an array of strings
154    aud: Audience,
155    /// Expiration time - identifies the expiration time on or after which the JWT must not be accepted
156    exp: i64,
157    /// Email address of the user (optional, provider-specific)
158    #[serde(skip_serializing_if = "Option::is_none")]
159    email: Option<String>,
160}

190/// Convert a JWK to a DecodingKey for jsonwebtoken
191fn jwk_to_decoding_key(
192    jwk: &openidconnect::core::CoreJsonWebKey,
193) -> Result<jsonwebtoken::DecodingKey> {
194    // Serialize the JWK to JSON to extract parameters
195    let jwk_json =
196        serde_json::to_value(jwk).map_err(|e| anyhow!("Failed to serialize JWK: {e:?}"))?;
197
198    // Extract n and e parameters
199    let n = jwk_json
200        .get("n")
201        .and_then(|v| v.as_str())
202        .ok_or_else(|| anyhow!("JWK missing 'n' parameter"))?;
203    let e = jwk_json
204        .get("e")
205        .and_then(|v| v.as_str())
206        .ok_or_else(|| anyhow!("JWK missing 'e' parameter"))?;
207
208    // Decode base64url encoded parameters
209    let n_bytes = base64::engine::general_purpose::URL_SAFE_NO_PAD
210        .decode(n.as_bytes())
211        .map_err(|e| anyhow!("Failed to decode 'n': {e:?}"))?;
212    let e_bytes = base64::engine::general_purpose::URL_SAFE_NO_PAD
213        .decode(e.as_bytes())
214        .map_err(|e| anyhow!("Failed to decode 'e': {e:?}"))?;
215
216    // Create RSA public key
217    let n_bigint = BigUint::from_bytes_be(&n_bytes);
218    let e_bigint = BigUint::from_bytes_be(&e_bytes);
219
220    let public_key = RsaPublicKey::new(n_bigint, e_bigint)
221        .map_err(|e| anyhow!("Failed to create RSA public key: {e:?}"))?;
222
223    // Convert to PEM format
224    let pem = public_key
225        .to_pkcs1_pem(rsa::pkcs1::LineEnding::LF)
226        .map_err(|e| anyhow!("Failed to encode public key as PEM: {e:?}"))?;
227
228    // Create DecodingKey
229    jsonwebtoken::DecodingKey::from_rsa_pem(pem.as_bytes())
230        .map_err(|e| anyhow!("Failed to create decoding key: {e:?}"))
231}

255impl OidcAuthProvider {
256    /// Create a new OIDC authentication provider
257    pub async fn new(config: OidcConfig) -> Result<Self> {
258        if config.issuers.is_empty() {
259            return Err(anyhow!("At least one OIDC issuer must be configured"));
260        }
261
262        let jwks_ttl = Duration::from_secs(config.jwks_refresh_interval_secs);
263        let mut clients = HashMap::new();
264
265        // Initialize a client for each configured issuer
266        for issuer_config in config.issuers {
267            let client = OidcIssuerClient::new(
268                issuer_config.issuer.clone(),
269                issuer_config.audience,
270                jwks_ttl,
271            )?;
272
273            clients.insert(issuer_config.issuer, Arc::new(client));
274        }
275
276        // Create token cache
277        let token_cache = Cache::builder()
278            .max_capacity(config.token_cache_size)
279            .time_to_live(Duration::from_secs(config.token_cache_ttl_secs))
280            .build();
281
282        // Load admin users from environment
283        let admin_users = load_admin_users();
284
285        Ok(Self {
286            clients,
287            token_cache,
288            admin_users,
289        })
290    }
291
292    fn is_admin(&self, subject: &str, email: Option<&str>) -> bool {
293        self.admin_users
294            .iter()
295            .any(|admin| admin == subject || email.map(|e| admin == e).unwrap_or(false))
296    }
297
298    /// Decode JWT payload without validation to extract issuer
299    ///
300    /// JWTs are structured as: header.payload.signature
301    /// Both header and payload are base64url-encoded JSON
302    fn decode_payload_unsafe(&self, token: &str) -> Result<Claims> {
303        let parts: Vec<&str> = token.split('.').collect();
304        if parts.len() != 3 {
305            return Err(anyhow!("Invalid JWT format"));
306        }
307
308        // Decode the payload (second part)
309        let payload_bytes = base64::engine::general_purpose::URL_SAFE_NO_PAD
310            .decode(parts[1].as_bytes())
311            .map_err(|e| anyhow!("Failed to decode JWT payload: {e:?}"))?;
312
313        let claims: Claims = serde_json::from_slice(&payload_bytes)
314            .map_err(|e| anyhow!("Failed to parse JWT claims: {e:?}"))?;
315
316        Ok(claims)
317    }
318
319    /// Validate an ID token and return authentication context
320    ///
321    /// This implementation follows OAuth 2.0 best practices by:
322    /// 1. Extracting kid from JWT header for direct key lookup
323    /// 2. Extracting issuer from JWT payload for direct client lookup
324    /// 3. Using O(1) lookups instead of O(n*m) iteration
325    /// 4. Eliminating timing side-channels
326    async fn validate_id_token(&self, token: &str) -> Result<AuthContext> {
327        // Step 1: Decode header (unsigned) to get key ID (kid)
328        let header = decode_header(token).map_err(|e| anyhow!("Invalid JWT header: {e:?}"))?;
329
330        let kid = header
331            .kid
332            .ok_or_else(|| anyhow!("JWT missing kid (key ID) in header"))?;
333
334        // Step 2: Decode payload (unsigned) to get issuer
335        let unverified_claims = self.decode_payload_unsafe(token)?;
336
337        // Step 3: Look up specific issuer client
338        let client = self
339            .clients
340            .get(&unverified_claims.iss)
341            .ok_or_else(|| anyhow!("Unknown issuer: {}", unverified_claims.iss))?;
342
343        // Step 4: Get JWKS and find specific key by kid
344        let jwks = client
345            .jwks_cache
346            .get()
347            .await
348            .map_err(|e| anyhow!("Failed to fetch JWKS: {e:?}"))?;
349
350        let key = jwks
351            .keys()
352            .iter()
353            .find(|k| k.key_id().map(|id| id.as_str()) == Some(kid.as_str()))
354            .ok_or_else(|| anyhow!("Key with kid '{}' not found in JWKS", kid))?;
355
356        // Step 5: Convert JWK to DecodingKey
357        let decoding_key = jwk_to_decoding_key(key)?;
358
359        // Step 6: Validate token with specific key and issuer
360        let mut validation = Validation::new(Algorithm::RS256);
361        // Don't validate audience yet - we'll do it manually
362        validation.validate_aud = false;
363        validation.set_issuer(&[&client.issuer]);
364
365        let token_data = decode::<Claims>(token, &decoding_key, &validation)
366            .map_err(|e| anyhow!("Token validation failed: {e:?}"))?;
367
368        let claims = token_data.claims;
369
370        // Step 7: Manually validate audience
371        if !claims.aud.contains(&client.audience) {
372            return Err(anyhow!("Invalid audience"));
373        }
374
375        // Step 8: Validate expiration
376        let expires_at = DateTime::from_timestamp(claims.exp, 0)
377            .ok_or_else(|| anyhow!("Invalid expiration timestamp"))?;
378
379        if expires_at < Utc::now() {
380            return Err(anyhow!("Token has expired"));
381        }
382
383        // Step 9: Check if user is admin
384        let is_admin = self.is_admin(&claims.sub, claims.email.as_deref());
385
386        Ok(AuthContext {
387            subject: claims.sub,
388            email: claims.email,
389            issuer: claims.iss,
390            expires_at: Some(expires_at),
391            auth_type: AuthType::Oidc,
392            is_admin,
393        })
394    }
395}

397#[async_trait::async_trait]
398impl AuthProvider for OidcAuthProvider {
399    async fn validate_token(&self, token: &str) -> Result<AuthContext> {
400        // Check token cache first
401        if let Some(cached) = self.token_cache.get(token).await {
402            return Ok((*cached).clone());
403        }
404
405        // Validate token
406        let auth_ctx = self.validate_id_token(token).await?;
407
408        // Cache the result
409        self.token_cache
410            .insert(token.to_string(), Arc::new(auth_ctx.clone()))
411            .await;
412
413        Ok(auth_ctx)
414    }
