1//! Authentication providers for Micromegas
2//!
3//! This crate provides authentication and authorization for Micromegas services.
4//! It supports multiple authentication methods:
5//!
6//! - **API Keys**: Simple bearer token authentication
7//! - **OIDC**: OpenID Connect authentication with automatic JWKS caching
8//!
9//! # Example: API Key Authentication
10//!
11//! ```rust
12//! use micromegas_auth::api_key::{ApiKeyAuthProvider, parse_key_ring};
13//! use micromegas_auth::types::AuthProvider;
14//!
15//! # async fn example() -> anyhow::Result<()> {
16//! let json = r#"[{"name": "user1", "key": "secret-key-123"}]"#;
17//! let keyring = parse_key_ring(json)?;
18//! let provider = ApiKeyAuthProvider::new(keyring);
19//!
20//! let auth_ctx = provider.validate_token("secret-key-123").await?;
21//! println!("Authenticated: {}", auth_ctx.subject);
22//! # Ok(())
23//! # }
24//! ```
25//!
26//! # Example: OIDC Authentication
27//!
28//! ```rust,no_run
29//! use micromegas_auth::oidc::{OidcAuthProvider, OidcConfig, OidcIssuer};
30//! use micromegas_auth::types::AuthProvider;
31//!
32//! # async fn example() -> anyhow::Result<()> {
33//! let config = OidcConfig {
34//!     issuers: vec![OidcIssuer {
35//!         issuer: "https://accounts.google.com".to_string(),
36//!         audience: "your-client-id.apps.googleusercontent.com".to_string(),
37//!     }],
38//!     jwks_refresh_interval_secs: 3600,
39//!     token_cache_size: 1000,
40//!     token_cache_ttl_secs: 300,
41//! };
42//!
43//! let provider = OidcAuthProvider::new(config).await?;
44//! let auth_ctx = provider.validate_token("id_token_here").await?;
45//! println!("Authenticated: {}", auth_ctx.subject);
46//! # Ok(())
47//! # }
48//! ```
