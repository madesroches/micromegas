32/// Deserializes a string into a `Key`.
33fn key_from_string<'de, D>(deserializer: D) -> Result<Key, D::Error>
34where
35    D: serde::Deserializer<'de>,
36{
37    let s: String = Deserialize::deserialize(deserializer)?;
38    Ok(Key::new(s))
39}

41/// Represents an entry in the keyring, mapping a key to a name.
42#[derive(Deserialize)]
43pub struct KeyRingEntry {
44    pub name: String,
45    #[serde(deserialize_with = "key_from_string")]
46    pub key: Key,
47}
48
49/// A map from `Key` to `String` (name).
50pub type KeyRing = HashMap<Key, String>;
51
52/// Parses a JSON string into a `KeyRing`.
53///
54/// The JSON string is expected to be an array of objects, each with a `name` and `key` field.
55pub fn parse_key_ring(json: &str) -> Result<KeyRing> {
56    let entries: Vec<KeyRingEntry> = serde_json::from_str(json)?;
57    let mut ring = KeyRing::new();
58    for entry in entries {
59        ring.insert(entry.key, entry.name);
60    }
61    Ok(ring)
62}

76#[async_trait::async_trait]
77impl AuthProvider for ApiKeyAuthProvider {
78    /// Validate an API key token using constant-time comparison
79    ///
80    /// This implementation protects against timing attacks by:
81    /// 1. Comparing the provided token against ALL keys in the keyring
82    /// 2. Using constant-time comparison from the `subtle` crate
83    /// 3. Always iterating through all keys regardless of match status
84    ///
85    /// This ensures the operation takes the same amount of time whether:
86    /// - The key is found early in the iteration
87    /// - The key is found late in the iteration
88    /// - The key is not found at all
89    async fn validate_token(&self, token: &str) -> Result<AuthContext> {
90        let token_bytes = token.as_bytes();
91        let mut found: Option<AuthContext> = None;
92
93        // Compare against all keys in constant time
94        // IMPORTANT: We iterate through ALL keys, even if we find a match,
95        // to ensure constant-time operation
96        for (stored_key, name) in &self.keyring {
97            let stored_bytes = stored_key.value.as_bytes();
98
99            // Constant-time comparison
100            // Returns 1 if equal, 0 if not equal
101            let matches = token_bytes.ct_eq(stored_bytes).unwrap_u8() == 1;
102
103            // Conditionally set the result without branching on the match
104            // If matches is true, we set found; if matches is false, found stays as-is
105            if matches {
106                found = Some(AuthContext {
107                    subject: name.clone(),
108                    email: None,
109                    issuer: "api_key".to_string(),
110                    expires_at: None,
111                    auth_type: AuthType::ApiKey,
112                    is_admin: false,
113                });
114            }
115            // Note: We do NOT break or return early - we continue checking all keys
116        }
117
118        found.ok_or_else(|| anyhow!("invalid API token"))
119    }
120}
