9/// Multi-provider authentication that tries API key first, then OIDC.
10///
11/// This provider allows supporting multiple authentication methods simultaneously.
12/// It tries providers in order until one succeeds:
13/// 1. API key (fast O(1) HashMap lookup)
14/// 2. OIDC (slower JWT validation with JWKS)
15///
16/// # Example
17///
18/// ```rust,no_run
19/// use micromegas_auth::api_key::{ApiKeyAuthProvider, parse_key_ring};
20/// use micromegas_auth::oidc::{OidcAuthProvider, OidcConfig, OidcIssuer};
21/// use micromegas_auth::multi::MultiAuthProvider;
22/// use std::sync::Arc;
23///
24/// # async fn example() -> anyhow::Result<()> {
25/// // Set up API key provider
26/// let keyring = parse_key_ring(r#"[{"name": "test", "key": "secret"}]"#)?;
27/// let api_key_provider = Arc::new(ApiKeyAuthProvider::new(keyring));
28///
29/// // Set up OIDC provider
30/// let oidc_config = OidcConfig {
31///     issuers: vec![OidcIssuer {
32///         issuer: "https://accounts.google.com".to_string(),
33///         audience: "your-app.apps.googleusercontent.com".to_string(),
34///     }],
35///     jwks_refresh_interval_secs: 3600,
36///     token_cache_size: 1000,
37///     token_cache_ttl_secs: 300,
38/// };
39/// let oidc_provider = Arc::new(OidcAuthProvider::new(oidc_config).await?);
40///
41/// // Create multi-provider
42/// let multi = MultiAuthProvider {
43///     api_key_provider: Some(api_key_provider),
44///     oidc_provider: Some(oidc_provider),
45/// };
46/// # Ok(())
47/// # }
48/// ```
49pub struct MultiAuthProvider {
50    /// Optional API key provider (tried first)
51    pub api_key_provider: Option<Arc<ApiKeyAuthProvider>>,
52    /// Optional OIDC provider (tried second)
53    pub oidc_provider: Option<Arc<OidcAuthProvider>>,
54}
55
56#[async_trait]
57impl AuthProvider for MultiAuthProvider {
58    async fn validate_token(&self, token: &str) -> anyhow::Result<AuthContext> {
59        // Try API key authentication first (fast path)
60        if let Some(api_key_provider) = &self.api_key_provider
61            && let Ok(auth_ctx) = api_key_provider.validate_token(token).await
62        {
63            return Ok(auth_ctx);
64        }
65
66        // Try OIDC authentication (slower, involves JWT validation)
67        if let Some(oidc_provider) = &self.oidc_provider
68            && let Ok(auth_ctx) = oidc_provider.validate_token(token).await
69        {
70            return Ok(auth_ctx);
71        }
72
73        anyhow::bail!("authentication failed with all providers")
74    }
75}

77#[cfg(test)]
78mod tests {
79    use super::*;
80    use crate::api_key::parse_key_ring;
81
82    #[tokio::test]
83    async fn test_multi_provider_api_key() {
84        let keyring = parse_key_ring(r#"[{"name": "test", "key": "secret"}]"#).unwrap();
85        let api_key_provider = Arc::new(ApiKeyAuthProvider::new(keyring));
86
87        let multi = MultiAuthProvider {
88            api_key_provider: Some(api_key_provider),
89            oidc_provider: None,
90        };
91
92        let result = multi.validate_token("secret").await;
93        assert!(result.is_ok());
94        let auth_ctx = result.unwrap();
95        assert_eq!(auth_ctx.subject, "test");
96    }
97
98    #[tokio::test]
99    async fn test_multi_provider_no_providers() {
100        let multi = MultiAuthProvider {
101            api_key_provider: None,
102            oidc_provider: None,
103        };
104
105        let result = multi.validate_token("any-token").await;
106        assert!(result.is_err());
107    }
108
109    #[tokio::test]
