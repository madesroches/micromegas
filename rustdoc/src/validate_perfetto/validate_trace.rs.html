1use anyhow::Result;

7fn main() -> Result<()> {

9    if args.len() != 2 {
10        eprintln!("Usage: {} <trace-file>", args[0]);
11        std::process::exit(1);
12    }
13
14    let trace_file = &args[1];
15    println!("Validating Perfetto trace: {}", trace_file);
16
17    let data = fs::read(trace_file)?;
18    println!("Trace file size: {} bytes", data.len());
19
20    let trace = Trace::decode(&data[..])?;
21    println!("Trace packets: {}", trace.packet.len());
22
23    let mut process_descriptors = 0;
24    let mut thread_descriptors = 0;
25    let mut async_track_descriptors = 0;
26    let mut track_events = 0;
27
28    for packet in &trace.packet {
29        if let Some(data) = &packet.data {
30            match data {
31                micromegas_perfetto::protos::trace_packet::Data::TrackDescriptor(track) => {
32                    if track.process.is_some() {
33                        process_descriptors += 1;
34                    } else if track.thread.is_some() {
35                        thread_descriptors += 1;
36                    } else if track.static_or_dynamic_name.is_some() {
37                        async_track_descriptors += 1;
38                    }
39                }
40                micromegas_perfetto::protos::trace_packet::Data::TrackEvent(_) => {
41                    track_events += 1;
42                }
43                _ => {}
44            }
45        }
46    }
47
48    println!("Process descriptors: {}", process_descriptors);
49    println!("Thread descriptors: {}", thread_descriptors);
50    println!("Async track descriptors: {}", async_track_descriptors);
51    println!("Track events: {}", track_events);
52
53    println!("âœ… Trace validation successful!");

56}
