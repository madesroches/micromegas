1//! Color UDFs for building packed RGBA `u32` colors from SQL.
2//!
3//! # API conventions
4//!
5//! These conventions apply to every color UDF in this crate. Once external SQL
6//! exists in the wild they are effectively frozen — any new color UDF must
7//! honour them, and behaviour-changing variants must adopt a distinct name.
8//!
9//! - **Packing.** Colors are packed `u32` in `0xRRGGBBAA` byte order: byte 0
10//!   (high byte) is red, byte 3 (low byte) is alpha. This matches the map
11//!   cell's decode in `MapViewer.tsx`.
12//! - **Component range.** Float inputs/outputs are in `[0.0, 1.0]`.
13//!   Out-of-range values are clamped at the byte boundary, not rejected.
14//! - **Alpha is straight, not premultiplied.** The alpha channel is treated
15//!   the same as RGB — it interpolates and quantizes the same way.
16//! - **Color space is sRGB-encoded 8-bit.** Lerps and other operations act
17//!   directly on sRGB byte values, which is what the GPU consumes and what
18//!   HLSL/Cg-style 8-bit color code does. Future perceptual-space variants
19//!   (e.g. `lerp_oklab`) must use an explicit suffix; unsuffixed names always
20//!   mean sRGB.
21//!
22//! Future-extension naming reservations (recorded so the API stays coherent
23//! as it grows):
24//! - Constructors by format: `rgba` (this module), `rgb`, `hsla`, `hsva`,
25//!   `color_from_hex`.
26//! - Operations on packed colors: `<op>_color` suffix — `lerp_color` (this
27//!   module), and (future) `mix_color`, `blend_color`, `tint_color`,
28//!   `saturate_color`.
29//! - Color-space-specific operations: `<op>_<space>` suffix — `lerp_oklab`,
30//!   `lerp_hsl`, etc.
31//! - Colormaps: `color_scale(name, t, alpha)` (this module) is the entry
32//!   point for sequential/diverging perceptual scales. Categorical scales
33//!   are reserved under a distinct name (e.g. `color_category(name, i,
34//!   alpha)`) because they take an integer index, not a `t`. Reversed
35//!   scales are reserved as either a `'<name>_r'` suffix or a sibling
36//!   `color_scale_reversed` UDF — pick when the first user asks. Direct
37//!   per-colormap UDFs (`viridis(t, alpha)`, …) are intentionally *not*
38//!   reserved; if needed later they can be added as aliases of
39//!   `color_scale`.
40
41/// `color_scale(name, t, alpha) -> UInt32`
42pub mod color_scale;
43/// `lerp_color(c1, c2, t) -> UInt32`
44pub mod lerp_color;
45/// `rgba(r, g, b, a) -> UInt32`
46pub mod rgba;

48/// Pack four 8-bit channels into a `0xRRGGBBAA` `u32`.
49#[inline]
50pub fn pack_rgba(r: u8, g: u8, b: u8, a: u8) -> u32 {
51    ((r as u32) << 24) | ((g as u32) << 16) | ((b as u32) << 8) | (a as u32)
52}
53
54/// Unpack a `0xRRGGBBAA` `u32` into `(r, g, b, a)`.
55#[inline]
56pub fn unpack_rgba(c: u32) -> (u8, u8, u8, u8) {
57    (
58        ((c >> 24) & 0xff) as u8,
59        ((c >> 16) & 0xff) as u8,
60        ((c >> 8) & 0xff) as u8,
61        (c & 0xff) as u8,
62    )
63}

65/// Quantize a normalized float to `0..=255` with round-half-up.
66///
67/// Clamps the scaled value (not the input) so the output invariant holds
68/// regardless of input range. ±∞ saturate via the clamp; NaN saturates to 0
69/// via Rust's saturating `f64 as u8` cast (`f64::clamp` propagates NaN, so
70/// the cast is what makes NaN inputs safe). Both safety nets are load-bearing
71/// — do not remove either one.
72#[inline]
73pub fn float_to_byte(f: f64) -> u8 {
74    (f * 255.0 + 0.5).clamp(0.0, 255.0) as u8
75}
76
77/// Round an already-in-`[0,255]` lerp result to a `u8`, half-up.
78///
79/// Shared with `lerp_color` so both UDFs use the same tie-breaking rule.
80#[inline]
81pub fn round_to_byte(f: f64) -> u8 {
82    (f + 0.5).clamp(0.0, 255.0) as u8
83}
