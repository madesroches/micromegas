30/// Configuration for the analytics web server.
31pub struct WebServerConfig {
32    /// HTTP port to listen on (default 3000).
33    pub port: u16,
34    /// Path to the built frontend `dist/` directory.
35    pub frontend_dir: String,
36    /// Base path for the app (e.g. `""` or `"/micromegas"`).
37    pub base_path: String,
38    /// CORS allowed origin (e.g. `"http://localhost:3000"`).
39    pub cors_origin: String,
40    /// PostgreSQL connection string for the web-app database.
41    pub app_db_string: String,
42    /// Optional object-store URI for maps blobs.
43    pub maps_uri: Option<String>,
44    /// Optional override for the maximum maps upload size in bytes.
45    pub max_upload_bytes: Option<usize>,
46    /// Disable OIDC/cookie auth (anonymous access, development only).
47    pub disable_auth: bool,
48    /// Environment variable name used to load the OIDC admin list.
49    ///
50    /// Standalone binary sets this to `"MICROMEGAS_ADMINS"`.
51    /// The monolith sets it to the analytics-scoped var (with fallback already resolved).
52    pub admin_var_name: String,
53}

59fn build_auth_state(config: &WebServerConfig) -> Result<Option<AuthState>> {
60    let oidc_config = OidcClientConfig::from_env()
61        .map_err(|e| anyhow::anyhow!("Failed to load OIDC client config: {e}"))?;
62
63    let cookie_domain = std::env::var("MICROMEGAS_COOKIE_DOMAIN").ok();
64    let secure_cookies = std::env::var("MICROMEGAS_SECURE_COOKIES")
65        .map(|v| v == "true" || v == "1")
66        .unwrap_or(false);
67
68    let state_signing_secret = std::env::var("MICROMEGAS_STATE_SECRET")
69        .context("MICROMEGAS_STATE_SECRET environment variable not set. Generate a secure random secret (e.g., openssl rand -base64 32)")?
70        .into_bytes();
71
72    Ok(Some(AuthState {
73        oidc_provider: Arc::new(tokio::sync::OnceCell::new()),
74        auth_provider: Arc::new(tokio::sync::OnceCell::new()),
75        config: oidc_config,
76        cookie_domain,
77        secure_cookies,
78        state_signing_secret,
79        base_path: config.base_path.clone(),
80        admin_var_name: config.admin_var_name.clone(),
81    }))
82}

84fn build_auth_routes(base_path: &str, auth_state: &Option<AuthState>) -> Router {
85    if let Some(state) = auth_state {
86        Router::new()
87            .route(
88                &format!("{base_path}/auth/login"),
89                get(crate::auth::auth_login),
90            )
91            .route(
92                &format!("{base_path}/auth/callback"),
93                get(crate::auth::auth_callback),
94            )
95            .route(
96                &format!("{base_path}/auth/refresh"),
97                post(crate::auth::auth_refresh),
98            )
99            .route(
100                &format!("{base_path}/auth/logout"),
101                post(crate::auth::auth_logout),
102            )
103            .route(&format!("{base_path}/auth/me"), get(crate::auth::auth_me))
104            .with_state(state.clone())
105    } else {
106        Router::new()
107            .route(&format!("{base_path}/auth/me"), get(auth_me_no_auth))
108            .route(
109                &format!("{base_path}/auth/logout"),
110                post(auth_logout_no_auth),
111            )
112    }
113}

123fn build_protected_routes(
124    base_path: &str,
125    auth_state: &Option<AuthState>,
126    app_db_pool: PgPool,
127    data_source_cache: DataSourceCache,
128    maps_state: maps::MapsState,
129) -> Router {
130    let routes = Router::new()
131        .route(
132            &format!("{base_path}/api/query-stream"),
133            post(stream_query::stream_query_handler),
134        )
135        .route(
136            &format!("{base_path}/api/screen-types"),
137            get(screens::list_screen_types),
138        )
139        .route(
140            &format!("{base_path}/api/screen-types/{{type_name}}/default"),
141            get(screens::get_default_config),
142        )
143        .route(
144            &format!("{base_path}/api/screens"),
145            get(screens::list_screens).post(screens::create_screen),
146        )
147        .route(
148            &format!("{base_path}/api/screens/{{name}}"),
149            get(screens::get_screen)
150                .put(screens::update_screen)
151                .delete(screens::delete_screen),
152        )
153        .route(
154            &format!("{base_path}/api/data-sources"),
155            get(data_sources::list_data_sources).post(data_sources::create_data_source),
156        )
157        .route(
158            &format!("{base_path}/api/data-sources/{{name}}"),
159            get(data_sources::get_data_source)
160                .put(data_sources::update_data_source)
161                .delete(data_sources::delete_data_source),
162        )
163        .route(
164            &format!("{base_path}/api/maps/catalog"),
165            get(maps::maps_catalog),
166        )
167        .route(
168            &format!("{base_path}/api/maps/blob/{{filename}}"),
169            put(maps::maps_upload)
170                .layer(DefaultBodyLimit::max(maps_state.max_upload_bytes))
171                .delete(maps::maps_delete),
172        )
173        .layer(Extension(app_db_pool))
174        .layer(Extension(data_source_cache))
175        .layer(Extension(maps_state))
176        .layer(middleware::from_fn(observability_middleware));
177
178    if let Some(state) = auth_state {
179        routes.layer(middleware::from_fn_with_state(
180            state.clone(),
181            crate::auth::cookie_auth_middleware,
182        ))
183    } else {
184        routes
185            .layer(Extension(AuthToken(String::new())))
186            .layer(Extension(ValidatedUser {
187                subject: "anonymous".to_string(),
188                email: None,
189                issuer: "local".to_string(),
190                is_admin: true,
191            }))
192    }
193}

229#[derive(Clone)]
230struct IndexState {
231    frontend_dir: String,
232    base_path: String,
233}
234
235async fn serve_index_with_config(
236    State(state): State<IndexState>,
237    _request: axum::extract::Request,
238) -> Result<impl IntoResponse, (StatusCode, String)> {
239    let index_path = format!("{}/index.html", state.frontend_dir);
240    let html = tokio::fs::read_to_string(&index_path).await.map_err(|e| {
241        (
242            StatusCode::INTERNAL_SERVER_ERROR,
243            format!("Failed to read index.html: {e}"),
244        )
245    })?;
246
247    let base_href = if state.base_path.is_empty() {
248        "/".to_string()
249    } else {
250        format!("{}/", state.base_path)
251    };
252    let injection = format!(
253        r#"<base href="{base_href}"><script>window.__MICROMEGAS_CONFIG__={{basePath:"{}"}}</script>"#,
254        state.base_path
255    );
256
257    let modified_html = html.replace("<head>", &format!("<head>{injection}"));
258
259    Ok((
260        [(header::CONTENT_TYPE, "text/html; charset=utf-8")],
261        modified_html,
262    ))
263}
264
265fn build_frontend(frontend_dir: &str, index_state: IndexState) -> Router {
266    let spa_fallback = get(serve_index_with_config).with_state(index_state.clone());
267    let serve_dir = ServeDir::new(frontend_dir).fallback(spa_fallback);
268
269    let index_handler = get(serve_index_with_config).with_state(index_state);
270    Router::new()
271        .route("/", index_handler.clone())
272        .route("/index.html", index_handler)
273        .fallback_service(serve_dir)
274}
275
276fn mount_frontend(
277    app: Router,
278    base_path: &str,
279    frontend: Router,
280    index_state: IndexState,
281) -> Router {
282    if base_path.is_empty() {
283        app.merge(frontend)
284    } else {
285        let app = app.nest(base_path, frontend);
286
287        let base_path_with_slash = format!("{base_path}/");
288        let index_handler = get(serve_index_with_config).with_state(index_state);
289        let app = app.route(&base_path_with_slash, index_handler);
290
291        let redirect_path = base_path.to_string();
292        app.layer(axum::middleware::from_fn(
293            move |req: axum::extract::Request, next: axum::middleware::Next| {
294                let base_path = redirect_path.clone();
295                async move {
296                    if req.uri().path() == base_path {
297                        return Redirect::permanent(&format!("{base_path}/")).into_response();
298                    }
299                    next.run(req).await.into_response()
300                }
301            },
302        ))
303    }
304}

306fn build_cors_layer(cors_origin: &str) -> Result<CorsLayer> {
307    let origin = cors_origin
308        .parse::<HeaderValue>()
309        .context("Invalid MICROMEGAS_WEB_CORS_ORIGIN format")?;
310
311    Ok(CorsLayer::new()
312        .allow_origin(origin)
313        .allow_methods([
314            Method::GET,
315            Method::POST,
316            Method::PUT,
317            Method::DELETE,
318            Method::OPTIONS,
319        ])
320        .allow_headers([header::CONTENT_TYPE, header::AUTHORIZATION])
321        .allow_credentials(true))
322}

369/// Start the analytics web server.
370///
371/// Binds to `config.port`, wires all routes, and shuts down gracefully when
372/// `shutdown` resolves.  This is called by both the standalone binary and the
373/// monolith.
374pub async fn run_web_server(
375    config: WebServerConfig,
376    shutdown: impl Future<Output = ()> + Send + 'static,
377    grace: Duration,
378) -> Result<()> {
379    let app_db_pool = sqlx::PgPool::connect(&config.app_db_string)
380        .await
381        .context("Failed to connect to micromegas_app database")?;
382    app_db::execute_migration(app_db_pool.clone()).await?;
383    info!("Connected to micromegas_app database");
384
385    let data_source_cache =
386        DataSourceCache::new(app_db_pool.clone(), std::time::Duration::from_secs(60));
387
388    let maps_store = maps::connect_maps_store(config.maps_uri.as_deref())
389        .context("Failed to connect to maps object store")?;
390    if maps_store.is_some() {
391        info!("Maps object store connected");
392    } else {
393        info!("MICROMEGAS_MAPS_OBJECT_STORE_URI not set — /api/maps/* will return 503");
394    }
395    let max_upload_bytes = config
396        .max_upload_bytes
397        .unwrap_or(maps::DEFAULT_MAX_UPLOAD_BYTES);
398    let maps_state = maps::MapsState::with_max_upload_bytes(maps_store, max_upload_bytes);
399
400    let auth_state = if config.disable_auth {
401        println!("WARNING: Authentication is disabled (--disable-auth)");
402        None
403    } else {
404        build_auth_state(&config)?
405    };
406
407    let app = Router::new()
408        .merge(build_public_routes(&config.base_path))
409        .merge(build_protected_routes(
410            &config.base_path,
411            &auth_state,
412            app_db_pool,
413            data_source_cache,
414            maps_state.clone(),
415        ))
416        .merge(build_auth_routes(&config.base_path, &auth_state));
417
418    let index_state = IndexState {
419        frontend_dir: config.frontend_dir.clone(),
420        base_path: config.base_path.clone(),
421    };
422    let frontend = build_frontend(&config.frontend_dir, index_state.clone());
423    let app = mount_frontend(app, &config.base_path, frontend, index_state);
424
425    let app = app.layer(CompressionLayer::new().gzip(true));
426    let app = app
427        .merge(build_protected_maps_blob_route(
428            &config.base_path,
429            &auth_state,
430            maps_state,
431        ))
432        .layer(build_cors_layer(&config.cors_origin)?);
433
434    let addr = format!("0.0.0.0:{}", config.port);
435    println!("Analytics web server starting on {addr}");
436    println!("CORS origin: {}", config.cors_origin);
437    if !config.base_path.is_empty() {
438        println!("Base path: {}", config.base_path);
439    }
440    println!(
441        "Authentication: {}",
442        if config.disable_auth {
443            "DISABLED"
444        } else {
445            "ENABLED"
446        }
447    );
448
449    let listener = tokio::net::TcpListener::bind(&addr).await?;
450    serve_axum_with_graceful_shutdown(
451        listener,
452        ServiceExt::<axum::extract::Request>::into_make_service_with_connect_info::<
453            std::net::SocketAddr,
454        >(app),
455        shutdown,
456        grace,
457    )
458    .await?;
459
460    Ok(())
