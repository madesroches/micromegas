28/// Request for streaming SQL query
29#[derive(Debug, Serialize, Deserialize)]
30pub struct StreamQueryRequest {
31    pub sql: String,
32    #[serde(default)]
33    pub params: HashMap<String, String>,
34    pub begin: Option<DateTime<Utc>>,
35    pub end: Option<DateTime<Utc>>,
36}

38/// Error codes for stream query errors
39#[derive(Debug, Clone, Copy, Serialize)]
40#[serde(rename_all = "SCREAMING_SNAKE_CASE")]
41pub enum ErrorCode {
42    InvalidSql,
43    ConnectionFailed,
44    Internal,
45    Forbidden,
46}
47
48/// Schema and batch frames use identical structure - size-prefixed binary
49#[derive(Serialize)]
50struct DataHeader {
51    #[serde(rename = "type")]
52    frame_type: &'static str,
53    size: usize,
54}
55
56/// Done frame to indicate successful completion
57#[derive(Serialize)]
58struct DoneFrame {
59    #[serde(rename = "type")]
60    frame_type: &'static str,
61}
62
63/// Error frame for streaming errors
64#[derive(Serialize)]
65struct ErrorFrame {
66    #[serde(rename = "type")]
67    frame_type: &'static str,
68    code: ErrorCode,
69    message: String,
70}

72/// Serialize a value to a JSON line (with trailing newline)
73fn json_line<T: Serialize>(value: &T) -> Bytes {
74    let mut json = serde_json::to_string(value).expect("serialization failed");
75    json.push('\n');
76    Bytes::from(json)
77}

79/// List of destructive functions that should be blocked in web queries
80const BLOCKED_FUNCTIONS: &[&str] = &[
81    "retire_partitions",
82    "retire_partition_by_metadata",
83    "retire_partition_by_file",
84];
85
86/// Check if the SQL query contains any blocked destructive functions
87pub fn contains_blocked_function(sql: &str) -> Option<&'static str> {
88    let sql_lower = sql.to_lowercase();
89    BLOCKED_FUNCTIONS
90        .iter()
91        .find(|&func| sql_lower.contains(func))
92        .copied()
93}

95/// Substitute macro variables in SQL query
96pub fn substitute_macros(sql: &str, params: &HashMap<String, String>) -> String {
97    let mut result = sql.to_string();
98    for (key, value) in params {
99        // Escape single quotes in values to prevent SQL injection
100        let escaped_value = value.replace('\'', "''");
101        // Replace $key with the escaped value
102        result = result.replace(&format!("${key}"), &escaped_value);
103    }
104    result
105}

107/// Encode a schema to Arrow IPC format
108///
109/// The tracker should be the same instance used for subsequent batch encoding
110/// to ensure dictionary IDs are consistent between schema and batches.
111pub fn encode_schema(
112    schema: &Schema,
113    tracker: &mut arrow_ipc::writer::DictionaryTracker,
114) -> Result<Vec<u8>> {
115    let mut buffer = Vec::new();
116    let data_gen = IpcDataGenerator::default();
117    let options = IpcWriteOptions::default();
118
119    let encoded = data_gen.schema_to_bytes_with_dictionary_tracker(schema, tracker, &options);
120    write_message(&mut buffer, encoded, &options).context("writing schema message")?;
121    Ok(buffer)
122}
123
124/// Encode a RecordBatch to Arrow IPC format
125pub fn encode_batch(
126    batch: &datafusion::arrow::array::RecordBatch,
127    tracker: &mut arrow_ipc::writer::DictionaryTracker,
128    compression: &mut CompressionContext,
129) -> Result<Vec<u8>> {
130    let mut buffer = Vec::new();
131    let data_gen = IpcDataGenerator::default();
132    let options = IpcWriteOptions::default();
133
134    let (encoded_dicts, encoded_batch) = data_gen
135        .encode(batch, tracker, &options, compression)
136        .context("encoding batch")?;
137
138    // Write dictionary batches first (if any)
139    for dict in encoded_dicts {
140        write_message(&mut buffer, dict, &options).context("writing dictionary message")?;
141    }
142
143    // Write the main batch
144    write_message(&mut buffer, encoded_batch, &options).context("writing batch message")?;
145
146    Ok(buffer)
147}

149/// Streaming SQL query endpoint using Arrow IPC protocol
150///
151/// Returns a stream of JSON-framed Arrow IPC messages:
152/// - `{"type":"schema","size":N}\n` followed by N bytes of schema IPC
153/// - `{"type":"batch","size":N}\n` followed by N bytes of batch IPC
154/// - `{"type":"done"}\n` on success
155/// - `{"type":"error","code":"..","message":"..}\n` on error
156#[span_fn]
157pub async fn stream_query_handler(
158    Extension(auth_token): Extension<AuthToken>,
159    Json(request): Json<StreamQueryRequest>,
160) -> impl IntoResponse {
161    info!(
162        "stream query sql={} params={:?} begin={:?} end={:?}",
163        request.sql, request.params, request.begin, request.end
164    );
165
166    // Check for blocked functions first (before starting the stream)
167    if let Some(blocked_func) = contains_blocked_function(&request.sql) {
168        let error_frame = ErrorFrame {
169            frame_type: "error",
170            code: ErrorCode::Forbidden,
171            message: format!(
172                "The function '{}' is not allowed in web queries for security reasons",
173                blocked_func
174            ),
175        };
176        let stream =
177            futures::stream::once(async move { Ok::<_, std::io::Error>(json_line(&error_frame)) });
178        return (
179            StatusCode::FORBIDDEN,
180            [(
181                header::CONTENT_TYPE,
182                "application/x-micromegas-arrow-stream",
183            )],
184            Body::from_stream(stream),
185        )
186            .into_response();
187    }
188
189    // Substitute macros
190    let sql = substitute_macros(&request.sql, &request.params);
191
192    // Build time range if provided
193    let time_range = match (request.begin, request.end) {
194        (Some(begin), Some(end)) => Some(TimeRange::new(begin, end)),
195        _ => None,
196    };
197
198    let stream = stream! {
199        // Create FlightSQL client
200        let client_factory = BearerFlightSQLClientFactory::new_with_client_type(
201            auth_token.0,
202            "web".to_string(),
203        );
204        let mut client = match client_factory.make_client().await {
205            Ok(c) => c,
206            Err(e) => {
207                yield Ok::<_, std::io::Error>(json_line(&ErrorFrame {
208                    frame_type: "error",
209                    code: ErrorCode::ConnectionFailed,
210                    message: e.to_string(),
211                }));
212                return;
213            }
214        };
215
216        // Start streaming query
217        let mut batch_stream = match client.query_stream(sql, time_range).await {
218            Ok(s) => s,
219            Err(e) => {
220                yield Ok(json_line(&ErrorFrame {
221                    frame_type: "error",
222                    code: ErrorCode::InvalidSql,
223                    message: e.to_string(),
224                }));
225                return;
226            }
227        };
228
229        // FlightSQL limitation: The schema is embedded in the first FlightData message,
230        // not sent separately. The FlightRecordBatchStream only populates its schema field
231        // after we read the first message. We must read the first batch here, extract the
232        // schema, send it to the client, then process this batch along with the rest.
233        let first_batch = batch_stream.next().await;
234
235        // Get schema from the stream (now available after reading first message)
236        let schema = match batch_stream.schema() {
237            Some(s) => s.clone(),
238            None => {
239                yield Ok(json_line(&ErrorFrame {
240                    frame_type: "error",
241                    code: ErrorCode::Internal,
242                    message: "No schema in response".to_string(),
243                }));
244                return;
245            }
246        };
247
248        // Track dictionaries and compression across schema and batches
249        // Must use same tracker for schema and batches to ensure dictionary IDs align
250        let mut dict_tracker = arrow_ipc::writer::DictionaryTracker::new(false);
251        let mut compression = CompressionContext::default();
252
253        // Encode and send schema
254        let schema_bytes = match encode_schema(&schema, &mut dict_tracker) {
255            Ok(bytes) => bytes,
256            Err(e) => {
257                yield Ok(json_line(&ErrorFrame {
258                    frame_type: "error",
259                    code: ErrorCode::Internal,
260                    message: format!("{e:#}"),
261                }));
262                return;
263            }
264        };
265
266        yield Ok(json_line(&DataHeader {
267            frame_type: "schema",
268            size: schema_bytes.len(),
269        }));
270        yield Ok(Bytes::from(schema_bytes));
271
272        // Helper to encode and yield a batch
273        macro_rules! yield_batch {
274            ($batch:expr) => {
275                let batch_bytes = match encode_batch(&$batch, &mut dict_tracker, &mut compression) {
276                    Ok(bytes) => bytes,
277                    Err(e) => {
278                        yield Ok(json_line(&ErrorFrame {
279                            frame_type: "error",
280                            code: ErrorCode::Internal,
281                            message: format!("{e:#}"),
282                        }));
283                        return;
284                    }
285                };
286                yield Ok(json_line(&DataHeader {
287                    frame_type: "batch",
288                    size: batch_bytes.len(),
289                }));
290                yield Ok(Bytes::from(batch_bytes));
291            };
292        }
293
294        // Process the first batch we read earlier (to get the schema)
295        if let Some(result) = first_batch {
296            match result {
297                Ok(batch) => {
298                    yield_batch!(batch);
299                }
300                Err(e) => {
301                    yield Ok(json_line(&ErrorFrame {
302                        frame_type: "error",
303                        code: ErrorCode::Internal,
304                        message: e.to_string(),
305                    }));
306                    return;
307                }
308            }
309        }
310
311        // Stream remaining batches
312        while let Some(result) = batch_stream.next().await {
313            match result {
314                Ok(batch) => {
315                    yield_batch!(batch);
316                }
317                Err(e) => {
318                    yield Ok(json_line(&ErrorFrame {
319                        frame_type: "error",
320                        code: ErrorCode::Internal,
321                        message: e.to_string(),
322                    }));
323                    return;
324                }
325            }
326        }
327
328        // Success
329        yield Ok(json_line(&DoneFrame { frame_type: "done" }));
330    };
331
332    (
333        [(
334            header::CONTENT_TYPE,
335            "application/x-micromegas-arrow-stream",
336        )],
337        Body::from_stream(stream),
338    )
339        .into_response()
340}
