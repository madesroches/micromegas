1//! Authentication endpoints for analytics-web-srv
2//!
3//! Implements OIDC authorization code flow with PKCE:
4//! - /auth/login - Initiate OIDC login
5//! - /auth/callback - Handle OIDC callback
6//! - /auth/refresh - Refresh tokens
7//! - /auth/logout - Clear session
8//! - /auth/me - Get current user info
9//!
10//! Security: All JWT tokens are fully validated (signature + claims) at this tier
11//! using the micromegas-auth crate with JWKS caching. Invalid tokens are rejected
12//! before forwarding requests to FlightSQL.
13

57/// OIDC client configuration
58#[derive(Debug, Clone, Deserialize)]
59pub struct OidcClientConfig {
60    /// OIDC provider issuer URL
61    pub issuer: String,
62    /// Client ID (public client)
63    pub client_id: String,
64    /// Redirect URI for callback
65    pub redirect_uri: String,
66}
67
68impl OidcClientConfig {
69    /// Load configuration from environment variables
70    ///
71    /// Required environment variables:
72    /// - MICROMEGAS_OIDC_CONFIG: JSON with "issuers" array (same format as FlightSQL server)
73    /// - MICROMEGAS_AUTH_REDIRECT_URI: OAuth callback URL
74    ///
75    /// Expected MICROMEGAS_OIDC_CONFIG format (uses micromegas-auth's OidcConfig):
76    /// {
77    ///   "issuers": [
78    ///     {
79    ///       "issuer": "https://...",
80    ///       "audience": "client-id"
81    ///     }
82    ///   ]
83    /// }
84    ///
85    /// Note: When multiple issuers are configured, the first issuer is used for
86    /// the OAuth login flow (you can only redirect to one provider). Token
87    /// validation via OidcAuthProvider will accept tokens from any configured issuer.
88    pub fn from_env() -> Result<Self> {
89        // Use the shared OidcConfig from micromegas-auth
90        let config = micromegas_auth::oidc::OidcConfig::from_env()?;
91
92        // Need at least one issuer
93        if config.issuers.is_empty() {
94            return Err(anyhow!(
95                "MICROMEGAS_OIDC_CONFIG must contain at least one issuer in the 'issuers' array"
96            ));
97        }
98
99        // Use the first issuer for OAuth login flow
100        // (token validation via OidcAuthProvider supports all issuers)
101        let issuer_config = &config.issuers[0];
102
103        if config.issuers.len() > 1 {
104            info!(
105                "Multiple OIDC issuers configured ({}). Using '{}' for OAuth login flow. \
106                 Token validation will accept tokens from all configured issuers.",
107                config.issuers.len(),
108                issuer_config.issuer
109            );
110        }
111
112        let redirect_uri = std::env::var("MICROMEGAS_AUTH_REDIRECT_URI")
113            .map_err(|_| anyhow!("MICROMEGAS_AUTH_REDIRECT_URI environment variable not set"))?;
114
115        Ok(OidcClientConfig {
116            issuer: issuer_config.issuer.clone(),
117            client_id: issuer_config.audience.clone(),
118            redirect_uri,
119        })
120    }
121}

131/// State for auth endpoints
132#[derive(Clone)]
133pub struct AuthState {
134    /// OIDC provider info (lazy initialized) - for OAuth flow
135    pub oidc_provider: Arc<tokio::sync::OnceCell<OidcProviderInfo>>,
136    /// OIDC auth provider (lazy initialized) - for JWT validation
137    pub auth_provider: Arc<tokio::sync::OnceCell<Arc<OidcAuthProvider>>>,
138    /// OIDC client configuration
139    pub config: OidcClientConfig,
140    /// Cookie domain (optional)
141    pub cookie_domain: Option<String>,
142    /// Whether we're in production (secure cookies)
143    pub secure_cookies: bool,
144    /// Secret for signing OAuth state parameters (HMAC-SHA256)
145    pub state_signing_secret: Vec<u8>,
146    /// Base path for cookies (e.g., "/micromegas"), defaults to "/"
147    pub base_path: String,
148}
149
150impl AuthState {
151    /// Returns the cookie path, using base_path or "/" if empty
152    pub fn cookie_path(&self) -> String {
153        if self.base_path.is_empty() {
154            "/".to_string()
155        } else {
156            self.base_path.clone()
157        }
158    }
159
160    pub async fn get_oidc_provider(&self) -> Result<&OidcProviderInfo> {
161        let config = self.config.clone();
162        self.oidc_provider
163            .get_or_try_init(|| async move {
164                let issuer_url = openidconnect::IssuerUrl::new(config.issuer.clone())
165                    .map_err(|e| anyhow!("Invalid issuer URL: {e:?}"))?;
166
167                let http_client = create_http_client()?;
168                let provider_metadata =
169                    CoreProviderMetadata::discover_async(issuer_url, &http_client)
170                        .await
171                        .map_err(|e| anyhow!("Failed to discover OIDC provider: {e:?}"))?;
172
173                let client_id = openidconnect::ClientId::new(config.client_id.clone());
174                let redirect_uri = openidconnect::RedirectUrl::new(config.redirect_uri.clone())
175                    .map_err(|e| anyhow!("Invalid redirect URI: {e:?}"))?;
176
177                Ok(OidcProviderInfo {
178                    metadata: Arc::new(provider_metadata),
179                    client_id,
180                    redirect_uri,
181                })
182            })
183            .await
184    }
185
186    pub fn build_oidc_client(&self, provider: &OidcProviderInfo) -> ConfiguredCoreClient {
187        openidconnect::core::CoreClient::from_provider_metadata(
188            (*provider.metadata).clone(),
189            provider.client_id.clone(),
190            None, // No client secret (public client with PKCE)
191        )
192        .set_redirect_uri(provider.redirect_uri.clone())
193    }
194
195    /// Get or initialize the OIDC auth provider for JWT validation
196    ///
197    /// The auth provider is lazy-initialized on first use and cached.
198    /// It uses the MICROMEGAS_OIDC_CONFIG environment variable for configuration,
199    /// which is the same format used by the FlightSQL server.
200    pub async fn get_auth_provider(&self) -> Result<&Arc<OidcAuthProvider>> {
201        self.auth_provider
202            .get_or_try_init(|| async {
203                let config = OidcConfig::from_env()?;
204                let provider = OidcAuthProvider::new(config).await?;
205                Ok(Arc::new(provider))
206            })
207            .await
208    }
209}

306/// Create a cookie with common settings
307pub fn create_cookie<'a>(
308    name: &'a str,
309    value: String,
310    max_age_secs: i64,
311    state: &AuthState,
312) -> Cookie<'a> {
313    let mut cookie = Cookie::build((name, value))
314        .http_only(true)
315        .secure(state.secure_cookies)
316        .same_site(SameSite::Lax)
317        .path(state.cookie_path())
318        .max_age(time::Duration::seconds(max_age_secs));
319
320    if let Some(domain) = &state.cookie_domain {
321        cookie = cookie.domain(domain.clone());
322    }
323
324    cookie.build()
325}
326
327/// Create an expired cookie to clear it
328pub fn clear_cookie<'a>(name: &'a str, state: &AuthState) -> Cookie<'a> {
329    let mut cookie = Cookie::build((name, ""))
330        .http_only(true)
331        .secure(state.secure_cookies)
332        .same_site(SameSite::Lax)
333        .path(state.cookie_path())
334        .max_age(time::Duration::seconds(0));
335
336    if let Some(domain) = &state.cookie_domain {
337        cookie = cookie.domain(domain.clone());
338    }
339
340    cookie.build()
341}

343/// GET /auth/login - Initiate OIDC login
344#[span_fn]
345pub async fn auth_login(
346    State(state): State<AuthState>,
347    Query(query): Query<LoginQuery>,
348) -> Result<impl IntoResponse, AuthApiError> {
349    let return_url = query.return_url.unwrap_or_else(|| "/".to_string());
350
351    // Validate return URL to prevent open redirect
352    if !validate_return_url(&return_url) {
353        return Err(AuthApiError::InvalidReturnUrl);
354    }
355
356    // Generate PKCE challenge
357    let (pkce_challenge, pkce_verifier) = PkceCodeChallenge::new_random_sha256();
358
359    // Generate state with nonce and return URL
360    let nonce = generate_nonce();
361    let oauth_state = OAuthState {
362        nonce: nonce.clone(),
363        return_url,
364        pkce_verifier: pkce_verifier.secret().to_string(),
365    };
366    // Sign the state with HMAC-SHA256 to prevent tampering
367    let state_signed = sign_state(&oauth_state, &state.state_signing_secret)
368        .map_err(|e| AuthApiError::Internal(format!("Failed to sign state: {e:?}")))?;
369
370    // Get OIDC provider and build client
371    let provider = state
372        .get_oidc_provider()
373        .await
374        .map_err(|e| AuthApiError::Internal(format!("Failed to get OIDC provider: {e:?}")))?;
375    let client = state.build_oidc_client(provider);
376
377    // Generate authorization URL
378    let (auth_url, _csrf_token, _nonce) = client
379        .authorize_url(
380            AuthenticationFlow::<CoreResponseType>::AuthorizationCode,
381            move || CsrfToken::new(state_signed.clone()),
382            Nonce::new_random,
383        )
384        .add_scope(Scope::new("openid".to_string()))
385        .add_scope(Scope::new("email".to_string()))
386        .add_scope(Scope::new("profile".to_string()))
387        .add_scope(Scope::new("offline_access".to_string()))
388        .set_pkce_challenge(pkce_challenge)
389        .url();
390
391    // Set cookie with nonce for validation
392    let cookie = create_cookie(OAUTH_STATE_COOKIE, nonce, 600, &state); // 10 minutes
393
394    Ok((
395        CookieJar::new().add(cookie),
396        Redirect::temporary(auth_url.as_str()),
397    ))
398}

400/// GET /auth/callback - Handle OIDC callback
401#[span_fn]
402pub async fn auth_callback(
403    State(state): State<AuthState>,
404    jar: CookieJar,
405    Query(query): Query<CallbackQuery>,
406) -> Result<impl IntoResponse, AuthApiError> {
407    // Verify and decode signed state parameter
408    let oauth_state = verify_state(&query.state, &state.state_signing_secret).map_err(|e| {
409        warn!("[auth_failure] reason=invalid_state details={e:?}");
410        AuthApiError::InvalidState
411    })?;
412
413    // Validate nonce from cookie
414    let cookie_nonce = jar
415        .get(OAUTH_STATE_COOKIE)
416        .ok_or_else(|| {
417            warn!("[auth_failure] reason=missing_oauth_state_cookie");
418            AuthApiError::InvalidState
419        })?
420        .value();
421
422    if cookie_nonce != oauth_state.nonce {
423        warn!("[auth_failure] reason=nonce_mismatch");
424        return Err(AuthApiError::InvalidState);
425    }
426
427    // Get OIDC provider and build client
428    let provider = state
429        .get_oidc_provider()
430        .await
431        .map_err(|e| AuthApiError::Internal(format!("Failed to get OIDC provider: {e:?}")))?;
432    let _client = state.build_oidc_client(provider);
433
434    // Exchange code for tokens using manual HTTP request
435    // Note: We don't use the openidconnect library's exchange_code() because:
436    // - Auth0 includes non-standard fields (e.g., updated_at) that cause parsing failures
437    // - The library's strict typing doesn't handle provider-specific extensions well
438    // - Manual HTTP gives us better error visibility and control over parsing
439    let http_client = create_http_client()
440        .map_err(|e| AuthApiError::Internal(format!("Failed to create HTTP client: {e:?}")))?;
441
442    let token_url = provider
443        .metadata
444        .token_endpoint()
445        .expect("token endpoint should exist");
446
447    let params = [
448        ("grant_type", "authorization_code"),
449        ("code", &query.code),
450        ("redirect_uri", &state.config.redirect_uri),
451        ("client_id", &state.config.client_id),
452        ("code_verifier", &oauth_state.pkce_verifier),
453    ];
454
455    // Note: Generic error messages are intentional to avoid leaking authentication details
456    // Detailed errors are logged server-side for debugging
457    let response = http_client
458        .post(token_url.as_str())
459        .form(&params)
460        .send()
461        .await
462        .map_err(|e| {
463            warn!("[auth_failure] reason=token_exchange_request_failed details={e:?}");
464            AuthApiError::TokenExchangeFailed
465        })?;
466
467    if !response.status().is_success() {
468        let status = response.status();
469        let body = response
470            .text()
471            .await
472            .unwrap_or_else(|_| "unknown".to_string());
473        warn!("[auth_failure] reason=token_exchange_failed status={status} body={body}");
474        return Err(AuthApiError::TokenExchangeFailed);
475    }
476
477    let token_response: serde_json::Value = response.json().await.map_err(|e| {
478        warn!("[auth_failure] reason=token_response_parse_failed details={e:?}");
479        AuthApiError::TokenExchangeFailed
480    })?;
481
482    // Extract tokens from JSON response
483    let id_token = token_response["id_token"]
484        .as_str()
485        .ok_or_else(|| {
486            warn!("[auth_failure] reason=missing_id_token");
487            AuthApiError::TokenExchangeFailed
488        })?
489        .to_string();
490
491    let refresh_token = token_response["refresh_token"]
492        .as_str()
493        .map(|s| s.to_string());
494
495    // Calculate expiration times
496    let access_token_expires = token_response["expires_in"]
497        .as_u64()
498        .map(|d| d as i64)
499        .unwrap_or(3600); // Default 1 hour
500
501    let refresh_token_expires = 30 * 24 * 3600; // 30 days
502
503    // Log successful login (extract subject from token for audit trail)
504    if let Some(sub) = extract_subject_from_token(&id_token) {
505        info!(
506            "[auth_success] event=login sub={sub} issuer={}",
507            state.config.issuer
508        );
509    }
510
511    // Create cookies
512    let mut new_jar = jar;
513    new_jar = new_jar.add(create_cookie(
514        ID_TOKEN_COOKIE,
515        id_token,
516        access_token_expires,
517        &state,
518    ));
519
520    if let Some(refresh) = refresh_token {
521        new_jar = new_jar.add(create_cookie(
522            REFRESH_TOKEN_COOKIE,
523            refresh,
524            refresh_token_expires,
525            &state,
526        ));
527    }
528
529    // Clear oauth state cookie
530    new_jar = new_jar.add(clear_cookie(OAUTH_STATE_COOKIE, &state));
531
532    // Redirect to return URL
533    Ok((new_jar, Redirect::temporary(&oauth_state.return_url)))
534}

536/// POST /auth/refresh - Refresh tokens
537#[span_fn]
538pub async fn auth_refresh(
539    State(state): State<AuthState>,
540    jar: CookieJar,
541) -> Result<impl IntoResponse, AuthApiError> {
542    // Get refresh token from cookie
543    let refresh_token = jar
544        .get(REFRESH_TOKEN_COOKIE)
545        .ok_or(AuthApiError::Unauthorized)?
546        .value()
547        .to_string();
548
549    // Get OIDC provider and build client
550    let provider = state
551        .get_oidc_provider()
552        .await
553        .map_err(|e| AuthApiError::Internal(format!("Failed to get OIDC provider: {e:?}")))?;
554    let _client = state.build_oidc_client(provider);
555
556    // Exchange refresh token for new tokens using manual HTTP request
557    // Note: Same reasoning as auth_callback - Auth0's non-standard fields break library parsing
558    let http_client = create_http_client()
559        .map_err(|e| AuthApiError::Internal(format!("Failed to create HTTP client: {e:?}")))?;
560
561    let token_url = provider
562        .metadata
563        .token_endpoint()
564        .expect("token endpoint should exist");
565
566    let params = [
567        ("grant_type", "refresh_token"),
568        ("refresh_token", &refresh_token),
569        ("client_id", &state.config.client_id),
570    ];
571
572    let response = http_client
573        .post(token_url.as_str())
574        .form(&params)
575        .send()
576        .await
577        .map_err(|e| {
578            warn!("[token_refresh_failure] reason=request_failed details={e:?}");
579            AuthApiError::Unauthorized
580        })?;
581
582    if !response.status().is_success() {
583        let status = response.status();
584        let body = response
585            .text()
586            .await
587            .unwrap_or_else(|_| "unknown".to_string());
588        warn!("[token_refresh_failure] reason=token_exchange_failed status={status} body={body}");
589        return Err(AuthApiError::Unauthorized);
590    }
591
592    let token_response: serde_json::Value = response.json().await.map_err(|e| {
593        warn!("[token_refresh_failure] reason=response_parse_failed details={e:?}");
594        AuthApiError::Unauthorized
595    })?;
596
597    // Extract new tokens from JSON response
598    let id_token = token_response["id_token"]
599        .as_str()
600        .ok_or_else(|| {
601            warn!("[token_refresh_failure] reason=missing_id_token");
602            AuthApiError::Unauthorized
603        })?
604        .to_string();
605
606    let refresh_token = token_response["refresh_token"]
607        .as_str()
608        .map(|s| s.to_string());
609
610    // Calculate expiration times
611    let id_token_expires = token_response["expires_in"]
612        .as_u64()
613        .map(|d| d as i64)
614        .unwrap_or(3600);
615
616    let refresh_token_expires = 30 * 24 * 3600; // 30 days
617
618    // Log successful token refresh
619    if let Some(sub) = extract_subject_from_token(&id_token) {
620        info!("[auth_success] event=token_refresh sub={sub}");
621    }
622
623    // Update cookies
624    let mut new_jar = jar;
625    new_jar = new_jar.add(create_cookie(
626        ID_TOKEN_COOKIE,
627        id_token,
628        id_token_expires,
629        &state,
630    ));
631
632    if let Some(refresh) = refresh_token {
633        new_jar = new_jar.add(create_cookie(
634            REFRESH_TOKEN_COOKIE,
635            refresh,
636            refresh_token_expires,
637            &state,
638        ));
639    }
640
641    Ok((new_jar, StatusCode::OK))
642}

654/// GET /auth/me - Get current user info
655///
656/// Returns user information from the validated JWT token.
657/// This endpoint must be behind the cookie_auth_middleware to ensure
658/// the token has been validated before extracting user info.
659///
660/// The user's `sub` (subject) and `email` come from the validated token claims.
661/// The `name` field is extracted directly from the JWT payload for display purposes.
662#[span_fn]
663pub async fn auth_me(
664    State(state): State<AuthState>,
665    jar: CookieJar,
666) -> Result<Json<UserInfo>, AuthApiError> {
667    // Get ID token from cookie
668    let id_token = jar
669        .get(ID_TOKEN_COOKIE)
670        .ok_or_else(|| {
671            debug!("no id_token cookie found");
672            AuthApiError::Unauthorized
673        })?
674        .value()
675        .to_string();
676
677    // Validate the token using the OIDC provider
678    let auth_provider = state.get_auth_provider().await.map_err(|e| {
679        warn!("[auth_failure] auth_provider_init_failed: {e:?}");
680        AuthApiError::Internal("Failed to initialize auth provider".to_string())
681    })?;
682
683    let parts = CookieTokenRequestParts {
684        token: id_token.clone(),
685    };
686
687    let auth_context = auth_provider.validate_request(&parts).await.map_err(|e| {
688        warn!("[auth_failure] {e}");
689        AuthApiError::InvalidToken
690    })?;
691
692    // Extract name from JWT payload (not in AuthContext)
693    // This is safe since we just validated the token
694    let name = extract_name_from_token(&id_token);
695
696    Ok(Json(UserInfo {
697        sub: auth_context.subject,
698        email: auth_context.email,
699        name,
700        is_admin: auth_context.is_admin,
701    }))
702}

704/// Extract the 'name' claim from a JWT payload
705///
706/// This is used by auth_me() to get the display name which isn't in AuthContext
707fn extract_name_from_token(token: &str) -> Option<String> {
708    let parts: Vec<&str> = token.split('.').collect();
709    if parts.len() != 3 {
710        return None;
711    }
712
713    let payload_bytes = base64::engine::general_purpose::URL_SAFE_NO_PAD
714        .decode(parts[1].as_bytes())
715        .ok()?;
716
717    let claims: IdTokenClaims = serde_json::from_slice(&payload_bytes).ok()?;
718    claims.name
719}
720
721/// Extract the 'sub' claim from a JWT payload for audit logging
722fn extract_subject_from_token(token: &str) -> Option<String> {
723    let parts: Vec<&str> = token.split('.').collect();
724    if parts.len() != 3 {
725        return None;
726    }
727
728    let payload_bytes = base64::engine::general_purpose::URL_SAFE_NO_PAD
729        .decode(parts[1].as_bytes())
730        .ok()?;
731
732    let claims: serde_json::Value = serde_json::from_slice(&payload_bytes).ok()?;
733    claims["sub"].as_str().map(|s| s.to_string())
734}

736/// Authentication API errors
737#[derive(Debug)]
738pub enum AuthApiError {
739    InvalidReturnUrl,
740    InvalidState,
741    TokenExchangeFailed,
742    Unauthorized,
743    InvalidToken,
744    Internal(String),
745}
746
747impl IntoResponse for AuthApiError {
748    fn into_response(self) -> Response {
749        let (status, message) = match self {
750            AuthApiError::InvalidReturnUrl => (StatusCode::BAD_REQUEST, "Invalid return URL"),
751            AuthApiError::InvalidState => (StatusCode::BAD_REQUEST, "Invalid OAuth state"),
752            AuthApiError::TokenExchangeFailed => {
753                (StatusCode::UNAUTHORIZED, "Token exchange failed")
754            }
755            AuthApiError::Unauthorized => (StatusCode::UNAUTHORIZED, "Unauthorized"),
756            AuthApiError::InvalidToken => (StatusCode::UNAUTHORIZED, "Invalid token"),
757            AuthApiError::Internal(msg) => {
758                error!("Auth internal error: {msg}");
759                (StatusCode::INTERNAL_SERVER_ERROR, "Internal server error")
760            }
761        };
762
763        (status, message).into_response()
764    }
765}

767/// Cookie-based authentication middleware with full JWT signature validation
768///
769/// Reads the ID token from httpOnly cookie, validates the signature using JWKS,
770/// and injects validated user info into request extensions.
771///
772/// Security: This middleware performs full JWT validation including:
773/// - Signature verification using cached JWKS from the OIDC provider
774/// - Issuer validation against configured issuers
775/// - Audience validation
776/// - Expiration check
777///
778/// Invalid or forged tokens are rejected before any downstream processing.
779///
780/// Note: We use the ID token (JWT) for FlightSQL API calls because:
781/// - ID tokens can be validated locally by both web tier and FlightSQL
782/// - This matches the Python API behavior which also uses ID tokens
783/// - Access tokens (JWE) would require token introspection endpoints
784#[span_fn]
785pub async fn cookie_auth_middleware(
786    State(state): State<AuthState>,
787    req: Request,
788    next: Next,
789) -> Result<Response, AuthApiError> {
790    // Extract cookies from request
791    let jar = CookieJar::from_headers(req.headers());
792
793    // Get ID token from cookie (JWT format for FlightSQL API calls)
794    let id_token = jar
795        .get(ID_TOKEN_COOKIE)
796        .ok_or_else(|| {
797            debug!("id_token cookie not found");
798            AuthApiError::Unauthorized
799        })?
800        .value()
801        .to_string();
802
803    // Get the OIDC auth provider (lazy initialized with JWKS caching)
804    let auth_provider = state.get_auth_provider().await.map_err(|e| {
805        warn!("[auth_failure] auth_provider_init_failed: {e:?}");
806        AuthApiError::Internal("Failed to initialize auth provider".to_string())
807    })?;
808
809    // Create request parts adapter for the cookie token
810    let parts = CookieTokenRequestParts {
811        token: id_token.clone(),
812    };
813
814    // Validate the token with full signature verification
815    let auth_context = auth_provider.validate_request(&parts).await.map_err(|e| {
816        warn!("[auth_failure] {e}");
817        AuthApiError::InvalidToken
818    })?;
819
820    // Log successful authentication (trace level to avoid noise on every request)
821    trace!(
822        "[auth_success] subject={} email={:?} issuer={} admin={}",
823        auth_context.subject, auth_context.email, auth_context.issuer, auth_context.is_admin
824    );
825
826    // Store token and validated user info in request extensions
827    let mut req = req;
828    req.extensions_mut().insert(AuthToken(id_token));
829    req.extensions_mut()
830        .insert(ValidatedUser::from(&auth_context));
831
832    // Continue to next middleware/handler
833    Ok(next.run(req).await)
834}
