1//! Map asset endpoints.
2//!
3//! GLBs are stored gzipped under a `.gz` suffix on the configured prefix —
4//! the suffix is the *only* signal we need on the read path. Catalog
5//! listing filters for it, GET tacks `.gz` onto the requested name and
6//! always responds with `Content-Encoding: gzip`, and PUT gzips the body
7//! before writing. That keeps the design portable: no
8//! `PutOptions::attributes`, so `object_store::LocalFileSystem` (which
9//! rejects attributes with `NotImplemented`) works the same as S3/GCS.
10

29/// Default max upload body size (256 MiB). Configurable via the
30/// `MICROMEGAS_MAPS_MAX_UPLOAD_BYTES` env var, surfaced on `MapsState`.
31pub const DEFAULT_MAX_UPLOAD_BYTES: usize = 256 * 1024 * 1024;
32
33/// Suffix appended to every stored object key. PUT writes `<name>.gz`,
34/// GET reads `<name>.gz`, and catalog listing only surfaces keys carrying
35/// this suffix (with the suffix stripped for display).
36const GZ_SUFFIX: &str = ".gz";
37
38/// The maps prefix is reserved for direct children — any subdirectory
39/// content is ignored. The configured store is treated as the source of
40/// truth: whatever flat `*.gz` files live there are the catalog.
41///
42/// In practice, axum's `{filename}` path capture is already a single
43/// segment (so `/` cannot reach the blob handler from the URL side), and
44/// `object_store::path::Path` keys are opaque (no `..` traversal).
45/// Filtering out keys that contain `/` keeps the catalog flat and stops
46/// nested objects from sneaking into the listing.
47pub fn is_direct_child(name: &str) -> bool {
48    !name.is_empty() && !name.contains('/')
49}
50
51/// The stored key for a user-facing filename: `level.glb` → `level.glb.gz`.
52fn stored_key(filename: &str) -> String {
53    format!("{filename}{GZ_SUFFIX}")
54}

56/// Shared state for the maps handlers.
57///
58/// Held as `Extension<MapsState>` to match the local convention (every
59/// existing protected route receives its dependencies through Extensions).
60#[derive(Clone)]
61pub struct MapsState {
62    pub store: Option<Arc<dyn ObjectStore>>,
63    /// Configured cap on the upload body, threaded through `MapsState` so
64    /// the router-build site can size the per-route `DefaultBodyLimit`
65    /// layer from a single source of truth. Not referenced inside handlers
66    /// — the 413 response from the layer is axum's plaintext default.
67    pub max_upload_bytes: usize,
68}
69
70impl MapsState {
71    pub fn new(store: Option<Arc<dyn ObjectStore>>) -> Self {
72        Self {
73            store,
74            max_upload_bytes: DEFAULT_MAX_UPLOAD_BYTES,
75        }
76    }
77
78    pub fn with_max_upload_bytes(store: Option<Arc<dyn ObjectStore>>, max: usize) -> Self {
79        Self {
80            store,
81            max_upload_bytes: max,
82        }
83    }
84}

86/// Connect to the store named by `MICROMEGAS_MAPS_OBJECT_STORE_URI`.
87///
88/// Returns `Ok(None)` when the env var is unset — the catalog endpoint
89/// will respond with 503 in that case.
90pub fn connect_maps_store(uri: Option<&str>) -> Result<Option<Arc<dyn ObjectStore>>> {
91    let Some(uri) = uri else {
92        return Ok(None);
93    };
94    let url = url::Url::parse(uri).context("parsing MICROMEGAS_MAPS_OBJECT_STORE_URI")?;
95    let (store, prefix) =
96        object_store::parse_url_opts(&url, std::env::vars().map(|(k, v)| (k.to_lowercase(), v)))
97            .context("connecting to MICROMEGAS_MAPS_OBJECT_STORE_URI")?;
98    Ok(Some(Arc::new(PrefixStore::new(store, prefix))))
99}

101/// Entry returned by the catalog endpoint. `size` is the on-disk
102/// (gzipped) byte count — that is what storage is billed against and what
103/// the GET response will set as `Content-Length`.
104#[derive(Debug, Serialize)]
105pub struct CatalogEntry {
106    pub file: String,
107    pub size: u64,
108    pub last_modified: DateTime<Utc>,
109}

111/// GET /api/maps/catalog
112///
113/// Lists every flat `*.gz` object directly under the configured prefix and
114/// strips the suffix for display. The catalog is the listing — there is no
115/// separate `maps.json` to keep in sync.
116#[span_fn]
117pub async fn maps_catalog(Extension(state): Extension<MapsState>) -> Response {
118    let Some(store) = state.store else {
119        return (
120            StatusCode::SERVICE_UNAVAILABLE,
121            "Maps endpoint not configured: set MICROMEGAS_MAPS_OBJECT_STORE_URI",
122        )
123            .into_response();
124    };
125
126    let mut entries: Vec<CatalogEntry> = Vec::new();
127    let mut list_stream = store.list(None);
128    while let Some(item) = list_stream.next().await {
129        match item {
130            Ok(meta) => {
131                let location = meta.location.as_ref();
132                if !is_direct_child(location) {
133                    continue;
134                }
135                let Some(display) = location.strip_suffix(GZ_SUFFIX) else {
136                    // Non-`.gz` entries are not part of the API surface —
137                    // a manual upload that skipped the PUT handler. Skip
138                    // it rather than mislead the UI.
139                    continue;
140                };
141                entries.push(CatalogEntry {
142                    file: display.to_string(),
143                    size: meta.size,
144                    last_modified: meta.last_modified,
145                });
146            }
147            Err(e) => {
148                error!("listing maps prefix: {e:?}");
149                return (StatusCode::INTERNAL_SERVER_ERROR, "Failed to list maps").into_response();
150            }
151        }
152    }
153
154    entries.sort_by(|a, b| a.file.cmp(&b.file));
155
156    (
157        [
158            (header::CONTENT_TYPE, "application/json"),
159            (header::CACHE_CONTROL, "no-cache"),
160        ],
161        Json(entries),
162    )
163        .into_response()
164}

166/// GET /api/maps/blob/{filename}
167///
168/// Streams the gzipped object stored at `{filename}.gz`. The axum
169/// `{filename}` path capture is a single segment, and
170/// `object_store::path::Path` keys are opaque — together those bound the
171/// request inside the configured `PrefixStore`. Anything that survives
172/// URL routing but still contains a `/` (e.g. percent-encoded forms
173/// decoded into the segment) is rejected as a defense-in-depth check.
174///
175/// The response is always `Content-Type: model/gltf-binary` +
176/// `Content-Encoding: gzip` — the suffix on the stored key is the
177/// contract; we don't read it back from object metadata.
178#[span_fn]
179pub async fn maps_blob(
180    Extension(state): Extension<MapsState>,
181    AxumPath(filename): AxumPath<String>,
182) -> Response {
183    if !is_direct_child(&filename) {
184        return (StatusCode::BAD_REQUEST, "Invalid filename").into_response();
185    }
186
187    let Some(store) = state.store else {
188        return (
189            StatusCode::SERVICE_UNAVAILABLE,
190            "Maps endpoint not configured: set MICROMEGAS_MAPS_OBJECT_STORE_URI",
191        )
192            .into_response();
193    };
194
195    let key = stored_key(&filename);
196    let get_result = match store.get(&ObjectPath::from(key.as_str())).await {
197        Ok(r) => r,
198        Err(object_store::Error::NotFound { .. }) => {
199            return (StatusCode::NOT_FOUND, "Map not found").into_response();
200        }
201        Err(e) => {
202            error!("fetching map blob '{filename}': {e:?}");
203            return (StatusCode::INTERNAL_SERVER_ERROR, "Failed to fetch map").into_response();
204        }
205    };
206
207    let mut headers = HeaderMap::new();
208    headers.insert(
209        header::CONTENT_TYPE,
210        HeaderValue::from_static("model/gltf-binary"),
211    );
212    headers.insert(header::CONTENT_ENCODING, HeaderValue::from_static("gzip"));
213    headers.insert(header::CONTENT_LENGTH, get_result.meta.size.into());
214    // `private`: the route requires cookie auth, so shared caches (CDNs,
215    // corporate proxies) must not store and re-serve responses to other users.
216    headers.insert(
217        header::CACHE_CONTROL,
218        HeaderValue::from_static("private, max-age=3600"),
219    );
220
221    let stream = get_result
222        .into_stream()
223        .map(|res| res.map_err(std::io::Error::other));
224
225    (headers, Body::from_stream(stream)).into_response()
226}

228/// JSON error body returned by the mutation handlers.
229#[derive(Debug, Serialize)]
230struct MapsErrorBody {
231    code: &'static str,
232    message: String,
233}
234
235fn maps_error(status: StatusCode, code: &'static str, message: impl Into<String>) -> Response {
236    let body = MapsErrorBody {
237        code,
238        message: message.into(),
239    };
240    (status, Json(body)).into_response()

253/// PUT /api/maps/blob/{filename}
254///
255/// Admin-only. Gzips the body server-side (or accepts a client-gzipped
256/// body verbatim) and stores it under `{filename}.gz`. The `.gz` suffix
257/// is server-owned: the user-facing filename must NOT end in `.gz`.
258#[span_fn]
259pub async fn maps_upload(
260    Extension(state): Extension<MapsState>,
261    // `AdminUser` is `FromRequestParts`, so this runs *before* the `Bytes`
262    // body extractor — a non-admin gets 403 without buffering the upload.
263    AdminUser(user): AdminUser,
264    AxumPath(filename): AxumPath<String>,
265    headers: HeaderMap,
266    body: Bytes,
267) -> Result<Response, AdminRequired> {
268    if !is_direct_child(&filename) || filename.ends_with(GZ_SUFFIX) {
269        return Ok(maps_error(
270            StatusCode::BAD_REQUEST,
271            "INVALID_FILENAME",
272            "Filename must be a single path segment and must not end in .gz",
273        ));
274    }
275
276    let content_type = headers
277        .get(header::CONTENT_TYPE)
278        .and_then(|v| v.to_str().ok())
279        .unwrap_or("");
280    if content_type != "model/gltf-binary" {
281        return Ok(maps_error(
282            StatusCode::UNSUPPORTED_MEDIA_TYPE,
283            "UNSUPPORTED_MEDIA_TYPE",
284            "Content-Type must be model/gltf-binary",
285        ));
286    }
287
288    let Some(store) = state.store else {
289        return Ok(maps_error(
290            StatusCode::SERVICE_UNAVAILABLE,
291            "MAPS_STORE_UNAVAILABLE",
292            "Maps endpoint not configured: set MICROMEGAS_MAPS_OBJECT_STORE_URI",
293        ));
294    };
295
296    // If the client pre-gzipped the body, store it verbatim. Otherwise
297    // gzip server-side so the stored object is always a gzip frame and
298    // the GET path can stream it untouched with `Content-Encoding: gzip`.
299    let client_pregzipped = headers
300        .get(header::CONTENT_ENCODING)
301        .and_then(|v| v.to_str().ok())
302        .map(|v| v.eq_ignore_ascii_case("gzip"))
303        .unwrap_or(false);
304
305    let stored: Bytes = if client_pregzipped {
306        body
307    } else {
308        // gzip is CPU-bound and can run for seconds on a 256 MiB upload at
309        // the default compression level — offload it to a blocking thread
310        // so other futures on this tokio worker keep making progress.
311        let to_encode = body;
312        let encode_result = tokio::task::spawn_blocking(move || -> std::io::Result<Vec<u8>> {
313            let mut encoder = GzEncoder::new(Vec::new(), Compression::default());
314            encoder.write_all(&to_encode)?;
315            encoder.finish()
316        })
317        .await;
318        match encode_result {
319            Ok(Ok(buf)) => Bytes::from(buf),
320            Ok(Err(e)) => {
321                error!("gzip failed for '{filename}': {e:?}");
322                return Ok(maps_error(
323                    StatusCode::INTERNAL_SERVER_ERROR,
324                    "GZIP_FAILED",
325                    "Failed to compress upload",
326                ));
327            }
328            Err(e) => {
329                error!("gzip task join failed for '{filename}': {e:?}");
330                return Ok(maps_error(
331                    StatusCode::INTERNAL_SERVER_ERROR,
332                    "GZIP_FAILED",
333                    "Failed to compress upload",
334                ));
335            }
336        }
337    };
338
339    let stored_size = stored.len() as u64;
340    let key = stored_key(&filename);
341
342    if let Err(e) = store
343        .put(&ObjectPath::from(key.as_str()), stored.into())
344        .await
345    {
346        error!("storing map '{filename}': {e:?}");
347        return Ok(maps_error(
348            StatusCode::INTERNAL_SERVER_ERROR,
349            "STORE_FAILED",
350            "Failed to store map",
351        ));
352    }
353
354    info!(
355        "Uploaded map '{filename}' ({stored_size} bytes gzipped) by {sub}",
356        sub = user.email.as_deref().unwrap_or(&user.subject)
357    );
358
359    Ok((
360        StatusCode::OK,
361        Json(UploadResponse {
362            file: filename,
363            size: stored_size,
364        }),
365    )
366        .into_response())
367}

369/// DELETE /api/maps/blob/{filename}
370///
371/// Admin-only. Idempotent — returns 204 whether or not the object existed,
372/// matching S3's native DELETE semantics and avoiding a useless `head()`
373/// round-trip just to manufacture a 404.
374#[span_fn]
375pub async fn maps_delete(
376    Extension(state): Extension<MapsState>,
377    AdminUser(user): AdminUser,
378    AxumPath(filename): AxumPath<String>,
379) -> Result<Response, AdminRequired> {
380    if !is_direct_child(&filename) {
381        return Ok(maps_error(
382            StatusCode::BAD_REQUEST,
383            "INVALID_FILENAME",
384            "Filename must be a single path segment",
385        ));
386    }
387
388    let Some(store) = state.store else {
389        return Ok(maps_error(
390            StatusCode::SERVICE_UNAVAILABLE,
391            "MAPS_STORE_UNAVAILABLE",
392            "Maps endpoint not configured: set MICROMEGAS_MAPS_OBJECT_STORE_URI",
393        ));
394    };
395
396    let key = stored_key(&filename);
397    match store.delete(&ObjectPath::from(key.as_str())).await {
398        Ok(()) | Err(object_store::Error::NotFound { .. }) => {
399            info!(
400                "Deleted map '{filename}' by {sub}",
401                sub = user.email.as_deref().unwrap_or(&user.subject)
402            );
403            Ok(StatusCode::NO_CONTENT.into_response())
404        }
405        Err(e) => {
406            error!("deleting map '{filename}': {e:?}");
407            Ok(maps_error(
408                StatusCode::INTERNAL_SERVER_ERROR,
409                "DELETE_FAILED",
410                "Failed to delete map",
411            ))
412        }
413    }
