45/// GET /auth/login - Initiate OIDC login
46#[span_fn]
47pub async fn auth_login(
48    State(state): State<AuthState>,
49    Query(query): Query<LoginQuery>,
50) -> Result<impl IntoResponse, AuthApiError> {
51    let return_url = query.return_url.unwrap_or_else(|| "/".to_string());
52
53    // Validate return URL to prevent open redirect
54    if !validate_return_url(&return_url) {
55        return Err(AuthApiError::InvalidReturnUrl);
56    }
57
58    // Generate PKCE challenge
59    let (pkce_challenge, pkce_verifier) = PkceCodeChallenge::new_random_sha256();
60
61    // Generate state with nonce and return URL
62    let nonce = generate_nonce();
63    let oauth_state = OAuthState {
64        nonce: nonce.clone(),
65        return_url,
66        pkce_verifier: pkce_verifier.secret().to_string(),
67    };
68    // Sign the state with HMAC-SHA256 to prevent tampering
69    let state_signed = sign_state(&oauth_state, &state.state_signing_secret)
70        .map_err(|e| AuthApiError::Internal(format!("Failed to sign state: {e:?}")))?;
71
72    // Get OIDC provider and build client
73    let provider = state
74        .get_oidc_provider()
75        .await
76        .map_err(|e| AuthApiError::Internal(format!("Failed to get OIDC provider: {e:?}")))?;
77    let client = provider.build_client();
78
79    // Generate authorization URL
80    let (auth_url, _csrf_token, _nonce) = client
81        .authorize_url(
82            AuthenticationFlow::<CoreResponseType>::AuthorizationCode,
83            move || CsrfToken::new(state_signed.clone()),
84            Nonce::new_random,
85        )
86        .add_scope(Scope::new("openid".to_string()))
87        .add_scope(Scope::new("email".to_string()))
88        .add_scope(Scope::new("profile".to_string()))
89        .add_scope(Scope::new("offline_access".to_string()))
90        .set_pkce_challenge(pkce_challenge)
91        .url();
92
93    // Set cookie with nonce for validation
94    let cookie = create_cookie(OAUTH_STATE_COOKIE, nonce, 600, &state); // 10 minutes
95
96    Ok((
97        CookieJar::new().add(cookie),
98        Redirect::temporary(auth_url.as_str()),
99    ))
100}

102/// GET /auth/callback - Handle OIDC callback
103#[span_fn]
104pub async fn auth_callback(
105    State(state): State<AuthState>,
106    jar: CookieJar,
107    Query(query): Query<CallbackQuery>,
108) -> Result<impl IntoResponse, AuthApiError> {
109    // Verify and decode signed state parameter
110    let oauth_state = verify_state(&query.state, &state.state_signing_secret).map_err(|e| {
111        warn!("[auth_failure] reason=invalid_state details={e:?}");
112        AuthApiError::InvalidState
113    })?;
114
115    // Validate nonce from cookie
116    let cookie_nonce = jar
117        .get(OAUTH_STATE_COOKIE)
118        .ok_or_else(|| {
119            warn!("[auth_failure] reason=missing_oauth_state_cookie");
120            AuthApiError::InvalidState
121        })?
122        .value();
123
124    if cookie_nonce != oauth_state.nonce {
125        warn!("[auth_failure] reason=nonce_mismatch");
126        return Err(AuthApiError::InvalidState);
127    }
128
129    // Get OIDC provider
130    let provider = state
131        .get_oidc_provider()
132        .await
133        .map_err(|e| AuthApiError::Internal(format!("Failed to get OIDC provider: {e:?}")))?;
134
135    // Exchange code for tokens using manual HTTP request
136    // Note: We don't use the openidconnect library's exchange_code() because:
137    // - Auth0 includes non-standard fields (e.g., updated_at) that cause parsing failures
138    // - The library's strict typing doesn't handle provider-specific extensions well
139    // - Manual HTTP gives us better error visibility and control over parsing
140    let http_client = create_http_client()
141        .map_err(|e| AuthApiError::Internal(format!("Failed to create HTTP client: {e:?}")))?;
142
143    let token_url = provider
144        .metadata
145        .token_endpoint()
146        .expect("token endpoint should exist");
147
148    let params = [
149        ("grant_type", "authorization_code"),
150        ("code", &query.code),
151        ("redirect_uri", &state.config.redirect_uri),
152        ("client_id", &state.config.client_id),
153        ("code_verifier", &oauth_state.pkce_verifier),
154    ];
155
156    // Note: Generic error messages are intentional to avoid leaking authentication details
157    // Detailed errors are logged server-side for debugging
158    let response = http_client
159        .post(token_url.as_str())
160        .form(&params)
161        .send()
162        .await
163        .map_err(|e| {
164            warn!("[auth_failure] reason=token_exchange_request_failed details={e:?}");
165            AuthApiError::TokenExchangeFailed
166        })?;
167
168    if !response.status().is_success() {
169        let status = response.status();
170        let body = response
171            .text()
172            .await
173            .unwrap_or_else(|_| "unknown".to_string());
174        warn!("[auth_failure] reason=token_exchange_failed status={status} body={body}");
175        return Err(AuthApiError::TokenExchangeFailed);
176    }
177
178    let token_response: serde_json::Value = response.json().await.map_err(|e| {
179        warn!("[auth_failure] reason=token_response_parse_failed details={e:?}");
180        AuthApiError::TokenExchangeFailed
181    })?;
182
183    // Extract tokens from JSON response
184    let id_token = token_response["id_token"]
185        .as_str()
186        .ok_or_else(|| {
187            warn!("[auth_failure] reason=missing_id_token");
188            AuthApiError::TokenExchangeFailed
189        })?
190        .to_string();
191
192    let refresh_token = token_response["refresh_token"]
193        .as_str()
194        .map(|s| s.to_string());
195
196    // Calculate expiration times
197    let access_token_expires = token_response["expires_in"]
198        .as_u64()
199        .map(|d| d as i64)
200        .unwrap_or(3600); // Default 1 hour
201
202    let refresh_token_expires = 30 * 24 * 3600; // 30 days
203
204    // Log successful login (extract subject from token for audit trail)
205    if let Some(sub) = extract_subject_from_token(&id_token) {
206        info!(
207            "[auth_success] event=login sub={sub} issuer={}",
208            state.config.issuer
209        );
210    }
211
212    // Create cookies
213    let mut new_jar = jar;
214    new_jar = new_jar.add(create_cookie(
215        ID_TOKEN_COOKIE,
216        id_token,
217        access_token_expires,
218        &state,
219    ));
220
221    if let Some(refresh) = refresh_token {
222        new_jar = new_jar.add(create_cookie(
223            REFRESH_TOKEN_COOKIE,
224            refresh,
225            refresh_token_expires,
226            &state,
227        ));
228    }
229
230    // Clear oauth state cookie
231    new_jar = new_jar.add(clear_cookie(OAUTH_STATE_COOKIE, &state));
232
233    // Redirect to return URL
234    Ok((new_jar, Redirect::temporary(&oauth_state.return_url)))
235}

237/// POST /auth/refresh - Refresh tokens
238#[span_fn]
239pub async fn auth_refresh(
240    State(state): State<AuthState>,
241    jar: CookieJar,
242) -> Result<impl IntoResponse, AuthApiError> {
243    // Get refresh token from cookie
244    let refresh_token = jar
245        .get(REFRESH_TOKEN_COOKIE)
246        .ok_or(AuthApiError::Unauthorized)?
247        .value()
248        .to_string();
249
250    // Get OIDC provider
251    let provider = state
252        .get_oidc_provider()
253        .await
254        .map_err(|e| AuthApiError::Internal(format!("Failed to get OIDC provider: {e:?}")))?;
255
256    // Exchange refresh token for new tokens using manual HTTP request
257    // Note: Same reasoning as auth_callback - Auth0's non-standard fields break library parsing
258    let http_client = create_http_client()
259        .map_err(|e| AuthApiError::Internal(format!("Failed to create HTTP client: {e:?}")))?;
260
261    let token_url = provider
262        .metadata
263        .token_endpoint()
264        .expect("token endpoint should exist");
265
266    let params = [
267        ("grant_type", "refresh_token"),
268        ("refresh_token", &refresh_token),
269        ("client_id", &state.config.client_id),
270    ];
271
272    let response = http_client
273        .post(token_url.as_str())
274        .form(&params)
275        .send()
276        .await
277        .map_err(|e| {
278            warn!("[token_refresh_failure] reason=request_failed details={e:?}");
279            AuthApiError::Unauthorized
280        })?;
281
282    if !response.status().is_success() {
283        let status = response.status();
284        let body = response
285            .text()
286            .await
287            .unwrap_or_else(|_| "unknown".to_string());
288        warn!("[token_refresh_failure] reason=token_exchange_failed status={status} body={body}");
289        return Err(AuthApiError::Unauthorized);
290    }
291
292    let token_response: serde_json::Value = response.json().await.map_err(|e| {
293        warn!("[token_refresh_failure] reason=response_parse_failed details={e:?}");
294        AuthApiError::Unauthorized
295    })?;
296
297    // Extract new tokens from JSON response
298    let id_token = token_response["id_token"]
299        .as_str()
300        .ok_or_else(|| {
301            warn!("[token_refresh_failure] reason=missing_id_token");
302            AuthApiError::Unauthorized
303        })?
304        .to_string();
305
306    let refresh_token = token_response["refresh_token"]
307        .as_str()
308        .map(|s| s.to_string());
309
310    // Calculate expiration times
311    let id_token_expires = token_response["expires_in"]
312        .as_u64()
313        .map(|d| d as i64)
314        .unwrap_or(3600);
315
316    let refresh_token_expires = 30 * 24 * 3600; // 30 days
317
318    // Log successful token refresh
319    if let Some(sub) = extract_subject_from_token(&id_token) {
320        info!("[auth_success] event=token_refresh sub={sub}");
321    }
322
323    // Update cookies
324    let mut new_jar = jar;
325    new_jar = new_jar.add(create_cookie(
326        ID_TOKEN_COOKIE,
327        id_token,
328        id_token_expires,
329        &state,
330    ));
331
332    if let Some(refresh) = refresh_token {
333        new_jar = new_jar.add(create_cookie(
334            REFRESH_TOKEN_COOKIE,
335            refresh,
336            refresh_token_expires,
337            &state,
338        ));
339    }
340
341    Ok((new_jar, StatusCode::OK))
342}

354/// GET /auth/me - Get current user info
355///
356/// Returns user information from the validated JWT token.
357/// This endpoint must be behind the cookie_auth_middleware to ensure
358/// the token has been validated before extracting user info.
359///
360/// The user's `sub` (subject) and `email` come from the validated token claims.
361/// The `name` field is extracted directly from the JWT payload for display purposes.
362#[span_fn]
363pub async fn auth_me(
364    State(state): State<AuthState>,
365    jar: CookieJar,
366) -> Result<Json<UserInfo>, AuthApiError> {
367    // Get ID token from cookie
368    let id_token = jar
369        .get(ID_TOKEN_COOKIE)
370        .ok_or_else(|| {
371            debug!("no id_token cookie found");
372            AuthApiError::Unauthorized
373        })?
374        .value()
375        .to_string();
376
377    // Validate the token using the OIDC provider
378    let auth_provider = state.get_auth_provider().await.map_err(|e| {
379        warn!("[auth_failure] auth_provider_init_failed: {e:?}");
380        AuthApiError::Internal("Failed to initialize auth provider".to_string())
381    })?;
382
383    let parts = CookieTokenRequestParts {
384        token: id_token.clone(),
385    };
386
387    let auth_context = auth_provider.validate_request(&parts).await.map_err(|e| {
388        warn!("[auth_failure] {e}");
389        AuthApiError::InvalidToken
390    })?;
391
392    // Extract name from JWT payload (not in AuthContext)
393    // This is safe since we just validated the token
394    let name = extract_name_from_token(&id_token);
395
396    Ok(Json(UserInfo {
397        sub: auth_context.subject,
398        email: auth_context.email,
399        name,
400        is_admin: auth_context.is_admin,
401    }))
402}

404/// Authentication API errors
405#[derive(Debug)]
406pub enum AuthApiError {
407    InvalidReturnUrl,
408    InvalidState,
409    TokenExchangeFailed,
410    Unauthorized,
411    InvalidToken,
412    Internal(String),
413}
414
415impl IntoResponse for AuthApiError {
416    fn into_response(self) -> Response {
417        let (status, message) = match self {
418            AuthApiError::InvalidReturnUrl => (StatusCode::BAD_REQUEST, "Invalid return URL"),
419            AuthApiError::InvalidState => (StatusCode::BAD_REQUEST, "Invalid OAuth state"),
420            AuthApiError::TokenExchangeFailed => {
421                (StatusCode::UNAUTHORIZED, "Token exchange failed")
422            }
423            AuthApiError::Unauthorized => (StatusCode::UNAUTHORIZED, "Unauthorized"),
424            AuthApiError::InvalidToken => (StatusCode::UNAUTHORIZED, "Invalid token"),
425            AuthApiError::Internal(msg) => {
426                error!("Auth internal error: {msg}");
427                (StatusCode::INTERNAL_SERVER_ERROR, "Internal server error")
428            }
429        };
430
431        (status, message).into_response()
432    }
433}

435/// Cookie-based authentication middleware with full JWT signature validation
436///
437/// Reads the ID token from the Authorization: Bearer header (preferred) or
438/// httpOnly cookie (fallback), validates the signature using JWKS,
439/// and injects validated user info into request extensions.
440///
441/// Security: This middleware performs full JWT validation including:
442/// - Signature verification using cached JWKS from the OIDC provider
443/// - Issuer validation against configured issuers
444/// - Audience validation
445/// - Expiration check
446///
447/// Invalid or forged tokens are rejected before any downstream processing.
448///
449/// Note: We use the ID token (JWT) for FlightSQL API calls because:
450/// - ID tokens can be validated locally by both web tier and FlightSQL
451/// - This matches the Python API behavior which also uses ID tokens
452/// - Access tokens (JWE) would require token introspection endpoints
453#[span_fn]
454pub async fn cookie_auth_middleware(
455    State(state): State<AuthState>,
456    req: Request,
457    next: Next,
458) -> Result<Response, AuthApiError> {
459    // Check Authorization: Bearer header first, then fall back to cookie
460    let id_token = if let Some(bearer) = req
461        .headers()
462        .get(axum::http::header::AUTHORIZATION)
463        .and_then(|h| h.to_str().ok())
464        .and_then(|h| h.strip_prefix("Bearer "))
465    {
466        bearer.to_string()
467    } else {
468        // Fall back to cookie
469        let jar = CookieJar::from_headers(req.headers());
470        jar.get(ID_TOKEN_COOKIE)
471            .ok_or_else(|| {
472                debug!("no Authorization header or id_token cookie found");
473                AuthApiError::Unauthorized
474            })?
475            .value()
476            .to_string()
477    };
478
479    // Get the OIDC auth provider (lazy initialized with JWKS caching)
480    let auth_provider = state.get_auth_provider().await.map_err(|e| {
481        warn!("[auth_failure] auth_provider_init_failed: {e:?}");
482        AuthApiError::Internal("Failed to initialize auth provider".to_string())
483    })?;
484
485    // Create request parts adapter for the cookie token
486    let parts = CookieTokenRequestParts {
487        token: id_token.clone(),
488    };
489
490    // Validate the token with full signature verification
491    let auth_context = auth_provider.validate_request(&parts).await.map_err(|e| {
492        warn!("[auth_failure] {e}");
493        AuthApiError::InvalidToken
494    })?;
495
496    // Log successful authentication (trace level to avoid noise on every request)
497    trace!(
498        "[auth_success] subject={} email={:?} issuer={} admin={}",
499        auth_context.subject, auth_context.email, auth_context.issuer, auth_context.is_admin
500    );
501
502    // Store token and validated user info in request extensions
503    let mut req = req;
504    req.extensions_mut().insert(AuthToken(id_token));
505    req.extensions_mut()
506        .insert(ValidatedUser::from(&auth_context));
507
508    // Continue to next middleware/handler
509    Ok(next.run(req).await)
510}

517/// Returned by `require_admin` when the user is not an admin.
518///
519/// Implements `IntoResponse` as 403 with a JSON `{ code, message }` body
520/// matching the data-sources error shape, so handlers can `?`-propagate
521/// it directly or convert into a domain error enum at the boundary.
522#[derive(Debug)]
523pub struct AdminRequired;
524
525impl IntoResponse for AdminRequired {
526    fn into_response(self) -> Response {
527        let body = serde_json::json!({
528            "code": "FORBIDDEN",
529            "message": "Admin access required",
530        });
531        (StatusCode::FORBIDDEN, Json(body)).into_response()
532    }
533}
534
535/// Returns `Ok(())` when the user is an admin, otherwise an `AdminRequired`
536/// error that renders as a 403 with `{ code: "FORBIDDEN", message: ... }`.
537pub fn require_admin(user: &ValidatedUser) -> Result<(), AdminRequired> {
538    if user.is_admin {
539        Ok(())
540    } else {
541        Err(AdminRequired)
542    }
543}

545/// Extractor that yields the validated user only when `is_admin` is true.
546///
547/// Why: `FromRequestParts` runs before any body extractor (which is
548/// `FromRequest`), so handlers that take a `Bytes` upload body can use
549/// this to short-circuit with 403 *before* the body is buffered into
550/// memory. Doing the admin check inside the handler body would mean a
551/// non-admin authenticated user could force the server to buffer up to
552/// `DefaultBodyLimit` bytes per request before getting rejected.
553pub struct AdminUser(pub ValidatedUser);
554
555impl<S: Send + Sync> FromRequestParts<S> for AdminUser {
556    type Rejection = AdminRequired;
557
558    async fn from_request_parts(parts: &mut Parts, _state: &S) -> Result<Self, Self::Rejection> {
559        let user = parts
560            .extensions
561            .get::<ValidatedUser>()
562            .cloned()
563            .ok_or(AdminRequired)?;
564        if user.is_admin {
565            Ok(AdminUser(user))
566        } else {
567            Err(AdminRequired)
568        }
569    }
