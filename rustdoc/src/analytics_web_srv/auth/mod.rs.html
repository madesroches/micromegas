1//! Authentication endpoints for analytics-web-srv
2//!
3//! Implements OIDC authorization code flow with PKCE:
4//! - /auth/login - Initiate OIDC login
5//! - /auth/callback - Handle OIDC callback
6//! - /auth/refresh - Refresh tokens
7//! - /auth/logout - Clear session
8//! - /auth/me - Get current user info
9//!
10//! Security: All JWT tokens are fully validated (signature + claims) at this tier
11//! using the micromegas-auth crate with JWKS caching. Invalid tokens are rejected
12//! before forwarding requests to FlightSQL.
13//!
14//! This module is split by concern:
15//! - [`config`] — web-specific OIDC client configuration.
16//! - [`state`] — shared `AuthState` and its lazily-initialized caches.
17//! - [`cookies`] — cookie helpers.
18//! - [`claims`] — validated-user / JWT claim types and extraction.
19//! - [`handlers`] — the Axum handlers, middleware, and extractors.
20//!
21//! Login-flow OIDC client construction (provider discovery + client
22//! building) lives in the `micromegas::auth::oidc_client` crate module, not
23//! here — this crate only consumes it.
24
25mod claims;
26mod config;
27mod cookies;
28mod handlers;
29mod state;
30
31pub use claims::{UserInfo, ValidatedUser};
32pub use config::OidcClientConfig;
33pub use cookies::{clear_cookie, create_cookie};
34pub use handlers::{
35    AdminRequired, AdminUser, AuthApiError, AuthToken, auth_callback, auth_login, auth_logout,
36    auth_me, auth_refresh, cookie_auth_middleware, require_admin,
37};
38pub use state::AuthState;
