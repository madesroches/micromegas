1use crate::app_db::DataSourceConfig;
2use anyhow::Result;
3use moka::future::Cache;
4use sqlx::PgPool;
5use std::sync::Arc;
6use std::time::Duration;
7
8/// Cached value: `Some` for an existing data source, `None` for a confirmed miss.
9type CachedEntry = Option<Arc<DataSourceConfig>>;
10
11/// In-memory cache mapping data source names to their configs.
12///
13/// - Lazy loading: entries are loaded from PG on first access
14/// - TTL expiry: entries expire after a configurable duration so updates from
15///   other processes are picked up
16/// - CRUD on this process: invalidate the cache entry so the next resolve
17///   fetches fresh from PG
18#[derive(Clone)]
19pub struct DataSourceCache {
20    cache: Cache<String, CachedEntry>,
21    pool: PgPool,
22}
23
24impl DataSourceCache {
25    pub fn new(pool: PgPool, ttl: Duration) -> Self {
26        let cache = Cache::builder()
27            .time_to_live(ttl)
28            .max_capacity(1000)
29            .build();
30        Self { cache, pool }
31    }
32
33    /// Resolve a data source name to its config.
34    /// Returns `None` if the data source does not exist.
35    pub async fn resolve(&self, name: &str) -> Result<Option<DataSourceConfig>> {
36        let pool = self.pool.clone();
37        let name_owned = name.to_string();
38
39        let result = self
40            .cache
41            .try_get_with::<_, anyhow::Error>(name_owned.clone(), async {
42                let row = sqlx::query_scalar::<_, serde_json::Value>(
43                    "SELECT config FROM data_sources WHERE name = $1",
44                )
45                .bind(&name_owned)
46                .fetch_optional(&pool)
47                .await
48                .map_err(|e| anyhow::anyhow!("database error: {e}"))?;
49
50                match row {
51                    Some(config_json) => {
52                        let config: DataSourceConfig = serde_json::from_value(config_json)
53                            .map_err(|e| anyhow::anyhow!("invalid config: {e}"))?;
54                        Ok(Some(Arc::new(config)))
55                    }
56                    None => Ok(None),
57                }
58            })
59            .await;
60
61        match result {
62            Ok(Some(config)) => Ok(Some((*config).clone())),
63            Ok(None) => Ok(None),
64            Err(e) => Err(anyhow::anyhow!("{e}")),
65        }
66    }
67
68    /// Invalidate a cache entry after create/update/delete.
69    pub async fn invalidate(&self, name: &str) {
70        self.cache.invalidate(name).await;
71    }
72}
