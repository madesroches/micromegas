1use anyhow::{Context, Result};
2use sqlx::Executor;

4async fn create_migration_table(tr: &mut sqlx::Transaction<'_, sqlx::Postgres>) -> Result<()> {
5    sqlx::query("CREATE TABLE migration(version INTEGER NOT NULL);")
6        .execute(&mut **tr)
7        .await
8        .with_context(|| "Creating migration table")?;
9    sqlx::query("INSERT INTO migration VALUES(1);")
10        .execute(&mut **tr)
11        .await
12        .with_context(|| "Recording the initial schema version")?;
13    Ok(())
14}

16async fn create_screens_table(tr: &mut sqlx::Transaction<'_, sqlx::Postgres>) -> Result<()> {
17    let sql = "
18        CREATE TABLE screens(
19            name VARCHAR(255) PRIMARY KEY,
20            screen_type VARCHAR(50) NOT NULL,
21            config JSONB NOT NULL,
22            created_by VARCHAR(255),
23            updated_by VARCHAR(255),
24            created_at TIMESTAMPTZ DEFAULT NOW(),
25            updated_at TIMESTAMPTZ DEFAULT NOW()
26        );
27        CREATE INDEX screens_screen_type ON screens(screen_type);
28        CREATE INDEX screens_created_at ON screens(created_at);
29    ";
30    tr.execute(sql)
31        .await
32        .with_context(|| "Creating screens table and indices")?;
33    Ok(())
34}
