30/// Hand-rolled `google.rpc.Status` proto. The OTLP/HTTP spec mandates this on 4xx/5xx
31/// responses ("The response body for all HTTP 4xx and HTTP 5xx responses MUST be a
32/// Protobuf-encoded Status message that describes the problem.").
33///
34/// `details` is `repeated google.protobuf.Any` per the upstream proto; we omit it in v1.
35#[derive(Clone, PartialEq, ::prost::Message)]
36pub struct Status {
37    #[prost(int32, tag = "1")]
38    pub code: i32,
39    #[prost(string, tag = "2")]
40    pub message: String,
41}
