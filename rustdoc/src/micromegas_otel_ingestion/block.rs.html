1//! Splitting an OTLP `Export*ServiceRequest` into per-Resource micromegas blocks.
2//!
3//! An `ExportRequest` may carry multiple resources (different services). We split at
4//! the Resource boundary so each block has an unambiguous `process_id`.

19/// A single per-resource block ready to be written. Carries everything the ingestion
20/// service needs to register the process + stream + block.
21pub struct PreparedBlock {
22    pub process_id: Uuid,
23    pub stream_id: Uuid,
24    pub block: Block,
25    pub signal: SignalKey,
26    pub resource_attrs: Vec<KeyValue>,
27    /// Number of records in the resource submessage (logs / metric data points / spans).
28    pub nb_records: i32,
29    /// Bounds derived from per-record timestamps. Used both for the `Block` envelope
30    /// and for the `processes.start_time` fallback when synthesizing a brand new process.
31    pub begin_time: DateTime<Utc>,
32    pub end_time: DateTime<Utc>,
33}
34
35/// Walks `ResourceLogs` to find min/max `time_unix_nano`. Falls back to
36/// `observed_time_unix_nano` when the per-record `time` is 0.
37///
38/// Records with no usable timestamp still count toward `count` even though the
39/// downstream processor drops them — keeps `block.nb_objects` consistent with
40/// the proto payload bytes (matching the metrics-side trade-off).
41fn logs_bounds(rl: &ResourceLogs) -> Option<(i64, i64, i32)> {
42    let mut min = i64::MAX;
43    let mut max = i64::MIN;
44    let mut count = 0i32;
45    for scope in &rl.scope_logs {
46        for record in &scope.log_records {
47            count += 1;
48            let t = if record.time_unix_nano != 0 {
49                record.time_unix_nano as i64
50            } else if record.observed_time_unix_nano != 0 {
51                record.observed_time_unix_nano as i64
52            } else {
53                continue;
54            };
55            min = min.min(t);
56            max = max.max(t);
57        }
58    }
59    if min == i64::MAX {
60        if count == 0 {
61            None
62        } else {
63            // All records have zero timestamps — fall back to wall clock at handler.
64            Some((0, 0, count))
65        }
66    } else {
67        Some((min, max, count))
68    }
69}

71/// Walks `ResourceSpans` for min(start_time) / max(end_time).
72///
73/// Spans with `start/end_time_unix_nano == 0` or wrong-sized `trace_id`/`span_id`
74/// still count toward `count` even though the downstream processor drops them —
75/// keeps `block.nb_objects` consistent with the proto payload bytes (matching
76/// the metrics-side trade-off).
77fn spans_bounds(rs: &ResourceSpans) -> Option<(i64, i64, i32)> {
78    let mut min = i64::MAX;
79    let mut max = i64::MIN;
80    let mut count = 0i32;
81    for scope in &rs.scope_spans {
82        for span in &scope.spans {
83            count += 1;
84            if span.start_time_unix_nano != 0 {
85                min = min.min(span.start_time_unix_nano as i64);
86            }
87            if span.end_time_unix_nano != 0 {
88                max = max.max(span.end_time_unix_nano as i64);
89            }
90        }
91    }
92    if count == 0 {
93        None
94    } else if min == i64::MAX || max == i64::MIN {
95        Some((0, 0, count))
96    } else {
97        Some((min, max, count))
98    }
99}

101/// Walks `ResourceMetrics` for min/max `time_unix_nano` across every Sum/Gauge/Histogram point.
102/// Histogram/ExponentialHistogram/Summary points still count toward bounds even though
103/// the v1 processor skips them — keeps block insert-time predicates consistent with
104/// payload contents.
105fn metrics_bounds(rm: &ResourceMetrics) -> Option<(i64, i64, i32)> {
106    use crate::proto::metric::Data;
107    let mut min = i64::MAX;
108    let mut max = i64::MIN;
109    let mut count = 0i32;
110    for scope in &rm.scope_metrics {
111        for metric in &scope.metrics {
112            match metric.data.as_ref() {
113                Some(Data::Sum(s)) => {
114                    for dp in &s.data_points {
115                        count += 1;
116                        let t = dp.time_unix_nano as i64;
117                        if t != 0 {
118                            min = min.min(t);
119                            max = max.max(t);
120                        }
121                    }
122                }
123                Some(Data::Gauge(g)) => {
124                    for dp in &g.data_points {
125                        count += 1;
126                        let t = dp.time_unix_nano as i64;
127                        if t != 0 {
128                            min = min.min(t);
129                            max = max.max(t);
130                        }
131                    }
132                }
133                Some(Data::Histogram(h)) => {
134                    for dp in &h.data_points {
135                        count += 1;
136                        let t = dp.time_unix_nano as i64;
137                        if t != 0 {
138                            min = min.min(t);
139                            max = max.max(t);
140                        }
141                    }
142                }
143                Some(Data::ExponentialHistogram(h)) => {
144                    for dp in &h.data_points {
145                        count += 1;
146                        let t = dp.time_unix_nano as i64;
147                        if t != 0 {
148                            min = min.min(t);
149                            max = max.max(t);
150                        }
151                    }
152                }
153                Some(Data::Summary(s)) => {
154                    for dp in &s.data_points {
155                        count += 1;
156                        let t = dp.time_unix_nano as i64;
157                        if t != 0 {
158                            min = min.min(t);
159                            max = max.max(t);
160                        }
161                    }
162                }
163                None => {}
164            }
165        }
166    }
167    if count == 0 {
168        None
169    } else if min == i64::MAX || max == i64::MIN {
170        Some((0, 0, count))
171    } else {
172        Some((min, max, count))
173    }
174}

176fn nanos_to_datetime(nanos: i64) -> DateTime<Utc> {
177    DateTime::from_timestamp_nanos(nanos)
178}

180/// Builds a single `PreparedBlock` from a per-resource payload.
181fn build_prepared_block(
182    payload_bytes: Vec<u8>,
183    resource_attrs: Vec<KeyValue>,
184    signal: SignalKey,
185    bounds: (i64, i64, i32),
186) -> PreparedBlock {
187    let (min_nanos, max_nanos, nb_records) = bounds;
188    let (begin_time, end_time) = if min_nanos == 0 && max_nanos == 0 {
189        let now = Utc::now();
190        (now, now)
191    } else {
192        (nanos_to_datetime(min_nanos), nanos_to_datetime(max_nanos))
193    };
194
195    let resource = crate::proto::Resource {
196        attributes: resource_attrs.clone(),
197        dropped_attributes_count: 0,
198        entity_refs: vec![],
199    };
200    let process_id = process_id_from_resource(Some(&resource));
201    let stream_id = stream_id_from_process_signal(process_id, signal);
202    let block_id = block_id_from_payload(&payload_bytes);
203
204    if is_degenerate_resource(&resource_attrs) {
205        // debug! rather than warn! — fires per resource per request, so a misconfigured
206        // client would otherwise log-flood. Operators chasing collapsed-identity bugs
207        // can switch the level on demand.
208        debug!(
209            "OTLP resource without host.id/host.name/process.pid/service.instance.id — \
210             multiple processes may collapse onto process_id={}",
211            process_id
212        );
213    }
214
215    let nb_objects = nb_records;
216    let begin_ticks = begin_time.timestamp_nanos_opt().unwrap_or(0);
217    let end_ticks = end_time.timestamp_nanos_opt().unwrap_or(0);
218
219    let block = Block {
220        block_id,
221        stream_id,
222        process_id,
223        begin_time: begin_time.to_rfc3339(),
224        begin_ticks,
225        end_time: end_time.to_rfc3339(),
226        end_ticks,
227        payload: BlockPayload {
228            dependencies: Vec::new(),
229            objects: payload_bytes,
230        },
231        object_offset: 0,
232        nb_objects,
233    };
234
235    PreparedBlock {
236        process_id,
237        stream_id,
238        block,
239        signal,
240        resource_attrs,
241        nb_records,
242        begin_time,
243        end_time,
244    }
245}

247/// Splits a logs request into per-resource blocks.
248pub fn split_logs(req: crate::proto::ExportLogsServiceRequest) -> Result<Vec<PreparedBlock>> {
249    let mut out = Vec::with_capacity(req.resource_logs.len());
250    for rl in req.resource_logs {
251        let Some(bounds) = logs_bounds(&rl) else {
252            continue;
253        };
254        let resource_attrs = rl
255            .resource
256            .as_ref()
257            .map(|r| r.attributes.clone())
258            .unwrap_or_default();
259        let payload_bytes = rl.encode_to_vec();
260        out.push(build_prepared_block(
261            payload_bytes,
262            resource_attrs,
263            SignalKey::Logs,
264            bounds,
265        ));
266    }
267    Ok(out)
268}

298            continue;
299        };

307            payload_bytes,
308            resource_attrs,
309            SignalKey::Traces,
310            bounds,
311        ));
312    }
313    Ok(out)
314}

316/// Pulls the well-known process attributes off a resource to build the row that goes
317/// in the `processes` table. Everything else gets folded into `properties` under
318/// `otel.resource.*` so it's queryable but doesn't bloat the typed columns.
319pub struct ProcessFromResource {
320    pub exe: String,
321    pub username: String,
322    pub computer: String,
323    pub distro: String,
324    pub cpu_brand: String,
325    pub start_time: DateTime<Utc>,
326    pub start_ticks: i64,
327    pub properties: Vec<Property>,
328}
329
330/// `processes` columns are declared `VARCHAR(255)`. OTel attributes have no length
331/// contract — `os.description` in particular can carry a full kernel banner well past
332/// 255 chars. Truncate by character (not byte) so we never split a multi-byte codepoint.
333/// Untruncated values would cause a Postgres "value too long" error → 503 → SDK retry loop.
334const PROCESS_FIELD_MAX_CHARS: usize = 255;
335
336fn truncate_for_db(s: String) -> String {
337    if s.chars().count() <= PROCESS_FIELD_MAX_CHARS {
338        s
339    } else {
340        s.chars().take(PROCESS_FIELD_MAX_CHARS).collect()
341    }
342}
343
344impl ProcessFromResource {
345    pub fn build(attrs: &[KeyValue], fallback_start: DateTime<Utc>) -> Self {
346        let svc_name = crate::identity::attr(attrs, "service.name")
347            .map(crate::identity::attr_to_string)
348            .unwrap_or_default();
349        let svc_ns = crate::identity::attr(attrs, "service.namespace")
350            .map(crate::identity::attr_to_string)
351            .unwrap_or_default();
352        let exe = if svc_ns.is_empty() {
353            svc_name
354        } else {
355            format!("{svc_ns}/{svc_name}")
356        };
357        let exe = truncate_for_db(exe);
358
359        let username = truncate_for_db(
360            crate::identity::attr(attrs, "user.name")
361                .map(crate::identity::attr_to_string)
362                .unwrap_or_default(),
363        );
364        let computer = truncate_for_db(
365            crate::identity::attr(attrs, "host.name")
366                .map(crate::identity::attr_to_string)
367                .unwrap_or_default(),
368        );
369        let distro = truncate_for_db(
370            crate::identity::attr(attrs, "os.description")
371                .map(crate::identity::attr_to_string)
372                .unwrap_or_default(),
373        );
374        let cpu_brand = truncate_for_db(
375            crate::identity::attr(attrs, "host.cpu.model.name")
376                .map(crate::identity::attr_to_string)
377                .unwrap_or_default(),
378        );
379
380        // Prefer the OTel-stable `process.creation.time`; accept legacy `process.start_time` as a fallback.
381        let start_time = crate::identity::attr(attrs, "process.creation.time")
382            .or_else(|| crate::identity::attr(attrs, "process.start_time"))
383            .and_then(|v| DateTime::parse_from_rfc3339(&attr_to_string(v)).ok())
384            .map(|dt| dt.with_timezone(&Utc))
385            .unwrap_or(fallback_start);
386        let start_ticks = start_time.timestamp_nanos_opt().unwrap_or(0);
387
388        // Properties: everything else, plus the well-known fields under otel.resource.* for queryability.
389        let mut properties = Vec::with_capacity(attrs.len());
390        for kv in attrs {
391            let value = kv.value.as_ref().map(attr_to_string).unwrap_or_default();
392            properties.push(Property::new(
393                std::sync::Arc::new(format!("otel.resource.{}", kv.key)),
394                std::sync::Arc::new(value),
395            ));
396        }
397
398        ProcessFromResource {
399            exe,
400            username,
401            computer,
402            distro,
403            cpu_brand,
404            start_time,
405            start_ticks,
406            properties,
407        }
408    }
