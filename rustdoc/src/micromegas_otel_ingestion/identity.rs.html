1//! Synthesizing micromegas Process / Stream / Block identity from OTLP `Resource` attributes.
2//!
3//! OTel has no `Process` object; just a `Resource` carrying `repeated KeyValue attributes`.
4//! We hash an OS-honest tuple of identifying attributes to a UUIDv5. Long-term stability of
5//! `process_id` values across upgrades is not a design goal — re-deriving existing ids is
6//! always acceptable, so the formula can be extended in-place when needed.
7
8use crate::proto::{AnyValue, KeyValue, Resource, any_value};
9use micromegas_tracing::prelude::*;
10use std::sync::Once;
11use uuid::{Uuid, uuid};
12
13/// `AnyValue.string_value_strindex` / `KeyValue.key_strindex` reference a
14/// `ProfilesDictionary.string_table` that exists **only** for the Profiling signal. Per the
15/// OTLP spec, receivers of logs/metrics/traces MUST treat these as absent. Warn once per
16/// process so a misconfigured profiling producer is noticeable without flooding the logs.
17fn warn_unexpected_strindex() {
18    static WARNED: Once = Once::new();
19    WARNED.call_once(|| {
20        warn!(
21            "ignoring profiling-only string-interning field on a non-profiling OTLP signal; \
22             treating it as absent (OTLP spec)"
23        );
24    });
25}

29pub const NS_OTEL_PROCESS_V1: Uuid = uuid!("80a447b8-fcdd-42a6-a613-f6c8719cd5fe");

37/// ASCII unit separator — used between concatenated string fields in identity formulas
38/// to prevent tuple-boundary collisions like `("abc", "")` vs `("ab", "c")`.
39const SEPARATOR: char = '\x1F';
40const SEPARATOR_STR: &str = "\x1F";
41
42/// OTel signal label used in stream-id derivation.
43#[derive(Debug, Clone, Copy)]
44pub enum SignalKey {
45    Logs,
46    Metrics,
47    Traces,
48}
49
50impl SignalKey {
51    pub fn as_str(&self) -> &'static str {
52        match self {
53            Self::Logs => "logs",
54            Self::Metrics => "metrics",
55            Self::Traces => "traces",
56        }
57    }
58}

60/// Convenience accessor — fetch one resource attribute by key, returning `None` when absent.
61///
62/// Matches on `kv.key` only; `kv.key_strindex` (profiling-only) is intentionally ignored, so an
63/// interned key (empty `key`) is treated as absent — the OTLP spec behavior for non-profiling
64/// signals.
65pub fn attr<'a>(attrs: &'a [KeyValue], key: &str) -> Option<&'a AnyValue> {
66    attrs
67        .iter()
68        .find(|kv| kv.key == key)
69        .and_then(|kv| kv.value.as_ref())
70}
71
72/// Returns the attribute's value rendered as a stable string.
73///
74/// - `string_value` → as-is
75/// - `int_value`    → decimal (some SDKs emit timestamps as int nanos)
76/// - `bool_value` / `double_value` / `bytes_value` / `array_value` / `kvlist_value` →
77///   their `Debug` form is unstable, so we just stringify with `format!`. In practice
78///   the resource attributes that feed identity are always strings or ints, but having
79///   a fallback prevents identity drift if an SDK emits something exotic.
80pub fn attr_to_string(v: &AnyValue) -> String {
81    match v.value.as_ref() {
82        Some(any_value::Value::StringValue(s)) => s.clone(),
83        Some(any_value::Value::IntValue(i)) => i.to_string(),
84        Some(any_value::Value::BoolValue(b)) => b.to_string(),
85        Some(any_value::Value::DoubleValue(d)) => d.to_string(),
86        Some(any_value::Value::BytesValue(b)) => format!("{b:?}"),
87        // Profiling-only string-table reference: no dictionary exists for this signal, so the
88        // index is meaningless. Treat as absent (per OTLP spec). NOTE: this value feeds the
89        // load-bearing UUIDv5 identity hash — never stringify the index back into it.
90        Some(any_value::Value::StringValueStrindex(_)) => {
91            warn_unexpected_strindex();
92            String::new()
93        }
94        Some(any_value::Value::ArrayValue(_)) | Some(any_value::Value::KvlistValue(_)) => {
95            // Structured values shouldn't appear in identity-bearing fields. If one does,
96            // hashing the Debug form is at least deterministic for a given prost version.
97            format!("{:?}", v.value)
98        }
99        None => String::new(),
100    }
101}

103/// Lower-case + trim. Applied to free-form string fields where the SDK may render
104/// the same logical value with different casing.
105fn norm(s: &str) -> String {
106    s.trim().to_ascii_lowercase()
107}
108
109/// Reads `attr` and returns the lower-cased + trimmed string value (or empty).
110fn attr_norm(attrs: &[KeyValue], key: &str) -> String {
111    attr(attrs, key)
112        .map(|v| norm(&attr_to_string(v)))
113        .unwrap_or_default()
114}
115
116/// Reads `attr` as-is (no case folding) — used for opaque values like `process.start_time`.
117fn attr_raw(attrs: &[KeyValue], key: &str) -> String {
118    attr(attrs, key).map(attr_to_string).unwrap_or_default()
119}

121/// Resolves the process owner's username.
122///
123/// `process.owner` is the OTel semantic-conventions attribute emitted by process resource
124/// detectors. When it is absent we fall back, in decreasing semantic closeness, to the
125/// process-scoped effective (`process.user.name`) and real (`process.real_user.name`) user
126/// names, then to the generic `user.name` for producers that only set that.
127pub fn process_owner_string(attrs: &[KeyValue]) -> String {
128    for key in [
129        "process.owner",
130        "process.user.name",
131        "process.real_user.name",
132        "user.name",
133    ] {
134        let s = attr_raw(attrs, key);
135        if !s.is_empty() {
136            return s;
137        }
138    }
139    String::new()
140}

142/// Resolves the OTel process-creation timestamp.
143///
144/// `process.creation.time` is the stable OTel semantic-conventions attribute and is
145/// what real SDKs emit; `process.start_time` is accepted as a fallback for any
146/// non-standard producer that still uses the older name.
147pub fn process_start_string(attrs: &[KeyValue]) -> String {
148    let s = attr_raw(attrs, "process.creation.time");
149    if !s.is_empty() {
150        return s;
151    }
152    attr_raw(attrs, "process.start_time")
153}

155/// Returns true when none of the four identifying fields are populated. Caller may want
156/// to log a warning so a degenerate-resource scenario doesn't silently collapse multiple
157/// processes onto one `process_id`.
158pub fn is_degenerate_resource(attrs: &[KeyValue]) -> bool {
159    attr_norm(attrs, "host.id").is_empty()
160        && attr_norm(attrs, "host.name").is_empty()
161        && attr_raw(attrs, "process.pid").is_empty()
162        && attr_norm(attrs, "service.instance.id").is_empty()
163}

165/// Derives `process_id` from a resource by hashing the identifying tuple under `NS_OTEL_PROCESS_V1`.
166///
167/// All fields pass through `attr_norm` (lower-case + trim) except `process.pid` and
168/// `process.creation.time` which are `attr_raw`. Field order (separated by `\x1F`):
169///
170///   host.id · host.name · process.pid · process.creation.time ·
171///   service.namespace · service.name · service.instance.id · process.owner ·
172///   os.type · os.version · os.name · os.description · os.build_id ·
173///   host.arch · host.type ·
174///   host.image.id · host.image.name · host.image.version ·
175///   host.cpu.model.id · host.cpu.model.name · host.cpu.family ·
176///   host.cpu.vendor.id · host.cpu.stepping · host.cpu.cache.l2.size ·
177///   service.version ·
178///   telemetry.sdk.name · telemetry.sdk.language · telemetry.sdk.version ·
179///   process.runtime.name · process.runtime.version · process.runtime.description
180///
181/// Fields are appended in-place under the same namespace UUID rather than bumping to `_V2` —
182/// the same pattern used when `process.owner` was added. Long-term stability of `process_id`
183/// values is not a design goal; re-deriving existing ids is always acceptable.
184pub fn process_id_from_resource(resource: Option<&Resource>) -> Uuid {
185    let attrs = resource.map(|r| r.attributes.as_slice()).unwrap_or(&[]);
186
187    let fields = [
188        attr_norm(attrs, "host.id"),
189        attr_norm(attrs, "host.name"),
190        attr_raw(attrs, "process.pid"),
191        process_start_string(attrs),
192        attr_norm(attrs, "service.namespace"),
193        attr_norm(attrs, "service.name"),
194        attr_norm(attrs, "service.instance.id"),
195        norm(&process_owner_string(attrs)),
196        // OS identity
197        attr_norm(attrs, "os.type"),
198        attr_norm(attrs, "os.version"),
199        attr_norm(attrs, "os.name"),
200        attr_norm(attrs, "os.description"),
201        attr_norm(attrs, "os.build_id"),
202        // Host hardware
203        attr_norm(attrs, "host.arch"),
204        attr_norm(attrs, "host.type"),
205        attr_norm(attrs, "host.image.id"),
206        attr_norm(attrs, "host.image.name"),
207        attr_norm(attrs, "host.image.version"),
208        attr_norm(attrs, "host.cpu.model.id"),
209        attr_norm(attrs, "host.cpu.model.name"),
210        attr_norm(attrs, "host.cpu.family"),
211        attr_norm(attrs, "host.cpu.vendor.id"),
212        attr_norm(attrs, "host.cpu.stepping"),
213        attr_norm(attrs, "host.cpu.cache.l2.size"),
214        // Service / SDK
215        attr_norm(attrs, "service.version"),
216        attr_norm(attrs, "telemetry.sdk.name"),
217        attr_norm(attrs, "telemetry.sdk.language"),
218        attr_norm(attrs, "telemetry.sdk.version"),
219        // Runtime
220        attr_norm(attrs, "process.runtime.name"),
221        attr_norm(attrs, "process.runtime.version"),
222        attr_norm(attrs, "process.runtime.description"),
223    ];
224
225    let key = fields.join(SEPARATOR_STR);
226    Uuid::new_v5(&NS_OTEL_PROCESS_V1, key.as_bytes())
227}

229/// Derives `stream_id` from `(process_id, signal)`. Max three streams per process.
230pub fn stream_id_from_process_signal(process_id: Uuid, signal: SignalKey) -> Uuid {
231    let key = format!("{process_id}{}{}", SEPARATOR, signal.as_str());
232    Uuid::new_v5(&NS_OTEL_STREAM_V1, key.as_bytes())
233}

235/// Derives `block_id` from the re-encoded protobuf bytes of one Resource submessage.
236/// `Uuid::new_v5` SHA-1s its input internally, so we don't pre-hash.
237pub fn block_id_from_payload(payload: &[u8]) -> Uuid {
238    Uuid::new_v5(&NS_OTEL_BLOCK_V1, payload)
