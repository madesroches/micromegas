1//! Errors surfaced by the OTLP ingestion adapter.
2//!
3//! Variants map onto HTTP status codes at the handler boundary:
4//!  - `Parse`        → 400 (malformed protobuf, malformed gzip)
5//!  - `Database`     → 503 (transient — client should retry per OTLP/HTTP spec)
6//!  - `Storage`      → 503 (transient)
7//!
8//! 415 (Content-Type / Content-Encoding) and 413 (body limit) are enforced upstream
9//! in the axum layer stack before the request reaches the OtelError surface.
10
11use micromegas_ingestion::web_ingestion_service::IngestionServiceError;
12use thiserror::Error;
13
14/// OTLP signal name (used purely for diagnostic messages).
15#[derive(Debug, Clone, Copy)]
16pub enum Signal {
17    Logs,
18    Metrics,
19    Traces,
20}
21
22impl Signal {
23    pub fn as_str(&self) -> &'static str {
24        match self {
25            Signal::Logs => "logs",
26            Signal::Metrics => "metrics",
27            Signal::Traces => "traces",
28        }
29    }
30}
31
32impl std::fmt::Display for Signal {
33    fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
34        f.write_str(self.as_str())
35    }
36}
37
38#[derive(Debug, Error)]
39pub enum OtelError {
40    /// Malformed protobuf, malformed gzip, or unrepresentable input.
41    /// Maps to 400.
42    #[error("OTLP parse error ({signal}): {message}")]
43    Parse { signal: Signal, message: String },
44
45    /// PostgreSQL transient failure. Maps to 503 + Retry-After.
46    #[error("OTLP database error ({signal}): {message}")]
47    Database { signal: Signal, message: String },
48
49    /// Object-store transient failure. Maps to 503 + Retry-After.
50    #[error("OTLP storage error ({signal}): {message}")]
51    Storage { signal: Signal, message: String },
52}
53
54impl OtelError {
55    pub fn signal(&self) -> Signal {
56        match self {
57            Self::Parse { signal, .. }
58            | Self::Database { signal, .. }
59            | Self::Storage { signal, .. } => *signal,
60        }
61    }
62
63    /// gRPC canonical `Code` for the embedded `google.rpc.Status` proto on error responses.
64    /// Despite the name, this travels over OTLP/HTTP — the spec just reuses `google.rpc.Status`
65    /// (and its gRPC code enum) as the error body format.
66    pub fn grpc_code(&self) -> i32 {
67        match self {
68            // INVALID_ARGUMENT = 3
69            Self::Parse { .. } => 3,
70            // UNAVAILABLE = 14
71            Self::Database { .. } | Self::Storage { .. } => 14,
72        }
73    }
74
75    /// HTTP status code for the response.
76    pub fn http_status(&self) -> u16 {
77        match self {
78            Self::Parse { .. } => 400,
79            Self::Database { .. } | Self::Storage { .. } => 503,
80        }
81    }
82
83    /// True when the OTLP/HTTP spec marks this status retryable.
84    pub fn is_retryable(&self) -> bool {
85        matches!(self, Self::Database { .. } | Self::Storage { .. })
86    }
87
88    /// Client-facing message for the `google.rpc.Status` body. Strips internal
89    /// detail (raw sqlx errors, object-store messages) that the full `Display`
90    /// form would otherwise leak — those still get logged server-side via
91    /// `error!`. `Parse` keeps its detail so clients can debug malformed
92    /// payloads, since prost / decoder messages don't reference server state.
93    pub fn public_message(&self) -> String {
94        match self {
95            Self::Parse { signal, message } => format!("OTLP parse error ({signal}): {message}"),
96            Self::Database { signal, .. } => format!("OTLP database error ({signal})"),
97            Self::Storage { signal, .. } => format!("OTLP storage error ({signal})"),
98        }
99    }
100}
101
102impl OtelError {
103    /// Wraps an `IngestionServiceError` with the OTLP signal of the request that
104    /// triggered it. Forces the caller to supply the signal at the conversion
105    /// site so the resulting label can't be mismatched against the route.
106    pub fn from_ingestion(err: IngestionServiceError, signal: Signal) -> Self {
107        match err {
108            IngestionServiceError::ParseError(m) => OtelError::Parse { signal, message: m },
109            IngestionServiceError::DatabaseError(m) => OtelError::Database { signal, message: m },
110            IngestionServiceError::StorageError(m) => OtelError::Storage { signal, message: m },
111        }
112    }
113}
