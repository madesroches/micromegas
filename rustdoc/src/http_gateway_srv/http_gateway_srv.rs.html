1use std::net::SocketAddr;
2use std::sync::Arc;
3
4use anyhow::Result;
5use axum::{Extension, Router};
6use clap::Parser;
7use micromegas::micromegas_main;
8use micromegas::servers;
9use micromegas::{axum, tracing::info};
10
11#[derive(Parser, Debug)]
12#[clap(name = "Micromegas http gateway server")]
13#[clap(about = "Micromegas http gateway server", version, author)]
14struct Cli {
15    #[clap(long, default_value = "0.0.0.0:3000")]
16    listen_endpoint_http: SocketAddr,
17}
18
19#[micromegas_main]
20async fn main() -> Result<()> {
21    let args = Cli::parse();
22
23    // Load header forwarding configuration
24    let header_config = Arc::new(servers::http_gateway::HeaderForwardingConfig::from_env()?);
25
26    let app = servers::http_gateway::register_routes(Router::new())
27        .layer(Extension(header_config))
28        .into_make_service_with_connect_info::<SocketAddr>();
29
30    let listener = tokio::net::TcpListener::bind(args.listen_endpoint_http).await?;
31    info!("Server running on {}", args.listen_endpoint_http);
32    axum::serve(listener, app).await?;
33    Ok(())
34}
