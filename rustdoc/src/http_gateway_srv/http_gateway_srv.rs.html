1use std::net::SocketAddr;
2
3use anyhow::Result;
4use axum::Router;
5use clap::Parser;
6use micromegas::micromegas_main;
7use micromegas::servers;
8use micromegas::{axum, tracing::info};
9
10#[derive(Parser, Debug)]
11#[clap(name = "Micromegas http gateway server")]
12#[clap(about = "Micromegas http gateway server", version, author)]
13struct Cli {
14    #[clap(long, default_value = "0.0.0.0:3000")]
15    listen_endpoint_http: SocketAddr,
16}
17
18#[micromegas_main]
19async fn main() -> Result<()> {
20    let args = Cli::parse();
21    let app = servers::http_gateway::register_routes(Router::new());
22    let listener = tokio::net::TcpListener::bind(args.listen_endpoint_http).await?;
23    info!("Server running on {}", args.listen_endpoint_http);
24    axum::serve(listener, app).await?;
25    Ok(())
26}
