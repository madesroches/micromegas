27/// Maximum number of ranges accepted in a single multi-range request. A
28/// parquet/block reader fetches at most a few thousand column chunks per file,
29/// so this is comfortably above legitimate use while bounding per-request work.
30const MAX_RANGES_PER_REQUEST: usize = 4096;
31
32/// Maximum size of a single NDJSON line in a `POST /prefetch` body. An item at
33/// the `MAX_RANGES_PER_REQUEST` cap serializes to roughly 100 KiB, so this
34/// leaves about 10x headroom while still bounding per-line memory; unlike a
35/// whole-body cap, this doesn't limit how many items a client can batch.
36const MAX_PREFETCH_LINE_BYTES: usize = 1024 * 1024; // 1 MiB

38pub const BYTES_PER_MEM_PERMIT: u64 = 1024 * 1024;

40/// Number of `mem_permits` (1 MiB each) needed to cover `bytes`.
41pub fn permits_for_bytes(bytes: u64) -> u32 {
42    bytes.div_ceil(BYTES_PER_MEM_PERMIT) as u32
43}
44
45/// Byte size of the fixed window a streaming request's memory charge is
46/// capped at: `permits_for_bytes(min(response size, this))`. Shared by the
47/// handlers' proportional per-stream charge below and by the startup guard
48/// in `object_cache_srv.rs`, which floors `--memory-budget-mb` at this value
49/// so a large read's charge can never exceed the whole budget — which would
50/// otherwise make `acquire_many_owned` hang forever instead of failing fast.
51pub fn stream_window_bytes(block_size: u64) -> u64 {
52    2 * DEMAND_WINDOW_BLOCKS * block_size
53}

55/// A response body wrapping a byte stream plus a memory-budget permit held
56/// for the body's entire lifetime: the permit is released whenever this value
57/// is dropped, whether that's after the body was fully sent or because the
58/// connection was aborted mid-stream. This is what makes the memory-budget
59/// guard cover the response's full lifetime rather than just the fetch
60/// window.
61struct PermitBody {
62    stream: BoxStream<'static, Result<Bytes, anyhow::Error>>,
63    _permit: OwnedSemaphorePermit,
64}

70        Pin::new(&mut self.get_mut().stream).poll_next(cx)
71    }
72}

84/// Wrap `inner` so that `on_complete` is called exactly once with the total
85/// bytes yielded, as soon as the payload is fully produced. When
86/// `expected_total` is known, the callback fires immediately BEFORE yielding
87/// the chunk that completes it — a `Content-Length`-framed HTTP body is
88/// considered complete by the transport once the declared byte count is
89/// written and is never polled again for a terminal `None`, so firing after
90/// the final `yield` (or on the terminal `None`) would never run in practice.
91/// A mid-stream `Err`, or a stream that ends before reaching `expected_total`,
92/// skips the callback — preserving the accepted under-reporting on truncation.
93/// When `expected_total` is `None` (length genuinely unknown up front), there
94/// is no "before completion" point to detect, so the callback instead fires
95/// on the terminal `None`, with whichever total was accumulated by then.
96fn count_bytes_served<F>(
97    mut inner: BoxStream<'static, Result<Bytes, anyhow::Error>>,
98    expected_total: Option<u64>,
99    on_complete: F,
100) -> BoxStream<'static, Result<Bytes, anyhow::Error>>
101where
102    F: FnOnce(u64) + Send + 'static,
103{
104    gen_stream! {
105        let mut total = 0u64;
106        let mut on_complete = Some(on_complete);
107        while let Some(item) = inner.next().await {
108            match &item {
109                Ok(chunk) => {
110                    total += chunk.len() as u64;
111                    // Fire BEFORE the final yield: the transport may never
112                    // poll us again once Content-Length is satisfied.
113                    if let Some(expected) = expected_total
114                        && total >= expected
115                        && let Some(f) = on_complete.take()
116                    {
117                        f(total);
118                    }
119                    yield item;
120                }
121                Err(_) => {
122                    yield item;
123                    return; // mid-stream error: skip the callback
124                }
125            }
126        }
127        // Fallback for streams with no known expected length: fire on the
128        // terminal `None` with whatever total was accumulated. When
129        // `expected_total` is `Some`, a stream that ends early (without an
130        // error) must NOT fire here — that's the accepted under-reporting
131        // case documented above, not a second chance to fire.
132        if expected_total.is_none()
133            && let Some(f) = on_complete.take()
134        {
135            f(total);
136        }
137    }
138    .boxed()
139}

141/// Interleave each range's 8-byte little-endian length prefix with its data
142/// chunks pulled from `inner`, matching the on-wire framing the client's
143/// length-prefixed reader expects. `range_lens` must have exactly one entry
144/// per non-degenerate range passed to the `stream_ranges` call that produced
145/// `inner`, in the same order, and `inner` must yield each range's bytes
146/// contiguously (see `RangeCache::stream_ranges`'s ordering guarantee).
147fn frame_ranges_stream(
148    mut inner: BoxStream<'static, Result<Bytes, anyhow::Error>>,
149    range_lens: Vec<u64>,
150) -> BoxStream<'static, Result<Bytes, anyhow::Error>> {
151    try_stream! {
152        for len in range_lens {
153            let mut prefix = BytesMut::with_capacity(8);
154            prefix.put_u64_le(len);
155            yield prefix.freeze();
156
157            let mut remaining = len;
158            while remaining > 0 {
159                let chunk = inner
160                    .next()
161                    .await
162                    .expect("stream_ranges under-yielded for a non-degenerate range")?;
163                remaining = remaining.saturating_sub(chunk.len() as u64);
164                yield chunk;
165            }
166        }
167    }
168    .boxed()
169}

171/// Map a response status to the `status` metric-tag value. Deliberately a
172/// small closed set (`"other"` covers everything else) so the `status`
173/// dimension stays bounded per the tagged-metric cardinality contract.
174fn status_label(status: StatusCode) -> &'static str {
175    match status {
176        StatusCode::OK => "200",
177        StatusCode::PARTIAL_CONTENT => "206",
178        StatusCode::BAD_REQUEST => "400",
179        StatusCode::NOT_FOUND => "404",
180        StatusCode::RANGE_NOT_SATISFIABLE => "416",
181        StatusCode::INTERNAL_SERVER_ERROR => "500",
182        StatusCode::SERVICE_UNAVAILABLE => "503",
183        _ => "other",
184    }
185}

187/// Thin wrapper around `head_handler_inner`, mirroring
188/// `get_range_handler`/`post_ranges_handler`: counts every outcome (not just
189/// the success path) with a `status`/`prefix`-tagged `object_cache_head_requests`.
190/// Before this, HEAD traffic had no direct counter and could only be inferred
191/// as a residual of the size/HEAD-tier metrics (#1280).
192pub async fn head_handler(
193    Path(key): Path<String>,
194    State(state): State<AppState>,
195) -> Result<Response, StatusCode> {
196    let prefix = state.cache.classify(&key);
197    let result = head_handler_inner(key, state).await;
198    let status = match &result {
199        Ok(resp) => resp.status(),
200        Err(code) => *code,
201    };
202    imetric!(
203        "object_cache_head_requests",
204        "count",
205        PropertySet::find_or_create(vec![
206            Property::new("status", status_label(status)),
207            Property::new("prefix", prefix),
208        ]),
209        1_u64
210    );
211    result
212}

215async fn head_handler_inner(key: String, state: AppState) -> Result<Response, StatusCode> {
216    if let Err(e) = validate_key(&key, &state.allowed_prefixes) {
217        warn!("rejected key {key}: {e}");
218        return Err(StatusCode::BAD_REQUEST);
219    }
220    match state.cache.size(&key).await {
221        Ok(size) => {
222            let response = Response::builder()
223                .status(StatusCode::OK)
224                .header("Content-Length", size.to_string())
225                .body(Body::empty())
226                .expect("build HEAD response");
227            Ok(response)
228        }
229        Err(e) => {
230            if is_not_found(&e) {
231                Err(StatusCode::NOT_FOUND)
232            } else {
233                error!("HEAD {key}: {e:?}");
234                Err(StatusCode::INTERNAL_SERVER_ERROR)
235            }
236        }
237    }
238}

270#[span_fn]
271async fn get_range_handler_inner(
272    key: String,
273    state: AppState,
274    headers: HeaderMap,
275) -> Result<Response, StatusCode> {
276    let request_start = Instant::now();
277    if let Err(e) = validate_key(&key, &state.allowed_prefixes) {
278        warn!("rejected key {key}: {e}");
279        return Err(StatusCode::BAD_REQUEST);
280    }
281
282    let file_size = match state.cache.size(&key).await {
283        Ok(s) => s,
284        Err(e) => {
285            if is_not_found(&e) {
286                return Err(StatusCode::NOT_FOUND);
287            }
288            error!("size {key}: {e:?}");
289            return Err(StatusCode::INTERNAL_SERVER_ERROR);
290        }
291    };
292
293    // A zero-byte object cannot be expressed as a satisfiable byte range, and
294    // `Content-Range: bytes 0-0/0` is not RFC 7233-valid for an empty entity.
295    // Serve it as a plain 200 with an empty body instead of a 206.
296    if file_size == 0 {
297        let response = Response::builder()
298            .status(StatusCode::OK)
299            .header("Content-Type", "application/octet-stream")
300            .header("Content-Length", "0")
301            .body(Body::empty())
302            .expect("build empty GET response");
303        return Ok(response);
304    }
305
306    let range_header = match headers.get("range").or_else(|| headers.get("Range")) {
307        Some(h) => h.to_str().unwrap_or("").to_string(),
308        None => format!("bytes=0-{}", file_size.saturating_sub(1)),
309    };
310
311    let byte_range = match parse_range_header(&range_header, file_size) {
312        Ok(r) => r,
313        Err(e) => {
314            warn!("bad Range header {range_header}: {e}");
315            return Err(StatusCode::BAD_REQUEST);
316        }
317    };
318
319    // An explicit range whose end exceeds the object size (or an open-ended
320    // range whose start is past EOF) is unsatisfiable per RFC 7233.
321    if byte_range.end > file_size || byte_range.start > file_size {
322        warn!("range {byte_range:?} out of bounds for {key} (file_size {file_size})");
323        return Err(StatusCode::RANGE_NOT_SATISFIABLE);
324    }
325
326    // An open-ended range at exactly EOF (`bytes=<file_size>-`) is a valid
327    // zero-length read. A zero-length entity has no satisfiable byte position to
328    // express in a 206 `Content-Range`, so serve it as a plain 200 with an empty
329    // body, mirroring the zero-byte-object case above, rather than falling
330    // through to `get_range` (which would build a malformed 206 header).
331    if byte_range.start == byte_range.end {
332        let response = Response::builder()
333            .status(StatusCode::OK)
334            .header("Content-Type", "application/octet-stream")
335            .header("Content-Length", "0")
336            .body(Body::empty())
337            .expect("build empty range response");
338        return Ok(response);
339    }
340
341    // Proportional memory charge: a small read charges close to its actual
342    // size, a large one clamps to the fixed streaming window, so per-request
343    // in-flight memory stays bounded regardless of how large the range is.
344    // The client falls back to the direct store on any non-2xx, so an
345    // oversized read still succeeds (just uncached).
346    let requested_bytes = byte_range.end - byte_range.start;
347    let window = stream_window_bytes(state.cache.block_size());
348    let permits_needed = permits_for_bytes(requested_bytes.min(window));
349    let mem_permit_wait_start = Instant::now();
350    let mem_permit = state
351        .mem_permits
352        .clone()
353        .acquire_many_owned(permits_needed)
354        .await
355        .expect("mem_permits semaphore is never closed");
356    fmetric!(
357        "object_cache_mem_permit_wait_ms",
358        "ms",
359        mem_permit_wait_start.elapsed().as_secs_f64() * 1000.0
360    );
361
362    // `_with_size` reuses the `file_size` already resolved above instead of
363    // resolving it again inside `stream_ranges`, so `range_cache_size_backend_hit`
364    // fires exactly once per ranged GET (see the plan's "Correctness fixes").
365    let mut inner = match state
366        .cache
367        .stream_ranges_with_size(
368            &key,
369            vec![byte_range.clone()],
370            file_size,
371            StreamRangesCaller::Range,
372        )
373        .await
374    {
375        Ok(s) => s.boxed(),
376        Err(e) => {
377            if let Some(RangeError::OutOfBounds { .. }) = e.downcast_ref::<RangeError>() {
378                warn!("range {byte_range:?} out of bounds for {key}: {e}");
379                return Err(StatusCode::RANGE_NOT_SATISFIABLE);
380            }
381            if is_not_found(&e) {
382                return Err(StatusCode::NOT_FOUND);
383            }
384            error!("get_range {key} {byte_range:?}: {e:?}");
385            return Err(StatusCode::INTERNAL_SERVER_ERROR);
386        }
387    };
388
389    // Commit-before-stream: await the first chunk before building the
390    // response, so a dead origin still surfaces as 500 rather than an
391    // aborted 200/206.
392    let first = match inner.next().await {
393        Some(Ok(chunk)) => chunk,
394        Some(Err(e)) => {
395            error!("get_range {key} {byte_range:?}: {e:?}");
396            return Err(StatusCode::INTERNAL_SERVER_ERROR);
397        }
398        None => unreachable!("stream_ranges yielded no chunks for a non-degenerate range"),
399    };
400
401    // Time to first byte, now that streaming (#1189/#1222) has landed --
402    // measured from handler entry to the point the first chunk is in hand.
403    fmetric!(
404        "object_cache_ttfb_ms",
405        "ms",
406        PropertySet::find_or_create(vec![Property::new("prefix", state.cache.classify(&key))]),
407        request_start.elapsed().as_secs_f64() * 1000.0
408    );
409
410    let key_for_log = key.clone();
411    let range_for_log = byte_range.clone();
412    let full = stream::once(async move { Ok::<Bytes, anyhow::Error>(first) })
413        .chain(inner)
414        .boxed();
415    let counted = count_bytes_served(full, Some(requested_bytes), move |bytes| {
416        imetric!("object_cache_get_bytes_served", "bytes", bytes);
417        debug!(
418            "GET {key_for_log} {}-{} served {bytes} bytes",
419            range_for_log.start,
420            range_for_log.end.saturating_sub(1)
421        );
422    });
423
424    let response = Response::builder()
425        .status(StatusCode::PARTIAL_CONTENT)
426        .header("Content-Type", "application/octet-stream")
427        .header("Content-Length", requested_bytes.to_string())
428        .header(
429            "Content-Range",
430            format!(
431                "bytes {}-{}/{}",
432                byte_range.start,
433                byte_range.end.saturating_sub(1),
434                file_size
435            ),
436        )
437        .body(permit_body(counted, mem_permit))
438        .expect("build GET response");
439    Ok(response)
440}

447/// Thin wrapper around `post_ranges_handler_inner`, mirroring
448/// `get_range_handler`'s wrapper: counts every outcome (not just success)

470            Property::new("status", status_label(status)),
471            Property::new("prefix", prefix),
472        ]),
473        1_u64

478#[span_fn]
479async fn post_ranges_handler_inner(
480    key: String,
481    state: AppState,
482    body: Bytes,
483) -> Result<Response, StatusCode> {
484    let request_start = Instant::now();
485    if let Err(e) = validate_key(&key, &state.allowed_prefixes) {
486        warn!("rejected key {key}: {e}");
487        return Err(StatusCode::BAD_REQUEST);
488    }
489
490    let req: RangesRequest = match serde_json::from_slice(&body) {
491        Ok(r) => r,
492        Err(e) => {
493            warn!("bad ranges JSON: {e}");
494            return Err(StatusCode::BAD_REQUEST);
495        }
496    };
497
498    // Bound the number of ranges to cap per-request work on this public
499    // authenticated endpoint.
500    if req.ranges.len() > MAX_RANGES_PER_REQUEST {
501        warn!(
502            "rejected {n} ranges for {key}: exceeds max {MAX_RANGES_PER_REQUEST}",
503            n = req.ranges.len()
504        );
505        return Err(StatusCode::BAD_REQUEST);
506    }
507
508    // Reject inverted/degenerate ranges (e.g. `[100, 50]` or `[50, 50]`),
509    // matching the single-range path's `parse_range_header` validation. An
510    // empty or backwards range would otherwise silently yield 0-length data.
511    // While iterating, sum the requested bytes for the proportional memory
512    // charge below, and track which blocks each range touches: a request
513    // with many scattered small ranges can transiently retain a whole block
514    // per range while charging only for the (much smaller) requested bytes,
515    // so the charge must also account for distinct blocks touched.
516    let block_size = state.cache.block_size();
517    let mut total_requested: u64 = 0;
518    let mut touched_blocks: HashSet<u64> = HashSet::new();
519    for &[s, e] in &req.ranges {
520        if s >= e {
521            warn!("rejected inverted range [{s}, {e}] for {key}");
522            return Err(StatusCode::BAD_REQUEST);
523        }
524        total_requested = total_requested.saturating_add(e - s);
525        touched_blocks.extend(blocks_for_range(s, e, block_size));
526    }
527
528    // Empty-ranges short-circuit, mirroring `get_ranges`'s own guard:
529    // `stream_ranges` always does a `size()` lookup up front regardless of
530    // `ranges`, so without this a `{"ranges":[]}` request against a missing
531    // key would flip from today's 200 (empty body) to 404. `_ranges_count` /
532    // `_ranges_bytes_served` are still emitted here to match the Ok-arm
533    // behavior below; `object_cache_ranges_requests` is emitted once by the
534    // `post_ranges_handler` wrapper for every outcome, so it is not repeated
535    // here.
536    if req.ranges.is_empty() {
537        imetric!("object_cache_ranges_count", "count", 0_u64);
538        imetric!("object_cache_ranges_bytes_served", "bytes", 0_u64);
539        let response = Response::builder()
540            .status(StatusCode::OK)
541            .header("Content-Type", "application/octet-stream")
542            .body(Body::empty())
543            .expect("build empty ranges response");
544        return Ok(response);
545    }
546
547    // Each range is written into the response body behind an 8-byte
548    // little-endian length prefix (see `frame_ranges_stream`); account for
549    // that overhead so the permit charge matches the actual framed size.
550    let framed_response_bytes = total_requested.saturating_add(8 * req.ranges.len() as u64);
551
552    // Proportional memory charge, capped at the fixed streaming window (see
553    // `get_range_handler` for the same pattern and its rationale). Charge for
554    // whichever is larger of the framed response size and the bytes backing
555    // the distinct blocks this request touches, so scattered small ranges
556    // spread across many blocks can't under-charge the shared memory budget.
557    let block_bytes = touched_blocks.len() as u64 * block_size;
558    let charge_bytes = framed_response_bytes.max(block_bytes);
559    let window = stream_window_bytes(block_size);
560    let permits_needed = permits_for_bytes(charge_bytes.min(window));
561    let mem_permit_wait_start = Instant::now();
562    let mem_permit = state
563        .mem_permits
564        .clone()
565        .acquire_many_owned(permits_needed)
566        .await
567        .expect("mem_permits semaphore is never closed");
568    fmetric!(
569        "object_cache_mem_permit_wait_ms",
570        "ms",
571        mem_permit_wait_start.elapsed().as_secs_f64() * 1000.0
572    );
573
574    let ranges: Vec<std::ops::Range<u64>> = req.ranges.iter().map(|&[s, e]| s..e).collect();
575    let range_lens: Vec<u64> = ranges.iter().map(|r| r.end - r.start).collect();
576    let range_count = ranges.len() as u64;
577
578    let inner = match state
579        .cache
580        .stream_ranges(&key, ranges.clone(), StreamRangesCaller::Ranges)
581        .await
582    {
583        Ok(s) => s.boxed(),
584        Err(e) => {
585            if let Some(RangeError::OutOfBounds { .. }) = e.downcast_ref::<RangeError>() {
586                warn!("ranges {ranges:?} out of bounds for {key}: {e}");
587                return Err(StatusCode::RANGE_NOT_SATISFIABLE);
588            }
589            if is_not_found(&e) {
590                return Err(StatusCode::NOT_FOUND);
591            }
592            error!("get_ranges {key}: {e:?}");
593            return Err(StatusCode::INTERNAL_SERVER_ERROR);
594        }
595    };
596
597    let mut framed = frame_ranges_stream(inner, range_lens);
598
599    // Commit-before-stream: await the first frame (always the first range's
600    // length prefix, since `req.ranges` is non-empty here) before building
601    // the response, so a dead origin still surfaces as 500 rather than an
602    // aborted 200.
603    let first = match framed.next().await {
604        Some(Ok(chunk)) => chunk,
605        Some(Err(e)) => {
606            error!("get_ranges {key}: {e:?}");
607            return Err(StatusCode::INTERNAL_SERVER_ERROR);
608        }
609        None => unreachable!("frame_ranges_stream always yields at least one prefix"),
610    };
611
612    fmetric!(
613        "object_cache_ttfb_ms",
614        "ms",
615        PropertySet::find_or_create(vec![Property::new("prefix", state.cache.classify(&key))]),
616        request_start.elapsed().as_secs_f64() * 1000.0
617    );
618
619    imetric!("object_cache_ranges_count", "count", range_count);
620
621    let key_for_log = key.clone();
622    let full = stream::once(async move { Ok::<Bytes, anyhow::Error>(first) })
623        .chain(framed)
624        .boxed();
625    let counted = count_bytes_served(full, Some(framed_response_bytes), move |bytes| {
626        imetric!("object_cache_ranges_bytes_served", "bytes", bytes);
627        debug!("POST ranges {key_for_log}: {range_count} ranges served {bytes} bytes");
628    });
629
630    let response = Response::builder()
631        .status(StatusCode::OK)
632        .header("Content-Type", "application/octet-stream")
633        .body(permit_body(counted, mem_permit))
634        .expect("build ranges response");
635    Ok(response)
636}

638/// Parse and process a single NDJSON line from a `POST /prefetch` body,
639/// updating the running counters. Returns `Err` only when the prefetch queue
640/// worker is gone, which aborts the whole request with `503`.
641fn process_prefetch_line(
642    line: &[u8],
643    state: &AppState,
644    accepted: &mut usize,
645    rejected: &mut usize,
646    dropped: &mut usize,
647) -> Result<(), StatusCode> {
648    if line.iter().all(|b| b.is_ascii_whitespace()) {
649        return Ok(());
650    }
651
652    let item: PrefetchItem = match serde_json::from_slice(line) {
653        Ok(item) => item,
654        Err(e) => {
655            warn!("bad prefetch line JSON: {e}");
656            *rejected += 1;
657            return Ok(());
658        }
659    };
660
661    if let Err(e) = validate_key(&item.key, &state.allowed_prefixes) {
662        warn!("rejected prefetch key {}: {e}", item.key);
663        *rejected += 1;
664        return Ok(());
665    }
666    let range_count = item.ranges.as_ref().map_or(0, |r| r.len());
667    if range_count > MAX_RANGES_PER_REQUEST {
668        warn!(
669            "rejected prefetch key {}: {range_count} ranges exceeds max {MAX_RANGES_PER_REQUEST}",
670            item.key
671        );
672        *rejected += 1;
673        return Ok(());
674    }
675    // `item.size` is intentionally NOT capped: the worker streams the
676    // block-index space in bounded windows (`prefetch_queue.rs`) rather
677    // than materializing it, so per-item work is bounded regardless of
678    // `size`, and an over-claimed `size` is bounded by stop-on-first-error
679    // once fills start. This lets legitimate multi-GB partitions warm.
680    // Absent/empty ranges = whole-object warm of [0, item.size), per the
681    // shared-type contract; only present ranges need bounds validation.
682    let has_invalid_range = item
683        .ranges
684        .iter()
685        .flatten()
686        .any(|&[s, e]| s >= e || e > item.size);
687    if has_invalid_range {
688        warn!(
689            "rejected prefetch key {}: inverted or out-of-bounds range for size {}",
690            item.key, item.size
691        );
692        *rejected += 1;
693        return Ok(());
694    }
695
696    match state.prefetch_tx.try_send(item) {
697        Ok(()) => *accepted += 1,
698        Err(TrySendError::Full(_)) => {
699            *dropped += 1;
700            imetric!("object_cache_prefetch_dropped", "count", 1_u64);
701        }
702        Err(TrySendError::Closed(_)) => {
703            error!("prefetch queue worker is gone");
704            return Err(StatusCode::SERVICE_UNAVAILABLE);
705        }
706    }
707    Ok(())
708}

710/// Accept a batch of keys to warm at prefetch priority and return
711/// immediately: fills are handed to the bounded queue in `AppState` and run
712/// asynchronously by the consumer task, so this handler never blocks on an
713/// origin fetch and never acquires a `mem_permit` (the response carries no
714/// object bytes; the fill's memory is already bounded by the scheduler).
715///
716/// The body is `application/x-ndjson`: one `PrefetchItem` JSON object per
717/// `\n`-terminated line, consumed incrementally as it arrives so the whole
718/// batch is never buffered. `MAX_PREFETCH_LINE_BYTES` bounds a single line;
719/// there is no whole-body size limit.
720#[span_fn]
721pub async fn prefetch_handler(
722    State(state): State<AppState>,
723    body: Body,
724) -> Result<Response, StatusCode> {
725    let mut stream = body.into_data_stream();
726    let mut buf = BytesMut::new();
727    let mut accepted = 0usize;
728    let mut rejected = 0usize;
729    let mut dropped = 0usize;
730
731    loop {
732        let chunk = match stream.next().await {
733            Some(Ok(chunk)) => chunk,
734            Some(Err(e)) => {
735                warn!("error reading prefetch body: {e}");
736                break;
737            }
738            None => break,
739        };
740        buf.extend_from_slice(&chunk);
741        while let Some(pos) = buf.iter().position(|&b| b == b'\n') {
742            let mut line = buf.split_to(pos + 1);
743            line.truncate(pos);
744            process_prefetch_line(&line, &state, &mut accepted, &mut rejected, &mut dropped)?;
745        }
746        if buf.len() > MAX_PREFETCH_LINE_BYTES {
747            warn!("prefetch line exceeds max {MAX_PREFETCH_LINE_BYTES} bytes");
748            return Err(StatusCode::BAD_REQUEST);
749        }
750    }
751
752    // A final line with no trailing newline is still processed; `buf` here
753    // is bounded by the same per-chunk check above.
754    if !buf.is_empty() {
755        process_prefetch_line(&buf, &state, &mut accepted, &mut rejected, &mut dropped)?;
756    }
757
758    imetric!("object_cache_prefetch_requests", "count", 1_u64);
759    imetric!(
760        "object_cache_prefetch_keys_enqueued",
761        "count",
762        accepted as u64
763    );
764    debug!("POST prefetch: accepted={accepted} rejected={rejected} dropped={dropped}");
765
766    let resp_body = PrefetchResponse {
767        accepted,
768        rejected,
769        dropped,
770    };
771    let response = Response::builder()
772        .status(StatusCode::ACCEPTED)
773        .header("Content-Type", "application/json")
774        .body(Body::from(
775            serde_json::to_vec(&resp_body).expect("serialize PrefetchResponse"),
776        ))
777        .expect("build prefetch response");
778    Ok(response)
779}
