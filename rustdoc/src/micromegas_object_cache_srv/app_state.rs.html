1use std::sync::Arc;
2
3use micromegas_object_cache::range_cache::RangeCache;
4use tokio::sync::Semaphore;
5
6#[derive(Clone)]
7pub struct AppState {
8    pub cache: RangeCache,
9    /// Prefixes a request key may fall under. Empty = allow every key; this is
10    /// only reachable via `--allow-all-prefixes`, since the server refuses to
11    /// start with an empty list otherwise.
12    pub allowed_prefixes: Vec<String>,
13    /// Cross-request bound on concurrently-assembled response bytes: one
14    /// permit per MiB. A handler acquires `ceil(bytes / 1 MiB)` permits before
15    /// assembling a response and holds them for the response body's full
16    /// lifetime.
17    pub mem_permits: Arc<Semaphore>,
18    /// Total capacity of `mem_permits`, in MiB. A request whose assembled size
19    /// would exceed this outright is rejected (413) rather than acquired
20    /// (which would otherwise block forever).
21    pub memory_budget_mb: u32,
22}
23
24impl AppState {
25    pub fn new(cache: RangeCache, allowed_prefixes: Vec<String>, memory_budget_mb: u32) -> Self {
26        Self {
27            cache,
28            allowed_prefixes,
29            mem_permits: Arc::new(Semaphore::new(memory_budget_mb as usize)),
30            memory_budget_mb,
31        }
32    }
33}
