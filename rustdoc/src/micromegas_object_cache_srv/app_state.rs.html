1use std::sync::Arc;
2
3use micromegas::object_cache::prefetch::PrefetchItem;
4use micromegas::object_cache::range_cache::RangeCache;
5use tokio::sync::{Semaphore, mpsc};
6
7#[derive(Clone)]
8pub struct AppState {
9    pub cache: RangeCache,
10    /// Prefixes a request key may fall under. Empty = allow every key; this is
11    /// only reachable via `--allow-all-prefixes`, since the server refuses to
12    /// start with an empty list otherwise.
13    pub allowed_prefixes: Vec<String>,
14    /// Cross-request bound on concurrent in-flight streaming windows: one
15    /// permit per MiB. A handler acquires `ceil(min(response size, the fixed
16    /// streaming window) / 1 MiB)` permits before starting to stream a
17    /// response and holds them for the response body's full lifetime — a
18    /// small response charges close to its actual size, a large one clamps
19    /// to the window, so the charge reflects in-flight window bytes rather
20    /// than the whole response size.
21    pub mem_permits: Arc<Semaphore>,
22    /// Total capacity of `mem_permits`, in MiB. The startup guard in
23    /// `object_cache_srv.rs` floors this at the fixed streaming window's
24    /// size, so a single large streaming request's charge can never exceed
25    /// the whole budget (which would otherwise block `acquire_many_owned`
26    /// forever).
27    pub memory_budget_mb: u32,
28    /// Sender side of the bounded prefetch queue; the consumer task (spawned
29    /// separately, see `prefetch_queue::spawn_prefetch_worker`) owns the
30    /// receiver. Every `AppState` clone (one per request) holds a clone of
31    /// this sender, so the channel closes only once the server shuts down.
32    pub prefetch_tx: mpsc::Sender<PrefetchItem>,
33}
34
35impl AppState {
36    pub fn new(
37        cache: RangeCache,
38        allowed_prefixes: Vec<String>,
39        memory_budget_mb: u32,
40        prefetch_tx: mpsc::Sender<PrefetchItem>,
41    ) -> Self {
42        Self {
43            cache,
44            allowed_prefixes,
45            mem_permits: Arc::new(Semaphore::new(memory_budget_mb as usize)),
46            memory_budget_mb,
47            prefetch_tx,
48        }
49    }
50}
