12/// Number of block indices per streamed window. Sized so a single window
13/// already spans several coalesced runs and can saturate the scheduler's
14/// `prefetch_permits` on its own, which is why `WINDOW_CONCURRENCY` can stay
15/// low.
16const WINDOW_BLOCKS: u64 = 64;
17
18/// In-flight `prefetch_blocks` windows per item. Total in-flight fills across
19/// the worker is `worker_concurrency * WINDOW_CONCURRENCY`, each bounded to
20/// `WINDOW_BLOCKS` — independent of `item.size`.
21const WINDOW_CONCURRENCY: usize = 1;
22
23/// Lazily chunk a block-index range into `WINDOW_BLOCKS`-sized windows. Never
24/// materializes the full range: representing `size == u64::MAX` costs
25/// nothing until windows are actually pulled.
26struct BlockWindows {
27    remaining: Range<u64>,
28}
29
30impl Iterator for BlockWindows {
31    type Item = Vec<u64>;
32
33    fn next(&mut self) -> Option<Vec<u64>> {
34        if self.remaining.is_empty() {
35            return None;
36        }
37        let take = WINDOW_BLOCKS.min(self.remaining.end - self.remaining.start);
38        let start = self.remaining.start;
39        let end = start + take;
40        self.remaining.start = end;
41        Some((start..end).collect())
42    }
43}

45/// Byte spans to warm for `item`: `[0, size)` for a whole-object warm (no or
46/// empty `ranges`), or the caller-supplied ranges as-is. The handler already
47/// validated each range against `item.size`.
48// The single-element `vec![0..item.size]` is a `Vec<Range<u64>>`, not the
49// `Vec<u64>` clippy's suggested `.collect()` fix would produce.
50#[allow(clippy::single_range_in_vec_init)]
51fn spans_for(item: &PrefetchItem) -> Vec<Range<u64>> {
52    match &item.ranges {
53        None => vec![0..item.size],
54        Some(rs) if rs.is_empty() => vec![0..item.size],
55        Some(rs) => rs.iter().map(|&[s, e]| s..e).collect(),
56    }
57}
58
59/// Lazy iterator over the block-index windows covering `item`. Empty spans
60/// (`size == 0`, or a degenerate `start >= end`) are skipped rather than
61/// passed to `blocks_for_range`, which underflows on `end == 0` and trips its
62/// `debug_assert!(start < end)` in test builds. No cross-window dedup is done
63/// even when supplied ranges overlap: the scheduler's `own_or_join` and the
64/// backend hit-path already dedup at the block level.
65fn lazy_windows(item: &PrefetchItem, block_size: u64) -> impl Iterator<Item = Vec<u64>> {
66    spans_for(item)
67        .into_iter()
68        .filter(|s| s.start < s.end)
69        .flat_map(move |s| BlockWindows {
70            remaining: blocks_for_range(s.start, s.end, block_size),
71        })
72}
73
74/// Stream `item`'s block-index windows through `prefetch_blocks`, stopping at
75/// the first error. This is the bound that replaces a per-item size cap: a
76/// window fully past the real EOF (an over-claimed `size`) fails at the
77/// origin, halting the stream within `WINDOW_CONCURRENCY` windows of the true
78/// end — `buffered` only advances the lazy `windows` iterator as it needs the
79/// next future, so on return the remaining windows are never generated.
80async fn warm_item(cache: &RangeCache, item: PrefetchItem, block_size: u64) {
81    let windows = lazy_windows(&item, block_size);
82    let mut stream = stream::iter(windows)
83        .map(|w| {
84            let cache = cache.clone();
85            let key = item.key.clone();
86            let size = item.size;
87            async move { cache.prefetch_blocks(&key, size, &w).await }
88        })
89        .buffered(WINDOW_CONCURRENCY);
90
91    let mut warmed_any = false;
92    while let Some(res) = stream.next().await {
93        if let Err(e) = res {
94            imetric!("object_cache_prefetch_fill_error", "count", 1_u64);
95            debug!("prefetch fill failed key={}: {e:?}", item.key);
96            return;
97        }
98        warmed_any = true;
99    }
100    // A no-op item (size == 0 or all-empty ranges) yields no windows; don't
101    // count it as a warmed key.
102    if warmed_any {
103        imetric!("object_cache_prefetch_keys_warmed", "count", 1_u64);
104    }
105}

107/// Spawn the bounded-queue prefetch consumer. The returned `Sender` is what
108/// `AppState` holds; the caller drives the pipeline to completion by dropping
109/// every clone of the sender and awaiting the returned `JoinHandle`, which
110/// resolves only once the channel has closed and every in-flight `warm_item`
111/// has finished (see the module docs on `prefetch_handler` for why this
112/// matters to tests).
113pub fn spawn_prefetch_worker(
114    cache: RangeCache,
115    queue_capacity: usize,
116    worker_concurrency: usize,
117) -> (mpsc::Sender<PrefetchItem>, JoinHandle<()>) {
118    let (tx, rx) = mpsc::channel::<PrefetchItem>(queue_capacity);
119
120    let handle = tokio::spawn(async move {
121        let block_size = cache.block_size();
122        ReceiverStream::new(rx)
123            .for_each_concurrent(worker_concurrency, |item| {
124                let cache = cache.clone();
125                async move { warm_item(&cache, item, block_size).await }
126            })
127            .await;
128    });
129
130    (tx, handle)
131}
