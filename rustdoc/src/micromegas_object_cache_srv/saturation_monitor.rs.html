1//! Periodic saturation gauges for the object cache: fetch-budget occupancy,
2//! in-flight entries, accounted RAM-tier usage, memory-budget occupancy,
3//! prefetch queue depth, host-level NIC throughput, and the foyer disk
4//! engine's own write-path throughput -- the signals #1206 calls out as
5//! missing for locating a bottleneck (a solid *counter* layer already
6//! exists, but no saturation/queue-depth signal). Modeled on
7//! `telemetry-sink::system_monitor::send_system_metrics_forever`: a
8//! background task that wakes on an interval and emits `imetric!`/`fmetric!`.
9//!
10//! The disk gauges used to come from `sysinfo::Disks`, but that reads 0 in
11//! the deployed container (the cache device/mount isn't enumerated there);
12//! they are now sourced from the foyer engine's own `Statistics` via
13//! `RangeCache::backend_disk_stats`, which measures the cache's own device
14//! I/O directly instead of relying on host disk enumeration.
15
16use std::sync::Arc;
17use std::time::Duration;
18
19use micromegas::object_cache::backend::BackendDiskStats;
20use micromegas::object_cache::prefetch::PrefetchItem;
21use micromegas::object_cache::range_cache::RangeCache;
22use micromegas::tracing::prelude::*;
23use sysinfo::Networks;
24use tokio::sync::{Semaphore, mpsc};
25use tokio::task::JoinHandle;
26
27/// How often the sampler wakes to emit gauges.
28///
29/// There's no existing cadence to match: the `sysinfo` sampler in
30/// `system_monitor.rs` sleeps for `sysinfo::MINIMUM_CPU_UPDATE_INTERVAL`
31/// (~200ms on Linux), but that floor exists only because it's the minimum
32/// needed for a valid CPU-usage delta -- it doesn't apply here. 5s is a
33/// telemetry-volume tradeoff instead: a shorter interval gives finer
34/// saturation resolution at more telemetry volume.
35const SAMPLE_INTERVAL: Duration = Duration::from_secs(5);
36
37/// Emit one round of saturation gauges. Split out from the spawn loop so it
38/// can be driven directly (see `saturation_monitor` tests). `pub` so the
39/// `object-cache-srv/tests/` integration-test crate can drive it directly.
40pub fn sample_once(
41    cache: &RangeCache,
42    mem_permits: &Semaphore,
43    memory_budget_mb: u32,
44    prefetch_tx: &mpsc::Sender<PrefetchItem>,
45    networks: &mut Networks,
46    prev_disk_stats: &mut Option<BackendDiskStats>,
47    interval_secs: f64,
48) {
49    let (shared_available, shared_total, prefetch_available, prefetch_total) =
50        cache.fetch_budget_stats();
51    imetric!(
52        "object_cache_fetch_shared_occupancy",
53        "count",
54        (shared_total - shared_available) as u64
55    );
56    imetric!(
57        "object_cache_fetch_shared_available",
58        "count",
59        shared_available as u64
60    );
61    imetric!(
62        "object_cache_fetch_prefetch_occupancy",
63        "count",
64        (prefetch_total - prefetch_available) as u64
65    );
66    imetric!(
67        "object_cache_fetch_prefetch_available",
68        "count",
69        prefetch_available as u64
70    );
71
72    imetric!(
73        "object_cache_inflight_entries",
74        "count",
75        cache.inflight_len() as u64
76    );
77
78    if let Some(ram_bytes) = cache.backend_ram_usage() {
79        imetric!(
80            "object_cache_ram_tier_usage_bytes",
81            "bytes",
82            ram_bytes as u64
83        );
84    }
85
86    if let Some(ram_entries) = cache.backend_ram_entries() {
87        imetric!("object_cache_ram_tier_entries", "count", ram_entries as u64);
88    }
89
90    let mem_available = mem_permits.available_permits() as u32;
91    let mem_occupancy = memory_budget_mb.saturating_sub(mem_available);
92    imetric!(
93        "object_cache_mem_budget_occupancy_mb",
94        "megabytes",
95        mem_occupancy as u64
96    );
97    imetric!(
98        "object_cache_mem_budget_available_mb",
99        "megabytes",
100        mem_available as u64
101    );
102
103    let queue_depth = prefetch_tx.max_capacity() - prefetch_tx.capacity();
104    imetric!(
105        "object_cache_prefetch_queue_depth",
106        "count",
107        queue_depth as u64
108    );
109
110    // NIC: the expected ceiling on the target im4gn.large (#1197) and
111    // currently unmeasured. `Networks::refresh` computes the byte delta
112    // since the previous refresh internally, so dividing by the sampler's
113    // fixed interval gives a rate.
114    networks.refresh(false);
115    let (rx_bytes, tx_bytes) = networks.list().values().fold((0u64, 0u64), |(rx, tx), n| {
116        (rx + n.received(), tx + n.transmitted())
117    });
118    fmetric!(
119        "object_cache_nic_rx_bytes_per_sec",
120        "bytes_per_sec",
121        rx_bytes as f64 / interval_secs
122    );
123    fmetric!(
124        "object_cache_nic_tx_bytes_per_sec",
125        "bytes_per_sec",
126        tx_bytes as f64 / interval_secs
127    );
128
129    // Foyer disk engine write-path throughput: the counters are cumulative
130    // since process start, so a rate needs a delta against the previous
131    // sample. `prev_disk_stats` starts `None` (no rate on the first tick),
132    // and stays `None` for a backend with no disk tier (e.g. `MemoryBackend`
133    // in tests), in which case nothing is emitted here.
134    let current_disk_stats = cache.backend_disk_stats();
135    if let (Some(prev), Some(current)) = (*prev_disk_stats, current_disk_stats) {
136        fmetric!(
137            "object_cache_foyer_disk_write_bytes_per_sec",
138            "bytes_per_sec",
139            current.write_bytes.saturating_sub(prev.write_bytes) as f64 / interval_secs
140        );
141        fmetric!(
142            "object_cache_foyer_disk_read_bytes_per_sec",
143            "bytes_per_sec",
144            current.read_bytes.saturating_sub(prev.read_bytes) as f64 / interval_secs
145        );
146        fmetric!(
147            "object_cache_foyer_disk_write_ios_per_sec",
148            "ops_per_sec",
149            current.write_ios.saturating_sub(prev.write_ios) as f64 / interval_secs
150        );
151        fmetric!(
152            "object_cache_foyer_disk_read_ios_per_sec",
153            "ops_per_sec",
154            current.read_ios.saturating_sub(prev.read_ios) as f64 / interval_secs
155        );
156    }
157    *prev_disk_stats = current_disk_stats;
158}

160/// Spawn the periodic saturation sampler, waking every `SAMPLE_INTERVAL`.
161/// Runs for the lifetime of the process; the returned handle is not expected
162/// to be awaited (the caller may drop it -- the spawned task keeps running
163/// detached, like `prefetch_queue::spawn_prefetch_worker`'s worker).
164pub fn spawn_saturation_monitor(
165    cache: RangeCache,
166    mem_permits: Arc<Semaphore>,
167    memory_budget_mb: u32,
168    prefetch_tx: mpsc::Sender<PrefetchItem>,
169) -> JoinHandle<()> {
170    tokio::spawn(async move {
171        let mut networks = Networks::new_with_refreshed_list();
172        let mut prev_disk_stats: Option<BackendDiskStats> = None;
173        loop {
174            tokio::time::sleep(SAMPLE_INTERVAL).await;
175            sample_once(
176                &cache,
177                &mem_permits,
178                memory_budget_mb,
179                &prefetch_tx,
180                &mut networks,
181                &mut prev_disk_stats,
182                SAMPLE_INTERVAL.as_secs_f64(),
183            );
184        }
185    })
186}
