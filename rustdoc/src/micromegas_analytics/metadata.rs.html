9/// Creates a `StreamInfo` from a database row.
10pub fn stream_from_row(row: &sqlx::postgres::PgRow) -> Result<StreamInfo> {
11    let dependencies_metadata_buffer: Vec<u8> = row.try_get("dependencies_metadata")?;
12    let dependencies_metadata: Vec<UserDefinedType> =
13        ciborium::from_reader(&dependencies_metadata_buffer[..])
14            .with_context(|| "decoding dependencies metadata")?;
15    let objects_metadata_buffer: Vec<u8> = row.try_get("objects_metadata")?;
16    let objects_metadata: Vec<UserDefinedType> =
17        ciborium::from_reader(&objects_metadata_buffer[..])
18            .with_context(|| "decoding objects metadata")?;
19    let tags: Vec<String> = row.try_get("tags")?;
20    let properties: Vec<Property> = row.try_get("properties")?;
21    Ok(StreamInfo {
22        stream_id: row.try_get("stream_id")?,
23        process_id: row.try_get("process_id")?,
24        dependencies_metadata,
25        objects_metadata,
26        tags,
27        properties: micromegas_telemetry::property::into_hashmap(properties),
28    })
29}

31/// Finds a stream by its ID.
32#[span_fn]
33pub async fn find_stream(
34    pool: &sqlx::Pool<sqlx::Postgres>,
35    stream_id: sqlx::types::Uuid,
36) -> Result<StreamInfo> {
37    let row = sqlx::query(
38        "SELECT stream_id, process_id, dependencies_metadata, objects_metadata, tags, properties
39         FROM streams
40         WHERE stream_id = $1
41         ;",
42    )
43    .bind(stream_id)
44    .fetch_one(pool)
45    .await
46    .with_context(|| "select from streams")?;
47    stream_from_row(&row)
48}

50/// Lists all streams for a given process that are tagged with a specific tag.
51pub async fn list_process_streams_tagged(
52    pool: &sqlx::Pool<sqlx::Postgres>,
53    process_id: sqlx::types::Uuid,
54    tag: &str,
55) -> Result<Vec<StreamInfo>> {
56    let stream_rows = sqlx::query(
57        "SELECT stream_id, process_id, dependencies_metadata, objects_metadata, tags, properties
58         FROM streams
59         WHERE process_id = $1
60         AND array_position(tags, $2) is not NULL
61         ;",
62    )
63    .bind(process_id)
64    .bind(tag)
65    .fetch_all(pool)
66    .await
67    .with_context(|| "fetching streams")?;
68    let mut streams = vec![];
69    for row in stream_rows {
70        let stream = stream_from_row(&row).with_context(|| "stream_from_row")?;
71        streams.push(stream);
72    }
73    Ok(streams)
74}

143/// Finds all blocks for a given stream within a given time range.
144#[span_fn]
145pub async fn find_stream_blocks_in_range(
146    connection: &mut sqlx::PgConnection,
147    stream_id: sqlx::types::Uuid,
148    begin_ticks: i64,
149    end_ticks: i64,
150) -> Result<Vec<BlockMetadata>> {
151    let rows = sqlx::query(
152        "SELECT block_id, stream_id, process_id, begin_time, begin_ticks, end_time, end_ticks, nb_objects, object_offset, payload_size, insert_time as block_insert_time
153         FROM blocks
154         WHERE stream_id = $1
155         AND begin_ticks <= $2
156         AND end_ticks >= $3
157         ORDER BY begin_ticks;",
158    )
159    .bind(stream_id)
160    .bind(end_ticks)
161    .bind(begin_ticks)
162    .fetch_all(connection)
163    .await
164    .with_context(|| "find_stream_blocks")?;
165    let mut blocks = Vec::new();
166    for r in rows {
167        blocks.push(block_from_row(&r)?);
168    }
169    Ok(blocks)
