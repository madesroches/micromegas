103/// A builder for creating a `RecordBatch` of log entries.
104pub struct LogEntriesRecordBuilder {
105    process_ids: StringDictionaryBuilder<Int16Type>,
106    stream_ids: StringDictionaryBuilder<Int16Type>,
107    block_ids: StringDictionaryBuilder<Int16Type>,
108    insert_times: PrimitiveBuilder<TimestampNanosecondType>,
109    exes: StringDictionaryBuilder<Int16Type>,
110    usernames: StringDictionaryBuilder<Int16Type>,
111    computers: StringDictionaryBuilder<Int16Type>,
112    times: PrimitiveBuilder<TimestampNanosecondType>,
113    targets: StringDictionaryBuilder<Int16Type>,
114    levels: PrimitiveBuilder<Int32Type>,
115    msgs: StringBuilder,
116    properties: ListBuilder<StructBuilder>,
117    process_properties: ListBuilder<StructBuilder>,
118}
119
120impl LogEntriesRecordBuilder {
121    pub fn with_capacity(capacity: usize) -> Self {
122        let prop_struct_fields = vec![
123            Field::new("key", DataType::Utf8, false),
124            Field::new("value", DataType::Utf8, false),
125        ];
126        let prop_field = Arc::new(Field::new(
127            "Property",
128            DataType::Struct(Fields::from(prop_struct_fields.clone())),
129            false,
130        ));
131        let props_builder = ListBuilder::new(StructBuilder::from_fields(
132            prop_struct_fields.clone(),
133            capacity,
134        ))
135        .with_field(prop_field.clone());
136        let process_props_builder =
137            ListBuilder::new(StructBuilder::from_fields(prop_struct_fields, capacity))
138                .with_field(prop_field);
139
140        Self {
141            process_ids: StringDictionaryBuilder::new(),
142            stream_ids: StringDictionaryBuilder::new(),
143            block_ids: StringDictionaryBuilder::new(),
144            insert_times: PrimitiveBuilder::with_capacity(capacity),
145            exes: StringDictionaryBuilder::new(),
146            usernames: StringDictionaryBuilder::new(),
147            computers: StringDictionaryBuilder::new(),
148            times: PrimitiveBuilder::with_capacity(capacity),
149            targets: StringDictionaryBuilder::new(),
150            levels: PrimitiveBuilder::with_capacity(capacity),
151            msgs: StringBuilder::new(),
152            properties: props_builder,
153            process_properties: process_props_builder,
154        }
155    }
156
157    pub fn get_time_range(&self) -> Option<(DateTime<Utc>, DateTime<Utc>)> {
158        if self.is_empty() {
159            return None;
160        }
161        // assuming that the events are in order
162        let slice = self.times.values_slice();
163        Some((
164            DateTime::from_timestamp_nanos(slice[0]),
165            DateTime::from_timestamp_nanos(slice[slice.len() - 1]),
166        ))
167    }
168
169    pub fn len(&self) -> i64 {
170        self.times.len() as i64
171    }
172
173    pub fn is_empty(&self) -> bool {
174        self.times.len() == 0
175    }
176
177    pub fn append(&mut self, row: &LogEntry) -> Result<()> {
178        self.process_ids
179            .append_value(format!("{}", row.process.process_id));
180        self.stream_ids.append_value(&*row.stream_id);
181        self.block_ids.append_value(&*row.block_id);
182        self.insert_times.append_value(row.insert_time);
183        self.exes.append_value(&row.process.exe);
184        self.usernames.append_value(&row.process.username);
185        self.computers.append_value(&row.process.computer);
186        self.times.append_value(row.time);
187        self.targets.append_value(&*row.target);
188        self.levels.append_value(row.level);
189        self.msgs.append_value(&*row.msg);
190        add_property_set_to_builder(&row.properties, &mut self.properties)?;
191        add_properties_to_builder(&row.process.properties, &mut self.process_properties)?;
192        Ok(())
193    }
194
195    pub fn finish(mut self) -> Result<RecordBatch> {
196        RecordBatch::try_new(
197            Arc::new(log_table_schema()),
198            vec![
199                Arc::new(self.process_ids.finish()),
200                Arc::new(self.stream_ids.finish()),
201                Arc::new(self.block_ids.finish()),
202                Arc::new(self.insert_times.finish().with_timezone_utc()),
203                Arc::new(self.exes.finish()),
204                Arc::new(self.usernames.finish()),
205                Arc::new(self.computers.finish()),
206                Arc::new(self.times.finish().with_timezone_utc()),
207                Arc::new(self.targets.finish()),
208                Arc::new(self.levels.finish()),
209                Arc::new(self.msgs.finish()),
210                Arc::new(self.properties.finish()),
211                Arc::new(self.process_properties.finish()),
212            ],
213        )
214        .with_context(|| "building record batch")
215    }
