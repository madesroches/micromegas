20/// Represents a single async span event record.
21/// Optimized for high-frequency data - process info can be joined when needed.
22#[derive(Debug, Clone)]
23pub struct AsyncEventRecord {
24    pub stream_id: Arc<String>,
25    pub block_id: Arc<String>,
26    pub time: i64,
27    pub event_type: Arc<String>,
28    pub span_id: i64,
29    pub parent_span_id: i64,
30    pub depth: u32,
31    pub name: Arc<String>,
32    pub filename: Arc<String>,
33    pub target: Arc<String>,
34    pub line: u32,
35}
36
37/// Returns the schema for the async events table.
38/// Optimized for high-frequency data - excludes process info that can be joined.
39pub fn async_events_table_schema() -> Schema {
40    Schema::new(vec![
41        Field::new(
42            "stream_id",
43            DataType::Dictionary(Box::new(DataType::Int16), Box::new(DataType::Utf8)),
44            false,
45        ),
46        Field::new(
47            "block_id",
48            DataType::Dictionary(Box::new(DataType::Int16), Box::new(DataType::Utf8)),
49            false,
50        ),
51        Field::new(
52            "time",
53            DataType::Timestamp(TimeUnit::Nanosecond, Some("+00:00".into())),
54            false,
55        ),
56        Field::new(
57            "event_type",
58            DataType::Dictionary(Box::new(DataType::Int16), Box::new(DataType::Utf8)),
59            false,
60        ),
61        Field::new("span_id", DataType::Int64, false),
62        Field::new("parent_span_id", DataType::Int64, false),
63        Field::new("depth", DataType::UInt32, false),
64        Field::new(
65            "name",
66            DataType::Dictionary(Box::new(DataType::Int16), Box::new(DataType::Utf8)),
67            false,
68        ),
69        Field::new(
70            "filename",
71            DataType::Dictionary(Box::new(DataType::Int16), Box::new(DataType::Utf8)),
72            false,
73        ),
74        Field::new(
75            "target",
76            DataType::Dictionary(Box::new(DataType::Int16), Box::new(DataType::Utf8)),
77            false,
78        ),
79        Field::new("line", DataType::UInt32, false),
80    ])
81}

83/// A builder for creating a `RecordBatch` of async event records.
84pub struct AsyncEventRecordBuilder {
85    stream_ids: StringDictionaryBuilder<Int16Type>,
86    block_ids: StringDictionaryBuilder<Int16Type>,
87    times: PrimitiveBuilder<TimestampNanosecondType>,
88    event_types: StringDictionaryBuilder<Int16Type>,
89    span_ids: PrimitiveBuilder<Int64Type>,
90    parent_span_ids: PrimitiveBuilder<Int64Type>,
91    depths: PrimitiveBuilder<UInt32Type>,
92    names: StringDictionaryBuilder<Int16Type>,
93    filenames: StringDictionaryBuilder<Int16Type>,
94    targets: StringDictionaryBuilder<Int16Type>,
95    lines: PrimitiveBuilder<UInt32Type>,
96}
97
98impl AsyncEventRecordBuilder {
99    pub fn with_capacity(capacity: usize) -> Self {
100        Self {
101            stream_ids: StringDictionaryBuilder::new(),
102            block_ids: StringDictionaryBuilder::new(),
103            times: PrimitiveBuilder::with_capacity(capacity),
104            event_types: StringDictionaryBuilder::new(),
105            span_ids: PrimitiveBuilder::with_capacity(capacity),
106            parent_span_ids: PrimitiveBuilder::with_capacity(capacity),
107            depths: PrimitiveBuilder::with_capacity(capacity),
108            names: StringDictionaryBuilder::new(),
109            filenames: StringDictionaryBuilder::new(),
110            targets: StringDictionaryBuilder::new(),
111            lines: PrimitiveBuilder::with_capacity(capacity),
112        }
113    }
114
115    pub fn get_time_range(&self) -> Option<TimeRange> {
116        if self.is_empty() {
117            return None;
118        }
119        // assuming that the events are in order
120        let slice = self.times.values_slice();
121        Some(TimeRange::new(
122            DateTime::from_timestamp_nanos(slice[0]),
123            DateTime::from_timestamp_nanos(slice[slice.len() - 1]),
124        ))
125    }
126
127    pub fn len(&self) -> i64 {
128        self.times.len() as i64
129    }
130
131    pub fn is_empty(&self) -> bool {
132        self.times.len() == 0
133    }
134
135    pub fn append(&mut self, record: &AsyncEventRecord) -> Result<()> {
136        self.stream_ids.append_value(&*record.stream_id);
137        self.block_ids.append_value(&*record.block_id);
138        self.times.append_value(record.time);
139        self.event_types.append_value(&*record.event_type);
140        self.span_ids.append_value(record.span_id);
141        self.parent_span_ids.append_value(record.parent_span_id);
142        self.depths.append_value(record.depth);
143        self.names.append_value(&*record.name);
144        self.filenames.append_value(&*record.filename);
145        self.targets.append_value(&*record.target);
146        self.lines.append_value(record.line);
147        Ok(())
148    }
149
150    pub fn finish(mut self) -> Result<RecordBatch> {
151        RecordBatch::try_new(
152            Arc::new(async_events_table_schema()),
153            vec![
154                Arc::new(self.stream_ids.finish()),
155                Arc::new(self.block_ids.finish()),
156                Arc::new(self.times.finish().with_timezone_utc()),
157                Arc::new(self.event_types.finish()),
158                Arc::new(self.span_ids.finish()),
159                Arc::new(self.parent_span_ids.finish()),
160                Arc::new(self.depths.finish()),
161                Arc::new(self.names.finish()),
162                Arc::new(self.filenames.finish()),
163                Arc::new(self.targets.finish()),
164                Arc::new(self.lines.finish()),
165            ],
166        )
167        .with_context(|| "building record batch")
168    }
169}
