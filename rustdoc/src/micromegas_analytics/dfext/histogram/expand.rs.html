19/// A DataFusion `TableFunctionImpl` that expands a histogram struct into rows of (bin_center, count).
20///
21/// Usage:
22/// ```sql
23/// SELECT bin_center, count
24/// FROM expand_histogram(
25///   (SELECT make_histogram(0.0, 100.0, 100, value)
26///    FROM measures WHERE name = 'cpu_usage')
27/// )
28/// ```
29#[derive(Debug)]
30pub struct ExpandHistogramTableFunction {}
31
32impl ExpandHistogramTableFunction {
33    pub fn new() -> Self {
34        Self {}
35    }
36}
37
38impl Default for ExpandHistogramTableFunction {
39    fn default() -> Self {
40        Self::new()
41    }
42}

44/// The source of histogram data - either a literal value or a subquery to evaluate.
45#[derive(Debug, Clone)]
46enum HistogramSource {
47    Literal(ScalarValue),
48    Subquery(Arc<LogicalPlan>),
49}
50
51impl TableFunctionImpl for ExpandHistogramTableFunction {
52    fn call(&self, args: &[Expr]) -> datafusion::error::Result<Arc<dyn TableProvider>> {
53        if args.len() != 1 {
54            return Err(DataFusionError::Plan(
55                "expand_histogram requires exactly one argument (a histogram)".into(),
56            ));
57        }
58
59        // Extract the histogram from the expression
60        let source = match &args[0] {
61            Expr::Literal(scalar, _metadata) => HistogramSource::Literal(scalar.clone()),
62            Expr::ScalarSubquery(subquery) => HistogramSource::Subquery(subquery.subquery.clone()),
63            other => {
64                return Err(DataFusionError::Plan(format!(
65                    "expand_histogram argument must be a histogram literal or subquery, got: {other:?}"
66                )));
67            }
68        };
69
70        Ok(Arc::new(ExpandHistogramTableProvider { source }))
71    }
72}

75    Arc::new(Schema::new(vec![

77        Field::new("count", DataType::UInt64, false),

81fn expand_histogram_to_batch(
82    histo_array: &HistogramArray,
83    index: usize,
84) -> Result<RecordBatch, DataFusionError> {
85    let start = histo_array.get_start(index)?;
86    let end = histo_array.get_end(index)?;
87    let bins = histo_array.get_bins(index)?;
88
89    let num_bins = bins.len();
90    if num_bins == 0 {
91        return Ok(RecordBatch::new_empty(output_schema()));
92    }
93
94    // Handle edge case where start == end (all values in a single point)
95    let bin_width = if (end - start).abs() < f64::EPSILON {
96        1.0 // Use unit width when range is zero
97    } else {
98        (end - start) / (num_bins as f64)
99    };
100
101    let mut bin_centers = Vec::with_capacity(num_bins);
102    let mut counts = Vec::with_capacity(num_bins);
103
104    for i in 0..num_bins {
105        let bin_center = start + (i as f64 + 0.5) * bin_width;
106        bin_centers.push(bin_center);
107        counts.push(bins.value(i));
108    }
109
110    let bin_center_array: ArrayRef = Arc::new(Float64Array::from(bin_centers));
111    let count_array: ArrayRef = Arc::new(UInt64Array::from(counts));
112
113    RecordBatch::try_new(output_schema(), vec![bin_center_array, count_array])
114        .map_err(|e| DataFusionError::External(e.into()))
115}
116
117fn extract_histogram_from_struct(
118    struct_array: &Arc<StructArray>,
119) -> Result<RecordBatch, DataFusionError> {
120    let histo_array = HistogramArray::new(struct_array.clone());
121    if histo_array.is_empty() {
122        return Ok(RecordBatch::new_empty(output_schema()));
123    }
124    expand_histogram_to_batch(&histo_array, 0)
125}
126
127fn scalar_to_batch(scalar: &ScalarValue) -> Result<RecordBatch, DataFusionError> {
128    if let ScalarValue::Struct(struct_array) = scalar {
129        extract_histogram_from_struct(struct_array)
130    } else {
131        Err(DataFusionError::Plan(format!(
132            "expand_histogram argument must be a struct (histogram), got: {:?}",
133            scalar.data_type()
134        )))
135    }
136}

138/// Table provider for expanding histogram data.
139#[derive(Debug)]
140pub struct ExpandHistogramTableProvider {
141    source: HistogramSource,
142}
143
144impl ExpandHistogramTableProvider {
145    /// Creates a new provider from a histogram scalar value.
146    pub fn from_scalar(scalar: ScalarValue) -> Result<Self, DataFusionError> {
147        if !matches!(scalar, ScalarValue::Struct(_)) {
148            return Err(DataFusionError::Plan(format!(
149                "expand_histogram argument must be a struct (histogram), got: {:?}",
150                scalar.data_type()
151            )));
152        }
153        Ok(Self {
154            source: HistogramSource::Literal(scalar),
155        })
156    }
157}
158
159#[async_trait]
160impl TableProvider for ExpandHistogramTableProvider {
161    fn as_any(&self) -> &dyn Any {
162        self
163    }
164
165    fn schema(&self) -> SchemaRef {
166        output_schema()
167    }
168
169    fn table_type(&self) -> TableType {
170        TableType::Temporary
171    }
172
173    async fn scan(
174        &self,
175        state: &dyn Session,
176        projection: Option<&Vec<usize>>,
177        _filters: &[Expr],
178        limit: Option<usize>,
179    ) -> datafusion::error::Result<Arc<dyn ExecutionPlan>> {
180        let mut record_batch = match &self.source {
181            HistogramSource::Literal(scalar) => scalar_to_batch(scalar)?,
182            HistogramSource::Subquery(plan) => {
183                // Execute the subquery to get the histogram scalar
184                let physical_plan = state.create_physical_plan(plan).await?;
185                let task_ctx = state.task_ctx();
186                let batches = datafusion::physical_plan::collect(physical_plan, task_ctx).await?;
187
188                if batches.is_empty() || batches[0].num_rows() == 0 {
189                    return Err(DataFusionError::Execution(
190                        "expand_histogram subquery returned no rows".into(),
191                    ));
192                }
193
194                let batch = &batches[0];
195                if batch.num_columns() != 1 {
196                    return Err(DataFusionError::Execution(format!(
197                        "expand_histogram subquery must return exactly one column, got {}",
198                        batch.num_columns()
199                    )));
200                }
201
202                // Extract the struct from the first row
203                let column = batch.column(0);
204                let struct_array = column.as_any().downcast_ref::<StructArray>().ok_or_else(
205                    || {
206                        DataFusionError::Execution(format!(
207                            "expand_histogram subquery must return a struct (histogram), got {:?}",
208                            column.data_type()
209                        ))
210                    },
211                )?;
212
213                let histo_array = HistogramArray::new(Arc::new(struct_array.clone()));
214                if histo_array.is_empty() {
215                    RecordBatch::new_empty(output_schema())
216                } else {
217                    expand_histogram_to_batch(&histo_array, 0)?
218                }
219            }
220        };
221
222        // Apply limit if specified
223        if let Some(n) = limit
224            && n < record_batch.num_rows()
225        {
226            record_batch = record_batch.slice(0, n);
227        }
228
229        let source = MemorySourceConfig::try_new(
230            &[vec![record_batch]],
231            self.schema(),
232            projection.map(|v| v.to_owned()),
233        )?;
234        Ok(DataSourceExec::from_data_source(source))
235    }
236}
