15fn parse_json_into_jsonb(values: &[ColumnarValue]) -> Result<ColumnarValue, DataFusionError> {
16    if values.len() != 1 {
17        return Err(DataFusionError::Execution(
18            "wrong number of arguments to jsonb_parse".into(),
19        ));
20    }
21    let src_arrays = ColumnarValue::values_to_arrays(values)?;
22    let string_array: &StringArray = src_arrays[0]
23        .as_any()
24        .downcast_ref::<_>()
25        .ok_or_else(|| DataFusionError::Execution("error casting json as StringArray".into()))?;
26    let mut builder = GenericBinaryBuilder::<i32>::new();
27    let mut buffer = vec![];
28    for index in 0..string_array.len() {
29        let src_value = string_array.value(index);
30        match parse_value(src_value.as_bytes()) {
31            Ok(parsed) => {
32                buffer.clear();
33                parsed.write_to_vec(&mut buffer);
34                builder.append_value(&buffer);
35            }
36            Err(e) => {
37                warn!("error parsing json={src_value} error={e:?}");
38                builder.append_null();
39            }
40        }
41    }
42    Ok(ColumnarValue::Array(Arc::new(builder.finish())))
43}
