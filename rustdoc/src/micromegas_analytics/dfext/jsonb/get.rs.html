11fn jsonb_get(values: &[ColumnarValue]) -> Result<ColumnarValue, DataFusionError> {
12    if values.len() != 2 {
13        return Err(DataFusionError::Execution(
14            "wrong number of arguments to jsonb_get".into(),
15        ));
16    }
17    let src_arrays = ColumnarValue::values_to_arrays(values)?;
18    let jsonb_array: &GenericBinaryArray<i32> =
19        src_arrays[0].as_any().downcast_ref::<_>().ok_or_else(|| {
20            DataFusionError::Execution("error casting jsonb as GenericBinaryArray".into())
21        })?;
22    let names: &StringArray = src_arrays[1].as_any().downcast_ref::<_>().ok_or_else(|| {
23        DataFusionError::Execution("error casting second argument as StringArray".into())
24    })?;
25    let mut builder = GenericBinaryBuilder::<i32>::new();
26    for index in 0..jsonb_array.len() {
27        let src_buffer = jsonb_array.value(index);
28        let name = names.value(index);
29        let jsonb = RawJsonb::new(src_buffer);
30        if let Some(value) = jsonb
31            .get_by_name(name, true)
32            .map_err(|e| DataFusionError::External(e.into()))?
33        {
34            builder.append_value(value.to_vec());
35        } else {
36            builder.append_null();
37        }
38    }
39    Ok(ColumnarValue::Array(Arc::new(builder.finish())))
40}

42/// Creates a user-defined function to get a value from a JSONB object by name.
43pub fn make_jsonb_get_udf() -> ScalarUDF {
