1use super::metadata_cache::MetadataCache;
2use super::reader_factory::ReaderFactory;
3use datafusion::execution::runtime_env::RuntimeEnv;

5use micromegas_tracing::prelude::*;
6use std::sync::Arc;

8/// Default metadata cache size in MB
9const DEFAULT_CACHE_SIZE_MB: u64 = 50;
10
11/// Bundles all runtime resources needed for lakehouse query execution.
12///
13/// This struct holds the data lake connection, metadata cache, and DataFusion runtime,
14/// providing a single context object that can be passed through the query path.
15#[derive(Clone)]
16pub struct LakehouseContext {
17    lake: Arc<DataLakeConnection>,
18    metadata_cache: Arc<MetadataCache>,
19    runtime: Arc<RuntimeEnv>,
20    reader_factory: Arc<ReaderFactory>,
21}
22
23impl LakehouseContext {
24    /// Creates a new lakehouse context with a default-sized metadata cache.
25    pub fn new(lake: Arc<DataLakeConnection>, runtime: Arc<RuntimeEnv>) -> Self {
26        let cache_mb = match std::env::var("MICROMEGAS_METADATA_CACHE_MB") {
27            Ok(s) => s.parse::<u64>().unwrap_or_else(|_| {
28                warn!(
29                    "Invalid MICROMEGAS_METADATA_CACHE_MB value '{s}', using default {DEFAULT_CACHE_SIZE_MB} MB"
30                );
31                DEFAULT_CACHE_SIZE_MB
32            }),
33            Err(_) => DEFAULT_CACHE_SIZE_MB,
34        };
35        let metadata_cache = Arc::new(MetadataCache::new(cache_mb * 1024 * 1024));
36        let reader_factory = Arc::new(ReaderFactory::new(
37            lake.blob_storage.inner(),
38            lake.db_pool.clone(),
39            metadata_cache.clone(),
40        ));
41        Self {
42            lake,
43            metadata_cache,
44            runtime,
45            reader_factory,
46        }
47    }
48
49    /// Creates a new lakehouse context with a custom metadata cache.
50    pub fn with_cache(
51        lake: Arc<DataLakeConnection>,
52        runtime: Arc<RuntimeEnv>,
53        metadata_cache: Arc<MetadataCache>,
54    ) -> Self {
55        let reader_factory = Arc::new(ReaderFactory::new(
56            lake.blob_storage.inner(),
57            lake.db_pool.clone(),
58            metadata_cache.clone(),
59        ));
60        Self {
61            lake,
62            metadata_cache,
63            runtime,
64            reader_factory,
65        }
66    }
67
68    /// Returns the data lake connection.
69    pub fn lake(&self) -> &Arc<DataLakeConnection> {
70        &self.lake
71    }
72
73    /// Returns the metadata cache.
74    pub fn metadata_cache(&self) -> &Arc<MetadataCache> {
75        &self.metadata_cache
76    }
77
78    /// Returns the DataFusion runtime environment.
79    pub fn runtime(&self) -> &Arc<RuntimeEnv> {
80        &self.runtime
81    }
82
83    /// Returns the shared `ReaderFactory`.
84    pub fn reader_factory(&self) -> &Arc<ReaderFactory> {
85        &self.reader_factory
86    }
87}

89impl std::fmt::Debug for LakehouseContext {

94    }
