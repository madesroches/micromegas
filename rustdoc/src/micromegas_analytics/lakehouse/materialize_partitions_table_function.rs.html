23/// A DataFusion `TableFunctionImpl` for materializing lakehouse partitions.
24#[derive(Debug)]
25pub struct MaterializePartitionsTableFunction {
26    runtime: Arc<RuntimeEnv>,
27    lake: Arc<DataLakeConnection>,
28    view_factory: Arc<ViewFactory>,
29}
30
31impl MaterializePartitionsTableFunction {
32    pub fn new(
33        runtime: Arc<RuntimeEnv>,
34        lake: Arc<DataLakeConnection>,
35        view_factory: Arc<ViewFactory>,
36    ) -> Self {
37        Self {
38            runtime,
39            lake,
40            view_factory,
41        }
42    }
43}
44
45async fn materialize_partitions_impl(
46    runtime: Arc<RuntimeEnv>,
47    lake: Arc<DataLakeConnection>,
48    view_factory: Arc<ViewFactory>,
49    view_name: &str,
50    insert_range: TimeRange,
51    partition_time_delta: TimeDelta,
52    logger: Arc<dyn Logger>,
53) -> anyhow::Result<()> {
54    let view = view_factory
55        .get_global_view(view_name)
56        .with_context(|| format!("can't find view {view_name}"))?;
57
58    let existing_partitions_all_views = Arc::new(
59        PartitionCache::fetch_overlapping_insert_range(&lake.db_pool, insert_range).await?,
60    );
61
62    materialize_partition_range(
63        existing_partitions_all_views,
64        runtime,
65        lake,
66        view,
67        insert_range,
68        partition_time_delta,
69        logger,
70    )
71    .await?;
72    Ok(())
73}
74
75impl TableFunctionImpl for MaterializePartitionsTableFunction {
76    fn call(&self, args: &[Expr]) -> datafusion::error::Result<Arc<dyn TableProvider>> {
77        // an alternative would be to use coerce & create_physical_expr
78        let Some(view_set_name) = args.first().map(exp_to_string).transpose()? else {
79            return plan_err!("Missing first argument, expected view_set_name: String");
80        };
81        let Some(begin) = args.get(1).map(exp_to_timestamp).transpose()? else {
82            return plan_err!("Missing 3rd argument, expected a UTC nanoseconds timestamp");
83        };
84        let Some(end) = args.get(2).map(exp_to_timestamp).transpose()? else {
85            return plan_err!("Missing 4th argument, expected a UTC nanoseconds timestamp");
86        };
87        let Some(delta) = args.get(3).map(exp_to_i64).transpose()? else {
88            return plan_err!("Missing 5th argument, expected a number of seconds(i64)");
89        };
90
91        let lake = self.lake.clone();
92        let view_factory = self.view_factory.clone();
93        let runtime = self.runtime.clone();
94
95        let spawner = move || {
96            let (tx, rx) = tokio::sync::mpsc::channel(100);
97            let logger = Arc::new(LogSender::new(tx));
98            tokio::spawn(async move {
99                if let Err(e) = materialize_partitions_impl(
100                    runtime,
101                    lake,
102                    view_factory,
103                    &view_set_name,
104                    TimeRange::new(begin, end),
105                    TimeDelta::seconds(delta),
106                    logger.clone(),
107                )
108                .await
109                .with_context(|| "materialize_partitions_impl")
110                {
111                    let msg = format!("{e:?}");
112                    let _ = logger.write_log_entry(msg.clone()).await;
113                    error!("{msg}");
114                }
115            });
116            rx
117        };
118
119        Ok(Arc::new(LogStreamTableProvider {
120            log_stream: Arc::new(TaskLogExecPlan::new(Box::new(spawner))),
121        }))
122    }
123}
