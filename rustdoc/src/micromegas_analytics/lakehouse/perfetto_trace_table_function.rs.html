18/// `PerfettoTraceTableFunction` generates Perfetto trace chunks from process telemetry data.
19///
20/// SQL Interface:
21/// ```sql
22/// SELECT chunk_id, chunk_data
23/// FROM perfetto_trace_chunks(
24///     'process_id',                              -- Process UUID (required)
25///     'span_types',                              -- 'thread', 'async', or 'both' (required)
26///     TIMESTAMP '2024-01-01T00:00:00Z',          -- Start time as UTC timestamp (required)
27///     TIMESTAMP '2024-01-01T01:00:00Z'           -- End time as UTC timestamp (required)
28/// ) ORDER BY chunk_id
29/// ```
30///
31/// Returns a table with schema:
32/// - chunk_id: Int32 - Sequential chunk identifier
33/// - chunk_data: Binary - Binary protobuf TracePacket data
34///
35#[derive(Debug)]
36pub struct PerfettoTraceTableFunction {
37    runtime: Arc<RuntimeEnv>,
38    lake: Arc<DataLakeConnection>,
39    object_store: Arc<dyn ObjectStore>,
40    view_factory: Arc<ViewFactory>,
41    part_provider: Arc<dyn QueryPartitionProvider>,
42}
43
44impl PerfettoTraceTableFunction {
45    pub fn new(
46        runtime: Arc<RuntimeEnv>,
47        lake: Arc<DataLakeConnection>,
48        object_store: Arc<dyn ObjectStore>,
49        view_factory: Arc<ViewFactory>,
50        part_provider: Arc<dyn QueryPartitionProvider>,
51    ) -> Self {
52        Self {
53            runtime,
54            lake,
55            object_store,
56            view_factory,
57            part_provider,
58        }
59    }
60
61    /// Create the output schema for the table function
62    pub fn output_schema() -> Arc<Schema> {
63        Arc::new(Schema::new(vec![
64            Field::new("chunk_id", DataType::Int32, false),
65            Field::new("chunk_data", DataType::Binary, false),
66        ]))
67    }
68}
69
70impl TableFunctionImpl for PerfettoTraceTableFunction {
71    #[span_fn]
72    fn call(&self, exprs: &[Expr]) -> datafusion::error::Result<Arc<dyn TableProvider>> {
73        // Parse process_id (arg 1)
74        let arg1 = exprs.first().map(exp_to_string);
75        let Some(Ok(process_id)) = arg1 else {
76            return plan_err!(
77                "First argument to perfetto_trace_chunks must be a string (the process ID), given {:?}",
78                arg1
79            );
80        };
81
82        // Parse span_types (arg 2)
83        let arg2 = exprs.get(1).map(exp_to_string);
84        let Some(Ok(span_types_str)) = arg2 else {
85            return plan_err!(
86                "Second argument to perfetto_trace_chunks must be a string ('thread', 'async', or 'both'), given {:?}",
87                arg2
88            );
89        };
90
91        let span_types = match span_types_str.as_str() {
92            "thread" => SpanTypes::Thread,
93            "async" => SpanTypes::Async,
94            "both" => SpanTypes::Both,
95            _ => {
96                return plan_err!(
97                    "span_types must be 'thread', 'async', or 'both', given: {}",
98                    span_types_str
99                );
100            }
101        };
102
103        // Parse start_time (arg 3) - expecting a timestamp expression
104        let arg3 = exprs.get(2).map(exp_to_timestamp);
105        let Some(Ok(start_time)) = arg3 else {
106            return plan_err!(
107                "Third argument to perfetto_trace_chunks must be a timestamp (start time), given {:?}",
108                arg3
109            );
110        };
111
112        // Parse end_time (arg 4) - expecting a timestamp expression
113        let arg4 = exprs.get(3).map(exp_to_timestamp);
114        let Some(Ok(end_time)) = arg4 else {
115            return plan_err!(
116                "Fourth argument to perfetto_trace_chunks must be a timestamp (end time), given {:?}",
117                arg4
118            );
119        };
120
121        // Create time range from parsed timestamps
122        let time_range = TimeRange {
123            begin: start_time,
124            end: end_time,
125        };
126
127        // Create the execution plan that will generate the trace chunks
128        let execution_plan = Arc::new(PerfettoTraceExecutionPlan::new(
129            Self::output_schema(),
130            process_id,
131            span_types,
132            time_range,
133            self.runtime.clone(),
134            self.lake.clone(),
135            self.object_store.clone(),
136            self.view_factory.clone(),
137            self.part_provider.clone(),
138        ));
139
140        // Wrap it in a TableProvider
141        Ok(Arc::new(PerfettoTraceTableProvider::new(execution_plan)))
142    }
143}
