27/// Statistics about a set of partitions.
28struct PartitionStats {
29    pub num_rows: i64,
30    pub min_event_time: DateTime<Utc>,
31    pub max_event_time: DateTime<Utc>,
32}
33
34fn compute_partition_stats(partitions: &[Partition]) -> Result<PartitionStats> {
35    // Filter out empty partitions before computing stats
36    let non_empty: Vec<_> = partitions.iter().filter(|p| !p.is_empty()).collect();
37
38    if non_empty.is_empty() {
39        anyhow::bail!(
40            "compute_partition_stats given only empty partitions (should be filtered at caller)"
41        );
42    }
43
44    let first = non_empty.first().unwrap();
45    let first_event_range = first
46        .event_time_range
47        .ok_or_else(|| anyhow::anyhow!("non-empty partition has no event_time_range"))?;
48
49    let state = PartitionStats {
50        num_rows: first.num_rows,
51        min_event_time: first_event_range.begin,
52        max_event_time: first_event_range.end,
53    };
54
55    non_empty
56        .iter()
57        .skip(1)
58        .try_fold(state, |state, part| -> Result<PartitionStats> {
59            let event_range = part
60                .event_time_range
61                .ok_or_else(|| anyhow::anyhow!("non-empty partition has no event_time_range"))?;
62            Ok(PartitionStats {
63                num_rows: state.num_rows + part.num_rows,
64                min_event_time: state.min_event_time.min(event_range.begin),
65                max_event_time: state.max_event_time.max(event_range.end),
66            })
67        })
68}

70/// Merges multiple partitions by splitting the work in batches to use less memory.
71/// The batches are based on event times.
72#[derive(Debug)]
73pub struct BatchPartitionMerger {
74    /// runtime: datafusion runtime
75    runtime: Arc<RuntimeEnv>,
76    /// file_schema: arrow schema of the parquet files
77    file_schema: Arc<Schema>,
78    /// view_factory: allows joins in merge query
79    view_factory: Arc<ViewFactory>,
80    /// session_configurator: allows custom tables in merge query
81    session_configurator: Arc<dyn SessionConfigurator>,
82    /// merge_batch_query: merge query with begin & end placeholders
83    merge_batch_query: String,
84    /// batch size to aim for
85    approx_nb_rows_per_batch: i64,
86}
87
88impl BatchPartitionMerger {
89    pub fn new(
90        runtime: Arc<RuntimeEnv>,
91        file_schema: Arc<Schema>,
92        view_factory: Arc<ViewFactory>,
93        session_configurator: Arc<dyn SessionConfigurator>,
94        merge_batch_query: String,
95        approx_nb_rows_per_batch: i64,
96    ) -> Self {
97        Self {
98            runtime,
99            file_schema,
100            view_factory,
101            session_configurator,
102            merge_batch_query,
103            approx_nb_rows_per_batch,
104        }
105    }
106}
107
108#[async_trait]
109impl PartitionMerger for BatchPartitionMerger {
110    #[span_fn]
111    async fn execute_merge_query(
112        &self,
113        lake: Arc<DataLakeConnection>,
114        partitions_to_merge: Arc<Vec<Partition>>,
115        partitions_all_views: Arc<PartitionCache>,
116    ) -> Result<SendableRecordBatchStream> {
117        info!("execute_merge_query");
118
119        // If all partitions are empty, return empty stream immediately
120        if partitions_to_merge.iter().all(|p| p.is_empty()) {
121            debug!("all partitions are empty, returning empty stream");
122            let builder = RecordBatchReceiverStreamBuilder::new(self.file_schema.clone(), 1);
123            return Ok(builder.build());
124        }
125
126        let stats = compute_partition_stats(partitions_to_merge.as_ref())?;
127        let nb_batches = ((stats.num_rows / self.approx_nb_rows_per_batch) + 1) as i32;
128        let batch_time_delta = ((stats.max_event_time - stats.min_event_time) / nb_batches)
129            + TimeDelta::nanoseconds(1);
130
131        let runtime = self.runtime.clone();
132        let file_schema = self.file_schema.clone();
133        let ctx = make_session_context(
134            runtime,
135            lake.clone(),
136            partitions_all_views,
137            None,
138            self.view_factory.clone(),
139            self.session_configurator.clone(),
140        )
141        .await?;
142        let src_table = PartitionedTableProvider::new(
143            file_schema,
144            lake.blob_storage.inner(),
145            partitions_to_merge,
146            lake.db_pool.clone(),
147        );
148        ctx.register_table(
149            TableReference::Bare {
150                table: "source".into(),
151            },
152            Arc::new(src_table),
153        )?;
154        let df_template = ctx.sql(&self.merge_batch_query).await.map_err(|e| {
155            DataFusionError::Execution(format!("building template for merge query: {e:?}"))
156        })?;
157
158        let mut builder = RecordBatchReceiverStreamBuilder::new(self.file_schema.clone(), 10);
159        let sender = builder.tx();
160        builder.spawn(async move {
161            let mut streams_stream = stream::iter(0..nb_batches)
162                .map(|i| {
163                    let begin = stats.min_event_time + (batch_time_delta * i);
164                    let end = begin + batch_time_delta;
165                    debug!("merging batch {begin} {end}");
166                    df_template
167                        .clone()
168                        .with_param_values(vec![
169                            ("begin", datetime_to_scalar(begin)),
170                            ("end", datetime_to_scalar(end)),
171                        ])
172                        .map(|df| async {
173                            tokio::spawn(df.execute_stream())
174                                .await
175                                .map_err(|e| DataFusionError::External(e.into()))
176                        })
177                })
178                .try_buffered(2);
179
180            while let Some(stream_res) = streams_stream.next().await {
181                let mut merge_stream = stream_res??;
182                let sender = sender.clone();
183                while let Some(rb_res) = merge_stream.next().await {
184                    if let Err(e) = sender.send(rb_res).await {
185                        error!("sending record batch: {e:?}");
186                    }
187                }
188            }
189            Ok(())
190        });
191        debug!("building merge stream");
192        Ok(builder.build())
193    }
194}
