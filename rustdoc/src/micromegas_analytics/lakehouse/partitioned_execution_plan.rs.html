17/// Creates a partitioned execution plan for scanning Parquet files.
18#[expect(clippy::too_many_arguments)]
19pub fn make_partitioned_execution_plan(
20    schema: SchemaRef,
21    object_store: Arc<dyn ObjectStore>,
22    state: &dyn Session,
23    projection: Option<&Vec<usize>>,
24    filters: &[Expr],
25    limit: Option<usize>,
26    partitions: Arc<Vec<Partition>>,
27    pool: sqlx::PgPool,
28) -> datafusion::error::Result<Arc<dyn ExecutionPlan>> {
29    let predicate = filters_to_predicate(schema.clone(), state, filters)?;
30
31    // Filter out empty partitions (num_rows = 0, file_path = None)
32    let mut file_group = vec![];
33    for part in &*partitions {
34        if !part.is_empty() {
35            let file_path = part.file_path.as_ref().ok_or_else(|| {
36                datafusion::error::DataFusionError::Internal(format!(
37                    "non-empty partition has no file_path: num_rows={}",
38                    part.num_rows
39                ))
40            })?;
41            file_group.push(PartitionedFile::new(file_path, part.file_size as u64));
42        }
43    }
44
45    // If all partitions are empty, return EmptyExec with projected schema
46    if file_group.is_empty() {
47        use datafusion::physical_plan::empty::EmptyExec;
48        let projected_schema = if let Some(projection) = projection {
49            Arc::new(schema.project(projection)?)
50        } else {
51            schema
52        };
53        return Ok(Arc::new(EmptyExec::new(projected_schema)));
54    }
55
56    let object_store_url = ObjectStoreUrl::parse("obj://lakehouse/").unwrap();
57    let reader_factory = Arc::new(ReaderFactory::new(object_store, pool));
58    let source = Arc::new(
59        ParquetSource::default()
60            .with_predicate(predicate)
61            .with_parquet_file_reader_factory(reader_factory),
62    );
63    let file_scan_config = FileScanConfigBuilder::new(object_store_url, schema, source)
64        .with_limit(limit)
65        .with_projection(projection.cloned())
66        .with_file_groups(vec![file_group.into()])
67        .build();
68    Ok(Arc::new(DataSourceExec::new(Arc::new(file_scan_config))))
69}
