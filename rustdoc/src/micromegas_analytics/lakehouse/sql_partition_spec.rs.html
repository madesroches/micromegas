72#[async_trait]
73impl PartitionSpec for SqlPartitionSpec {
74    fn is_empty(&self) -> bool {
75        self.record_count < 1
76    }
77
78    fn get_source_data_hash(&self) -> Vec<u8> {
79        self.record_count.to_le_bytes().to_vec()
80    }
81
82    async fn write(&self, lake: Arc<DataLakeConnection>, logger: Arc<dyn Logger>) -> Result<()> {
83        if self.record_count == 0 {
84            return Ok(());
85        }
86        let desc = format!(
87            "[{}, {}] {} {}",
88            self.view_metadata.view_set_name,
89            self.view_metadata.view_instance_id,
90            self.insert_range.begin.to_rfc3339(),
91            self.insert_range.end.to_rfc3339()
92        );
93        logger.write_log_entry(format!("writing {desc}")).await?;
94        let df = self.ctx.sql(&self.extract_query).await?;
95        let mut stream = df.execute_stream().await?;
96
97        let (tx, rx) = tokio::sync::mpsc::channel(1);
98        let join_handle = tokio::spawn(write_partition_from_rows(
99            lake.clone(),
100            self.view_metadata.clone(),
101            self.schema.clone(),
102            self.insert_range,
103            self.get_source_data_hash(),
104            rx,
105            logger.clone(),
106        ));
107
108        while let Some(rb_res) = stream.next().await {
109            let rb = self.transformer.transform(rb_res?).await?;
110            let (mintime, maxtime) = min_max_time_dataframe(
111                self.ctx.read_batch(rb.clone())?,
112                &self.min_event_time_column,
113                &self.max_event_time_column,
114            )
115            .await?;
116            tx.send(PartitionRowSet {
117                min_time_row: mintime,
118                max_time_row: maxtime,
119                rows: rb,
120            })
121            .await?;
122        }
123        drop(tx);
124        join_handle.await??;
125        Ok(())
126    }
127}

129/// Fetches a `SqlPartitionSpec` by executing a count query and an extract query.
130#[allow(clippy::too_many_arguments)]
131pub async fn fetch_sql_partition_spec(
132    ctx: SessionContext,
133    transformer: Arc<dyn RecordBatchTransformer>,
134    schema: Arc<Schema>,
135    count_src_sql: String,
136    extract_query: String,
137    min_event_time_column: Arc<String>,
138    max_event_time_column: Arc<String>,
139    view_metadata: ViewMetadata,
140    insert_range: TimeRange,
141) -> Result<SqlPartitionSpec> {
142    let df = ctx.sql(&count_src_sql).await?;
143    let batches: Vec<RecordBatch> = df.collect().await?;
144    if batches.len() != 1 {
145        anyhow::bail!("fetch_sql_partition_spec: query should return a single batch");
146    }
147    let rb = &batches[0];
148    let count_column: &Int64Array = typed_column_by_name(rb, "count")?;
149    if count_column.len() != 1 {
150        anyhow::bail!("fetch_sql_partition_spec: query should return a single row");
151    }
152    let count = count_column.value(0);
153    if count > 0 {
154        trace!(
155            "fetch_sql_partition_spec for view {}, count={count}",
156            &*view_metadata.view_set_name
157        );
158    }
159    Ok(SqlPartitionSpec::new(
160        ctx,
161        transformer,
162        schema,
163        extract_query,
164        min_event_time_column,
165        max_event_time_column,
166        view_metadata,
167        insert_range,
168        count,
169    ))
170}
