55/// A view of log entries.
56#[derive(Debug)]
57pub struct LogView {
58    view_set_name: Arc<String>,
59    view_instance_id: Arc<String>,
60    process_id: Option<sqlx::types::Uuid>,
61}
62
63impl LogView {
64    pub fn new(view_instance_id: &str) -> Result<Self> {
65        let process_id = if view_instance_id == "global" {
66            None
67        } else {
68            Some(Uuid::parse_str(view_instance_id).with_context(|| "Uuid::parse_str")?)
69        };
70
71        Ok(Self {
72            view_set_name: Arc::new(String::from(VIEW_SET_NAME)),
73            view_instance_id: Arc::new(view_instance_id.into()),
74            process_id,
75        })
76    }
77}

80impl View for LogView {
81    fn get_view_set_name(&self) -> Arc<String> {

89    async fn make_batch_partition_spec(
90        &self,
91        runtime: Arc<RuntimeEnv>,
92        lake: Arc<DataLakeConnection>,
93        existing_partitions: Arc<PartitionCache>,
94        insert_range: TimeRange,
95    ) -> Result<Arc<dyn PartitionSpec>> {
96        if *self.view_instance_id != "global" {
97            anyhow::bail!("not supported for jit queries... should it?");
98        }
99        let source_data = Arc::new(
100            fetch_partition_source_data(runtime, lake, existing_partitions, insert_range, "log")
101                .await
102                .with_context(|| "fetch_partition_source_data")?,
103        );
104        Ok(Arc::new(BlockPartitionSpec {
105            view_metadata: ViewMetadata {
106                view_set_name: self.view_set_name.clone(),
107                view_instance_id: self.view_instance_id.clone(),
108                file_schema_hash: self.get_file_schema_hash(),
109            },
110            schema: self.get_file_schema(),
111            insert_range,
112            source_data,
113            block_processor: Arc::new(LogBlockProcessor {}),
114        }))
115    }
116
117    fn get_file_schema_hash(&self) -> Vec<u8> {
118        vec![SCHEMA_VERSION]
119    }
120
121    fn get_file_schema(&self) -> Arc<Schema> {
122        Arc::new(log_table_schema())
123    }
124
125    async fn jit_update(
126        &self,
127        runtime: Arc<RuntimeEnv>,
128        lake: Arc<DataLakeConnection>,
129        query_range: Option<TimeRange>,
130    ) -> Result<()> {
131        if *self.view_instance_id == "global" {
132            // this view instance is updated using the deamon
133            return Ok(());
134        }
135        let process = Arc::new(
136            find_process(
137                &lake.db_pool,
138                &self
139                    .process_id
140                    .with_context(|| "getting a view's process_id")?,
141            )
142            .await
143            .with_context(|| "find_process")?,
144        );
145        let query_range =
146            query_range.unwrap_or_else(|| TimeRange::new(process.start_time, chrono::Utc::now()));
147
148        let blocks_view = BlocksView::new()?;
149        let all_partitions = generate_process_jit_partitions(
150            &JitPartitionConfig::default(),
151            runtime.clone(),
152            lake.clone(),
153            &blocks_view,
154            &query_range,
155            process.clone(),
156            "log",
157        )
158        .await
159        .with_context(|| "generate_process_jit_partitions")?;
160        let view_meta = ViewMetadata {
161            view_set_name: self.get_view_set_name(),
162            view_instance_id: self.get_view_instance_id(),
163            file_schema_hash: self.get_file_schema_hash(),
164        };
165
166        for part in all_partitions {
167            if !is_jit_partition_up_to_date(&lake.db_pool, view_meta.clone(), &part).await? {
168                write_partition_from_blocks(
169                    lake.clone(),
170                    view_meta.clone(),
171                    self.get_file_schema(),
172                    part,
173                    Arc::new(LogBlockProcessor {}),
174                )
175                .await
176                .with_context(|| "write_partition_from_blocks")?;
177            }
178        }
179        Ok(())
180    }

182    fn make_time_filter(&self, begin: DateTime<Utc>, end: DateTime<Utc>) -> Result<Vec<Expr>> {
