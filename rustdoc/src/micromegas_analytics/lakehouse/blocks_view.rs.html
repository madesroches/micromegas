28/// A view of the `blocks` table, providing access to telemetry block metadata.
29#[derive(Debug)]
30pub struct BlocksView {
31    view_set_name: Arc<String>,
32    view_instance_id: Arc<String>,
33    data_sql: Arc<String>,
34}
35
36impl BlocksView {
37    pub fn new() -> Result<Self> {
38        let data_sql = Arc::new(String::from(
39            r#"SELECT block_id, streams.stream_id, processes.process_id, blocks.begin_time, blocks.begin_ticks, blocks.end_time, blocks.end_ticks, blocks.nb_objects, blocks.object_offset, blocks.payload_size, blocks.insert_time,
40           streams.dependencies_metadata, streams.objects_metadata, streams.tags, streams.properties, streams.insert_time as stream_insert_time,
41           processes.start_time, processes.start_ticks, processes.tsc_frequency, processes.exe, processes.username, processes.realname, processes.computer, processes.distro, processes.cpu_brand, processes.insert_time as process_insert_time, processes.parent_process_id, processes.properties as process_properties
42         FROM blocks, streams, processes
43         WHERE blocks.stream_id = streams.stream_id
44         AND blocks.process_id = processes.process_id
45         AND blocks.insert_time >= $1
46         AND blocks.insert_time < $2
47         ORDER BY blocks.insert_time, blocks.block_id
48         ;"#,
49        ));
50        Ok(Self {
51            view_set_name: Arc::new(String::from(VIEW_SET_NAME)),
52            view_instance_id: Arc::new(String::from(VIEW_INSTANCE_ID)),
53            data_sql,
54        })
55    }
56}

58#[async_trait]
59impl View for BlocksView {
60    fn get_view_set_name(&self) -> Arc<String> {
61        self.view_set_name.clone()
62    }
63
64    fn get_view_instance_id(&self) -> Arc<String> {
65        self.view_instance_id.clone()

67
68    async fn make_batch_partition_spec(
69        &self,
70        _runtime: Arc<RuntimeEnv>,
71        lake: Arc<DataLakeConnection>,
72        _existing_partitions: Arc<PartitionCache>,
73        insert_range: TimeRange,
74    ) -> Result<Arc<dyn PartitionSpec>> {
75        let view_meta = ViewMetadata {
76            view_set_name: self.get_view_set_name(),
77            view_instance_id: self.get_view_instance_id(),
78            file_schema_hash: self.get_file_schema_hash(),
79        };
80        let source_count_query = "
81             SELECT COUNT(*) as count
82             FROM blocks, streams, processes
83             WHERE blocks.stream_id = streams.stream_id
84             AND blocks.process_id = processes.process_id
85             AND blocks.insert_time >= $1
86             AND blocks.insert_time < $2
87             ;";
88        Ok(Arc::new(
89            fetch_metadata_partition_spec(
90                &lake.db_pool,
91                source_count_query,
92                self.data_sql.clone(),
93                view_meta,
94                self.get_file_schema(),
95                insert_range,
96                self.get_time_bounds(),
97            )
98            .await
99            .with_context(|| "fetch_metadata_partition_spec")?,
100        ))
101    }
102
103    fn get_file_schema_hash(&self) -> Vec<u8> {
104        blocks_file_schema_hash()
105    }
106
107    fn get_file_schema(&self) -> Arc<Schema> {
108        Arc::new(blocks_view_schema())
109    }
110
111    async fn jit_update(
112        &self,
113        _runtime: Arc<RuntimeEnv>,
114        _lake: Arc<DataLakeConnection>,
115        _query_range: Option<TimeRange>,
116    ) -> Result<()> {
117        if *self.view_instance_id == "global" {
118            // this view instance is updated using the deamon
119            return Ok(());
120        }
121        anyhow::bail!("not supported");
122    }
123
124    fn make_time_filter(&self, begin: DateTime<Utc>, end: DateTime<Utc>) -> Result<Vec<Expr>> {
125        Ok(vec![
126            col("begin_time").lt_eq(lit(datetime_to_scalar(end))),
127            col("insert_time").gt_eq(lit(datetime_to_scalar(begin))),
128        ])
129    }
130
131    fn get_time_bounds(&self) -> Arc<dyn DataFrameTimeBounds> {
132        //todo: make more robust, by changing to [ min(begin, insert), max(end, insert) ]
133        Arc::new(NamedColumnsTimeBounds::new(
134            BEGIN_TIME_COLUMN.clone(),
135            INSERT_TIME_COLUMN.clone(),
136        ))
137    }
138
139    fn get_update_group(&self) -> Option<i32> {
140        Some(1000)
141    }
142
143    fn get_max_partition_time_delta(&self, strategy: &PartitionCreationStrategy) -> TimeDelta {
144        match strategy {
145            PartitionCreationStrategy::Abort | PartitionCreationStrategy::CreateFromSource => {
146                TimeDelta::hours(1)
147            }
148            PartitionCreationStrategy::MergeExisting(_partitions) => TimeDelta::days(1),
149        }
150    }
