16/// An async scalar UDF that deletes duplicate blocks within the query time range.
17///
18/// A duplicate is defined as multiple rows with the same `block_id`. This function
19/// keeps the row with the earliest `insert_time` and deletes all others.
20///
21/// The time range is passed via the constructor (not as SQL arguments), similar to
22/// `ViewInstanceTableFunction`.
23#[derive(Debug)]
24pub struct DeleteDuplicateBlocks {
25    signature: Signature,
26    lake: Arc<DataLakeConnection>,
27    query_range: Option<TimeRange>,
28}

83impl AsyncScalarUDFImpl for DeleteDuplicateBlocks {

86        &self,
87        args: ScalarFunctionArgs,
88    ) -> datafusion::error::Result<ColumnarValue> {
89        let args = ColumnarValue::values_to_arrays(&args.args)?;
90        if !args.is_empty() {
91            return internal_err!("delete_duplicate_blocks expects no arguments");
92        }
93
94        let Some(range) = &self.query_range else {
95            return internal_err!("delete_duplicate_blocks requires a query time range to be set");

98        let deleted_count = delete_duplicate_blocks(&self.lake.db_pool, *range)
99            .await
100            .map_err(|e| DataFusionError::Execution(format!("Failed to delete duplicates: {e}")))?;
101
102        let mut builder = StringBuilder::with_capacity(1, 64);
103        builder.append_value(format!("Deleted {deleted_count} duplicate blocks"));
104
105        Ok(ColumnarValue::Array(Arc::new(builder.finish())))
106    }

127/// Deletes duplicate blocks within the specified time range.
128///
129/// A duplicate is defined as multiple rows with the same `block_id`. This function
130/// keeps the row with the earliest `insert_time` and deletes all others.
131///
132/// Returns the number of deleted duplicate blocks.
133#[span_fn]
134pub async fn delete_duplicate_blocks(
135    pool: &sqlx::PgPool,
136    time_range: TimeRange,
137) -> anyhow::Result<u64> {
138    let mut transaction = pool.begin().await?;
139
140    // Delete duplicates, keeping the row with the earliest insert_time for each block_id.
141    // Note: The time range is used to identify block_ids that have duplicates, but once
142    // identified, ALL duplicate rows for that block_id are deleted regardless of their
143    // insert_time. This ensures complete cleanup of duplicates even if some copies
144    // exist outside the query range.
145    let delete_result = sqlx::query(
146        "WITH dups AS (
147            SELECT block_id, MIN(insert_time) as keep_time
148            FROM blocks
149            WHERE insert_time >= $1 AND insert_time < $2
150            GROUP BY block_id
151            HAVING COUNT(*) > 1
152        )
153        DELETE FROM blocks b
154        USING dups d
155        WHERE b.block_id = d.block_id
156        AND b.insert_time != d.keep_time",
157    )
158    .bind(time_range.begin)
159    .bind(time_range.end)
160    .execute(&mut *transaction)
161    .await?;
162
163    let deleted_count = delete_result.rows_affected();
164
165    transaction.commit().await?;
166
167    if deleted_count > 0 {
168        info!(
169            "Deleted {deleted_count} duplicate blocks in range [{}, {})",
170            time_range.begin, time_range.end
171        );
172    }
173
174    Ok(deleted_count)
175}
