61    part_provider: Arc<dyn QueryPartitionProvider>,
62    query_range: Option<TimeRange>,

65) -> Result<()> {

74}

76/// query_partitions_context returns a context to run queries using the partitions as the "source" table
77#[span_fn]
78pub async fn query_partitions_context(
79    runtime: Arc<RuntimeEnv>,
80    reader_factory: Arc<ReaderFactory>,
81    object_store: Arc<dyn object_store::ObjectStore>,
82    schema: SchemaRef,
83    partitions: Arc<Vec<Partition>>,
84) -> Result<SessionContext> {
85    let table = PartitionedTableProvider::new(schema, reader_factory, partitions);
86    let object_store_url = ObjectStoreUrl::parse("obj://lakehouse/").unwrap();
87    let ctx = SessionContext::new_with_config_rt(SessionConfig::default(), runtime);
88    ctx.register_object_store(object_store_url.as_ref(), object_store);
89    ctx.register_table(
90        TableReference::Bare {
91            table: "source".into(),
92        },
93        Arc::new(table),
94    )?;
95    register_extension_functions(&ctx);
96    Ok(ctx)
97}

99// query_partitions returns a dataframe, leaving the option of streaming the results
100#[span_fn]
101pub async fn query_partitions(
102    runtime: Arc<RuntimeEnv>,
103    reader_factory: Arc<ReaderFactory>,
104    object_store: Arc<dyn object_store::ObjectStore>,
105    schema: SchemaRef,
106    partitions: Arc<Vec<Partition>>,
107    sql: &str,
108) -> Result<DataFrame> {
109    let ctx =
110        query_partitions_context(runtime, reader_factory, object_store, schema, partitions).await?;
111    Ok(ctx.sql(sql).await?)
112}

114/// register functions that are part of the lakehouse architecture
115#[span_fn]
116pub fn register_lakehouse_functions(
117    ctx: &SessionContext,
118    lakehouse: Arc<LakehouseContext>,
119    part_provider: Arc<dyn QueryPartitionProvider>,
120    query_range: Option<TimeRange>,
121    view_factory: Arc<ViewFactory>,
122) {
123    ctx.register_udtf(
124        "view_instance",
125        Arc::new(ViewInstanceTableFunction::new(
126            lakehouse.clone(),
127            view_factory.clone(),
128            part_provider.clone(),
129            query_range,
130        )),
131    );
132    ctx.register_udtf(
133        "list_partitions",
134        Arc::new(ListPartitionsTableFunction::new(lakehouse.lake().clone())),
135    );
136    ctx.register_udtf(
137        "list_view_sets",
138        Arc::new(ListViewSetsTableFunction::new(view_factory.clone())),
139    );
140    ctx.register_udtf(
141        "retire_partitions",
142        Arc::new(RetirePartitionsTableFunction::new(lakehouse.lake().clone())),
143    );
144    ctx.register_udtf(
145        "perfetto_trace_chunks",
146        Arc::new(PerfettoTraceTableFunction::new(
147            lakehouse.clone(),
148            view_factory.clone(),
149            part_provider.clone(),
150        )),
151    );
152    ctx.register_udtf(
153        "materialize_partitions",
154        Arc::new(MaterializePartitionsTableFunction::new(
155            lakehouse.clone(),
156            view_factory.clone(),
157        )),
158    );
159    ctx.register_udf(
160        AsyncScalarUDF::new(Arc::new(GetPayload::new(lakehouse.lake().clone()))).into_scalar_udf(),
161    );
162    ctx.register_udf(make_retire_partition_by_file_udf(lakehouse.lake().clone()).into_scalar_udf());
163    ctx.register_udf(
164        make_retire_partition_by_metadata_udf(lakehouse.lake().clone()).into_scalar_udf(),
165    );
166    ctx.register_udf(
167        make_delete_duplicate_blocks_udf(lakehouse.lake().clone(), query_range).into_scalar_udf(),
168    );
169    ctx.register_udf(
170        make_delete_duplicate_processes_udf(lakehouse.lake().clone(), query_range)
171            .into_scalar_udf(),
172    );
173    ctx.register_udf(
174        make_delete_duplicate_streams_udf(lakehouse.lake().clone(), query_range).into_scalar_udf(),
175    );
176}

218#[span_fn]
219pub async fn make_session_context(
220    lakehouse: Arc<LakehouseContext>,
221    part_provider: Arc<dyn QueryPartitionProvider>,
222    query_range: Option<TimeRange>,
223    view_factory: Arc<ViewFactory>,
224    configurator: Arc<dyn SessionConfigurator>,
225) -> Result<SessionContext> {
226    // Disable page index reading for backward compatibility with legacy Parquet files
227    // Legacy files may have incomplete ColumnIndex metadata (missing null_pages field)
228    // which causes errors in DataFusion 51+ with Arrow 57.0 when reading page indexes
229    let config = SessionConfig::default()
230        .set_bool("datafusion.execution.parquet.enable_page_index", false)
231        .with_information_schema(true);
232    let ctx = SessionContext::new_with_config_rt(config, lakehouse.runtime().clone());
233    if let Some(range) = &query_range {
234        ctx.add_analyzer_rule(Arc::new(TableScanRewrite::new(*range)));
235    }
236    let object_store_url = ObjectStoreUrl::parse("obj://lakehouse/").unwrap();
237    let object_store = lakehouse.lake().blob_storage.inner();
238    ctx.register_object_store(object_store_url.as_ref(), object_store);
239    let reader_factory = lakehouse.reader_factory().clone();
240    register_functions(
241        &ctx,
242        lakehouse.clone(),
243        part_provider.clone(),
244        query_range,
245        view_factory.clone(),
246    );
247    for view in view_factory.get_global_views() {
248        register_table(
249            lakehouse.clone(),
250            reader_factory.clone(),
251            part_provider.clone(),
252            query_range,
253            &ctx,
254            view.clone(),
255        )
256        .await?;
257    }
258    // Apply custom configuration
259    configurator.configure(&ctx).await?;
260    Ok(ctx)
261}

263#[span_fn]
264pub async fn query(
265    lakehouse: Arc<LakehouseContext>,
266    part_provider: Arc<dyn QueryPartitionProvider>,
267    query_range: Option<TimeRange>,
268    sql: &str,
269    view_factory: Arc<ViewFactory>,
270    configurator: Arc<dyn SessionConfigurator>,
271) -> Result<Answer> {
272    info!("query sql={sql}");
273    let ctx = make_session_context(
274        lakehouse,
275        part_provider,
276        query_range,
277        view_factory,
278        configurator,
279    )
280    .await
281    .with_context(|| "make_session_context")?;
282    let df = ctx.sql(sql).await?;
283    let schema = df.schema().inner().clone();
284    let batches: Vec<RecordBatch> = df.collect().await?;
285    Ok(Answer::new(schema, batches))
286}
