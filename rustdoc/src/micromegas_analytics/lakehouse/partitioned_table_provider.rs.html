15/// A DataFusion `TableProvider` for a set of pre-defined partitions.
16pub struct PartitionedTableProvider {
17    schema: SchemaRef,
18    object_store: Arc<dyn ObjectStore>,
19    partitions: Arc<Vec<Partition>>,
20    pool: PgPool,
21}
22
23impl std::fmt::Debug for PartitionedTableProvider {
24    fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
25        f.debug_struct("PartitionedTableProvider")
26            .field("schema", &self.schema)
27            .field("partitions_count", &self.partitions.len())
28            .finish()
29    }
30}
31
32impl PartitionedTableProvider {
33    pub fn new(
34        schema: SchemaRef,
35        object_store: Arc<dyn ObjectStore>,
36        partitions: Arc<Vec<Partition>>,
37        pool: PgPool,
38    ) -> Self {
39        Self {
40            schema,
41            object_store,
42            partitions,
43            pool,
44        }
45    }
46}
47
48#[async_trait]
49impl TableProvider for PartitionedTableProvider {
50    fn as_any(&self) -> &dyn Any {
51        self
52    }
53
54    fn schema(&self) -> SchemaRef {
55        self.schema.clone()
56    }
57
58    fn table_type(&self) -> TableType {
59        TableType::Base
60    }
61
62    async fn scan(
63        &self,
64        state: &dyn Session,
65        projection: Option<&Vec<usize>>,
66        filters: &[Expr],
67        limit: Option<usize>,
68    ) -> datafusion::error::Result<Arc<dyn ExecutionPlan>> {
69        make_partitioned_execution_plan(
70            self.schema(),
71            self.object_store.clone(),
72            state,
73            projection,
74            filters,
75            limit,
76            self.partitions.clone(),
77            self.pool.clone(),
78        )
79    }
80
81    /// Tell DataFusion to push filters down to the scan method
82    fn supports_filters_pushdown(
83        &self,
84        filters: &[&Expr],
85    ) -> datafusion::error::Result<Vec<TableProviderFilterPushDown>> {
86        // Inexact because the pruning can't handle all expressions and pruning
87        // is not done at the row level -- there may be rows in returned files
88        // that do not pass the filter
89        Ok(vec![TableProviderFilterPushDown::Inexact; filters.len()])
90    }
