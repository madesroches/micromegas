17/// BlockProcessor transforms a single block of telemetry into a set of rows
18#[async_trait]
19pub trait BlockProcessor: Send + Sync + Debug {
20    /// Processes a single block of telemetry.
21    async fn process(
22        &self,
23        blob_storage: Arc<BlobStorage>,
24        src_block: Arc<PartitionSourceBlock>,
25    ) -> Result<Option<PartitionRowSet>>;
26}

28/// BlockPartitionSpec processes blocks individually and out of order
29/// which works fine for measures & log entries
30#[derive(Debug)]
31pub struct BlockPartitionSpec {
32    pub view_metadata: ViewMetadata,
33    pub schema: Arc<Schema>,
34    pub insert_range: TimeRange,
35    pub source_data: Arc<dyn PartitionBlocksSource>,
36    pub block_processor: Arc<dyn BlockProcessor>,
37}
38
39#[async_trait]
40impl PartitionSpec for BlockPartitionSpec {
41    fn is_empty(&self) -> bool {
42        self.source_data.is_empty()
43    }
44
45    fn get_source_data_hash(&self) -> Vec<u8> {
46        self.source_data.get_source_data_hash()
47    }
48
49    async fn write(&self, lake: Arc<DataLakeConnection>, logger: Arc<dyn Logger>) -> Result<()> {
50        let desc = format!(
51            "[{}, {}] {} {}",
52            self.view_metadata.view_set_name,
53            self.view_metadata.view_instance_id,
54            self.insert_range.begin.to_rfc3339(),
55            self.insert_range.end.to_rfc3339()
56        );
57        logger.write_log_entry(format!("writing {desc}")).await?;
58
59        logger
60            .write_log_entry(format!(
61                "reading {} blocks",
62                self.source_data.get_nb_blocks()
63            ))
64            .await?;
65
66        // Allow empty source data - write_partition_from_rows will create
67        // an empty partition record if no data is sent through the channel
68        let (tx, rx) = tokio::sync::mpsc::channel(1);
69        let join_handle = tokio::spawn(write_partition_from_rows(
70            lake.clone(),
71            self.view_metadata.clone(),
72            self.schema.clone(),
73            self.insert_range,
74            self.source_data.get_source_data_hash(),
75            rx,
76            logger.clone(),
77        ));
78
79        // If source data is empty, just close the channel to create an empty partition
80        if self.source_data.is_empty() {
81            drop(tx);
82            join_handle.await??;
83            return Ok(());
84        }
85
86        let max_size = self.source_data.get_max_payload_size() as usize;
87        let mut nb_tasks = (100 * 1024 * 1024) / max_size; // try to download up to 100 MB of payloads
88        nb_tasks = nb_tasks.clamp(1, 64);
89
90        let mut stream = self
91            .source_data
92            .get_blocks_stream()
93            .await
94            .map(|src_block_res| async {
95                let src_block = src_block_res.with_context(|| "get_blocks_stream")?;
96                let block_processor = self.block_processor.clone();
97                let blob_storage = lake.blob_storage.clone();
98                let handle = tokio::spawn(async move {
99                    block_processor
100                        .process(blob_storage, src_block)
101                        .await
102                        .with_context(|| "processing source block")
103                });
104                handle.await.with_context(|| "handle.await")?
105            })
106            .buffer_unordered(nb_tasks);
107
108        while let Some(res_opt_rows) = stream.next().await {
109            match res_opt_rows {
110                Err(e) => {
111                    error!("{e:?}");
112                    logger.write_log_entry(format!("{e:?}")).await?;
113                }
114                Ok(Some(row_set)) => {
115                    tx.send(row_set).await?;
116                }
117                Ok(None) => {
118                    debug!("empty block");
119                }
120            }
121        }
122        drop(tx);
123        join_handle.await??;
124        Ok(())
125    }
126}
