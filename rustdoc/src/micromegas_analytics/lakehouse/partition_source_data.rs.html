33/// Represents a single block of source data for a partition.
34#[derive(Debug)]
35pub struct PartitionSourceBlock {
36    pub block: BlockMetadata,
37    pub stream: Arc<StreamMetadata>,
38    pub process: Arc<ProcessMetadata>,
39}
40
41/// A trait for providing blocks of source data for partitions.
42#[async_trait]
43pub trait PartitionBlocksSource: Sync + Send + Debug {
44    /// Returns true if there are no blocks in the source.
45    fn is_empty(&self) -> bool;
46    /// Returns the number of blocks in the source.
47    fn get_nb_blocks(&self) -> i64;
48    /// Returns the maximum payload size of the blocks in the source.
49    fn get_max_payload_size(&self) -> i64;
50    /// Returns a hash of the source data.
51    fn get_source_data_hash(&self) -> Vec<u8>;
52    /// Returns a stream of the source blocks.
53    async fn get_blocks_stream(&self) -> BoxStream<'static, Result<Arc<PartitionSourceBlock>>>;
54}
55
56/// A `PartitionBlocksSource` implementation that stores blocks in memory.
57#[derive(Debug)]
58pub struct SourceDataBlocksInMemory {
59    pub blocks: Vec<Arc<PartitionSourceBlock>>,
60    pub block_ids_hash: Vec<u8>,
61}
62
63#[async_trait]
64impl PartitionBlocksSource for SourceDataBlocksInMemory {
65    fn is_empty(&self) -> bool {
66        self.blocks.is_empty()
67    }
68
69    fn get_nb_blocks(&self) -> i64 {
70        self.blocks.len() as i64
71    }
72
73    fn get_max_payload_size(&self) -> i64 {
74        let mut max_size = self.blocks[0].block.payload_size;
75        for block in &self.blocks {
76            max_size = max_size.max(block.block.payload_size);
77        }
78        max_size
79    }
80
81    fn get_source_data_hash(&self) -> Vec<u8> {
82        self.block_ids_hash.clone()
83    }
84
85    async fn get_blocks_stream(&self) -> BoxStream<'static, Result<Arc<PartitionSourceBlock>>> {
86        let stream = futures::stream::iter(self.blocks.clone()).map(Ok);
87        stream.boxed()
88    }
89}

91/// A `PartitionBlocksSource` implementation that fetches blocks from a DataFusion DataFrame.
92#[derive(Debug)]
93pub struct SourceDataBlocks {
94    pub blocks_dataframe: DataFrame,
95    pub object_count: i64,
96    pub block_count: i64,
97    pub max_payload: i64,
98}
99
100#[async_trait]
101impl PartitionBlocksSource for SourceDataBlocks {
102    fn is_empty(&self) -> bool {
103        self.object_count == 0
104    }
105
106    fn get_nb_blocks(&self) -> i64 {
107        self.block_count
108    }
109
110    fn get_max_payload_size(&self) -> i64 {
111        self.max_payload
112    }
113
114    fn get_source_data_hash(&self) -> Vec<u8> {
115        self.object_count.to_le_bytes().to_vec()
116    }
117
118    async fn get_blocks_stream(&self) -> BoxStream<'static, Result<Arc<PartitionSourceBlock>>> {
119        let df = self.blocks_dataframe.clone();
120        Box::pin(try_stream! {
121            let mut stream = df.execute_stream().await?;
122            while let Some(res) = stream.next().await {
123                let b = res.with_context(|| "fetching blocks query results")?;
124                let block_id_column = string_column_by_name(&b, "block_id")?;
125                let stream_id_column = string_column_by_name(&b, "stream_id")?;
126                let process_id_column = string_column_by_name(&b, "process_id")?;
127                let begin_time_column: &TimestampNanosecondArray = typed_column_by_name(&b, "begin_time")?;
128                let begin_ticks_column: &Int64Array = typed_column_by_name(&b, "begin_ticks")?;
129                let end_time_column: &TimestampNanosecondArray = typed_column_by_name(&b, "end_time")?;
130                let end_ticks_column: &Int64Array = typed_column_by_name(&b, "end_ticks")?;
131                let nb_objects_column: &Int32Array = typed_column_by_name(&b, "nb_objects")?;
132                let object_offset_column: &Int64Array = typed_column_by_name(&b, "object_offset")?;
133                let payload_size_column: &Int64Array = typed_column_by_name(&b, "payload_size")?;
134                let block_insert_time_column: &TimestampNanosecondArray =
135                    typed_column_by_name(&b, "insert_time")?;
136                let dependencies_metadata_column: &BinaryArray =
137                    typed_column_by_name(&b, "streams.dependencies_metadata")?;
138                let objects_metadata_column: &BinaryArray =
139                    typed_column_by_name(&b, "streams.objects_metadata")?;
140                let stream_tags_column: &GenericListArray<i32> = typed_column_by_name(&b, "streams.tags")?;
141                let stream_properties_accessor = properties_column_by_name(&b, "streams.properties")?;
142
143                let process_start_time_column: &TimestampNanosecondArray =
144                    typed_column_by_name(&b, "processes.start_time")?;
145                let process_start_ticks_column: &Int64Array =
146                    typed_column_by_name(&b, "processes.start_ticks")?;
147                let process_tsc_freq_column: &Int64Array =
148                    typed_column_by_name(&b, "processes.tsc_frequency")?;
149                let process_exe_column = string_column_by_name(&b, "processes.exe")?;
150                let process_username_column = string_column_by_name(&b, "processes.username")?;
151                let process_realname_column = string_column_by_name(&b, "processes.realname")?;
152                let process_computer_column = string_column_by_name(&b, "processes.computer")?;
153                let process_distro_column = string_column_by_name(&b, "processes.distro")?;
154                let process_cpu_column = string_column_by_name(&b, "processes.cpu_brand")?;
155                let process_parent_column = string_column_by_name(&b, "processes.parent_process_id")?;
156                let process_properties_accessor = properties_column_by_name(&b, "processes.properties")?;
157                for ir in 0..b.num_rows() {
158                    let block_insert_time = block_insert_time_column.value(ir);
159                    let stream_id = Uuid::parse_str(stream_id_column.value(ir))?;
160                    let process_id = Uuid::parse_str(process_id_column.value(ir))?;
161                    let block = BlockMetadata {
162                        block_id: Uuid::parse_str(block_id_column.value(ir))?,
163                        stream_id,
164                        process_id,
165                        begin_time: DateTime::from_timestamp_nanos(begin_time_column.value(ir)),
166                        end_time: DateTime::from_timestamp_nanos(end_time_column.value(ir)),
167                        begin_ticks: begin_ticks_column.value(ir),
168                        end_ticks: end_ticks_column.value(ir),
169                        nb_objects: nb_objects_column.value(ir),
170                        payload_size: payload_size_column.value(ir),
171                        object_offset: object_offset_column.value(ir),
172                        insert_time: DateTime::from_timestamp_nanos(block_insert_time),
173                    };
174
175                    let dependencies_metadata = dependencies_metadata_column.value(ir);
176                    let objects_metadata = objects_metadata_column.value(ir);
177                    let stream_tags = stream_tags_column
178                        .value(ir)
179                        .as_any()
180                        .downcast_ref::<StringArray>()
181                        .with_context(|| "casting stream_tags")?
182                        .iter()
183                        .map(|item| String::from(item.unwrap_or_default()))
184                        .collect();
185                    let stream_properties_jsonb = stream_properties_accessor.jsonb_value(ir)?;
186                    let stream = StreamMetadata {
187                        process_id,
188                        stream_id,
189                        dependencies_metadata: ciborium::from_reader(dependencies_metadata)
190                            .with_context(|| "decoding dependencies_metadata")?,
191                        objects_metadata: ciborium::from_reader(objects_metadata)
192                            .with_context(|| "decoding objects_metadata")?,
193                        tags: stream_tags,
194                        properties: Arc::new(stream_properties_jsonb),
195                    };
196                    let process_properties_jsonb = process_properties_accessor.jsonb_value(ir)?;
197                    let parent_value = process_parent_column.value(ir);
198                    let parent_process_id = if parent_value.is_empty() {
199                        None
200                    } else {
201                        Some(Uuid::parse_str(parent_value).with_context(|| "parsing parent process_id")?)
202                    };
203
204                    let process = ProcessMetadata {
205                        process_id,
206                        exe: process_exe_column.value(ir).into(),
207                        username: process_username_column.value(ir).into(),
208                        realname: process_realname_column.value(ir).into(),
209                        computer: process_computer_column.value(ir).into(),
210                        distro: process_distro_column.value(ir).into(),
211                        cpu_brand: process_cpu_column.value(ir).into(),
212                        tsc_frequency: process_tsc_freq_column.value(ir),
213                        start_time: DateTime::from_timestamp_nanos(process_start_time_column.value(ir)),
214                        start_ticks: process_start_ticks_column.value(ir),
215                        parent_process_id,
216                        properties: Arc::new(process_properties_jsonb),
217                    };
218                    yield Arc::new(PartitionSourceBlock {
219                        block,
220                        stream: Arc::new(stream),
221                        process: Arc::new(process),
222                    });
223                }
224            }
225
226        })
227    }

237/// Fetches partition source data from the data lake.
238pub async fn fetch_partition_source_data(
239    runtime: Arc<RuntimeEnv>,
240    lake: Arc<DataLakeConnection>,
241    existing_partitions: Arc<PartitionCache>,
242    insert_range: TimeRange,
243    source_stream_tag: &str,
244) -> Result<SourceDataBlocks> {
245    let begin_rfc = insert_range.begin.to_rfc3339();
246    let end_rfc = insert_range.end.to_rfc3339();
247    let sql = format!(
248        r#"
249          SELECT block_id, stream_id, process_id, begin_time, begin_ticks, end_time, end_ticks, nb_objects,
250              object_offset, payload_size, insert_time,
251              "streams.dependencies_metadata", "streams.objects_metadata", "streams.tags", "streams.properties",
252              "processes.start_time", "processes.start_ticks", "processes.tsc_frequency", "processes.exe",
253              "processes.username", "processes.realname", "processes.computer", "processes.distro", "processes.cpu_brand",
254              "processes.parent_process_id", "processes.properties"
255          FROM source
256          WHERE array_has( "streams.tags", '{source_stream_tag}' )
257          AND insert_time >= '{begin_rfc}'
258          AND insert_time < '{end_rfc}'
259          ;"#
260    );
261    let block_partitions = existing_partitions
262        .filter("blocks", "global", &blocks_file_schema_hash(), insert_range)
263        .partitions;
264    let df = query_partitions(
265        runtime,
266        lake.clone(),
267        Arc::new(blocks_view_schema()),
268        Arc::new(block_partitions),
269        &sql,
270    )
271    .await
272    .with_context(|| "blocks query")?;
273    let blocks_stats_df = df.clone().aggregate(
274        vec![],
275        vec![
276            sum(col("nb_objects")),
277            count_all(),
278            max(col("payload_size")),
279        ],
280    )?;
281    let blocks_stats_rbs = blocks_stats_df.collect().await?;
282    if blocks_stats_rbs.len() != 1 {
283        anyhow::bail!("nb_objects_rbs has size {}", blocks_stats_rbs.len());
284    }
285    if blocks_stats_rbs[0].num_rows() != 1 {
286        anyhow::bail!(
287            "nb_objects_rbs[0] has size {}",
288            blocks_stats_rbs[0].num_rows()
289        );
290    }
291    let sub_nb_objects_column: &Int64Array = typed_column(&blocks_stats_rbs[0], 0)?;
292    let object_count = sub_nb_objects_column.value(0);
293    let block_count_column: &Int64Array = typed_column(&blocks_stats_rbs[0], 1)?;
294    let block_count = block_count_column.value(0);
295    let max_payload_size_column: &Int64Array = typed_column(&blocks_stats_rbs[0], 2)?;
296    let max_payload = max_payload_size_column.value(0);
297
298    Ok(SourceDataBlocks {
299        blocks_dataframe: df,
300        object_count,
301        block_count,
302        max_payload,
303    })
