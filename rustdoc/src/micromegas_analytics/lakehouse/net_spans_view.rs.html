70/// A view of network bandwidth spans (Connection / Object / Property / RPC).
71#[derive(Debug)]
72pub struct NetSpansView {
73    view_set_name: Arc<String>,
74    view_instance_id: Arc<String>,
75    process_id: sqlx::types::Uuid,
76    view_factory: Arc<ViewFactory>,
77}
78
79impl NetSpansView {
80    pub fn new(view_instance_id: &str, view_factory: Arc<ViewFactory>) -> Result<Self> {
81        if view_instance_id == "global" {
82            anyhow::bail!("NetSpansView does not support global view access");
83        }
84        let process_id = Uuid::parse_str(view_instance_id).with_context(|| "Uuid::parse_str")?;
85        Ok(Self {
86            view_set_name: Arc::new(String::from(VIEW_SET_NAME)),
87            view_instance_id: Arc::new(view_instance_id.into()),
88            process_id,
89            view_factory,
90        })
91    }
92}

115/// Writes a partition from a set of blocks.
116#[span_fn]
117async fn write_partition(
118    lake: Arc<DataLakeConnection>,
119    view_meta: ViewMetadata,
120    schema: Arc<Schema>,
121    convert_ticks: &ConvertTicks,
122    spec: &SourceDataBlocksInMemory,
123    process_id: Arc<String>,
124) -> Result<()> {
125    let nb_events = hash_to_object_count(&spec.block_ids_hash)? as usize;
126    info!("nb_events: {nb_events}");
127    let mut record_builder = NetSpanRecordBuilder::with_capacity(nb_events / 2);
128    if spec.blocks.is_empty() {
129        anyhow::bail!("empty partition spec");
130    }
131    let min_insert_time = spec.blocks[0].block.insert_time;
132    let max_insert_time = spec.blocks[spec.blocks.len() - 1].block.insert_time;
133
134    let (tx, rx) = tokio::sync::mpsc::channel(1);
135    let null_response_writer = Arc::new(ResponseWriter::new(None));
136    let join_handle = spawn_with_context(write_partition_from_rows(
137        lake.clone(),
138        view_meta,
139        schema,
140        TimeRange::new(min_insert_time, max_insert_time),
141        spec.block_ids_hash.clone(),
142        rx,
143        null_response_writer,
144    ));
145
146    // A process has exactly one net stream (the Unreal NetTraceWriter creates a
147    // single NetStream tagged "net"), so all blocks here share the same stream.
148    // Validate the invariant so a future regression surfaces early instead of
149    // silently tagging rows with the wrong stream_id.
150    let stream = spec.blocks[0].stream.clone();
151    for b in &spec.blocks {
152        anyhow::ensure!(
153            b.stream.stream_id == stream.stream_id,
154            "net_spans partition contains multiple streams ({} and {}); expected one per process",
155            stream.stream_id,
156            b.stream.stream_id,
157        );
158    }
159
160    // Split on time gaps: blocks are contiguous in ticks by construction (the
161    // flush point's `Now` is shared between the closing and new block), so a
162    // gap means a block was lost — don't stitch the span stack across it.
163    let mut blocks_to_process: Vec<BlockMetadata> = vec![];
164    let mut last_end: Option<i64> = None;
165    for block in &spec.blocks {
166        let contiguous = last_end
167            .map(|e| block.block.begin_ticks == e)
168            .unwrap_or(true);
169        if !contiguous {
170            append_net_span_tree(
171                &mut record_builder,
172                convert_ticks,
173                &blocks_to_process,
174                lake.blob_storage.clone(),
175                &stream,
176                process_id.clone(),
177            )
178            .await?;
179            blocks_to_process = vec![];
180        }
181        blocks_to_process.push(block.block.clone());
182        last_end = Some(block.block.end_ticks);
183    }
184    if !blocks_to_process.is_empty() {
185        append_net_span_tree(
186            &mut record_builder,
187            convert_ticks,
188            &blocks_to_process,
189            lake.blob_storage.clone(),
190            &stream,
191            process_id.clone(),
192        )
193        .await?;
194    }
195
196    let min_time_row = convert_ticks.delta_ticks_to_time(spec.blocks[0].block.begin_ticks);
197    let max_time_row =
198        convert_ticks.delta_ticks_to_time(spec.blocks[spec.blocks.len() - 1].block.end_ticks);
199    let rows_time_range = record_builder
200        .get_time_range()
201        .unwrap_or(TimeRange::new(min_time_row, max_time_row));
202    let nb_rows = record_builder.len();
203    let rows = record_builder
204        .finish()
205        .with_context(|| "record_builder.finish()")?;
206    info!("writing {} rows", nb_rows);
207    if nb_rows > 0 {
208        tx.send(PartitionRowSet {
209            rows_time_range,
210            rows,
211        })
212        .await?;
213    }
214    drop(tx);
215    join_handle.await??;
216    Ok(())
217}

219/// Rebuilds the partition if it's missing or out of date.
220#[span_fn]
221async fn update_partition(
222    lake: Arc<DataLakeConnection>,
223    view_meta: ViewMetadata,
224    schema: Arc<Schema>,
225    convert_ticks: &ConvertTicks,
226    spec: &SourceDataBlocksInMemory,
227    process_id: Arc<String>,
228) -> Result<()> {
229    if is_jit_partition_up_to_date(&lake.db_pool, view_meta.clone(), spec).await? {
230        return Ok(());
231    }
232    write_partition(lake, view_meta, schema, convert_ticks, spec, process_id)
233        .await
234        .with_context(|| "write_partition")?;
235    Ok(())
236}

265    #[span_fn]
266    async fn jit_update(
267        &self,
268        lakehouse: Arc<LakehouseContext>,
269        query_range: Option<TimeRange>,
270    ) -> Result<()> {
271        let (process, last_block_end_ticks, last_block_end_time) = find_process_with_latest_timing(
272            lakehouse.clone(),
273            self.view_factory.clone(),
274            &self.process_id,
275            query_range,
276        )
277        .await
278        .with_context(|| "find_process_with_latest_timing")?;
279
280        let process = Arc::new(process);
281        let query_range =
282            query_range.unwrap_or_else(|| TimeRange::new(process.start_time, last_block_end_time));
283
284        let convert_ticks = make_time_converter_from_latest_timing(
285            &process,
286            last_block_end_ticks,
287            last_block_end_time,
288        )
289        .with_context(|| "make_time_converter_from_latest_timing")?;
290
291        let blocks_view = BlocksView::new()?;
292        let all_partitions = generate_process_jit_partitions(
293            &JitPartitionConfig::default(),
294            lakehouse.clone(),
295            &blocks_view,
296            &query_range,
297            process.clone(),
298            NET_STREAM_TAG,
299        )
300        .await
301        .with_context(|| "generate_process_jit_partitions")?;
302
303        let process_id_str = Arc::new(self.process_id.to_string());
304        for part in &all_partitions {
305            update_partition(
306                lakehouse.lake().clone(),
307                ViewMetadata {
308                    view_set_name: self.get_view_set_name(),
309                    view_instance_id: self.get_view_instance_id(),
310                    file_schema_hash: self.get_file_schema_hash(),
311                },
312                self.get_file_schema(),
313                &convert_ticks,
314                part,
315                process_id_str.clone(),
316            )
317            .await
318            .with_context(|| "update_partition")?;
319        }
320        Ok(())
321    }
