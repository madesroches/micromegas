44/// A DataFusion `TableProvider` for listing view sets with their current schema information.
45#[derive(Debug)]
46pub struct ListViewSetsTableProvider {
47    pub view_factory: Arc<ViewFactory>,
48}
49
50#[async_trait]
51impl TableProvider for ListViewSetsTableProvider {
52    fn as_any(&self) -> &dyn Any {
53        self
54    }
55
56    fn schema(&self) -> SchemaRef {
57        Arc::new(Schema::new(vec![
58            Field::new("view_set_name", DataType::Utf8, false),
59            Field::new("current_schema_hash", DataType::Binary, false),
60            Field::new("schema", DataType::Utf8, false),
61            Field::new("has_view_maker", DataType::Boolean, false),
62            Field::new("global_instance_available", DataType::Boolean, false),
63        ]))
64    }
65
66    fn table_type(&self) -> TableType {
67        TableType::Temporary
68    }
69
70    async fn scan(
71        &self,
72        _state: &dyn Session,
73        projection: Option<&Vec<usize>>,
74        _filters: &[Expr],
75        _limit: Option<usize>,
76    ) -> datafusion::error::Result<Arc<dyn ExecutionPlan>> {
77        // Get current schema versions from the view factory
78        let schema_infos =
79            list_view_sets(&self.view_factory).map_err(|e| DataFusionError::External(e.into()))?;
80
81        // Convert to Arrow arrays
82        let view_set_names: Vec<String> = schema_infos
83            .iter()
84            .map(|info| info.view_set_name.clone())
85            .collect();
86        let schema_hashes: Vec<&[u8]> = schema_infos
87            .iter()
88            .map(|info| info.current_schema_hash.as_slice())
89            .collect();
90        let schemas: Vec<String> = schema_infos
91            .iter()
92            .map(|info| info.schema.clone())
93            .collect();
94        let has_view_makers: Vec<bool> = schema_infos
95            .iter()
96            .map(|info| info.has_view_maker)
97            .collect();
98        let global_instances: Vec<bool> = schema_infos
99            .iter()
100            .map(|info| info.global_instance_available)
101            .collect();
102
103        let view_set_name_array: ArrayRef = Arc::new(StringArray::from(view_set_names));
104        let schema_hash_array: ArrayRef = Arc::new(BinaryArray::from(schema_hashes));
105        let schema_array: ArrayRef = Arc::new(StringArray::from(schemas));
106        let has_view_maker_array: ArrayRef = Arc::new(BooleanArray::from(has_view_makers));
107        let global_instance_array: ArrayRef = Arc::new(BooleanArray::from(global_instances));
108
109        let columns = vec![
110            view_set_name_array,
111            schema_hash_array,
112            schema_array,
113            has_view_maker_array,
114            global_instance_array,
115        ];
116
117        let record_batch = RecordBatch::try_new(self.schema(), columns)
118            .map_err(|e| DataFusionError::External(e.into()))?;
119
120        Ok(MemorySourceConfig::try_new_exec(
121            &[vec![record_batch]],
122            self.schema(),
123            projection.map(|v| v.to_owned()),
124        )?)
125    }
126}
