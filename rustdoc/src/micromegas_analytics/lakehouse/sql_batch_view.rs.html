37/// SQL-defined view updated in batch
38#[derive(Debug)]
39pub struct SqlBatchView {
40    view_set_name: Arc<String>,
41    view_instance_id: Arc<String>,
42    min_event_time_column: Arc<String>,
43    max_event_time_column: Arc<String>,
44    count_src_query: Arc<String>,
45    extract_query: Arc<String>,
46    merge_partitions_query: Arc<String>,
47    schema: Arc<Schema>,
48    merger: Arc<dyn PartitionMerger>,
49    view_factory: Arc<ViewFactory>,
50    session_configurator: Arc<dyn SessionConfigurator>,
51    update_group: Option<i32>,
52    max_partition_delta_from_source: TimeDelta,
53    max_partition_delta_from_merge: TimeDelta,
54}
55
56impl SqlBatchView {
57    #[expect(clippy::too_many_arguments)]
58    /// # Arguments
59    ///
60    /// * `runtime` - datafusion runtime
61    /// * `view_set_name` - name of the table
62    /// * `min_event_time_column` - min(column) should result in the first timestamp in a dataframe
63    /// * `max_event_time_column` - max(column) should result in the last timestamp in a dataframe
64    /// * `count_src_query` - used to count the rows of the underlying data to know if a cached partition is up to date
65    /// * `extract_query` - used to extract the source data into a cached partition
66    /// * `merge_partitions_query` - used to merge multiple partitions into a single one (and user queries which are one multiple partitions by default)
67    /// * `lake` - data lake
68    /// * `view_factory` - all views accessible to the `count_src_query`
69    /// * `session_configurator` - configurator for registering custom tables (e.g., JSON files)
70    /// * `update_group` - tells the daemon which view should be materialized and in what order
71    pub async fn new(
72        runtime: Arc<RuntimeEnv>,
73        view_set_name: Arc<String>,
74        min_event_time_column: Arc<String>,
75        max_event_time_column: Arc<String>,
76        count_src_query: Arc<String>,
77        extract_query: Arc<String>,
78        merge_partitions_query: Arc<String>,
79        lake: Arc<DataLakeConnection>,
80        view_factory: Arc<ViewFactory>,
81        session_configurator: Arc<dyn SessionConfigurator>,
82        update_group: Option<i32>,
83        max_partition_delta_from_source: TimeDelta,
84        max_partition_delta_from_merge: TimeDelta,
85        merger_maker: Option<&MergerMaker>,
86    ) -> Result<Self> {
87        let null_part_provider = Arc::new(NullPartitionProvider {});
88        let lakehouse = Arc::new(LakehouseContext::new(lake.clone(), runtime.clone()));
89        let ctx = make_session_context(
90            lakehouse,
91            null_part_provider,
92            None,
93            view_factory.clone(),
94            session_configurator.clone(),
95        )
96        .await
97        .with_context(|| "make_session_context")?;
98        let now_str = Utc::now().to_rfc3339();
99        let sql = extract_query
100            .replace("{begin}", &now_str)
101            .replace("{end}", &now_str);
102        let extracted_df = ctx.sql(&sql).await?;
103        let schema = extracted_df.schema().inner().clone();
104        let session_configurator_for_merger = session_configurator.clone();
105        let merger = merger_maker.unwrap_or(&|_runtime, schema| {
106            let merge_query = Arc::new(merge_partitions_query.replace("{source}", "source"));
107            Arc::new(QueryMerger::new(
108                view_factory.clone(),
109                session_configurator_for_merger.clone(),
110                schema,
111                merge_query,
112            ))
113        })(runtime.clone(), schema.clone());
114
115        Ok(Self {
116            view_set_name,
117            view_instance_id: Arc::new(String::from("global")),
118            min_event_time_column,
119            max_event_time_column,
120            count_src_query,
121            extract_query,
122            merge_partitions_query,
123            schema,
124            merger,
125            view_factory,
126            session_configurator,
127            update_group,
128            max_partition_delta_from_source,
129            max_partition_delta_from_merge,
130        })
131    }
132}

134#[async_trait]
135impl View for SqlBatchView {
136    fn get_view_set_name(&self) -> Arc<String> {
137        self.view_set_name.clone()
138    }
139
140    fn get_view_instance_id(&self) -> Arc<String> {
141        self.view_instance_id.clone()
142    }
143
144    async fn make_batch_partition_spec(
145        &self,
146        lakehouse: Arc<LakehouseContext>,
147        existing_partitions: Arc<PartitionCache>,
148        insert_range: TimeRange,
149    ) -> Result<Arc<dyn PartitionSpec>> {
150        let view_meta = ViewMetadata {
151            view_set_name: self.get_view_set_name(),
152            view_instance_id: self.get_view_instance_id(),
153            file_schema_hash: self.get_file_schema_hash(),
154        };
155        let partitions_in_range = Arc::new(existing_partitions.filter_insert_range(insert_range));
156        let ctx = make_session_context(
157            lakehouse,
158            partitions_in_range.clone(),
159            None,
160            self.view_factory.clone(),
161            self.session_configurator.clone(),
162        )
163        .await
164        .with_context(|| "make_session_context")?;
165
166        let count_src_sql = self
167            .count_src_query
168            .replace("{begin}", &insert_range.begin.to_rfc3339())
169            .replace("{end}", &insert_range.end.to_rfc3339());
170
171        let extract_sql = self
172            .extract_query
173            .replace("{begin}", &insert_range.begin.to_rfc3339())
174            .replace("{end}", &insert_range.end.to_rfc3339());
175
176        Ok(Arc::new(
177            fetch_sql_partition_spec(
178                ctx,
179                Arc::new(TrivialRecordBatchTransformer {}),
180                self.get_time_bounds(),
181                self.schema.clone(),
182                count_src_sql,
183                extract_sql,
184                view_meta,
185                insert_range,
186            )
187            .await
188            .with_context(|| "fetch_sql_partition_spec")?,
189        ))
190    }
191
192    fn get_file_schema_hash(&self) -> Vec<u8> {
193        let mut hasher = DefaultHasher::new();
194        self.schema.hash(&mut hasher);
195        hasher.finish().to_le_bytes().to_vec()
196    }
197
198    fn get_file_schema(&self) -> Arc<Schema> {
199        self.schema.clone()
200    }
201
202    async fn jit_update(
203        &self,
204        _lakehouse: Arc<LakehouseContext>,
205        _query_range: Option<TimeRange>,
206    ) -> Result<()> {
207        Ok(())
208    }
209
210    fn make_time_filter(&self, begin: DateTime<Utc>, end: DateTime<Utc>) -> Result<Vec<Expr>> {
211        Ok(vec![
212            col(&*self.min_event_time_column).lt_eq(lit(datetime_to_scalar(end))),
213            col(&*self.max_event_time_column).gt_eq(lit(datetime_to_scalar(begin))),
214        ])
215    }
216
217    fn get_time_bounds(&self) -> Arc<dyn DataFrameTimeBounds> {
218        Arc::new(NamedColumnsTimeBounds::new(
219            self.min_event_time_column.clone(),
220            self.max_event_time_column.clone(),
221        ))
222    }
223
224    async fn register_table(&self, ctx: &SessionContext, table: MaterializedView) -> Result<()> {
225        let view_name = self.get_view_set_name().to_string();
226        let partitions_table_name = format!("__{view_name}__partitions");
227        ctx.register_table(
228            TableReference::Bare {
229                table: partitions_table_name.clone().into(),
230            },
231            Arc::new(table),
232        )?;
233        let df = ctx
234            .sql(
235                &self
236                    .merge_partitions_query
237                    .replace("{source}", &partitions_table_name),
238            )
239            .await?;
240        ctx.register_table(
241            TableReference::Bare {
242                table: view_name.into(),
243            },
244            df.into_view(),
245        )?;
246        Ok(())
247    }
248
249    async fn merge_partitions(
250        &self,
251        lakehouse: Arc<LakehouseContext>,
252        partitions_to_merge: Arc<Vec<Partition>>,
253        partitions_all_views: Arc<PartitionCache>,
254    ) -> Result<SendableRecordBatchStream> {
255        let res = self
256            .merger
257            .execute_merge_query(lakehouse, partitions_to_merge, partitions_all_views)
258            .await;
259        if let Err(e) = &res {
260            error!("{e:?}");
261        }
262        res
263    }
264
265    fn get_update_group(&self) -> Option<i32> {
266        self.update_group
267    }
268
269    fn get_max_partition_time_delta(&self, strategy: &PartitionCreationStrategy) -> TimeDelta {
270        match strategy {
271            PartitionCreationStrategy::Abort | PartitionCreationStrategy::CreateFromSource => {
272                self.max_partition_delta_from_source
273            }
274            PartitionCreationStrategy::MergeExisting(_partitions) => {
275                self.max_partition_delta_from_merge
276            }
277        }
278    }
