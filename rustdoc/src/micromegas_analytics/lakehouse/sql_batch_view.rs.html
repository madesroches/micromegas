36/// SQL-defined view updated in batch
37#[derive(Debug)]
38pub struct SqlBatchView {
39    runtime: Arc<RuntimeEnv>,
40    view_set_name: Arc<String>,
41    view_instance_id: Arc<String>,
42    min_event_time_column: Arc<String>,
43    max_event_time_column: Arc<String>,
44    count_src_query: Arc<String>,
45    extract_query: Arc<String>,
46    merge_partitions_query: Arc<String>,
47    schema: Arc<Schema>,
48    merger: Arc<dyn PartitionMerger>,
49    view_factory: Arc<ViewFactory>,
50    session_configurator: Arc<dyn SessionConfigurator>,
51    update_group: Option<i32>,
52    max_partition_delta_from_source: TimeDelta,
53    max_partition_delta_from_merge: TimeDelta,
54}
55
56impl SqlBatchView {
57    #[expect(clippy::too_many_arguments)]
58    /// # Arguments
59    ///
60    /// * `runtime` - datafusion runtime
61    /// * `view_set_name` - name of the table
62    /// * `min_event_time_column` - min(column) should result in the first timestamp in a dataframe
63    /// * `max_event_time_column` - max(column) should result in the last timestamp in a dataframe
64    /// * `count_src_query` - used to count the rows of the underlying data to know if a cached partition is up to date
65    /// * `extract_query` - used to extract the source data into a cached partition
66    /// * `merge_partitions_query` - used to merge multiple partitions into a single one (and user queries which are one multiple partitions by default)
67    /// * `lake` - data lake
68    /// * `view_factory` - all views accessible to the `count_src_query`
69    /// * `session_configurator` - configurator for registering custom tables (e.g., JSON files)
70    /// * `update_group` - tells the daemon which view should be materialized and in what order
71    pub async fn new(
72        runtime: Arc<RuntimeEnv>,
73        view_set_name: Arc<String>,
74        min_event_time_column: Arc<String>,
75        max_event_time_column: Arc<String>,
76        count_src_query: Arc<String>,
77        extract_query: Arc<String>,
78        merge_partitions_query: Arc<String>,
79        lake: Arc<DataLakeConnection>,
80        view_factory: Arc<ViewFactory>,
81        session_configurator: Arc<dyn SessionConfigurator>,
82        update_group: Option<i32>,
83        max_partition_delta_from_source: TimeDelta,
84        max_partition_delta_from_merge: TimeDelta,
85        merger_maker: Option<&MergerMaker>,
86    ) -> Result<Self> {
87        let null_part_provider = Arc::new(NullPartitionProvider {});
88        let ctx = make_session_context(
89            runtime.clone(),
90            lake,
91            null_part_provider,
92            None,
93            view_factory.clone(),
94            session_configurator.clone(),
95        )
96        .await
97        .with_context(|| "make_session_context")?;
98        let now_str = Utc::now().to_rfc3339();
99        let sql = extract_query
100            .replace("{begin}", &now_str)
101            .replace("{end}", &now_str);
102        let extracted_df = ctx.sql(&sql).await?;
103        let schema = extracted_df.schema().inner().clone();
104        let session_configurator_for_merger = session_configurator.clone();
105        let merger = merger_maker.unwrap_or(&|runtime, schema| {
106            let merge_query = Arc::new(merge_partitions_query.replace("{source}", "source"));
107            Arc::new(QueryMerger::new(
108                runtime,
109                view_factory.clone(),
110                session_configurator_for_merger.clone(),
111                schema,
112                merge_query,
113            ))
114        })(runtime.clone(), schema.clone());
115
116        Ok(Self {
117            runtime,
118            view_set_name,
119            view_instance_id: Arc::new(String::from("global")),
120            min_event_time_column,
121            max_event_time_column,
122            count_src_query,
123            extract_query,
124            merge_partitions_query,
125            schema,
126            merger,
127            view_factory,
128            session_configurator,
129            update_group,
130            max_partition_delta_from_source,
131            max_partition_delta_from_merge,
132        })
133    }
134}

136#[async_trait]
137impl View for SqlBatchView {
138    fn get_view_set_name(&self) -> Arc<String> {
139        self.view_set_name.clone()
140    }
141
142    fn get_view_instance_id(&self) -> Arc<String> {
143        self.view_instance_id.clone()
144    }
145
146    async fn make_batch_partition_spec(
147        &self,
148        _runtime: Arc<RuntimeEnv>,
149        lake: Arc<DataLakeConnection>,
150        existing_partitions: Arc<PartitionCache>,
151        insert_range: TimeRange,
152    ) -> Result<Arc<dyn PartitionSpec>> {
153        let view_meta = ViewMetadata {
154            view_set_name: self.get_view_set_name(),
155            view_instance_id: self.get_view_instance_id(),
156            file_schema_hash: self.get_file_schema_hash(),
157        };
158        let partitions_in_range = Arc::new(existing_partitions.filter_insert_range(insert_range));
159        let ctx = make_session_context(
160            self.runtime.clone(),
161            lake.clone(),
162            partitions_in_range.clone(),
163            None,
164            self.view_factory.clone(),
165            self.session_configurator.clone(),
166        )
167        .await
168        .with_context(|| "make_session_context")?;
169
170        let count_src_sql = self
171            .count_src_query
172            .replace("{begin}", &insert_range.begin.to_rfc3339())
173            .replace("{end}", &insert_range.end.to_rfc3339());
174
175        let extract_sql = self
176            .extract_query
177            .replace("{begin}", &insert_range.begin.to_rfc3339())
178            .replace("{end}", &insert_range.end.to_rfc3339());
179
180        Ok(Arc::new(
181            fetch_sql_partition_spec(
182                ctx,
183                Arc::new(TrivialRecordBatchTransformer {}),
184                self.get_time_bounds(),
185                self.schema.clone(),
186                count_src_sql,
187                extract_sql,
188                view_meta,
189                insert_range,
190            )
191            .await
192            .with_context(|| "fetch_sql_partition_spec")?,
193        ))
194    }
195
196    fn get_file_schema_hash(&self) -> Vec<u8> {
197        let mut hasher = DefaultHasher::new();
198        self.schema.hash(&mut hasher);
199        hasher.finish().to_le_bytes().to_vec()
200    }
201
202    fn get_file_schema(&self) -> Arc<Schema> {
203        self.schema.clone()
204    }
205
206    async fn jit_update(
207        &self,
208        _runtime: Arc<RuntimeEnv>,
209        _lake: Arc<DataLakeConnection>,
210        _query_range: Option<TimeRange>,
211    ) -> Result<()> {
212        Ok(())
213    }
214
215    fn make_time_filter(&self, begin: DateTime<Utc>, end: DateTime<Utc>) -> Result<Vec<Expr>> {
216        Ok(vec![
217            col(&*self.min_event_time_column).lt_eq(lit(datetime_to_scalar(end))),
218            col(&*self.max_event_time_column).gt_eq(lit(datetime_to_scalar(begin))),
219        ])
220    }
221
222    fn get_time_bounds(&self) -> Arc<dyn DataFrameTimeBounds> {
223        Arc::new(NamedColumnsTimeBounds::new(
224            self.min_event_time_column.clone(),
225            self.max_event_time_column.clone(),
226        ))
227    }
228
229    async fn register_table(&self, ctx: &SessionContext, table: MaterializedView) -> Result<()> {
230        let view_name = self.get_view_set_name().to_string();
231        let partitions_table_name = format!("__{view_name}__partitions");
232        ctx.register_table(
233            TableReference::Bare {
234                table: partitions_table_name.clone().into(),
235            },
236            Arc::new(table),
237        )?;
238        let df = ctx
239            .sql(
240                &self
241                    .merge_partitions_query
242                    .replace("{source}", &partitions_table_name),
243            )
244            .await?;
245        ctx.register_table(
246            TableReference::Bare {
247                table: view_name.into(),
248            },
249            df.into_view(),
250        )?;
251        Ok(())
252    }
253
254    async fn merge_partitions(
255        &self,
256        _runtime: Arc<RuntimeEnv>,
257        lake: Arc<DataLakeConnection>,
258        partitions_to_merge: Arc<Vec<Partition>>,
259        partitions_all_views: Arc<PartitionCache>,
260    ) -> Result<SendableRecordBatchStream> {
261        let res = self
262            .merger
263            .execute_merge_query(lake, partitions_to_merge, partitions_all_views)
264            .await;
265        if let Err(e) = &res {
266            error!("{e:?}");
267        }
268        res
269    }
270
271    fn get_update_group(&self) -> Option<i32> {
272        self.update_group
273    }
274
275    fn get_max_partition_time_delta(&self, strategy: &PartitionCreationStrategy) -> TimeDelta {
276        match strategy {
277            PartitionCreationStrategy::Abort | PartitionCreationStrategy::CreateFromSource => {
278                self.max_partition_delta_from_source
279            }
280            PartitionCreationStrategy::MergeExisting(_partitions) => {
281                self.max_partition_delta_from_merge
282            }
283        }
284    }
