26/// A trait for defining a partition specification.
27#[async_trait]
28pub trait PartitionSpec: Send + Sync + Debug {
29    /// Returns true if the partition is empty.
30    fn is_empty(&self) -> bool;
31    /// Returns a hash of the source data.
32    fn get_source_data_hash(&self) -> Vec<u8>;
33    /// Writes the partition to the data lake.
34    async fn write(&self, lake: Arc<DataLakeConnection>, logger: Arc<dyn Logger>) -> Result<()>;
35}

45/// A trait for defining a view.
46#[async_trait]
47pub trait View: std::fmt::Debug + Send + Sync {
48    /// name of the table from the user's perspective
49    fn get_view_set_name(&self) -> Arc<String>;
50
51    /// get_view_instance_id can be a process_id, a stream_id or 'global'.
52    fn get_view_instance_id(&self) -> Arc<String>;
53
54    /// make_batch_partition_spec determines what should be found in an up to date partition.
55    /// The resulting PartitionSpec can be used to validate existing partitions are create a new one.
56    async fn make_batch_partition_spec(
57        &self,
58        runtime: Arc<RuntimeEnv>,
59        lake: Arc<DataLakeConnection>,
60        existing_partitions: Arc<PartitionCache>,
61        insert_range: TimeRange,
62    ) -> Result<Arc<dyn PartitionSpec>>;
63
64    /// get_file_schema_hash returns a hash (can be a version number, version string, etc.) that allows
65    /// to identify out of date partitions.
66    fn get_file_schema_hash(&self) -> Vec<u8>;
67
68    /// get_file_schema returns the schema of the partition file in object storage
69    fn get_file_schema(&self) -> Arc<Schema>;
70
71    /// jit_update creates or updates process-specific partitions before a query
72    async fn jit_update(
73        &self,
74        runtime: Arc<RuntimeEnv>,
75        lake: Arc<DataLakeConnection>,
76        query_range: Option<TimeRange>,
77    ) -> Result<()>;
78
79    /// make_time_filter returns a set of expressions that will filter out the rows of the partition
80    /// outside the time range requested.
81    fn make_time_filter(&self, _begin: DateTime<Utc>, _end: DateTime<Utc>) -> Result<Vec<Expr>>;
82
83    // a view must provide a way to compute the time bounds of a DataFrame corresponding to its schema
84    fn get_time_bounds(&self) -> Arc<dyn DataFrameTimeBounds>;
85
86    /// register the table in the SessionContext
87    async fn register_table(&self, ctx: &SessionContext, table: MaterializedView) -> Result<()> {
88        let view_set_name = self.get_view_set_name().to_string();
89        ctx.register_table(
90            TableReference::Bare {
91                table: view_set_name.into(),
92            },
93            Arc::new(table),
94        )?;
95        Ok(())
96    }
97
98    async fn merge_partitions(
99        &self,
100        runtime: Arc<RuntimeEnv>,
101        lake: Arc<DataLakeConnection>,
102        partitions_to_merge: Arc<Vec<Partition>>,
103        partitions_all_views: Arc<PartitionCache>,
104    ) -> Result<SendableRecordBatchStream> {
105        let merge_query = Arc::new(String::from("SELECT * FROM source;"));
106        let empty_view_factory = Arc::new(ViewFactory::new(vec![]));
107        let merger = QueryMerger::new(
108            runtime,
109            empty_view_factory,
110            Arc::new(NoOpSessionConfigurator),
111            self.get_file_schema(),
112            merge_query,
113        );
114        merger
115            .execute_merge_query(lake, partitions_to_merge, partitions_all_views)
116            .await
117    }
118
119    /// tells the daemon which view should be materialized and in what order
120    fn get_update_group(&self) -> Option<i32>;
121
122    /// allow the view to subdivide the requested partition
123    fn get_max_partition_time_delta(&self, _strategy: &PartitionCreationStrategy) -> TimeDelta {
124        TimeDelta::days(1)
125    }
126}

135    }
