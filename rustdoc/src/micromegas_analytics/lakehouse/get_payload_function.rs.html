18/// A scalar UDF that retrieves the payload of a block from the data lake.
19#[derive(Debug)]
20pub struct GetPayload {
21    signature: Signature,
22    lake: Arc<DataLakeConnection>,
23}
24
25impl PartialEq for GetPayload {
26    fn eq(&self, other: &Self) -> bool {
27        self.signature == other.signature
28    }
29}
30
31impl Eq for GetPayload {}
32
33impl std::hash::Hash for GetPayload {
34    fn hash<H: std::hash::Hasher>(&self, state: &mut H) {
35        self.signature.hash(state);
36    }
37}
38
39impl GetPayload {
40    pub fn new(lake: Arc<DataLakeConnection>) -> Self {
41        Self {
42            signature: Signature::exact(
43                vec![DataType::Utf8, DataType::Utf8, DataType::Utf8],
44                Volatility::Immutable,
45            ),
46            lake,
47        }
48    }
49}

76#[async_trait]
77impl AsyncScalarUDFImpl for GetPayload {
78    async fn invoke_async_with_args(
79        &self,
80        args: ScalarFunctionArgs,
81    ) -> datafusion::error::Result<ColumnarValue> {
82        let args = ColumnarValue::values_to_arrays(&args.args)?;
83        if args.len() != 3 {
84            return internal_err!("wrong number of arguments to get_payload()");
85        }
86        let process_ids = args[0]
87            .as_any()
88            .downcast_ref::<StringArray>()
89            .ok_or_else(|| {
90                DataFusionError::Execution("downcasting process_ids in GetPayload".into())
91            })?
92            .clone();
93        let stream_ids = args[1]
94            .as_any()
95            .downcast_ref::<StringArray>()
96            .ok_or_else(|| {
97                DataFusionError::Execution("downcasting stream_ids in GetPayload".into())
98            })?
99            .clone();
100        let block_ids = args[2]
101            .as_any()
102            .downcast_ref::<StringArray>()
103            .ok_or_else(|| {
104                DataFusionError::Execution("downcasting block_ids in GetPayload".into())
105            })?
106            .clone();
107        let lake = self.lake.clone();
108        let mut stream = futures::stream::iter(0..process_ids.len())
109            .map(|i| {
110                let process_id = process_ids.value(i);
111                let stream_id = stream_ids.value(i);
112                let block_id = block_ids.value(i);
113                let obj_path = format!("blobs/{process_id}/{stream_id}/{block_id}");
114                let lake = lake.clone();
115                tokio::spawn(async move { lake.blob_storage.read_blob(&obj_path).await })
116            })
117            .buffered(10);
118        let mut result_builder = BinaryBuilder::with_capacity(block_ids.len(), 1024 * 1024);
119        while let Some(res) = stream.next().await {
120            result_builder.append_value(
121                res.map_err(|e| {
122                    DataFusionError::Execution(format!("error downloading payload: {e:?}"))
123                })?
124                .map_err(|e| {
125                    DataFusionError::Execution(format!("error downloading payload: {e:?}"))
126                })?,
127            );
128        }
129        Ok(ColumnarValue::Array(Arc::new(result_builder.finish())))
130    }
131}
