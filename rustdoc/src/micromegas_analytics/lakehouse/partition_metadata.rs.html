10/// Load partition metadata by file path from the dedicated metadata table
11#[span_fn]
12pub async fn load_partition_metadata(
13    pool: &PgPool,
14    file_path: &str,
15) -> Result<Arc<ParquetMetaData>> {
16    let row = sqlx::query("SELECT metadata FROM partition_metadata WHERE file_path = $1")
17        .bind(file_path)
18        .fetch_one(pool)
19        .await
20        .with_context(|| format!("loading metadata for file: {}", file_path))?;
21
22    let metadata_bytes: Vec<u8> = row.try_get("metadata")?;
23    let metadata = parse_parquet_metadata(&Bytes::from(metadata_bytes))
24        .with_context(|| format!("parsing metadata for file: {}", file_path))?;
25    Ok(Arc::new(metadata))
26}

28/// Delete multiple partition metadata entries in a single transaction
29/// Uses PostgreSQL's ANY() with array to avoid placeholder limits
30#[span_fn]
31pub async fn delete_partition_metadata_batch(
32    tr: &mut sqlx::Transaction<'_, sqlx::Postgres>,
33    file_paths: &[String],
34) -> Result<()> {
35    if file_paths.is_empty() {
36        return Ok(());
37    }
38
39    // Use PostgreSQL's ANY() with array - no placeholder limits
40    let result = sqlx::query("DELETE FROM partition_metadata WHERE file_path = ANY($1)")
41        .bind(file_paths)
42        .execute(&mut **tr)
43        .await
44        .with_context(|| format!("deleting {} metadata entries", file_paths.len()))?;
45
46    debug!("deleted {} metadata entries", result.rows_affected());
47    Ok(())
48}
