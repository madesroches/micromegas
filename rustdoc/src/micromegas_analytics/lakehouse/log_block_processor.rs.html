14/// A `BlockProcessor` implementation for processing log blocks.
15#[derive(Debug)]
16pub struct LogBlockProcessor {}
17
18#[async_trait]
19impl BlockProcessor for LogBlockProcessor {
20    async fn process(
21        &self,
22        blob_storage: Arc<BlobStorage>,
23        src_block: Arc<PartitionSourceBlock>,
24    ) -> Result<Option<PartitionRowSet>> {
25        let convert_ticks =
26            make_time_converter_from_block_meta(&src_block.process, &src_block.block)?;
27        let nb_log_entries = src_block.block.nb_objects;
28        let mut record_builder = LogEntriesRecordBuilder::with_capacity(nb_log_entries as usize);
29
30        for_each_log_entry_in_block(
31            blob_storage,
32            &convert_ticks,
33            src_block.process.clone(),
34            &src_block.stream,
35            &src_block.block,
36            |log_entry| {
37                record_builder.append(&log_entry)?;
38                Ok(true) // continue
39            },
40        )
41        .await
42        .with_context(|| "for_each_log_entry_in_block")?;
43
44        if let Some(time_range) = record_builder.get_time_range() {
45            let record_batch = record_builder.finish()?;
46            Ok(Some(PartitionRowSet {
47                min_time_row: time_range.0,
48                max_time_row: time_range.1,
49                rows: record_batch,
50            }))
51        } else {
52            Ok(None)
53        }
54    }
55}
