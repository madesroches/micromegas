23/// A custom [`ParquetFileReaderFactory`] that handles opening parquet files
24/// from object storage, and loads metadata on-demand.
25pub struct ReaderFactory {

27    pool: PgPool,
28}

42async fn load_parquet_metadata(filename: &str, pool: &PgPool) -> Result<Arc<ParquetMetaData>> {
43    // Load metadata on-demand using the dedicated metadata table
44    load_partition_metadata(pool, filename)
45        .await
46        .with_context(|| format!("[reader_factory] loading metadata for {filename}"))
47}

49impl ParquetFileReaderFactory for ReaderFactory {
50    fn create_reader(
51        &self,
52        _partition_index: usize,
53        file_meta: FileMeta,
54        metadata_size_hint: Option<usize>,
55        _metrics: &ExecutionPlanMetricsSet,
56    ) -> datafusion::error::Result<Box<dyn AsyncFileReader + Send>> {
57        // todo: don't ignore metrics, report performance of the reader
58        let filename = file_meta.location().to_string();
59        let object_store = Arc::clone(&self.object_store);
60        let mut inner = ParquetObjectReader::new(object_store, file_meta.location().clone());
61        if let Some(hint) = metadata_size_hint {
62            inner = inner.with_footer_size_hint(hint)
63        };
64
65        Ok(Box::new(ParquetReader {
66            filename,
67            pool: self.pool.clone(),
68            metadata: Arc::new(Mutex::new(None)), // Load on-demand in get_metadata()
69            inner,
70        }))
71    }
72}

74/// A wrapper around a `ParquetObjectReader` that loads metadata on-demand.
75pub struct ParquetReader {
76    pub filename: String,
77    pub pool: PgPool,
78    pub metadata: Arc<Mutex<Option<Arc<ParquetMetaData>>>>, // Thread-safe cached metadata
79    pub inner: ParquetObjectReader,
80}
81
82impl AsyncFileReader for ParquetReader {
83    fn get_bytes(
84        &mut self,
85        range: Range<u64>,
86    ) -> BoxFuture<'_, datafusion::parquet::errors::Result<Bytes>> {
87        // debug!("ParquetReader::get_bytes {}", &self.filename);
88        self.inner.get_bytes(range)
89    }
90
91    fn get_byte_ranges(
92        &mut self,
93        ranges: Vec<Range<u64>>,
94    ) -> BoxFuture<'_, datafusion::parquet::errors::Result<Vec<Bytes>>> {
95        // debug!("ParquetReader::get_byte_ranges {}", &self.filename);
96        self.inner.get_byte_ranges(ranges)
97    }
98
99    fn get_metadata(
100        &mut self,
101        _options: Option<&ArrowReaderOptions>,
102    ) -> BoxFuture<'_, datafusion::parquet::errors::Result<Arc<ParquetMetaData>>> {
103        let metadata_cache = self.metadata.clone();
104        let pool = self.pool.clone();
105        let filename = self.filename.clone();
106
107        Box::pin(async move {
108            // Check if we already have metadata cached
109            {
110                let lock = metadata_cache.lock().await;
111                if let Some(metadata) = &*lock {
112                    debug!("reusing cached metadata");
113                    return Ok(metadata.clone());
114                }
115            }
116
117            // Load metadata from database
118            let metadata = load_parquet_metadata(&filename, &pool)
119                .await
120                .map_err(|e| datafusion::parquet::errors::ParquetError::External(e.into()))?;
121
122            // Cache the metadata for future calls
123            {
124                let mut lock = metadata_cache.lock().await;
125                *lock = Some(metadata.clone());
126            }
127
128            Ok(metadata)
129        })
130    }
