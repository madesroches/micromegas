14/// Creates an empty record batch with an empty schema.
15pub fn make_empty_record_batch() -> RecordBatch {
16    let mut list_builder = ListBuilder::new(StructBuilder::from_fields([], 0));
17    let array = list_builder.finish();
18    as_struct_array(array.values()).into()
19}

21/// Parses Parquet metadata from a byte slice.
22pub fn parse_parquet_metadata(bytes: &Bytes) -> Result<ParquetMetaData> {
23    ParquetMetaDataReader::decode_metadata(bytes).with_context(|| "parsing ParquetMetaData")
24}
25
26/// Serializes Parquet metadata to a byte slice.
27pub fn serialize_parquet_metadata(pmd: &ParquetMetaData) -> Result<bytes::Bytes> {
28    let mut buffer_writer = BytesMut::new().writer();
29    let md_writer = ParquetMetaDataWriter::new(&mut buffer_writer, pmd);
30    md_writer
31        .finish()
32        .with_context(|| "closing ParquetMetaDataWriter")?;
33    Ok(buffer_writer.into_inner().freeze())
34}
