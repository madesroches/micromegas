1/// Records a sync span as two thread events
2///
3/// # Examples
4///
5/// ```
6/// use micromegas_tracing::span_scope;
7///
8/// # fn main() {
9/// span_scope!("scope");
10/// # }
11/// ```
12#[macro_export]
13macro_rules! span_scope {
14    ($scope_name:ident, $name:expr) => {
15        static $scope_name: $crate::spans::SpanMetadata = $crate::spans::SpanMetadata {
16            name: $name,
17            location: $crate::spans::SpanLocation {
18                lod: $crate::levels::Verbosity::Max,
19                target: module_path!(),
20                module_path: module_path!(),
21                file: file!(),
22                line: line!(),
23            },
24        };
25        let guard_named = $crate::guards::ThreadSpanGuard::new(&$scope_name);
26    };
27    ($name:expr) => {
28        $crate::span_scope!(_METADATA_NAMED, $name);
29    };
30}

117macro_rules! imetric {

123                unit: $unit,
124                target: module_path!(),

126                line: line!(),

128        $crate::dispatch::int_metric(&METRIC_METADATA, $value);
129    }};
130    ($name:expr, $unit:expr, $properties:expr, $value:expr) => {{
131        static METRIC_METADATA: $crate::metrics::StaticMetricMetadata =
132            $crate::metrics::StaticMetricMetadata {
133                lod: $crate::levels::Verbosity::Max,

135                unit: $unit,
136                target: module_path!(),
137                file: file!(),
138                line: line!(),
139            };
140        $crate::dispatch::tagged_integer_metric(&METRIC_METADATA, $properties, $value);
141    }};
142}

185/// The standard logging macro.
186///
187/// This macro will generically log with the specified `Level` and `format!`
188/// based argument list.
189///
190/// # Examples
191///
192/// ```
193/// use micromegas_tracing::prelude::*;
194///
195/// # fn main() {
196/// let data = (42, "Forty-two");
197/// let private_data = "private";
198///
199/// log!(Level::Error, "Received errors: {}, {}", data.0, data.1);
200/// log!(target: "app_events", Level::Warn, "App warning: {}, {}, {}",
201///     data.0, data.1, private_data);
202/// # }
203/// ```
204#[macro_export]
205macro_rules! log {
206    (target: $target:expr, $lvl:expr, $($arg:tt)+) => ({
207        static LOG_DESC: $crate::logs::LogMetadata = $crate::logs::LogMetadata {
208            level: $lvl,
209            level_filter: std::sync::atomic::AtomicU32::new($crate::logs::FILTER_LEVEL_UNSET_VALUE),
210            fmt_str: $crate::__first_arg!($($arg)+),
211            target: $target,
212            module_path: $crate::__log_module_path!(),
213            file: file!(),
214            line: line!(),
215        };
216        if $lvl <= $crate::levels::STATIC_MAX_LEVEL && $lvl <= $crate::levels::max_level() {
217            $crate::dispatch::log(&LOG_DESC, format_args!($($arg)+));
218        }
219    });
220    ($lvl:expr, properties: $properties:expr, $($arg:tt)+) => ({
221        static LOG_DESC: $crate::logs::LogMetadata = $crate::logs::LogMetadata {
222            level: $lvl,
223            level_filter: std::sync::atomic::AtomicU32::new($crate::logs::FILTER_LEVEL_UNSET_VALUE),
224            fmt_str: $crate::__first_arg!($($arg)+),
225            target: $crate::__log_module_path!(),
226            module_path: $crate::__log_module_path!(),
227            file: file!(),
228            line: line!(),
229        };
230        if $lvl <= $crate::levels::STATIC_MAX_LEVEL && $lvl <= $crate::levels::max_level() {
231            $crate::dispatch::log_tagged(&LOG_DESC, $properties, format_args!($($arg)+));
232        }
233    });
234    ($lvl:expr, $($arg:tt)+) => ($crate::log!(target: $crate::__log_module_path!(), $lvl, $($arg)+));
235}

373/// Determines if a message logged at the specified level in that module will
374/// be logged.
375///
376/// This can be used to avoid expensive computation of log message arguments if
377/// the message would be ignored anyway.
378///
379/// # Examples
380///
381/// ```
382/// use micromegas_tracing::prelude::*;
383///
384/// # fn foo() {
385/// if log_enabled!(Level::Debug) {
386///     let data = expensive_call();
387///     debug!("expensive debug data: {} {}", data.x, data.y);
388/// }
389/// if log_enabled!(target: "Global", Level::Debug) {
390///    let data = expensive_call();
391///    debug!(target: "Global", "expensive debug data: {} {}", data.x, data.y);
392/// }
393/// # }
394/// # struct Data { x: u32, y: u32 }
395/// # fn expensive_call() -> Data { Data { x: 0, y: 0 } }
396/// # fn main() {}
397/// ```
398#[macro_export(local_inner_macros)]
399macro_rules! log_enabled {
400    (target: $target:expr, $lvl:expr) => {{
401        let lvl = $lvl;
402        static LOG_ENABLED_METADATA: $crate::logs::LogMetadata = $crate::logs::LogMetadata {
403            level: $lvl,
404            level_filter: std::sync::atomic::AtomicU32::new($crate::logs::FILTER_LEVEL_UNSET_VALUE),
405            fmt_str: "",
406            target: $target,
407            module_path: $crate::__log_module_path!(),
408            file: $crate::__log_file!(),
409            line: $crate::__log_line!(),
410        };
411        lvl <= $crate::levels::STATIC_MAX_LEVEL
412            && lvl <= $crate::levels::max_level()
413            && $crate::dispatch::log_enabled(&LOG_ENABLED_METADATA)
414    }};
415    ($lvl:expr) => {
416        $crate::log_enabled!(target: $crate::__log_module_path!(), $lvl)
417    };
418}
