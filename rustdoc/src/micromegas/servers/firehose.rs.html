1//! Kinesis Data Firehose HTTP Endpoint Delivery route for `telemetry-ingestion-srv`.
2//!
3//! Exposes `POST /ingestion/otlp/v1/metrics/firehose` so a CloudWatch Metric Stream can
4//! push metrics into micromegas as **Metric Stream → Firehose → micromegas**, with no
5//! Lambda, no Kinesis Data Stream, and no collector process in between. Firehose is a
6//! dumb managed pipe: it wraps each delivered record (in OpenTelemetry 1.0.0 output mode,
7//! an OTLP `ExportMetricsServiceRequest` protobuf) in a small JSON envelope and expects a
8//! fixed ack shape back.
9//!
10//! Shared Firehose transport plumbing (auth, ack shape, request-id parsing) lives in
11//! `firehose_common` — this module only knows about the metrics-specific decode/ingest
12//! calls.
13//!
14//! Once a record's bytes are extracted from the envelope, they are the exact protobuf the
15//! existing `handler::ingest_metrics` already handles — no new identity, block, split, or
16//! write logic.
17
18use super::firehose_common::{firehose_auth_middleware, firehose_response, request_id_from};
19use super::ingestion_limits::apply_ingestion_body_limits;
20use axum::Extension;
21use axum::Router;
22use axum::http::{HeaderMap, StatusCode};
23use axum::middleware;
24use axum::response::Response;
25use axum::routing::post;
26use micromegas_auth::types::AuthProvider;
27use micromegas_ingestion::web_ingestion_service::WebIngestionService;
28use micromegas_otel_ingestion::{Signal, handler};
29use micromegas_tracing::prelude::*;
30use std::sync::Arc;
31
32async fn firehose_handler(
33    Extension(service): Extension<Arc<WebIngestionService>>,
34    headers: HeaderMap,
35    body: bytes::Bytes,
36) -> Response {
37    let mut request_id = request_id_from(&headers);
38    let envelope = match handler::decode_firehose_envelope(&body, Signal::Metrics) {
39        Ok(e) => e,
40        Err(err) => {
41            error!("firehose decode error: {err}");
42            let status = StatusCode::from_u16(err.http_status())
43                .unwrap_or(StatusCode::INTERNAL_SERVER_ERROR);
44            return firehose_response(status, &request_id, Some(&err.public_message()));
45        }
46    };
47    if request_id.is_empty() {
48        request_id = envelope.request_id.clone(); // header preferred; body requestId is fallback
49    }
50    match handler::ingest_firehose_metrics(service, envelope.records).await {
51        Ok(()) => firehose_response(StatusCode::OK, &request_id, None),
52        Err(err) => {
53            error!("firehose ingest error: {err}");
54            let status = StatusCode::from_u16(err.http_status())
55                .unwrap_or(StatusCode::INTERNAL_SERVER_ERROR);
56            firehose_response(status, &request_id, Some(&err.public_message()))
57        }
58    }
59}

61/// Builds the Firehose sub-router: route + service extension + optional Firehose-auth
62/// layer + shared ingestion body limits (gzip + 20 MiB wire / 300 MiB decompressed).
63///
64/// Deliberately not merged into `protected_app` — it must not sit under the global Bearer
65/// `auth_middleware`, since Firehose can only send its credential via
66/// `X-Amz-Firehose-Access-Key`. Auth is applied only when `auth_provider` is `Some`,
67/// matching every other ingestion route's dev-mode-open behavior.
68pub fn firehose_router(
69    service: Arc<WebIngestionService>,
70    auth_provider: Option<Arc<dyn AuthProvider>>,
71) -> Router {
72    let mut router = Router::new()
73        .route(
74            "/ingestion/otlp/v1/metrics/firehose",
75            post(firehose_handler),
76        )
77        .layer(Extension(service));
78    if let Some(provider) = auth_provider {
79        router = router.layer(middleware::from_fn(move |req, next| {
80            firehose_auth_middleware(provider.clone(), req, next)
81        }));
82    }
83    apply_ingestion_body_limits(router)
84}
