1//! Shared body-limit / decompression layers for ingestion routes carrying large,
2//! potentially-compressed bodies (OTLP and webhook). Factored out so every such
3//! router applies identical limits instead of duplicating the constants.
4
5use axum::Router;
6use axum::extract::DefaultBodyLimit;
7use tower_http::decompression::RequestDecompressionLayer;
8use tower_http::limit::RequestBodyLimitLayer;
9
10/// 20 MiB matches the OTel Collector `confighttp.max_request_body_size` default —
11/// anything an SDK is willing to send under the conventional Collector cap fits here too.
12/// Applies to compressed wire bytes (the `RequestBodyLimitLayer` runs outside the
13/// decompression layer).
14pub(crate) const INGESTION_BODY_LIMIT_BYTES: usize = 20 * 1024 * 1024;
15
16/// Cap on the decompressed body size the handler will materialize. Without this,
17/// a malicious gzip payload up to `INGESTION_BODY_LIMIT_BYTES` could expand at gzip's
18/// worst-case ratio (~1000×) and OOM the server. Sized at 15× the wire cap to
19/// cover legitimate protobuf compression (commonly observed up to 10×) with
20/// headroom, while still bounding the worst case to a survivable allocation.
21pub(crate) const INGESTION_DECOMPRESSED_BODY_LIMIT_BYTES: usize = 300 * 1024 * 1024;
22
23/// `Retry-After` value (in seconds) on retryable 503 responses. Conservative default —
24/// tune based on observed recovery times.
25pub(crate) const RETRY_AFTER_SECONDS: u32 = 30;
26
27/// Applies the shared body-limit + gzip-decompression layers to `router`.
28///
29/// Layer order, outermost → innermost (request travels through them top to bottom):
30///  1. `DefaultBodyLimit::max(300 MiB)` — caps the post-decompression bytes the
31///     handler's `Bytes` extractor will materialize, defending against gzip-bomb
32///     expansion that the wire-byte limit can't see.
33///  2. `RequestBodyLimitLayer(20 MiB)` — caps the *compressed* wire bytes;
34///     enforced before decompression, returning 413 on oversize.
35///  3. `RequestDecompressionLayer` — gzip-decodes the body before the handler.
36///  4. handler.
37pub(crate) fn apply_ingestion_body_limits(router: Router) -> Router {
38    router
39        .layer(RequestDecompressionLayer::new().gzip(true))
40        .layer(RequestBodyLimitLayer::new(INGESTION_BODY_LIMIT_BYTES))
41        .layer(DefaultBodyLimit::max(
42            INGESTION_DECOMPRESSED_BODY_LIMIT_BYTES,
43        ))
44}
