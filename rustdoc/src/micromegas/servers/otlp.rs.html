1//! OTLP/HTTP route registration for `telemetry-ingestion-srv`.
2//!
3//! Exposes three routes that match the OTLP/HTTP spec:
4//!  - `POST /ingestion/otlp/v1/logs`
5//!  - `POST /ingestion/otlp/v1/metrics`
6//!  - `POST /ingestion/otlp/v1/traces`
7//!
8//! The OTLP sub-router applies its own 20 MiB body limit (matching the OTel Collector
9//! `confighttp.max_request_body_size` default) plus gzip request decompression,
10//! independent of the parent router's 100 MiB limit on `/ingestion/insert_block`.
11//!
12//! Per OTLP/HTTP spec, success responses are 2xx with an `Export*ServiceResponse` proto;
13//! 4xx/5xx responses are protobuf-encoded `google.rpc.Status` messages, with a
14//! `Retry-After` header on retryable 503 responses.
15

32/// 20 MiB matches the OTel Collector `confighttp.max_request_body_size` default —
33/// anything an SDK is willing to send under the conventional Collector cap fits here too.
34/// Applies to compressed wire bytes (the `RequestBodyLimitLayer` runs outside the
35/// decompression layer).
36const OTLP_BODY_LIMIT_BYTES: usize = 20 * 1024 * 1024;
37
38/// Cap on the decompressed body size the handler will materialize. Without this,
39/// a malicious gzip payload up to `OTLP_BODY_LIMIT_BYTES` could expand at gzip's
40/// worst-case ratio (~1000×) and OOM the server. Sized at 15× the wire cap to
41/// cover legitimate protobuf compression (commonly observed up to 10×) with
42/// headroom, while still bounding the worst case to a survivable allocation.
43const OTLP_DECOMPRESSED_BODY_LIMIT_BYTES: usize = 300 * 1024 * 1024;
44
45/// `Retry-After` value (in seconds) on retryable 503 responses. Conservative default —
46/// tune based on observed recovery times.
47const RETRY_AFTER_SECONDS: u32 = 30;
48
49const CONTENT_TYPE_PROTOBUF: &str = "application/x-protobuf";
50
51/// Examines the `Content-Type` header. The spec allows parameters
52/// (e.g. `application/x-protobuf; charset=utf-8`), so we parse rather than string-compare.
53fn check_content_type(headers: &HeaderMap) -> Result<(), OtlpHttpError> {
54    let Some(ct) = headers.get(header::CONTENT_TYPE) else {
55        return Err(OtlpHttpError::WrongContentType);
56    };
57    let Ok(ct) = ct.to_str() else {
58        return Err(OtlpHttpError::WrongContentType);
59    };
60    let media = ct
61        .split(';')
62        .next()
63        .unwrap_or("")
64        .trim()
65        .to_ascii_lowercase();
66    if media == CONTENT_TYPE_PROTOBUF {
67        Ok(())
68    } else {
69        Err(OtlpHttpError::WrongContentType)
70    }
71}

73/// Internal error type covering both pre-handler validation failures (415) and
74/// post-handler `OtelError`s. Each variant maps to a single HTTP response shape
75/// (status code, optional `Retry-After`, `google.rpc.Status` body).
76enum OtlpHttpError {
77    WrongContentType,
78    Otel(OtelError),
79}
80
81impl OtlpHttpError {
82    fn into_otlp_response(self) -> Response {
83        match self {
84            OtlpHttpError::WrongContentType => build_error_response(
85                StatusCode::UNSUPPORTED_MEDIA_TYPE,
86                3, // INVALID_ARGUMENT
87                "Content-Type must be application/x-protobuf",
88                false,
89            ),
90            OtlpHttpError::Otel(err) => {
91                let retryable = err.is_retryable();
92                let status = match err.http_status() {
93                    400 => StatusCode::BAD_REQUEST,
94                    503 => StatusCode::SERVICE_UNAVAILABLE,
95                    other => {
96                        StatusCode::from_u16(other).unwrap_or(StatusCode::INTERNAL_SERVER_ERROR)
97                    }
98                };
99                let code = err.grpc_code();
100                // Detailed error (includes raw sqlx / object-store messages) is
101                // logged server-side; only the sanitized public form goes to the
102                // client to avoid leaking backend internals.
103                error!("OTLP error: {}", err);
104                build_error_response(status, code, &err.public_message(), retryable)
105            }
106        }
107    }
108}
109
110fn build_error_response(status: StatusCode, code: i32, message: &str, retryable: bool) -> Response {
111    let proto_status = micromegas_otel_ingestion::proto::Status {
112        code,
113        message: message.to_string(),
114    };
115    let body = proto_status.encode_to_vec();
116    let mut response = Response::builder()
117        .status(status)
118        .header(
119            header::CONTENT_TYPE,
120            HeaderValue::from_static(CONTENT_TYPE_PROTOBUF),
121        )
122        .body(Body::from(body))
123        .expect("building OTLP error response");
124    if retryable && let Ok(value) = HeaderValue::from_str(&RETRY_AFTER_SECONDS.to_string()) {
125        response.headers_mut().insert(header::RETRY_AFTER, value);
126    }
127    response
128}
129
130fn proto_response<M: Message>(msg: M) -> Response {
131    let body = msg.encode_to_vec();
132    Response::builder()
133        .status(StatusCode::OK)
134        .header(
135            header::CONTENT_TYPE,
136            HeaderValue::from_static(CONTENT_TYPE_PROTOBUF),
137        )
138        .body(Body::from(body))
139        .expect("building OTLP success response")
140}

142async fn logs_handler(
143    Extension(service): Extension<Arc<WebIngestionService>>,
144    headers: HeaderMap,
145    body: bytes::Bytes,
146) -> Response {
147    if let Err(e) = check_content_type(&headers) {
148        return e.into_otlp_response();
149    }
150    match handler::ingest_logs(service, body).await {
151        Ok(resp) => proto_response(resp),
152        Err(e) => OtlpHttpError::Otel(e).into_otlp_response(),
153    }
154}

184/// Builds a sub-Router carrying the three OTLP routes plus the body-limit and
185/// gzip-decompression layers scoped to those routes.
186///
187/// Layer order, outermost → innermost (request travels through them top to bottom):
188///  1. `DefaultBodyLimit::max(300 MiB)` — caps the post-decompression bytes the
189///     handler's `Bytes` extractor will materialize, defending against gzip-bomb
190///     expansion that the wire-byte limit can't see.
191///  2. `RequestBodyLimitLayer(20 MiB)` — caps the *compressed* wire bytes;
192///     enforced before decompression, returning 413 on oversize.
193///  3. `RequestDecompressionLayer` — gzip-decodes the body before the handler.
194///  4. handler.
195pub fn otlp_router() -> Router {
196    Router::new()
197        .route("/ingestion/otlp/v1/logs", post(logs_handler))
198        .route("/ingestion/otlp/v1/metrics", post(metrics_handler))
199        .route("/ingestion/otlp/v1/traces", post(traces_handler))
200        .layer(RequestDecompressionLayer::new().gzip(true))
201        .layer(RequestBodyLimitLayer::new(OTLP_BODY_LIMIT_BYTES))
202        .layer(DefaultBodyLimit::max(OTLP_DECOMPRESSED_BODY_LIMIT_BYTES))
203}
