1use micromegas_auth::types::AuthProvider;
2use micromegas_tracing::prelude::*;
3use std::sync::Arc;
4use tonic::{Request, Status};
5
6/// Checks the authentication of a Tonic request using an `AuthProvider`.
7///
8/// This function checks for a `Bearer` token in the `Authorization` header
9/// and validates it using the provided authentication provider.
10pub async fn check_auth(
11    req: Request<()>,
12    auth_provider: &Arc<dyn AuthProvider>,
13) -> Result<Request<()>, Status> {
14    let metadata = req.metadata();
15    let authorization = metadata
16        .get(http::header::AUTHORIZATION.as_str())
17        .ok_or_else(|| {
18            trace!("missing authorization header");
19            Status::unauthenticated("missing authorization header")
20        })?
21        .to_str()
22        .map_err(|_e| {
23            warn!("error parsing authorization header");
24            Status::unauthenticated("error parsing authorization header")
25        })?;
26    let bearer = "Bearer ";
27    if !authorization.starts_with(bearer) {
28        warn!("Invalid auth header");
29        return Err(Status::unauthenticated("Invalid auth header"));
30    }
31    let token = &authorization[bearer.len()..];
32
33    let auth_ctx = auth_provider.validate_token(token).await.map_err(|e| {
34        warn!("authentication failed: {e}");
35        Status::unauthenticated("invalid token")
36    })?;
37
38    info!(
39        "authenticated: subject={} email={:?} issuer={} admin={}",
40        auth_ctx.subject, auth_ctx.email, auth_ctx.issuer, auth_ctx.is_admin
41    );
42
43    let mut req = req;
44    req.extensions_mut().insert(auth_ctx);
45    Ok(req)
46}
