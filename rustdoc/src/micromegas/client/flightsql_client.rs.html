9/// Represents a prepared statement with its schema and query string.
10#[derive(Debug)]
11pub struct PreparedStatement {
12    pub schema: SchemaRef,
13    pub query: String,
14}
15
16/// Micromegas FlightSQL client
17pub struct Client {
18    inner: FlightSqlServiceClient<Channel>,
19}
20
21impl Client {
22    /// Creates a new client from a grpc channel
23    ///
24    /// # Arguments
25    ///
26    /// * `channel` - The gRPC channel to use for communication.
27    pub fn new(channel: Channel) -> Self {
28        let inner = FlightSqlServiceClient::new(channel);
29        Self { inner }
30    }
31
32    /// Returns a mutable reference to the inner `FlightSqlServiceClient`.
33    ///
34    /// This allows direct access to the underlying FlightSQL client for advanced operations.
35    pub fn inner_mut(&mut self) -> &mut FlightSqlServiceClient<Channel> {
36        &mut self.inner
37    }
38
39    fn set_query_range(&mut self, query_range: Option<TimeRange>) {
40        self.inner.set_header(
41            "query_range_begin",
42            query_range.map_or(String::from(""), |r| r.begin.to_rfc3339()),
43        );
44        self.inner.set_header(
45            "query_range_end",
46            query_range.map_or(String::from(""), |r| r.end.to_rfc3339()),
47        );
48    }
49
50    /// Executes a SQL query and returns the results as a vector of `RecordBatch`.
51    ///
52    /// # Arguments
53    ///
54    /// * `sql` - The SQL query string to execute.
55    /// * `query_range` - An optional `TimeRange` to filter the query results by time.
56    pub async fn query(
57        &mut self,
58        sql: String,
59        query_range: Option<TimeRange>,
60    ) -> Result<Vec<RecordBatch>> {
61        let mut record_batch_stream = self.query_stream(sql, query_range).await?;
62        let mut batches = vec![];
63        while let Some(batch_res) = record_batch_stream.next().await {
64            batches.push(batch_res?);
65        }
66        Ok(batches)
67    }
68
69    /// Executes a SQL query and returns the results as a `FlightRecordBatchStream`.
70    ///
71    /// This function is useful for processing large query results incrementally.
72    ///
73    /// # Arguments
74    ///
75    /// * `sql` - The SQL query string to execute.
76    /// * `query_range` - An optional `TimeRange` to filter the query results by time.
77    pub async fn query_stream(
78        &mut self,
79        sql: String,
80        query_range: Option<TimeRange>,
81    ) -> Result<FlightRecordBatchStream> {
82        self.set_query_range(query_range);
83        let info = self.inner.execute(sql, None).await?;
84        let ticket = info.endpoint[0]
85            .ticket
86            .clone()
87            .with_context(|| "reading ticket from endpoint")?;
88        let flight_data_stream = self.inner.do_get(ticket).await?.into_inner();
89        Ok(FlightRecordBatchStream::new(flight_data_stream))
90    }
91
92    /// Prepares a SQL statement and returns a `PreparedStatement`.
93    ///
94    /// This function allows to compute the schema of a query without actually executing it.
95    ///
96    /// # Arguments
97    ///
98    /// * `sql` - The SQL query string to prepare.
99    pub async fn prepare_statement(&mut self, sql: String) -> Result<PreparedStatement> {
100        self.set_query_range(None);
101        let prepared = self.inner.prepare(sql.clone(), None).await?;
102        Ok(PreparedStatement {
103            schema: Arc::new(prepared.dataset_schema()?.clone()),
104            query: sql,
105        })
106    }
107}
