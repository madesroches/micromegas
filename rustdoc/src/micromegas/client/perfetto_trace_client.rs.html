12async fn get_process_exe(
13    process_id: &str,
14    client: &mut Client,
15    query_range: TimeRange,
16) -> Result<String> {
17    let sql_process = format!(
18        r#"
19        SELECT "processes.exe" as exe
20        FROM blocks
21        WHERE process_id = '{process_id}'
22        LIMIT 1
23        "#
24    );
25    let batches = client.query(sql_process, Some(query_range)).await?;
26    if batches.len() != 1 || batches[0].num_rows() != 1 {
27        anyhow::bail!("process not found");
28    }
29    let exes: &StringArray = typed_column_by_name(&batches[0], "exe")?;
30    Ok(exes.value(0).to_owned())
31}

33/// Formats a Perfetto trace from the telemetry data.
34///
35/// This function queries the FlightSQL server for process, thread, and span information
36/// and formats it into a Perfetto trace protobuf.
37pub async fn format_perfetto_trace(
38    client: &mut Client,
39    process_id: &str,
40    query_range: TimeRange,
41) -> Result<Vec<u8>> {
42    let mut writer = Writer::new(process_id);
43    let exe = get_process_exe(process_id, client, query_range).await?;
44    writer.append_process_descriptor(&exe);
45
46    let sql_streams = format!(
47        r#"
48        SELECT stream_id,
49               property_get("streams.properties", 'thread-name') as thread_name,
50               property_get("streams.properties", 'thread-id') as thread_id
51        FROM blocks
52        WHERE process_id = '{process_id}'
53        AND array_has("streams.tags", 'cpu')
54        GROUP BY stream_id, thread_name, thread_id
55    "#
56    );
57
58    let batches = client.query(sql_streams, Some(query_range)).await?;
59    for b in batches {
60        let stream_id_column: &StringArray = typed_column_by_name(&b, "stream_id")?;
61        let thread_name_column: &StringArray = typed_column_by_name(&b, "thread_name")?;
62        let thread_id_column: &StringArray = typed_column_by_name(&b, "thread_id")?;
63        for row in 0..b.num_rows() {
64            let stream_id = stream_id_column.value(row);
65            let thread_name = thread_name_column.value(row);
66            eprintln!("stream_id={stream_id} thread_name={thread_name}");
67            let thread_id = thread_id_column.value(row);
68            writer.append_thread_descriptor(stream_id, thread_id.parse::<i32>()?, thread_name);
69
70            let sql_spans = format!(
71                r#"
72                SELECT id, parent, depth, hash, begin, end, duration, name, target, filename, line
73                FROM view_instance('thread_spans', '{stream_id}');
74            "#
75            );
76            let span_batches = client.query(sql_spans, Some(query_range)).await?;
77            for b in span_batches {
78                let begins: &TimestampNanosecondArray = typed_column_by_name(&b, "begin")?;
79                let ends: &TimestampNanosecondArray = typed_column_by_name(&b, "end")?;
80                let names: &StringArray = typed_column_by_name(&b, "name")?;
81                let targets: &StringArray = typed_column_by_name(&b, "target")?;
82                let filenames: &StringArray = typed_column_by_name(&b, "filename")?;
83                let lines: &UInt32Array = typed_column_by_name(&b, "line")?;
84                for row in 0..b.num_rows() {
85                    writer.append_span(
86                        begins.value(row) as u64,
87                        ends.value(row) as u64,
88                        names.value(row),
89                        targets.value(row),
90                        filenames.value(row),
91                        lines.value(row),
92                    );
93                }
94            }
95        }
96    }
97
98    Ok(writer.into_trace().encode_to_vec())
99}

101/// Writes a Perfetto trace to a file.
102///
103/// This function calls `format_perfetto_trace` to generate the trace data
104/// and then writes it to the specified output file.
105pub async fn write_perfetto_trace(
106    client: &mut Client,
107    process_id: &str,
108    begin: DateTime<Utc>,
109    end: DateTime<Utc>,
110    out_filename: &str,
111) -> Result<()> {
112    let buf = format_perfetto_trace(client, process_id, TimeRange::new(begin, end)).await?;
113    let mut file = File::create(out_filename).await?;
114    file.write_all(&buf).await?;
115    Ok(())
116}
