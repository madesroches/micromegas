17/// Fail fast if the cache server can't be reached, so reads fall back to the
18/// direct store instead of stalling on a hung connection.
19const CACHE_CONNECT_TIMEOUT: Duration = Duration::from_secs(2);
20/// Overall per-request timeout; a slow cache surfaces as an error and triggers
21/// the existing fallback-to-direct path.
22const CACHE_REQUEST_TIMEOUT: Duration = Duration::from_secs(15);

67    /// Issue a range GET and return the body bytes along with the full object
68    /// size parsed from the `Content-Range: bytes {start}-{end}/{size}` response
69    /// header. The 206 response already carries the full size, so callers can
70    /// avoid a separate HEAD round-trip. When the header is absent or
71    /// unparseable, the size is returned as `None` and the caller should fall
72    /// back to a `head_size` lookup.
73    ///
74    /// An open-ended `end` (`None`) requests `bytes={start}-`, i.e. from `start`
75    /// to the end of the object, which the cache server resolves against the
76    /// true object size.
77    async fn get_range_bytes(
78        &self,
79        location: &Path,
80        start: u64,
81        end: Option<u64>,
82    ) -> Result<(Bytes, Option<u64>)> {
83        let url = self.obj_url(location);
84        let range_header = match end {
85            Some(end) => format!("bytes={}-{}", start, end.saturating_sub(1)),
86            None => format!("bytes={start}-"),
87        };
88        let resp = self
89            .add_auth(self.http.get(&url))
90            .header("Range", range_header)
91            .send()
92            .await
93            .with_context(|| "sending GET to cache")?;
94        if !resp.status().is_success() {
95            return Err(anyhow!("cache GET {url} status {}", resp.status()));
96        }
97        let object_size = parse_content_range_size(resp.headers());
98        let data = resp.bytes().await.with_context(|| "reading GET response")?;
99        Ok((data, object_size))
100    }
101
102    /// Resolve the full object size for a ranged read, preferring the size from
103    /// the GET's `Content-Range` header and falling back to a HEAD only when the
104    /// header was missing or unparseable.
105    async fn resolve_size(&self, location: &Path, from_range: Option<u64>) -> Result<u64> {
106        match from_range {
107            Some(size) => Ok(size),
108            None => self.head_size(location).await,
109        }
110    }

112    /// Issue an unranged GET and build a streaming `GetResult`, mapping the
113    /// response body to `GetResultPayload::Stream` so the whole object is never
114    /// buffered in memory (matching how the direct store streams the body). The
115    /// object size comes from the `Content-Length` header, which is required to
116    /// populate `meta.size` and the `0..size` range without reading the body.
117    async fn get_full_stream(&self, location: &Path) -> Result<GetResult> {
118        let url = self.obj_url(location);
119        let resp = self
120            .add_auth(self.http.get(&url))
121            .send()
122            .await
123            .with_context(|| "sending GET to cache")?;
124        if !resp.status().is_success() {
125            return Err(anyhow!("cache GET {url} status {}", resp.status()));
126        }
127        let size = resp
128            .content_length()
129            .ok_or_else(|| anyhow!("missing Content-Length in GET response"))?;
130        Ok(stream_get_result(location, resp, size))
131    }
132
133    async fn head_size(&self, location: &Path) -> Result<u64> {
134        let url = self.obj_url(location);
135        let resp = self
136            .add_auth(self.http.head(&url))
137            .send()
138            .await
139            .with_context(|| "sending HEAD to cache")?;
140        if !resp.status().is_success() {
141            return Err(anyhow!("cache HEAD {url} status {}", resp.status()));
142        }
143        resp.headers()
144            .get("content-length")
145            .and_then(|v| v.to_str().ok())
146            .and_then(|s| s.parse::<u64>().ok())
147            .ok_or_else(|| anyhow!("missing Content-Length in HEAD response"))
148    }
149}

151/// Parse the full object size from a `Content-Range: bytes {start}-{end}/{size}`
152/// header (the suffix after `/`). Returns `None` when the header is absent or
153/// not in the expected form (e.g. `bytes */size` or an unparseable value).
154fn parse_content_range_size(headers: &reqwest::header::HeaderMap) -> Option<u64> {
155    headers
156        .get(reqwest::header::CONTENT_RANGE)
157        .and_then(|v| v.to_str().ok())
158        .and_then(|s| s.rsplit('/').next())
159        .and_then(|size| size.trim().parse::<u64>().ok())
160}

168/// Build a streaming `GetResult` for an unranged GET, wrapping the reqwest
169/// response body in `GetResultPayload::Stream` so the object is delivered in
170/// chunks rather than buffered whole. `object_size` (from `Content-Length`)
171/// populates `meta.size` and the `0..object_size` range.
172fn stream_get_result(location: &Path, resp: reqwest::Response, object_size: u64) -> GetResult {
173    let meta = ObjectMeta {
174        location: location.clone(),
175        last_modified: chrono::Utc::now(),
176        size: object_size,
177        e_tag: None,
178        version: None,
179    };
180    let body = resp
181        .bytes_stream()
182        .map_err(|e| object_store::Error::Generic {
183            store: "CacheClientStore",
184            source: Box::new(e),
185        });
186    GetResult {
187        payload: GetResultPayload::Stream(Box::pin(body)),
188        meta,
189        range: 0..object_size,
190        attributes: Attributes::default(),
191    }
192}
193
194/// Build a `GetResult` for a ranged GET. `range` is the slice actually returned
195/// while `object_size` is the full object size (per the `ObjectMeta` contract).
196///
197/// When no bytes are returned (e.g. a 0-byte object, or a range starting at or
198/// beyond EOF), the requested `range` may lie outside the object; report an
199/// empty `0..0` range so it always matches `data.len()` and stays within
200/// `object_size`, satisfying the `GetResult` invariant.
201fn ranged_get_result(
202    location: &Path,
203    data: Bytes,
204    range: Range<u64>,
205    object_size: u64,
206) -> GetResult {
207    let range = if data.is_empty() { 0..0 } else { range };
208    build_get_result(location, data, range, object_size)
209}
210
211fn build_get_result(
212    location: &Path,
213    data: Bytes,
214    range: Range<u64>,
215    object_size: u64,
216) -> GetResult {
217    let meta = ObjectMeta {
218        location: location.clone(),
219        last_modified: chrono::Utc::now(),
220        size: object_size,
221        e_tag: None,
222        version: None,
223    };
224    let payload = GetResultPayload::Stream(Box::pin(stream::once(async move { Ok(data) })));
225    GetResult {
226        payload,
227        meta,
228        range,
229        attributes: Attributes::default(),
230    }
231}

252    async fn get_opts(
253        &self,
254        location: &Path,
255        options: GetOptions,
256    ) -> object_store::Result<GetResult> {
257        use object_store::GetRange;
258
259        // The cache HTTP protocol can't convey conditional/version preconditions,
260        // so any such request must go straight to the direct store to preserve
261        // the expected 412/304 semantics.
262        if options.if_match.is_some()
263            || options.if_none_match.is_some()
264            || options.if_modified_since.is_some()
265            || options.if_unmodified_since.is_some()
266            || options.version.is_some()
267        {
268            return self.direct.get_opts(location, options).await;
269        }
270
271        // A head-only request needs metadata, not the body: return an empty
272        // payload with the true object size instead of streaming the object.
273        if options.head {
274            let result: Result<GetResult> = match self.head_size(location).await {
275                Ok(size) => Ok(ranged_get_result(location, Bytes::new(), 0..0, size)),
276                Err(e) => Err(e),
277            };
278            return match result {
279                Ok(r) => Ok(r),
280                Err(e) => {
281                    // Falling back to the direct store is a by-design graceful
282                    // degradation path (cache restarting/unreachable), not an
283                    // error: keep it at debug and let the fallback metric (which
284                    // is what dashboards alert on) carry the signal, so a cache
285                    // outage doesn't flood logs with one warning per read.
286                    imetric!("range_cache_client_fallback", "count", 1_u64);
287                    debug!("cache miss for {location} (head), falling back to direct: {e}");
288                    self.direct.get_opts(location, options).await
289                }
290            };
291        }
292
293        let result: Result<GetResult> = match &options.range {
294            None => self.get_full_stream(location).await,
295            // Issue the range GET first and read the full object size from the
296            // 206 `Content-Range` header, avoiding a preceding HEAD round-trip.
297            // `resolve_size` only falls back to a HEAD if that header was absent
298            // or unparseable.
299            Some(GetRange::Bounded(r)) => {
300                match self.get_range_bytes(location, r.start, Some(r.end)).await {
301                    Ok((data, size_hint)) => match self.resolve_size(location, size_hint).await {
302                        Ok(size) => {
303                            let len = data.len() as u64;
304                            Ok(ranged_get_result(
305                                location,
306                                data,
307                                r.start..r.start + len,
308                                size,
309                            ))
310                        }
311                        Err(e) => Err(e),
312                    },
313                    Err(e) => Err(e),
314                }
315            }
316            Some(GetRange::Offset(offset)) => {
317                let start = *offset;
318                // Open-ended range: the server resolves `-` against the true
319                // object size, returned to us via `Content-Range`.
320                match self.get_range_bytes(location, start, None).await {
321                    Ok((data, size_hint)) => match self.resolve_size(location, size_hint).await {
322                        Ok(size) => {
323                            let len = data.len() as u64;
324                            Ok(ranged_get_result(location, data, start..start + len, size))
325                        }
326                        Err(e) => Err(e),
327                    },
328                    Err(e) => Err(e),
329                }
330            }
331            // Suffix reads need the object size up front to compute the start
332            // offset, since the cache server's Range parser does not accept the
333            // `bytes=-N` suffix form. A HEAD is unavoidable here.
334            Some(GetRange::Suffix(suffix)) => match self.head_size(location).await {
335                Ok(size) => {
336                    let start = size.saturating_sub(*suffix);
337                    self.get_range_bytes(location, start, Some(size))
338                        .await
339                        .map(|(data, _)| {
340                            let len = data.len() as u64;
341                            ranged_get_result(location, data, start..start + len, size)
342                        })
343                }
344                Err(e) => Err(e),
345            },
346        };
347
348        match result {
349            Ok(r) => Ok(r),
350            Err(e) => {
351                imetric!("range_cache_client_fallback", "count", 1_u64);
352                debug!("cache miss for {location}, falling back to direct: {e}");
353                self.direct.get_opts(location, options).await
354            }
355        }
356    }

358    async fn get_ranges(
359        &self,
360        location: &Path,
361        ranges: &[Range<u64>],
362    ) -> object_store::Result<Vec<Bytes>> {
363        if ranges.is_empty() {
364            return Ok(vec![]);
365        }
366
367        let url = format!(
368            "{}/ranges/{}",
369            self.cache_base_url.trim_end_matches('/'),
370            location.as_ref()
371        );
372        let ranges_json: Vec<[u64; 2]> = ranges.iter().map(|r| [r.start, r.end]).collect();
373        let body = json!({ "ranges": ranges_json }).to_string();
374
375        let resp = match self
376            .add_auth(
377                self.http
378                    .post(&url)
379                    .header("Content-Type", "application/json")
380                    .body(body),
381            )
382            .send()
383            .await
384        {
385            Ok(r) if r.status().is_success() => r,
386            Ok(r) => {
387                imetric!("range_cache_client_fallback", "count", 1_u64);
388                debug!("cache ranges {url} status {}, falling back", r.status());
389                return self.direct.get_ranges(location, ranges).await;
390            }
391            Err(e) => {
392                imetric!("range_cache_client_fallback", "count", 1_u64);
393                debug!("cache ranges request failed: {e}, falling back to direct");
394                return self.direct.get_ranges(location, ranges).await;
395            }
396        };
397
398        let mut data = match resp.bytes().await {
399            Ok(b) => b,
400            Err(e) => {
401                imetric!("range_cache_client_fallback", "count", 1_u64);
402                debug!("reading ranges response failed: {e}, falling back to direct");
403                return self.direct.get_ranges(location, ranges).await;
404            }
405        };
406
407        let mut results = Vec::with_capacity(ranges.len());
408        for _ in 0..ranges.len() {
409            if data.remaining() < 8 {
410                // A truncated/garbled framing from our own cache is a protocol
411                // violation (unexpected), unlike the clean miss/outage paths
412                // above — keep this at warn.
413                imetric!("range_cache_client_fallback", "count", 1_u64);
414                warn!("truncated ranges response, falling back to direct");
415                return self.direct.get_ranges(location, ranges).await;
416            }
417            let len = data.get_u64_le() as usize;
418            if data.remaining() < len {
419                imetric!("range_cache_client_fallback", "count", 1_u64);
420                warn!("truncated ranges response body, falling back to direct");
421                return self.direct.get_ranges(location, ranges).await;
422            }
423            results.push(data.copy_to_bytes(len));
424        }
425
426        Ok(results)
427    }
