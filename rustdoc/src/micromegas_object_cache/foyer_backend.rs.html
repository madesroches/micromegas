9pub struct FoyerBackend {
10    cache: HybridCache<String, Bytes>,
11}
12
13impl FoyerBackend {
14    pub async fn new(dir: &str, ram_bytes: usize, disk_bytes: usize) -> Result<Self> {
15        Self::new_with_shards(dir, ram_bytes, disk_bytes, 8).await
16    }
17
18    pub async fn new_with_shards(
19        dir: &str,
20        ram_bytes: usize,
21        disk_bytes: usize,
22        shards: usize,
23    ) -> Result<Self> {
24        ensure!(shards > 0, "shards must be > 0");
25        let cache = HybridCacheBuilder::new()
26            .memory(ram_bytes)
27            .with_weighter(|_key: &String, value: &Bytes| value.len())
28            .with_shards(shards)
29            // Pin the RAM tier to LRU explicitly: only LRU maps
30            // `CacheHint::Low` to a low-priority eviction hint in foyer 0.14.x
31            // (Lfu/S3Fifo/Fifo silently discard it). This is the crate's
32            // current default; pinning it defensively guards `FillHint`
33            // against a future foyer default change.
34            .with_eviction_config(LruConfig::default())
35            .storage(Engine::Large)
36            .with_device_options(DirectFsDeviceOptions::new(dir).with_capacity(disk_bytes))
37            .build()
38            .await?;
39        Ok(Self { cache })
40    }
41
42    pub async fn close(&self) -> Result<()> {
43        self.cache.close().await?;
44        Ok(())
45    }
46}
47
48impl From<FillHint> for CacheHint {
49    fn from(hint: FillHint) -> Self {
50        match hint {
51            FillHint::Demand => CacheHint::Normal,
52            FillHint::Prefetch => CacheHint::Low,
53        }
54    }
55}
56
57#[async_trait]
58impl RangeCacheBackend for FoyerBackend {
59    async fn get(&self, key: &str) -> Option<Bytes> {
60        match self.cache.obtain(key.to_string()).await {
61            Ok(Some(entry)) => Some(entry.value().clone()),
62            Ok(None) => None,
63            // A backend (disk/IO) error must not fail the read: treat it as a
64            // miss so the caller falls back to origin, but surface it as a
65            // metric + log so a degraded SSD volume is observable rather than
66            // silently inflating origin traffic.
67            Err(e) => {
68                imetric!("range_cache_backend_error", "count", 1_u64);
69                warn!("range_cache backend get error key={key}: {e}");
70                None
71            }
72        }
73    }
74
75    async fn put(&self, key: String, value: Bytes, hint: FillHint) {
76        self.cache.insert_with_hint(key, value, hint.into());
77    }
78}
