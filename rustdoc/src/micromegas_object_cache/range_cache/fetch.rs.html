24/// Concurrency for probing the cache backend for block hits before going to
25/// origin. Backend probes do real disk I/O on a foyer RAM-tier miss, so a
26/// large read probed sequentially would serialize hundreds of disk reads.
27const BACKEND_PROBE_CONCURRENCY: usize = 16;
28
29/// Result of phase 1 (`probe_blocks`): the backend-hit blocks (`Demand` only)
30/// and the sorted, deduplicated set of blocks that must be fetched from
31/// origin.
32struct ProbeOutcome {
33    hits: HashMap<u64, Bytes>,
34    missing: Vec<u64>,
35}
36
37/// Result of phase 2 (`register_missing`): the single-flight registration of
38/// every missing block.
39struct RegisteredBlocks {
40    /// The missing blocks this call owns (is responsible for fetching from
41    /// origin); a sorted subsequence of the `missing` slice passed in.
42    owned: Vec<u64>,
43    /// The in-flight entry for every missing block, owned or joined, keyed by
44    /// block index.
45    entries: HashMap<u64, Arc<InFlight>>,
46    /// The batch handle for a prefetch call (`None` on the demand path); lets a
47    /// demand joiner promote every sibling of a scattered prefetch call.
48    batch: Option<Arc<BatchState>>,
49}
50
51/// One coalesced run of contiguous missing blocks filled by a single origin
52/// GET. Bundles the block-index `range` with the in-flight `entries` and cache
53/// `keys` for those blocks; `entries[i]` and `keys[i]` both correspond to block
54/// index `range.start + i`, an alignment the type keeps together rather than
55/// leaving to three parallel `Vec`s.
56struct CoalescedRun {
57    range: Range<u64>,
58    entries: Vec<Arc<InFlight>>,
59    keys: Vec<String>,
60}
61
62impl RangeCache {
63    /// Fetch `indices` (block indices within `key`, sized `file_size`) at the
64    /// given priority, returning the bytes for every requested block on the
65    /// `Demand` path. On the `Prefetch` path bytes are written to the backend
66    /// and dropped as each owned run completes; the returned map is always
67    /// empty, which is what keeps the prefetch peak bounded by
68    /// `prefetch_concurrency * max_coalesced_get_bytes` rather than the full
69    /// request size.
70    pub(super) async fn fetch_blocks(
71        &self,
72        key: &str,
73        file_size: u64,
74        indices: &[u64],
75        prio: Priority,
76    ) -> Result<HashMap<u64, Bytes>> {
77        if indices.is_empty() {
78            return Ok(HashMap::new());
79        }
80
81        // Classify once per call (a longest-prefix string match), not once
82        // per block probe: `prefix_tags` is reused across every probe and
83        // every coalesced run below.
84        let prefix_tags = self.classify_tags(key);
85        let block_tag = prefix_tags.prefix;
86        let (run_demand_tag, run_prefetch_tag) =
87            (prefix_tags.prefix_demand, prefix_tags.prefix_prefetch);
88
89        let ProbeOutcome { hits, missing } = self
90            .probe_blocks(key, file_size, indices, prio, block_tag)
91            .await;
92        if missing.is_empty() {
93            return Ok(hits);
94        }
95
96        let RegisteredBlocks {
97            owned,
98            entries,
99            batch: _batch,
100        } = self.register_missing(key, &missing, prio);
101
102        let hint = match prio {
103            Priority::Demand => FillHint::Demand,
104            Priority::Prefetch => FillHint::Prefetch,
105        };
106
107        // `owned` is a sorted subsequence of `missing` (itself sorted), so no
108        // extra sort is needed before coalescing.
109        for range in coalesce_runs(&owned, self.block_size, self.max_coalesced_get_bytes) {
110            let run_entries: Vec<Arc<InFlight>> = (range.start..range.end)
111                .map(|idx| entries.get(&idx).expect("owned entry present").clone())
112                .collect();
113            let run_keys: Vec<String> = (range.start..range.end)
114                .map(|idx| format!("blk:{}:{key}:{idx}", self.ns))
115                .collect();
116            let run = CoalescedRun {
117                range,
118                entries: run_entries,
119                keys: run_keys,
120            };
121            self.spawn_run_fetch(key, file_size, run, hint, run_demand_tag, run_prefetch_tag);
122        }
123
124        if prio == Priority::Prefetch {
125            join_prefetch(entries).await?;
126            return Ok(HashMap::new());
127        }
128
129        join_demand(entries, &missing, hits).await
130    }
131
132    /// Phase 1: probe the backend for every requested block with bounded
133    /// concurrency (a foyer RAM-tier miss falls through to async disk I/O,
134    /// and probing a large read one block at a time would serialize those
135    /// disk reads), healing a length-mismatched cached block by treating it
136    /// as missing, and partitioning into hits (`Demand` only) and missing
137    /// (sorted, deduplicated).
138    async fn probe_blocks(
139        &self,
140        key: &str,
141        file_size: u64,
142        indices: &[u64],
143        prio: Priority,
144        block_tag: &'static PropertySet,
145    ) -> ProbeOutcome {
146        let mut hits: HashMap<u64, Bytes> = HashMap::new();
147        let mut missing: Vec<u64> = Vec::new();
148        {
149            // The futures are collected eagerly (not mapped lazily inside the
150            // stream) so the resulting future stays `Send`-inferable across
151            // `tokio::spawn` (rustc's HRTB limitation with borrowed closures).
152            let probe_futs: Vec<_> = indices
153                .iter()
154                .map(|&idx| {
155                    let block_key = format!("blk:{}:{key}:{idx}", self.ns);
156                    let backend = self.backend.clone();
157                    // Computed eagerly (ahead of the `backend.get` call) so it
158                    // can be passed in as the promotion-gate contract: a
159                    // backend must not copy a length-mismatched value into a
160                    // faster tier. See `RangeCacheBackend::get`'s doc.
161                    let expected_range = block_byte_range(idx, self.block_size, file_size);
162                    let expected_len = expected_range.end - expected_range.start;
163                    async move {
164                        imetric!("range_cache_block_request", "count", block_tag, 1_u64);
165                        let probe = instrument_named!(
166                            backend.get(&block_key, expected_len),
167                            "range_cache_backend_read"
168                        )
169                        .await;
170                        (idx, probe, expected_len)
171                    }
172                })
173                .collect();
174            let mut probes = stream::iter(probe_futs).buffer_unordered(BACKEND_PROBE_CONCURRENCY);
175            while let Some((idx, probe, expected_len)) = probes.next().await {
176                match probe {
177                    Some(data) => {
178                        // A hit's length must match the block's true byte span.
179                        // A mismatch means a prior prefetch stored this block
180                        // under an undersized caller-supplied `size` (or the
181                        // origin object changed size): treat it as a miss so
182                        // it gets refetched and overwritten with the correct
183                        // bytes, healing the poisoned entry. Defense in depth:
184                        // the backend itself must already refuse to promote a
185                        // mismatched value (the `expected_len` passed above),
186                        // so this should never trigger in practice.
187                        if data.len() as u64 != expected_len {
188                            imetric!("range_cache_block_len_mismatch", "count", 1_u64);
189                            warn!(
190                                "range_cache block length mismatch key={key} idx={idx} \
191                                 expected={expected_len} got={}",
192                                data.len()
193                            );
194                            missing.push(idx);
195                            continue;
196                        }
197                        imetric!("range_cache_block_backend_hit", "count", block_tag, 1_u64);
198                        // A prefetch has nothing to do with an already-cached
199                        // block, so drop the bytes immediately instead of
200                        // accumulating every hit for the whole call.
201                        if prio == Priority::Demand {
202                            hits.insert(idx, data);
203                        }
204                    }
205                    None => missing.push(idx),
206                }
207            }
208        }
209        missing.sort_unstable();
210        missing.dedup();
211        ProbeOutcome { hits, missing }
212    }
213
214    /// Phase 2: register every missing block in the single-flight scheduler,
215    /// splitting into `owned` (this call is responsible for fetching it from
216    /// origin) and joined (some other call already owns it). `missing` is
217    /// taken by reference so `fetch_blocks` retains ownership of the `Vec`
218    /// for the `join_demand` call in phase 4.
219    fn register_missing(&self, key: &str, missing: &[u64], prio: Priority) -> RegisteredBlocks {
220        // Only prefetch calls carry a batch handle: it's what lets a demand
221        // joiner promote every sibling of a scattered prefetch call, not
222        // just the one block it happened to touch (see `promote_whole_batch`).
223        let batch = if prio == Priority::Prefetch {
224            Some(Arc::new(BatchState::new(missing.len())))
225        } else {
226            None
227        };
228
229        let mut owned: Vec<u64> = Vec::new();
230        let mut entries: HashMap<u64, Arc<InFlight>> = HashMap::with_capacity(missing.len());
231        for &idx in missing {
232            let block_key = format!("blk:{}:{key}:{idx}", self.ns);
233            match self.scheduler.own_or_join(block_key, prio, batch.clone()) {
234                Ownership::Owner(entry) => {
235                    owned.push(idx);
236                    entries.insert(idx, entry);
237                }
238                Ownership::Joiner(entry) => {
239                    entries.insert(idx, entry);
240                }
241            }
242        }
243        // Every member of this batch (owned or joined) was recorded as a
244        // promotion sibling inside `own_or_join`, under the inflight lock,
245        // so no entry became joinable before the batch list held it.
246        RegisteredBlocks {
247            owned,
248            entries,
249            batch,
250        }
251    }
252
253    /// Phase 3: spawn the detached task that owns one coalesced run of
254    /// `run_entries`/`run_keys`: acquire a run permit, issue one
255    /// `origin.get_range`, validate the run length, split the buffer into
256    /// per-block chunks, write each to the backend, fulfill each entry, and
257    /// remove the run's keys from the scheduler. Spawned as a detached task
258    /// so a cancelled caller (e.g. an axum request future dropped on client
259    /// disconnect) can never strand the other owned blocks or any joiners
260    /// waiting on this run. `FulfillGuard` covers the panic path.
261    fn spawn_run_fetch(
262        &self,
263        key: &str,
264        file_size: u64,
265        run: CoalescedRun,
266        hint: FillHint,
267        run_demand_tag: &'static PropertySet,
268        run_prefetch_tag: &'static PropertySet,
269    ) {
270        // Capture a `RangeCache` clone (cheap: all fields are `Arc`s or `Copy`)
271        // rather than cloning each field the task needs individually, mirroring
272        // the pattern in `stream_ranges_inner`. It also lets the success tail be
273        // a `RangeCache` method that reads `block_size`/`backend` from `self`.
274        let cache = self.clone();
275        let block_size = self.block_size;
276        let key_owned = key.to_string();
277
278        tokio::spawn(async move {
279            let guard = FulfillGuard::new(
280                cache.scheduler.clone(),
281                run.keys
282                    .iter()
283                    .cloned()
284                    .zip(run.entries.iter().cloned())
285                    .collect(),
286            );
287            let permit_wait_start = Instant::now();
288            let permit = instrument_named!(
289                acquire_run_permit(&cache.scheduler, &run.entries),
290                "range_cache_fetch_permit_wait"
291            )
292            .await;
293            // The run's effective priority, resolved once the permit is
294            // acquired so a promotion racing the wait is reflected: used
295            // to tag every latency/hit-rate signal this run emits.
296            let class = effective_priority(&run.entries).class_label();
297            let run_class_tag = if class == metric_tags::CLASS_DEMAND {
298                run_demand_tag
299            } else {
300                run_prefetch_tag
301            };
302            fmetric!(
303                "range_cache_fetch_permit_wait_ms",
304                "ms",
305                metric_tags::class_tags(class),
306                permit_wait_start.elapsed().as_secs_f64() * 1000.0
307            );
308
309            let byte_start = run.range.start * block_size;
310            let byte_end = block_byte_range(run.range.end - 1, block_size, file_size).end;
311            let path = Path::from(key_owned.as_str());
312            let origin_get_start = Instant::now();
313            let outcome = instrument_named!(
314                cache.origin.get_range(&path, byte_start..byte_end),
315                "range_cache_origin_get"
316            )
317            .await;
318            fmetric!(
319                "range_cache_origin_get_ms",
320                "ms",
321                metric_tags::class_tags(class),
322                origin_get_start.elapsed().as_secs_f64() * 1000.0
323            );
324            drop(permit);
325
326            match outcome {
327                Ok(data) => {
328                    cache
329                        .fulfill_run_success(
330                            &run,
331                            data,
332                            byte_start..byte_end,
333                            &key_owned,
334                            run_class_tag,
335                            hint,
336                        )
337                        .await;
338                }
339                Err(e) => {
340                    let shared: Arc<anyhow::Error> = Arc::new(anyhow::Error::from(e));
341                    for entry in &run.entries {
342                        entry.fulfill(Err(shared.clone()));
343                    }
344                }
345            }
346            for k in &run.keys {
347                cache.scheduler.remove_entry(k);
348            }
349            guard.disarm();
350        });
351    }
352
353    /// Success-path tail of `spawn_run_fetch`'s per-run task: validate the
354    /// origin GET's length against the run's expected byte span (a short read
355    /// is an explicit error, never silently clipped — see the inline comment
356    /// below), then split the buffer into per-block chunks, writing each to the
357    /// backend and fulfilling its entry.
358    async fn fulfill_run_success(
359        &self,
360        run: &CoalescedRun,
361        data: Bytes,
362        byte_range: Range<u64>,
363        key_owned: &str,
364        run_class_tag: &'static PropertySet,
365        hint: FillHint,
366    ) {
367        let block_size = self.block_size;
368        let range = &run.range;
369        let run_len = (range.end - range.start) as usize;
370        // The backend-hit path above heals a length mismatch (an undersized
371        // cached entry, or the origin object having changed size) by treating
372        // it as a miss and refetching. A true origin fetch has nowhere further
373        // to fall back to, so a short read here — which `object_store`'s
374        // `GetRange::Bounded` returns without error when the object is shorter
375        // than requested — must surface as an explicit error instead of being
376        // silently clipped by `assemble_range`, which would otherwise
377        // under-yield bytes and either corrupt a `Content-Length`-declared
378        // response or trip `frame_ranges_stream`'s under-yield `.expect()`.
379        let expected_run_bytes = byte_range.end - byte_range.start;
380        if data.len() as u64 != expected_run_bytes {
381            imetric!("range_cache_origin_run_len_mismatch", "count", 1_u64);
382            warn!(
383                "range_cache origin fetch length mismatch key={key_owned} \
384                 run={range:?} expected={expected_run_bytes} got={}",
385                data.len()
386            );
387            let shared: Arc<anyhow::Error> = Arc::new(anyhow!(
388                "origin object changed size mid-fetch: key={key_owned} \
389                 run={range:?} expected {expected_run_bytes} bytes, got {} bytes",
390                data.len()
391            ));
392            for entry in &run.entries {
393                entry.fulfill(Err(shared.clone()));
394            }
395        } else {
396            imetric!(
397                "range_cache_origin_block_fetch",
398                "count",
399                run_class_tag,
400                run_len as u64
401            );
402            imetric!(
403                "range_cache_origin_block_bytes",
404                "bytes",
405                run_class_tag,
406                data.len() as u64
407            );
408            debug!(
409                "range_cache origin fetch key={key_owned} run={range:?} bytes={}",
410                data.len()
411            );
412            for i in 0..run_len {
413                let offset = i as u64 * block_size;
414                let local_start = offset as usize;
415                let local_end = (offset + block_size).min(data.len() as u64) as usize;
416                let chunk = data.slice(local_start..local_end);
417                self.backend
418                    .put(run.keys[i].clone(), chunk.clone(), hint)
419                    .await;
420                run.entries[i].fulfill(Ok(chunk));
421            }
422        }
423    }
424}

426/// Phase 4, demand path: join every entry in index order, returning the
427/// populated hits map (backend hits from phase 1 plus the freshly-fetched
428/// missing blocks).
429async fn join_demand(
430    entries: HashMap<u64, Arc<InFlight>>,
431    missing: &[u64],
432    mut hits: HashMap<u64, Bytes>,
433) -> Result<HashMap<u64, Bytes>> {
434    let joined = join_all(missing.iter().map(|idx| {
435        let entry = entries
436            .get(idx)
437            .expect("entry present for every missing index")
438            .clone();
439        let idx = *idx;
440        async move { (idx, entry.join().await) }
441    }))
442    .await;
443
444    for (idx, r) in joined {
445        let data = r.map_err(|e| reconstruct_shared_error(&e))?;
446        hits.insert(idx, data);
447    }
448    Ok(hits)
449}

451/// Phase 4, prefetch path: join entries as each completes (not in index
452/// order), dropping the joined bytes and the `Arc<InFlight>` right away. The
453/// watch channel inside an entry retains the fulfilled `Bytes` — a slice
454/// sharing its whole run's GET buffer — for as long as the entry is alive,
455/// so holding every entry until all runs finished (as the demand path's
456/// `join_all` does) would let the prefetch peak reach the full request size
457/// instead of the documented `prefetch_concurrency * max_coalesced_get_bytes`
458/// bound. Any demand joiner that registered before completion still gets the
459/// bytes: it holds its own `Arc<InFlight>` and reads the channel
460/// independently of this early drop.
461async fn join_prefetch(entries: HashMap<u64, Arc<InFlight>>) -> Result<()> {
462    let mut joins: FuturesUnordered<_> = entries
463        .into_values()
464        .map(|entry| async move { entry.join().await.map(|_bytes| ()) })
465        .collect();
466    while let Some(r) = joins.next().await {
467        r.map_err(|e| reconstruct_shared_error(&e))?;
468    }
469    Ok(())
