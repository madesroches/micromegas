44/// Relative urgency of an origin fetch. Lower is more urgent.
45#[derive(Copy, Clone, Debug, PartialEq, Eq)]

47enum Priority {
48    Demand = 0,
49    Prefetch = 1,
50}

52impl Priority {
53    fn from_u8(v: u8) -> Self {
54        if v == Priority::Demand as u8 {

57            Priority::Prefetch

62/// The set of entries submitted in one prefetch call. Used only when
63/// `promote_whole_batch` is enabled: a demand joiner into any sibling entry
64/// promotes every other entry in the batch, not just the one it joined.
65///
66/// Siblings are tracked by `Weak<InFlight>`, captured once each entry is
67/// created or joined, rather than by key string: a key can be removed from
68/// `FetchScheduler::inflight` and later reused by an unrelated fetch (e.g.
69/// after eviction), and promoting by key alone would then spuriously promote
70/// that new, logically unrelated `InFlight`.
71struct BatchState {
72    entries: StdMutex<Vec<Weak<InFlight>>>,
73}

77/// One outstanding origin fetch (a single block or a `size()` head), shared
78/// across every concurrent caller asking for the same key.
79struct InFlight {
80    priority: AtomicU8,
81    promote: Notify,
82    result: watch::Sender<Option<FetchResult>>,
83    batch: Option<Arc<BatchState>>,
84    /// Set by the first `fulfill()` call. Lets a `FulfillGuard`'s panic-path
85    /// fallback tell which entries in a partially-completed run already got
86    /// a real result, so it never clobbers one with a synthesized error.
87    fulfilled: std::sync::atomic::AtomicBool,
88}
89
90impl InFlight {
91    fn new(priority: Priority, batch: Option<Arc<BatchState>>) -> Self {
92        let (result, _rx) = watch::channel(None);
93        Self {
94            priority: AtomicU8::new(priority as u8),
95            promote: Notify::new(),
96            result,
97            batch,
98            fulfilled: std::sync::atomic::AtomicBool::new(false),
99        }
100    }
101
102    fn priority(&self) -> Priority {
103        Priority::from_u8(self.priority.load(Ordering::Acquire))
104    }
105
106    fn promote_to_demand(&self) {
107        self.priority
108            .store(Priority::Demand as u8, Ordering::Release);
109        self.promote.notify_one();
110    }
111
112    /// Deliver `result` to every joiner. Idempotent: only the first call
113    /// actually sends — later calls (notably a `FulfillGuard`'s panic-path
114    /// fallback racing a fetch that in fact completed normally) are no-ops,
115    /// so a real result is never overwritten by a synthesized error.
116    fn fulfill(&self, result: FetchResult) {
117        if self.fulfilled.swap(true, Ordering::AcqRel) {
118            return;
119        }
120        // A send error just means every waiter already dropped its
121        // receiver, which is harmless.
122        let _ = self.result.send(Some(result));
123    }
124
125    /// Wait for the fetch to complete, returning immediately if it already
126    /// has. Safe against the "subscribe after send" race: a fresh `watch`
127    /// receiver's initial value counts as unseen until `borrow_and_update`
128    /// checks it, so a result sent before this call is still observed here.
129    async fn join(&self) -> FetchResult {
130        let mut rx = self.result.subscribe();
131        loop {
132            if let Some(r) = rx.borrow_and_update().clone() {
133                return r;
134            }
135            if rx.changed().await.is_err() {
136                return Err(Arc::new(anyhow!(
137                    "in-flight fetch entry dropped without a result"
138                )));
139            }
140        }
141    }
142}

149/// Owns the in-flight single-flight map and the priority-aware origin-fetch
150/// budget. Replaces the two moka caches that used to live directly on
151/// `RangeCache`.
152struct FetchScheduler {
153    inflight: StdMutex<HashMap<String, Arc<InFlight>>>,
154    /// Bounds total concurrent origin GETs (blocks + size heads don't count
155    /// against this; only run GETs acquire it).
156    shared_permits: Arc<Semaphore>,
157    /// Prefetch runs must additionally hold one of these; sized to
158    /// `total - demand_reserved` so demand always finds a free shared permit.
159    prefetch_permits: Arc<Semaphore>,
160    promote_whole_batch: bool,
161}
162
163impl FetchScheduler {
164    fn new(total: usize, demand_reserved: usize, promote_whole_batch: bool) -> Self {
165        assert!(total > 0, "fetch concurrency total must be > 0");
166        // Strictly less: `demand_reserved == total` would leave the prefetch
167        // semaphore with zero permits, hanging every prefetch run forever.
168        assert!(
169            demand_reserved < total,
170            "demand_reserved ({demand_reserved}) must be < total ({total})"
171        );
172        Self {
173            inflight: StdMutex::new(HashMap::new()),
174            shared_permits: Arc::new(Semaphore::new(total)),
175            prefetch_permits: Arc::new(Semaphore::new(total - demand_reserved)),
176            promote_whole_batch,
177        }
178    }
179
180    /// Look up `key` in the in-flight map: become the owner if absent, or a
181    /// joiner if present. A demand joiner into a prefetch-priority entry
182    /// promotes it (and, if `promote_whole_batch`, its batch siblings) to
183    /// demand so it competes for reserved capacity instead of sitting behind
184    /// other prefetch work.
185    ///
186    /// The entry is registered in `batch` (owner or joiner alike) while the
187    /// inflight lock is still held, i.e. before any concurrent demand joiner
188    /// can find the entry and run `promote_batch_siblings` over a
189    /// partially-populated list.
190    fn own_or_join(
191        &self,
192        key: String,
193        prio: Priority,
194        batch: Option<Arc<BatchState>>,
195    ) -> Ownership {
196        let mut promote_batch: Option<(Arc<BatchState>, Arc<InFlight>)> = None;
197        let ownership = {
198            let mut map = self.inflight.lock().expect("inflight lock");
199            if let Some(existing) = map.get(&key) {
200                if prio == Priority::Demand && existing.priority() == Priority::Prefetch {
201                    existing.promote_to_demand();
202                    if self.promote_whole_batch
203                        && let Some(bs) = existing.batch.clone()
204                    {
205                        promote_batch = Some((bs, existing.clone()));
206                    }
207                }
208                if let Some(bs) = &batch {
209                    bs.entries
210                        .lock()
211                        .expect("batch lock")
212                        .push(Arc::downgrade(existing));
213                }
214                Ownership::Joiner(existing.clone())
215            } else {
216                let entry = Arc::new(InFlight::new(prio, batch.clone()));
217                if let Some(bs) = &batch {
218                    bs.entries
219                        .lock()
220                        .expect("batch lock")
221                        .push(Arc::downgrade(&entry));
222                }
223                map.insert(key.clone(), entry.clone());
224                Ownership::Owner(entry)
225            }
226        };
227        if let Some((bs, skip)) = promote_batch {
228            self.promote_batch_siblings(&bs, &skip);
229        }
230        ownership
231    }
232
233    /// Promote every sibling in `batch` other than `skip` to demand priority.
234    /// Siblings are resolved through `Weak<InFlight>` references captured at
235    /// batch-membership time, not by re-looking up their key in `inflight`:
236    /// the key may since have been removed and reused by an unrelated fetch,
237    /// and identity (not the key string) is what must match here.
238    fn promote_batch_siblings(&self, batch: &BatchState, skip: &Arc<InFlight>) {
239        let entries = batch.entries.lock().expect("batch lock");
240        for weak in entries.iter() {
241            if let Some(entry) = weak.upgrade()
242                && !Arc::ptr_eq(&entry, skip)
243            {
244                entry.promote_to_demand();
245            }
246        }
247    }
248
249    fn remove_entry(&self, key: &str) {
250        self.inflight.lock().expect("inflight lock").remove(key);
251    }
252}

254/// Scope guard held by the task that owns one or more in-flight entries
255/// (either a single `size()` head or the members of one coalesced block
256/// run). If the owning task exits normally, it calls `disarm()` after
257/// fulfilling every entry and removing it from `FetchScheduler::inflight`.
258///
259/// If instead the task panics (e.g. `Bytes::slice` on a short origin read)
260/// tokio catches the unwind at the task boundary and silently drops the
261/// result, but this guard's `Drop` still runs during that unwind: it
262/// fulfills any not-yet-fulfilled entry with an error (the `fulfilled` flag
263/// on `InFlight` makes this a no-op for entries that already got a real
264/// result) and removes every entry from the map. Without this, a panicking
265/// owner would leave `fulfill()` never called, hanging every joiner
266/// (including the owner itself) forever and leaking the entry permanently.
267struct FulfillGuard {
268    scheduler: Arc<FetchScheduler>,
269    entries: Vec<(String, Arc<InFlight>)>,
270    armed: bool,
271}
272
273impl FulfillGuard {
274    fn new(scheduler: Arc<FetchScheduler>, entries: Vec<(String, Arc<InFlight>)>) -> Self {
275        Self {
276            scheduler,
277            entries,
278            armed: true,
279        }
280    }
281
282    /// Call once the normal completion path has fulfilled and removed every
283    /// entry, so the guard's `Drop` becomes a no-op.
284    fn disarm(mut self) {
285        self.armed = false;
286    }
287}
288
289impl Drop for FulfillGuard {
290    fn drop(&mut self) {
291        if !self.armed {
292            return;
293        }
294        warn!(
295            "fetch task exited without completing normally (likely a panic); \
296             fulfilling {} in-flight entries with an error",
297            self.entries.len()
298        );
299        for (key, entry) in &self.entries {
300            entry.fulfill(Err(Arc::new(anyhow!(
301                "fetch task exited without producing a result (panic during fetch)"
302            ))));
303            self.scheduler.remove_entry(key);
304        }
305    }
306}

308/// Held for the duration of one origin GET; dropping it frees the slot(s) for
309/// the next waiter. Fields are never read, only held for their `Drop` effect.
310struct RunPermit {
311    _shared: OwnedSemaphorePermit,
312    _prefetch: Option<OwnedSemaphorePermit>,
313}

315/// Resolves when any of `entries`' priority has been promoted since this call
316/// started. Constructing the `Notified` future(s) up front (before the caller
317/// re-checks priority) is what makes this race-free: tokio guarantees a
318/// `notify_one()` that happens after a `Notified` is created is never missed,
319/// even if that `Notified` is later dropped without completing.
320async fn any_entry_promoted(entries: &[Arc<InFlight>]) {
321    match entries {
322        [] => std::future::pending::<()>().await,
323        [only] => only.promote.notified().await,
324        many => {
325            let futs: Vec<_> = many
326                .iter()
327                .map(|e| Box::pin(e.promote.notified()))
328                .collect();
329            futures::future::select_all(futs).await;
330        }
331    }
332}

334/// Acquire the permit(s) needed to run one coalesced GET covering `entries`,
335/// honoring promotion: the run's effective priority is the most urgent
336/// (minimum) of its entries', re-checked every time a promotion wakes this
337/// loop so a promotion mid-wait drops the prefetch-class requirement.
338async fn acquire_run_permit(scheduler: &FetchScheduler, entries: &[Arc<InFlight>]) -> RunPermit {
339    loop {
340        let all_prefetch = entries.iter().all(|e| e.priority() == Priority::Prefetch);
341        if !all_prefetch {
342            let shared = scheduler
343                .shared_permits
344                .clone()
345                .acquire_owned()
346                .await
347                .expect("shared_permits semaphore is never closed");
348            return RunPermit {
349                _shared: shared,
350                _prefetch: None,
351            };
352        }
353
354        tokio::select! {
355            prefetch = scheduler.prefetch_permits.clone().acquire_owned() => {
356                let prefetch = prefetch.expect("prefetch_permits semaphore is never closed");
357                tokio::select! {
358                    shared = scheduler.shared_permits.clone().acquire_owned() => {
359                        let shared = shared.expect("shared_permits semaphore is never closed");
360                        return RunPermit { _shared: shared, _prefetch: Some(prefetch) };
361                    }
362                    _ = any_entry_promoted(entries) => {
363                        drop(prefetch);
364                        continue;
365                    }
366                }
367            }
368            _ = any_entry_promoted(entries) => {
369                continue;
370            }
371        }
372    }
373}

375/// Reconstruct an owned error from a shared in-flight failure. `anyhow::Error`
376/// is not `Clone`, so a joiner cannot move it out of the `Arc`; and it must
377/// not be stringified, or a missing key would stop downcasting to
378/// `object_store::Error::NotFound` and would surface as a 500 instead of a
379/// 404 (see `validation::is_not_found`).
380fn reconstruct_shared_error(shared: &Arc<anyhow::Error>) -> anyhow::Error {
381    if let Some(object_store::Error::NotFound { path, source }) =
382        shared.downcast_ref::<object_store::Error>()
383    {
384        let rebuilt = object_store::Error::NotFound {
385            path: path.clone(),
386            source: Box::new(std::io::Error::new(
387                std::io::ErrorKind::NotFound,
388                source.to_string(),
389            )),
390        };
391        return anyhow::Error::from(rebuilt);
392    }
393    // `{shared:?}` keeps the full context chain (and backtrace, if captured)
394    // so joiners' 500 log lines are as informative as the owner's.
395    anyhow!("{shared:?}")
396}

404/// Range-aware read cache over an origin object store.
405///
406/// # Cache invalidation
407///
408/// This cache assumes object keys are **write-once and content-addressed**: a
409/// given key always maps to the same bytes for the lifetime of the object. The
410/// size and block caches therefore carry no TTL, etag, or generation in their
411/// keys and are never invalidated. Overwriting an existing key with different
412/// content would cause stale size/block data to be served indefinitely. This is
413/// safe for micromegas lake objects (blocks, parquet) which are never
414/// overwritten; do not point this cache at a mutable namespace.
415///
416/// # In-flight map and priority
417///
418/// Concurrent fetches of the same block or size are collapsed via an
419/// in-flight map (`FetchScheduler`): the first caller to ask for a key
420/// becomes its owner and issues the origin request (spawned as a detached
421/// task, so a cancelled caller never strands the others waiting on it);
422/// every other concurrent caller joins and observes the same result.
423/// Contiguous missing blocks the owner controls are coalesced into one
424/// `origin.get_range` per run. Every origin GET is either `Demand` or
425/// `Prefetch` priority; a demand caller joining a prefetch-priority fetch
426/// promotes it (see `own_or_join`), so a late demand read is never stuck
427/// behind unrelated prefetch traffic.
428#[derive(Clone)]
429pub struct RangeCache {
430    origin: Arc<dyn ObjectStore>,
431    backend: Arc<dyn RangeCacheBackend>,
432    block_size: u64,
433    ns: String,
434    scheduler: Arc<FetchScheduler>,
435    max_coalesced_get_bytes: u64,
436}
437

464    /// Size in bytes of one cache block. Every distinct block a request
465    /// touches is fetched and held whole, so callers gating memory (e.g. the
466    /// server's cross-request budget) need this to account for amplification
467    /// from small scattered ranges.
468    pub fn block_size(&self) -> u64 {
469        self.block_size
470    }
471
472    #[span_fn]
473    pub async fn size(&self, key: &str) -> Result<u64> {
474        // The cache key carries no etag/version: see the module docs — keys
475        // are assumed write-once and content-addressed, so a cached size is
476        // never invalidated.
477        let meta_key = format!("meta:{}:{key}", self.ns);
478
479        if let Some(data) = self.backend.get(&meta_key).await
480            && data.len() == 8
481        {
482            imetric!("range_cache_size_backend_hit", "count", 1_u64);
483            return decode_size(&data);
484        }
485
486        match self
487            .scheduler
488            .own_or_join(meta_key.clone(), Priority::Demand, None)
489        {
490            Ownership::Owner(entry) => {
491                let origin = self.origin.clone();
492                let backend = self.backend.clone();
493                let scheduler = self.scheduler.clone();
494                let key_owned = key.to_string();
495                let meta_key_owned = meta_key.clone();
496                let task_entry = entry.clone();
497                tokio::spawn(async move {
498                    let guard = FulfillGuard::new(
499                        scheduler.clone(),
500                        vec![(meta_key_owned.clone(), task_entry.clone())],
501                    );
502                    imetric!("range_cache_origin_head", "count", 1_u64);
503                    match origin.head(&Path::from(key_owned.as_str())).await {
504                        Ok(object_meta) => {
505                            let size = object_meta.size;
506                            debug!("range_cache origin head key={key_owned} size={size}");
507                            let size_bytes = Bytes::from(size.to_le_bytes().to_vec());
508                            backend
509                                .put(meta_key_owned.clone(), size_bytes.clone(), FillHint::Demand)
510                                .await;
511                            task_entry.fulfill(Ok(size_bytes));
512                        }
513                        Err(e) => {
514                            task_entry.fulfill(Err(Arc::new(anyhow::Error::from(e))));
515                        }
516                    }
517                    scheduler.remove_entry(&meta_key_owned);
518                    guard.disarm();
519                });
520                let data = entry
521                    .join()
522                    .await
523                    .map_err(|e| reconstruct_shared_error(&e))?;
524                decode_size(&data)
525            }
526            Ownership::Joiner(entry) => {
527                let data = entry
528                    .join()
529                    .await
530                    .map_err(|e| reconstruct_shared_error(&e))?;
531                decode_size(&data)
532            }
533        }
534    }

536    /// Fetch `indices` (block indices within `key`, sized `file_size`) at the
537    /// given priority, returning the bytes for every requested block on the
538    /// `Demand` path. On the `Prefetch` path bytes are written to the backend
539    /// and dropped as each owned run completes; the returned map is always
540    /// empty, which is what keeps the prefetch peak bounded by
541    /// `prefetch_concurrency * max_coalesced_get_bytes` rather than the full
542    /// request size.
543    async fn fetch_blocks(
544        &self,
545        key: &str,
546        file_size: u64,
547        indices: &[u64],
548        prio: Priority,
549    ) -> Result<HashMap<u64, Bytes>> {
550        if indices.is_empty() {
551            return Ok(HashMap::new());
552        }
553
554        let mut hits: HashMap<u64, Bytes> = HashMap::new();
555        let mut missing: Vec<u64> = Vec::new();
556        {
557            // Probe the backend with bounded concurrency: a foyer RAM-tier
558            // miss falls through to async disk I/O, and probing a large read
559            // one block at a time would serialize those disk reads. The
560            // futures are collected eagerly (not mapped lazily inside the
561            // stream) so the resulting future stays `Send`-inferable across
562            // `tokio::spawn` (rustc's HRTB limitation with borrowed closures).
563            let probe_futs: Vec<_> = indices
564                .iter()
565                .map(|&idx| {
566                    let block_key = format!("blk:{}:{key}:{idx}", self.ns);
567                    let backend = self.backend.clone();
568                    async move {
569                        imetric!("range_cache_block_request", "count", 1_u64);
570                        (idx, backend.get(&block_key).await)
571                    }
572                })
573                .collect();
574            let mut probes = stream::iter(probe_futs).buffer_unordered(BACKEND_PROBE_CONCURRENCY);
575            while let Some((idx, probe)) = probes.next().await {
576                match probe {
577                    Some(data) => {
578                        imetric!("range_cache_block_backend_hit", "count", 1_u64);
579                        // A prefetch has nothing to do with an already-cached
580                        // block, so drop the bytes immediately instead of
581                        // accumulating every hit for the whole call.
582                        if prio == Priority::Demand {
583                            hits.insert(idx, data);
584                        }
585                    }
586                    None => missing.push(idx),
587                }
588            }
589        }
590
591        if missing.is_empty() {
592            return Ok(hits);
593        }
594        missing.sort_unstable();
595        missing.dedup();
596
597        // Only prefetch calls carry a batch handle: it's what lets a demand
598        // joiner promote every sibling of a scattered prefetch call, not just
599        // the one block it happened to touch (see `promote_whole_batch`).
600        let batch = if prio == Priority::Prefetch {
601            Some(Arc::new(BatchState {
602                entries: StdMutex::new(Vec::with_capacity(missing.len())),
603            }))
604        } else {
605            None
606        };
607
608        let mut owned: Vec<u64> = Vec::new();
609        let mut entries: HashMap<u64, Arc<InFlight>> = HashMap::with_capacity(missing.len());
610        for &idx in &missing {
611            let block_key = format!("blk:{}:{key}:{idx}", self.ns);
612            match self.scheduler.own_or_join(block_key, prio, batch.clone()) {
613                Ownership::Owner(entry) => {
614                    owned.push(idx);
615                    entries.insert(idx, entry);
616                }
617                Ownership::Joiner(entry) => {
618                    entries.insert(idx, entry);
619                }
620            }
621        }
622        // Every member of this batch (owned or joined) was recorded as a
623        // promotion sibling inside `own_or_join`, under the inflight lock,
624        // so no entry became joinable before the batch list held it.
625
626        let hint = match prio {
627            Priority::Demand => FillHint::Demand,
628            Priority::Prefetch => FillHint::Prefetch,
629        };
630
631        // `owned` is a sorted subsequence of `missing` (itself sorted), so no
632        // extra sort is needed before coalescing.
633        for run in coalesce_runs(&owned, self.block_size, self.max_coalesced_get_bytes) {
634            let run_len = (run.end - run.start) as usize;
635            let run_entries: Vec<Arc<InFlight>> = (run.start..run.end)
636                .map(|idx| entries.get(&idx).expect("owned entry present").clone())
637                .collect();
638            let run_keys: Vec<String> = (run.start..run.end)
639                .map(|idx| format!("blk:{}:{key}:{idx}", self.ns))
640                .collect();
641            let origin = self.origin.clone();
642            let backend = self.backend.clone();
643            let scheduler = self.scheduler.clone();
644            let block_size = self.block_size;
645            let key_owned = key.to_string();
646
647            // Spawned as a detached task so a cancelled caller (e.g. an axum
648            // request future dropped on client disconnect) can never strand
649            // the other owned blocks or any joiners waiting on this run.
650            tokio::spawn(async move {
651                let guard = FulfillGuard::new(
652                    scheduler.clone(),
653                    run_keys
654                        .iter()
655                        .cloned()
656                        .zip(run_entries.iter().cloned())
657                        .collect(),
658                );
659                let permit = acquire_run_permit(&scheduler, &run_entries).await;
660                let byte_start = run.start * block_size;
661                let byte_end = block_byte_range(run.end - 1, block_size, file_size).end;
662                let path = Path::from(key_owned.as_str());
663                let outcome = origin.get_range(&path, byte_start..byte_end).await;
664                drop(permit);
665
666                match outcome {
667                    Ok(data) => {
668                        imetric!("range_cache_origin_block_fetch", "count", run_len as u64);
669                        imetric!("range_cache_origin_block_bytes", "bytes", data.len() as u64);
670                        debug!(
671                            "range_cache origin fetch key={key_owned} run={run:?} bytes={}",
672                            data.len()
673                        );
674                        for i in 0..run_len {
675                            let offset = i as u64 * block_size;
676                            let local_start = offset as usize;
677                            let local_end = (offset + block_size).min(data.len() as u64) as usize;
678                            let chunk = data.slice(local_start..local_end);
679                            backend.put(run_keys[i].clone(), chunk.clone(), hint).await;
680                            run_entries[i].fulfill(Ok(chunk));
681                        }
682                    }
683                    Err(e) => {
684                        let shared: Arc<anyhow::Error> = Arc::new(anyhow::Error::from(e));
685                        for entry in &run_entries {
686                            entry.fulfill(Err(shared.clone()));
687                        }
688                    }
689                }
690                for k in &run_keys {
691                    scheduler.remove_entry(k);
692                }
693                guard.disarm();
694            });
695        }
696
697        if prio == Priority::Prefetch {
698            // Join entries as each completes (not in index order), dropping
699            // the joined bytes and the `Arc<InFlight>` right away. The watch
700            // channel inside an entry retains the fulfilled `Bytes` — a slice
701            // sharing its whole run's GET buffer — for as long as the entry
702            // is alive, so holding every entry until all runs finished (as
703            // the demand path's `join_all` does) would let the prefetch peak
704            // reach the full request size instead of the documented
705            // `prefetch_concurrency * max_coalesced_get_bytes` bound. Any
706            // demand joiner that registered before completion still gets the
707            // bytes: it holds its own `Arc<InFlight>` and reads the channel
708            // independently of this early drop.
709            let mut joins: FuturesUnordered<_> = entries
710                .into_values()
711                .map(|entry| async move { entry.join().await.map(|_bytes| ()) })
712                .collect();
713            while let Some(r) = joins.next().await {
714                r.map_err(|e| reconstruct_shared_error(&e))?;
715            }
716            return Ok(HashMap::new());
717        }
718
719        let joined = join_all(missing.iter().map(|idx| {
720            let entry = entries
721                .get(idx)
722                .expect("entry present for every missing index")
723                .clone();
724            let idx = *idx;
725            async move { (idx, entry.join().await) }
726        }))
727        .await;
728
729        for (idx, r) in joined {
730            let data = r.map_err(|e| reconstruct_shared_error(&e))?;
731            hits.insert(idx, data);
732        }
733        Ok(hits)
734    }

736    #[span_fn]
737    pub async fn get_range(&self, key: &str, range: Range<u64>) -> Result<Bytes> {
738        let file_size = match self.size(key).await {
739            Ok(s) => s,
740            Err(e) => {
741                imetric!("range_cache_get_range_error", "count", 1_u64);
742                return Err(e);
743            }
744        };
745
746        let start = range.start;
747        let end = range.end;
748
749        if end > file_size {
750            imetric!("range_cache_get_range_error", "count", 1_u64);
751            return Err(RangeError::OutOfBounds {
752                requested_end: end,
753                file_size,
754            }
755            .into());
756        }
757
758        if start >= end {
759            return Ok(Bytes::new());
760        }
761
762        let blk_indices = blocks_for_range(start, end, self.block_size);
763        let indices: Vec<u64> = (blk_indices.start..blk_indices.end).collect();
764
765        let block_map = match self
766            .fetch_blocks(key, file_size, &indices, Priority::Demand)
767            .await
768        {
769            Ok(m) => m,
770            Err(e) => {
771                imetric!("range_cache_get_range_error", "count", 1_u64);
772                return Err(e);
773            }
774        };
775
776        let blocks: Vec<(u64, Bytes)> = indices
777            .into_iter()
778            .map(|idx| {
779                let data = block_map
780                    .get(&idx)
781                    .cloned()
782                    .expect("fetch_blocks returns every requested index");
783                (idx, data)
784            })
785            .collect();
786        Ok(assemble_range(&blocks, self.block_size, start, end))
787    }

789    #[span_fn]
790    pub async fn get_ranges(&self, key: &str, ranges: &[Range<u64>]) -> Result<Vec<Bytes>> {
791        if ranges.is_empty() {
792            return Ok(vec![]);
793        }
794
795        // Propagate the size-lookup error unwrapped so the underlying
796        // `object_store::Error` (notably `NotFound`) survives the downcast in
797        // callers, matching `get_range` and the single-GET endpoint.
798        let file_size = match self.size(key).await {
799            Ok(s) => s,
800            Err(e) => {
801                imetric!("range_cache_get_ranges_error", "count", 1_u64);
802                return Err(e);
803            }
804        };
805
806        let mut all_block_indices = BTreeSet::new();
807        for r in ranges {
808            let start = r.start;
809            let end = r.end;
810            if end > file_size {
811                imetric!("range_cache_get_ranges_error", "count", 1_u64);
812                return Err(RangeError::OutOfBounds {
813                    requested_end: end,
814                    file_size,
815                }
816                .into());
817            }
818            if start < end {
819                let blk = blocks_for_range(start, end, self.block_size);
820                for idx in blk.start..blk.end {
821                    all_block_indices.insert(idx);
822                }
823            }
824        }
825
826        let indices: Vec<u64> = all_block_indices.into_iter().collect();
827        let block_map = match self
828            .fetch_blocks(key, file_size, &indices, Priority::Demand)
829            .await
830        {
831            Ok(m) => m,
832            Err(e) => {
833                imetric!("range_cache_get_ranges_error", "count", 1_u64);
834                return Err(e);
835            }
836        };
837
838        let mut result = Vec::with_capacity(ranges.len());
839        for r in ranges {
840            let start = r.start;
841            let end = r.end;
842            if start >= end {
843                result.push(Bytes::new());
844                continue;
845            }
846            let blk = blocks_for_range(start, end, self.block_size);
847            let blocks: Vec<(u64, Bytes)> = (blk.start..blk.end)
848                .filter_map(|idx| block_map.get(&idx).map(|d| (idx, d.clone())))
849                .collect();
850            result.push(assemble_range(&blocks, self.block_size, start, end));
851        }
852
853        Ok(result)
854    }

856    /// Warm the cache for `ranges` at `Prefetch` priority without returning
857    /// any bytes. The HTTP surface for this (endpoint + client method) is
858    /// #1198; this is the priority-carrying core it builds on. Public (rather
859    /// than crate-private) so integration tests under `tests/` — which
860    /// compile as a separate crate — can exercise the promotion behavior
861    /// described in the fetch-rework plan.
862    pub async fn prefetch_ranges(&self, key: &str, ranges: &[Range<u64>]) -> Result<()> {
863        if ranges.is_empty() {
864            return Ok(());
865        }
866        let file_size = self.size(key).await?;
867        let mut all_block_indices = BTreeSet::new();
868        for r in ranges {
869            let start = r.start;
870            let end = r.end;
871            if end > file_size {
872                return Err(RangeError::OutOfBounds {
873                    requested_end: end,
874                    file_size,
875                }
876                .into());
877            }
878            if start < end {
879                let blk = blocks_for_range(start, end, self.block_size);
880                all_block_indices.extend(blk.start..blk.end);
881            }
882        }
883        self.prefetch_blocks(
884            key,
885            file_size,
886            &all_block_indices.into_iter().collect::<Vec<_>>(),
887        )
888        .await
889    }

891    /// Warm the cache for the given block indices at `Prefetch` priority.
892    pub async fn prefetch_blocks(&self, key: &str, file_size: u64, indices: &[u64]) -> Result<()> {
893        self.fetch_blocks(key, file_size, indices, Priority::Prefetch)
894            .await?;
895        Ok(())
896    }
