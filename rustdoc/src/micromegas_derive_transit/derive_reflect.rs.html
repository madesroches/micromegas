6fn metadata_from_type(t: &syn::Type) -> (QuoteRes, bool) {
7    match t {
8        syn::Type::Array(_) => panic!("Array field type not supported"),
9        syn::Type::BareFn(fun) => (quote! {#fun}, true),
10        syn::Type::Group(_) => panic!("Group field type not supported"),
11        syn::Type::ImplTrait(_) => panic!("ImplTrait field type not supported"),
12        syn::Type::Infer(_) => panic!("Infer field type not supported"),
13        syn::Type::Macro(_) => panic!("Macro field type not supported"),
14        syn::Type::Never(_) => panic!("Never field type not supported"),
15        syn::Type::Paren(_) => panic!("Paren field type not supported"),
16        syn::Type::Path(type_path) => (quote! {#type_path}, false),
17        syn::Type::Ptr(pointer_type) => (quote! {#pointer_type}, true),
18        syn::Type::Reference(reference) => (quote! {#reference}, true),
19        syn::Type::Slice(_) => panic!("Slice field type not supported"),
20        syn::Type::TraitObject(_) => panic!("TraitObject field type not supported"),
21        syn::Type::Tuple(_) => panic!("Tuple field type not supported"),
22        syn::Type::Verbatim(_) => panic!("Verbatim field type not supported"),
23        unknown_field_type => {
24            panic!("Unexpected field type: {unknown_field_type:?}")
25        }
26    }
27}

29pub fn derive_reflect_impl(input: TokenStream) -> TokenStream {
30    let ast = parse_macro_input!(input as DeriveInput);
31    let udt_identifier = ast.ident.clone();
32    let udt_name = ast.ident.to_string();
33    let mut members = Vec::new();
34
35    match ast.data {
36        syn::Data::Struct(s) => match s.fields {
37            syn::Fields::Named(named_fields) => {
38                for field in named_fields.named {
39                    let field_name = field.ident.unwrap().to_string();
40                    let (field_type, is_reference) = metadata_from_type(&field.ty);
41                    members.push((field_name, field_type, is_reference));
42                }
43            }
44            syn::Fields::Unnamed(_) => panic!("only named fields are supported"),
45            syn::Fields::Unit => panic!("unit fields not expected"),
46        },
47        syn::Data::Enum(_) => panic!("enums not supported"),
48        syn::Data::Union(_) => panic!("bunions not supported"),
49    }
50
51    let mut secondary_types = vec![];
52    for m in &members {
53        let member_type = &m.1;
54        let is_reference = &m.2;
55        let type_name = member_type.to_string();
56        if *is_reference
57            || type_name == "i64"
58            || type_name == "u64"
59            || type_name == "u32"
60            || type_name == "i32"
61            || type_name == "f64"
62        {
63            continue;
64        }
65        secondary_types.push(member_type.clone());
66    }
67
68    let secondary_types_toks = secondary_types.iter().map(|t| {
69        quote! {
70            #t::reflect(),
71        }
72    });
73
74    let members_toks = members.iter().map(|m| {
75        let member_name = &m.0;
76        let member_ident = format_ident!("{}", &m.0);
77        let member_type = &m.1;
78        let member_type_name = member_type.to_string();
79        let is_reference = &m.2;
80        quote! {
81            micromegas_transit::Member{ name: std::sync::Arc::new(String::from(#member_name)),
82                    type_name: String::from(#member_type_name),
83                    offset: memoffset::offset_of!(#udt_identifier,#member_ident),
84                    size: std::mem::size_of::<#member_type>(),
85                    is_reference: #is_reference,
86        },
87        }
88    });
89
90    TokenStream::from(quote! {
91        impl micromegas_transit::Reflect for #udt_identifier{
92            fn reflect() -> micromegas_transit::UserDefinedType{
93                micromegas_transit::UserDefinedType{
94                    name: std::sync::Arc::new(String::from(#udt_name)),
95                    size: std::mem::size_of::<#udt_identifier>(),
96                    members: vec![#(#members_toks)*],
97                    is_reference: false,
98                    secondary_udts: vec![#(#secondary_types_toks)*],
99                }
100            }
101        }
102    })
103}
