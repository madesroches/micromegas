9#[derive(Parser, Debug)]
10#[command(name = "telemetry-generator")]
11#[command(about = "Generates telemetry data including async spans")]
12struct Args {
13    /// Duration to run the generator in seconds

15    duration: u64,

17    /// Number of async tasks to spawn
18    #[arg(long, default_value = "5")]

23    threads: usize,

26// Sync span function example
27#[span_fn]
28fn sync_operation(value: i32) -> i32 {
29    info!("performing sync operation with value: {value}");
30    std::thread::sleep(Duration::from_millis(50));
31    value * 2
32}

34// Async span function example
35#[span_fn]
36async fn async_operation(delay_ms: u64) -> String {
37    info!("starting async operation with {delay_ms}ms delay");
38    sleep(Duration::from_millis(delay_ms)).await;
39    format!("completed after {delay_ms}ms")
40}

42// Nested async function
43#[span_fn]
44async fn nested_async_operation() {

46    info!("first async result: {result1}");
47
48    let result2 = async_operation(150).await;
49    info!("second async result: {result2}");

52// Manual span instrumentation example
53async fn manual_async_work() {
54    static_span_desc!(WORK_DESC, "manual_work");
55
56    async {
57        info!("doing manual async work");
58        sleep(Duration::from_millis(75)).await;
59        info!("manual work completed");
60    }
61    .instrument(&WORK_DESC)
62    .await;
63}

65// Multi-threaded async function that creates spans across different threads
66#[span_fn]
67async fn multi_threaded_async_work(task_id: usize) -> String {
68    info!("starting multi-threaded task {task_id}");
69
70    // Create nested async work that may be scheduled on different threads
71    let subtask1 = async_subtask(task_id, "validation").await;
72    let subtask2 = async_subtask(task_id, "computation").await;
73
74    // Simulate some async I/O that can cause thread migration
75    sleep(Duration::from_millis(30 + (task_id as u64 * 10))).await;
76
77    info!("multi-threaded task {task_id} completed");
78    format!("task_{task_id}_result: {subtask1} + {subtask2}")
79}

81// Async subtask that can be scheduled on any thread
82#[span_fn]
83async fn async_subtask(task_id: usize, operation: &str) -> String {
84    info!("executing subtask {operation} for task {task_id}");
85
86    // Variable delay to create different scheduling patterns
87    let delay = 20 + (task_id as u64 * 5);
88    sleep(Duration::from_millis(delay)).await;
89
90    format!("{operation}_{task_id}")
91}

93// Concurrent async task with potential for work stealing
94#[span_fn]
95async fn concurrent_async_task(operation: &str, base_delay: u64) -> String {
96    info!("starting concurrent operation: {operation}");
97
98    // Create parent-child async span relationship
99    let preparation = prepare_async_operation(operation).await;
100    let execution = execute_async_operation(operation, base_delay).await;
101    let cleanup = cleanup_async_operation(operation).await;
102
103    info!("concurrent operation {operation} completed");
104    format!("{operation}: {preparation} -> {execution} -> {cleanup}")
105}

107#[span_fn]
108async fn prepare_async_operation(operation: &str) -> String {
109    info!("preparing {operation}");
110    sleep(Duration::from_millis(15)).await;
111    format!("prepared_{operation}")
112}

114#[span_fn]
115async fn execute_async_operation(operation: &str, delay: u64) -> String {
116    info!("executing {operation} with {delay}ms delay");
117    sleep(Duration::from_millis(delay)).await;
118    format!("executed_{operation}")
119}

121#[span_fn]
122async fn cleanup_async_operation(operation: &str) -> String {
123    info!("cleaning up {operation}");

125    format!("cleaned_{operation}")
126}

128#[micromegas_main]
129async fn main() -> Result<(), Box<dyn std::error::Error>> {
130    let args = Args::parse();
131
132    info!(
133        "hello from generator - creating multi-threaded async spans (duration={}s, async_tasks={}, threads={})",
134        args.duration, args.async_tasks, args.threads
135    );
136
137    // Generate metrics
138    imetric!("Frame Time", "ticks", 1000);
139    fmetric!("Frame Time", "ticks", 1.0);
140    imetric!("Memory Usage", "bytes", 2048);
141    fmetric!("CPU Usage", "percent", 75.5);
142
143    // Generate sync span events
144    let sync_result = sync_operation(42);
145    info!("sync operation result: {sync_result}");
146
147    // Generate async span events
148    let async_result = async_operation(200).await;
149    info!("async operation result: {async_result}");
150
151    // Generate nested async spans
152    nested_async_operation().await;
153
154    // Generate manual instrumentation spans
155    manual_async_work().await;
156
157    // Generate multi-threaded async spans to create cross-stream async events
158    info!("starting multi-threaded async operations");
159
160    let mut handles = vec![];
161
162    // Create multiple async tasks that run on different threads (using args.async_tasks)
163    for i in 0..args.async_tasks {
164        let handle = tokio::spawn(async move { multi_threaded_async_work(i).await });
165        handles.push(handle);
166    }
167
168    // Create concurrent async tasks with work-stealing potential (using args.threads)
169    let operations = [
170        "database_query",
171        "api_call",
172        "file_processing",
173        "cache_lookup",
174        "network_request",
175        "disk_io",
176    ];
177    let concurrent_tasks: Vec<_> = (0..args.threads)
178        .map(|i| {
179            let op_name = operations[i % operations.len()];
180            let base_delay = 60 + (i as u64 * 20);
181            tokio::spawn(concurrent_async_task(op_name, base_delay))
182        })
183        .collect();
184
185    // Wait for all tasks to complete
186    for handle in handles {
187        if let Ok(result) = handle.await {
188            info!("multi-threaded task completed: {result}");
189        }
190    }
191
192    for task in concurrent_tasks {
193        if let Ok(result) = task.await {
194            info!("concurrent task result: {result}");
195        }
196    }
197
198    // Run for the specified duration
199    info!(
200        "running for {} seconds to generate continuous telemetry",
201        args.duration
202    );
203    tokio::time::sleep(Duration::from_secs(args.duration)).await;
204
205    info!("generator completed successfully with multi-threaded async spans");
206    Ok(())
207}
