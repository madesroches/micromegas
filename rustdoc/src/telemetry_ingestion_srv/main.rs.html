1//! Telemetry Ingestion Server
2//!
3//! Accepts telemetry data through http, stores the metadata in postgresql and the
4//! raw event payload in the object store.
5//!
6//! Env variables:
7//!  - `MICROMEGAS_SQL_CONNECTION_STRING` : to connect to postgresql
8//!  - `MICROMEGAS_OBJECT_STORE_URI` : to write the payloads
9//!  - `MICROMEGAS_API_KEYS` : (optional) JSON array of API keys
10//!  - `MICROMEGAS_OIDC_CONFIG` : (optional) OIDC configuration JSON
11

34#[derive(Parser, Debug)]
35#[clap(name = "Telemetry Ingestion Server")]
36#[clap(about = "Telemetry Ingestion Server", version, author)]
37struct Cli {
38    #[clap(long, default_value = "127.0.0.1:8081")]
39    listen_endpoint_http: SocketAddr,
40
41    /// Disable authentication (development mode only)
42    #[clap(long)]
43    disable_auth: bool,
44}
45
46/// Serves the HTTP ingestion service.
47///
48/// This function sets up the Axum router, applies middleware, and starts the HTTP server.
49async fn serve_http(
50    args: &Cli,
51    lake: DataLakeConnection,
52    auth_provider: Option<Arc<dyn AuthProvider>>,
53) -> Result<(), Box<dyn std::error::Error>> {
54    use axum::routing::get;
55
56    let service = Arc::new(WebIngestionService::new(lake));
57
58    // Health check endpoint (no auth required)
59    let health_router =
60        Router::new().route("/health", get(|| async { axum::http::StatusCode::OK }));
61
62    // Protected routes (require auth)
63    let mut protected_app = servers::ingestion::register_routes(Router::new())
64        .layer(DefaultBodyLimit::disable())
65        .layer(RequestBodyLimitLayer::new(100 * 1024 * 1024))
66        .layer(Extension(service));
67
68    // Add authentication middleware if enabled
69    let auth_enabled = auth_provider.is_some();
70    if let Some(provider) = auth_provider {
71        info!("Authentication enabled - API key and/or OIDC");
72        protected_app = protected_app.layer(middleware::from_fn(move |req, next| {
73            auth_middleware(provider.clone(), req, next)
74        }));
75    } else {
76        warn!("Authentication disabled - development mode only!");
77    }
78
79    // Merge health check (public) with protected routes
80    let mut app = health_router.merge(protected_app);
81
82    // Add observability middleware last (outer layer)
83    app = app.layer(middleware::from_fn(observability_middleware));
84
85    let listener = tokio::net::TcpListener::bind(args.listen_endpoint_http)
86        .await
87        .unwrap();
88    info!(
89        "serving on {} with authentication={}",
90        args.listen_endpoint_http, auth_enabled
91    );
92    axum::serve(
93        listener,
94        app.into_make_service_with_connect_info::<SocketAddr>(),
95    )
96    .await
97    .unwrap();
98
99    Ok(())
100}

102#[micromegas_main(interop_max_level = "info")]
103async fn main() -> Result<(), Box<dyn std::error::Error>> {
104    let args = Cli::parse();
105    let connection_string = std::env::var("MICROMEGAS_SQL_CONNECTION_STRING")
106        .with_context(|| "reading MICROMEGAS_SQL_CONNECTION_STRING")?;
107    let object_store_uri = std::env::var("MICROMEGAS_OBJECT_STORE_URI")
108        .with_context(|| "reading MICROMEGAS_OBJECT_STORE_URI")?;
109    let data_lake = connect_to_remote_data_lake(&connection_string, &object_store_uri).await?;
110
111    // Initialize authentication providers (same pattern as flight-sql-srv)
112    let auth_required = !args.disable_auth;
113    let auth_provider: Option<Arc<dyn AuthProvider>> = if auth_required {
114        // Initialize API key provider if configured
115        let api_key_provider = match std::env::var("MICROMEGAS_API_KEYS") {
116            Ok(keys_json) => {
117                let keyring = parse_key_ring(&keys_json)?;
118                info!("API key authentication enabled");
119                Some(Arc::new(ApiKeyAuthProvider::new(keyring)))
120            }
121            Err(_) => {
122                info!("MICROMEGAS_API_KEYS not set - API key auth disabled");
123                None
124            }
125        };
126
127        // Initialize OIDC provider if configured
128        let oidc_provider = match OidcConfig::from_env() {
129            Ok(config) => {
130                info!("Initializing OIDC authentication");
131                Some(Arc::new(OidcAuthProvider::new(config).await?))
132            }
133            Err(e) => {
134                info!("OIDC not configured ({e}) - OIDC auth disabled");
135                None
136            }
137        };
138
139        // Create multi-provider if either is configured
140        if api_key_provider.is_some() || oidc_provider.is_some() {
141            Some(Arc::new(MultiAuthProvider {
142                api_key_provider,
143                oidc_provider,
144            }) as Arc<dyn AuthProvider>)
145        } else {
146            return Err("Authentication required but no auth providers configured. \
147                 Set MICROMEGAS_API_KEYS or MICROMEGAS_OIDC_CONFIG, \
148                 or use --disable-auth for development"
149                .into());
150        }
151    } else {
152        info!("Authentication disabled (--disable_auth)");
153        None
154    };
155
156    serve_http(&args, data_lake, auth_provider).await?;
157    Ok(())
158}
