217        /// Automatically configure authentication from environment variables.
218        ///
219        /// Checks for authentication configuration in this order:
220        /// 1. API key authentication via `MICROMEGAS_INGESTION_API_KEY`
221        /// 2. OIDC client credentials via `MICROMEGAS_OIDC_TOKEN_ENDPOINT`,
222        ///    `MICROMEGAS_OIDC_CLIENT_ID`, and `MICROMEGAS_OIDC_CLIENT_SECRET`
223        /// 3. Falls back to no authentication (TrivialRequestDecorator)
224        ///
225        /// # Example
226        ///
227        /// ```rust,no_run
228        /// use micromegas_telemetry_sink::TelemetryGuardBuilder;
229        ///
230        /// // Set environment variable
231        /// unsafe {
232        ///     std::env::set_var("MICROMEGAS_INGESTION_API_KEY", "secret-key-123");
233        /// }
234        ///
235        /// // Builder automatically configures API key authentication
236        /// let _guard = TelemetryGuardBuilder::default()
237        ///     .with_auth_from_env()
238        ///     .build()
239        ///     .expect("Failed to build telemetry guard");
240        /// ```
241        #[must_use]
242        pub fn with_auth_from_env(mut self) -> Self {
243            use crate::api_key_decorator::ApiKeyRequestDecorator;
244            use crate::oidc_client_credentials_decorator::OidcClientCredentialsDecorator;
245
246            // Try API key authentication first
247            if let Ok(decorator) = ApiKeyRequestDecorator::from_env() {
248                info!("Configured telemetry sink with API key authentication");
249                self.telemetry_make_request_decorator = Box::new(move || Arc::new(decorator));
250                return self;
251            }
252
253            // Try OIDC client credentials authentication
254            if let Ok(decorator) = OidcClientCredentialsDecorator::from_env() {
255                info!("Configured telemetry sink with OIDC client credentials authentication");
256                self.telemetry_make_request_decorator = Box::new(move || Arc::new(decorator));
257                return self;
258            }
259
260            // No authentication configured - use trivial decorator (no-op)
261            info!(
262                "Telemetry sink authentication not configured - sending unauthenticated requests"
263            );
264            self
265        }

308        pub fn build(self) -> anyhow::Result<TelemetryGuard> {
309            let target_max_level: Vec<_> = self
310                .target_max_levels
311                .into_iter()
312                .filter(|(key, _val)| key != "MAX_LEVEL")
313                .map(|(key, val)| {
314                    (
315                        key,
316                        LevelFilter::from_str(val.as_str()).unwrap_or(LevelFilter::Off),
317                    )
318                })
319                .collect();
320
321            let guard = {
322                lazy_static::lazy_static! {
323                    static ref GLOBAL_WEAK_GUARD: Mutex<Weak<TracingSystemGuard>> = Mutex::new(Weak::new());
324                }
325                let mut weak_guard = GLOBAL_WEAK_GUARD.lock().unwrap();
326                let weak = &mut *weak_guard;
327
328                if let Some(arc) = weak.upgrade() {
329                    arc
330                } else {
331                    let mut sinks: Vec<(LevelFilter, BoxedEventSink)> = vec![];
332                    let telemetry_sink_url = self
333                        .telemetry_sink_url
334                        .or_else(|| std::env::var("MICROMEGAS_TELEMETRY_URL").ok())
335                        .filter(|url| !url.trim().is_empty());
336
337                    if let Some(url) = telemetry_sink_url {
338                        let metadata_retry = self.telemetry_metadata_retry.unwrap_or_else(|| {
339                            tokio_retry::strategy::ExponentialBackoff::from_millis(10).take(3)
340                        });
341                        let blocks_retry = self.telemetry_blocks_retry.unwrap_or_else(|| {
342                            tokio_retry::strategy::ExponentialBackoff::from_millis(10).take(3)
343                        });
344                        sinks.push((
345                            self.telemetry_sink_max_level,
346                            Box::new(HttpEventSink::new(
347                                &url,
348                                self.max_queue_size,
349                                metadata_retry,
350                                blocks_retry,
351                                self.telemetry_make_request_decorator,
352                            )),
353                        ));
354                    }
355                    if self.local_sink_enabled {
356                        sinks.push((self.local_sink_max_level, Box::new(LocalEventSink::new())));
357                    }
358                    let mut extra_sinks = self.extra_sinks.into_values().collect();
359                    sinks.append(&mut extra_sinks);
360
361                    let sink: BoxedEventSink = Box::new(CompositeSink::new(
362                        sinks,
363                        target_max_level,
364                        self.max_level_override,
365                    ));
366
367                    // the composite sink inits micromegas_tracing::levels::set_max_level, which install_log_interop needs
368                    if self.install_log_capture {
369                        install_log_interop(self.interop_max_level_override);
370                    }
371                    if self.install_tracing_capture {
372                        install_tracing_interop(self.interop_max_level_override);
373                    }
374
375                    let arc = Arc::<TracingSystemGuard>::new(TracingSystemGuard::new(
376                        self.logs_buffer_size,
377                        self.metrics_buffer_size,
378                        self.threads_buffer_size,
379                        sink.into(),
380                        self.process_properties,
381                        std::env::var("MICROMEGAS_ENABLE_CPU_TRACING")
382                            .map(|v| v == "true")
383                            .unwrap_or(false), // Default to disabled for minimal overhead
384                    )?);
385
386                    if self.system_metrics_enabled {
387                        spawn_system_monitor();
388                    }
389
390                    *weak = Arc::<TracingSystemGuard>::downgrade(&arc);
391                    arc
392                }
393            };
394            // order here is important
395            Ok(TelemetryGuard {
396                _guard: guard,
397                _thread_guard: TracingThreadGuard::new(),
398            })
399        }
400    }

402    pub struct TelemetryGuard {
403        // note we rely here on the drop order being the same as the declaration order
404        _thread_guard: TracingThreadGuard,
405        _guard: Arc<TracingSystemGuard>,
406    }

408    impl TelemetryGuard {
409        pub fn new() -> anyhow::Result<Self> {
410            TelemetryGuardBuilder::default().build()
411        }
